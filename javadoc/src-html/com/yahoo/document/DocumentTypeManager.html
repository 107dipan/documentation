020/**
021 * The DocumentTypeManager keeps track of the document types registered in
022 * the Vespa common repository.
023 * <p>
024 * The DocumentTypeManager is also responsible for registering a FieldValue
025 * factory for each data type a field can have. The Document object
026 * uses this factory to serialize and deserialize the various datatypes.
027 * The factory could also be used to expand the functionality of various
028 * datatypes, for instance displaying the data type in human-readable form
029 * or as XML.
030 *
031 * @author Thomas Gundersen
032 */
033public class DocumentTypeManager {
034
035    private final static Logger log = Logger.getLogger(DocumentTypeManager.class.getName());
036    private ConfigSubscriber subscriber;

038    // *Configured data types* (not built-in/primitive) indexed by their id
039    //
040    // *Primitive* data types are always available and have a single id.
041    // 
042    // *Built-in dynamic* types: The tensor type.
043    // Any tensor type has the same id and is always available just like primitive types.
044    // However, unlike primitive types, each tensor type is a separate DataType instance
045    // (which carries additional type information (detailedType) supplied at runtime).
046    // The reason for this is that we want the data type to stay the same when we change the detailed tensor type
047    // to maintain compatibility on (some) tensor type changes.
048    private Map<Integer, DataType> dataTypes = new LinkedHashMap<>();
049    private Map<DataTypeName, DocumentType> documentTypes = new LinkedHashMap<>();
050    private AnnotationTypeRegistry annotationTypeRegistry = new AnnotationTypeRegistry();
051

073    private void registerDefaultDataTypes() {
074        DocumentType superDocType = DataType.DOCUMENT;
075        dataTypes.put(superDocType.getId(), superDocType);
076        documentTypes.put(superDocType.getDataTypeName(), superDocType);
077
078        Class<? extends DataType> dataTypeClass = DataType.class;
079        for (java.lang.reflect.Field field : dataTypeClass.getFields()) {
080            if (Modifier.isStatic(field.getModifiers()) && Modifier.isFinal(field.getModifiers())) {
081                if (DataType.class.isAssignableFrom(field.getType())) {
082                    try {
083                        //these are all static final DataTypes listed in DataType:
084                        DataType type = (DataType) field.get(null);
085                        register(type);
086                    } catch (IllegalAccessException e) {
087                        //ignore
088                    }
089                }
090            }
091        }
092        for (AnnotationType type : AnnotationTypes.ALL_TYPES) {
093            annotationTypeRegistry.register(type);
094        }
095    }

097    public boolean hasDataType(String name) {
098        if (name.startsWith("tensor(")) return true; // built-in dynamic: Always present
099        for (DataType type : dataTypes.values()) {
100            if (type.getName().equalsIgnoreCase(name)) {
101                return true;
102            }
103        }
104        return false;
105    }
106
107    public boolean hasDataType(int code) {
108        if (code == DataType.tensorDataTypeCode) return true; // built-in dynamic: Always present
109        return dataTypes.containsKey(code);
110    }

112    public DataType getDataType(String name) {
113        if (name.startsWith("tensor(")) // built-in dynamic
114            return new TensorDataType(TensorType.fromSpec(name));
115        
116        List<DataType> foundTypes = new ArrayList<>();
117        for (DataType type : dataTypes.values()) {
118            if (type.getName().equalsIgnoreCase(name)) {
119                foundTypes.add(type);
120            }
121        }
122        if (foundTypes.isEmpty()) {
123            throw new IllegalArgumentException("No datatype named " + name);
124        } else if (foundTypes.size() == 1) {
125            return foundTypes.get(0);
126        } else {
127            //the found types are probably documents or structs, sort them by type
128            Collections.sort(foundTypes, new Comparator<DataType>() {
129                public int compare(DataType first, DataType second) {
130                    if (first instanceof StructuredDataType && !(second instanceof StructuredDataType)) {
131                        return 1;
132                    } else if (!(first instanceof StructuredDataType) && second instanceof StructuredDataType) {
133                        return -1;
134                    } else {
135                        return 0;
136                    }
137                }
138            });
139        }
140        return foundTypes.get(0);
141    }

145    /**
146     * Return a data type instance
147     * 
148     * @param code the code of the data type to return, which must be either built in or present in this manager
149     * @param detailedType detailed type information, or the empty string if none
150     * @return the appropriate DataType instance
151     */
152    public DataType getDataType(int code, String detailedType) {
153        if (code == DataType.tensorDataTypeCode) // built-in dynamic
154            return new TensorDataType(TensorType.fromSpec(detailedType));
155
156        DataType type = dataTypes.get(code);
157        if (type == null) {
158            StringBuilder types=new StringBuilder();
159            for (Integer key : dataTypes.keySet()) {
160                types.append(key).append(" ");
161            }
162            throw new IllegalArgumentException("No datatype with code " + code + ". Registered type ids: " + types);
163        } else {
164            return type;
165        }
166    }

168    DataType getDataTypeAndReturnTemporary(int code, String detailedType) {
169        if (hasDataType(code)) {
170            return getDataType(code, detailedType);
171        }
172        return new TemporaryDataType(code, detailedType);
173    }

184    /**
185     * Register a single datatype. Re-registering an existing, but equal, datatype is ok.
186     * 
187     * @param type The datatype to register
188     */
189    void registerSingleType(DataType type) {
190        if (type instanceof TensorDataType) return; // built-in dynamic: Created on the fly
191        if (dataTypes.containsKey(type.getId())) {
192            DataType existingType = dataTypes.get(type.getId());
193            if (((type instanceof TemporaryDataType) || (type instanceof TemporaryStructuredDataType))
194                && !((existingType instanceof TemporaryDataType) || (existingType instanceof TemporaryStructuredDataType))) {
195                //we're trying to register a temporary type over a permanent one, don't do that:
196                return;
197            } else if ((existingType == type || existingType.equals(type))
198                    && !(existingType instanceof TemporaryDataType)
199                    && !(type instanceof TemporaryDataType)
200                    && !(existingType instanceof TemporaryStructuredDataType)
201                    && !(type instanceof TemporaryStructuredDataType)) { // Shortcut to improve speed
202                // Oki. Already registered.
203                return;
204            } else if (type instanceof DocumentType && dataTypes.get(type.getId()) instanceof DocumentType) {
205                /*
206                DocumentType newInstance = (DocumentType) type;
207                DocumentType oldInstance = (DocumentType) dataTypes.get(type.getId());
208                TODO fix tests
209                */
210                log.warning("Document type " + existingType + " is not equal to document type attempted registered " + type
211                        + ", but have same name. OVERWRITING TYPE as many tests currently does this. "
212                        + "Fix tests so we can throw exception here.");
213                //throw new IllegalStateException("Datatype " + existingType + " is not equal to datatype attempted registered "
214                //        + type + ", but already uses id " + type.getId());
215            } else if ((existingType instanceof TemporaryDataType) || (existingType instanceof TemporaryStructuredDataType)) {
216                //removing temporary type to be able to register correct type
217                dataTypes.remove(existingType.getId());
218            } else {
219                throw new IllegalStateException("Datatype " + existingType + " is not equal to datatype attempted registered "
220                                                + type + ", but already uses id " + type.getId());
221            }
222        }
223
224        if (type instanceof DocumentType) {
225            DocumentType docType = (DocumentType) type;
226            if (docType.getInheritedTypes().size() == 0) {
227                docType.inherit(documentTypes.get(new DataTypeName("document")));
228            }
229            documentTypes.put(docType.getDataTypeName(), docType);
230        }
231        dataTypes.put(type.getId(), type);
232        type.setRegistered();
233    }

316    void replaceTemporaryTypes() {
317        for (DataType type : dataTypes.values()) {
318            List<DataType> seenStructs = new LinkedList<>();
319            replaceTemporaryTypes(type, seenStructs);
320        }
321    }
322
323    private void replaceTemporaryTypes(DataType type, List<DataType> seenStructs) {
324        if (type instanceof WeightedSetDataType) {
325            replaceTemporaryTypesInWeightedSet((WeightedSetDataType) type, seenStructs);
326        } else if (type instanceof MapDataType) {
327            replaceTemporaryTypesInMap((MapDataType) type, seenStructs);
328        } else if (type instanceof CollectionDataType) {
329            replaceTemporaryTypesInCollection((CollectionDataType) type, seenStructs);
330        } else if (type instanceof StructDataType) {
331            replaceTemporaryTypesInStruct((StructDataType) type, seenStructs);
332        } else if (type instanceof PrimitiveDataType) {
333            //OK because these types are always present
334        } else if (type instanceof AnnotationReferenceDataType) {
335            //OK because this type is always present
336        } else if (type instanceof DocumentType) {
337            //OK because this type is always present
338        } else if (type instanceof TensorDataType) {
339            //OK because this type is always present
340        } else if (type instanceof ReferenceDataType) {
341            replaceTemporaryTypeInReference((ReferenceDataType) type);
342        } else if (type instanceof TemporaryDataType) {
343            throw new IllegalStateException("TemporaryDataType registered in DocumentTypeManager, BUG!!");
344        } else {
345            log.warning("Don't know how to replace temporary data types in " + type);
346        }
347    }
348
349    @SuppressWarnings("deprecation")
350    private void replaceTemporaryTypesInStruct(StructDataType structDataType, List<DataType> seenStructs) {
351        seenStructs.add(structDataType);
352        for (Field field : structDataType.getFieldsThisTypeOnly()) {
353            DataType fieldType = field.getDataType();
354            if (fieldType instanceof TemporaryDataType) {
355                field.setDataType(getDataType(fieldType.getCode(), ((TemporaryDataType)fieldType).getDetailedType()));
356            } else {
357                if (!seenStructs.contains(fieldType)) {
358                    replaceTemporaryTypes(fieldType, seenStructs);
359                }
360            }
361        }
362    }
363
364    private void replaceTemporaryTypeInReference(ReferenceDataType referenceDataType) {
365        if (referenceDataType.getTargetType() instanceof TemporaryStructuredDataType) {
366            referenceDataType.setTargetType((DocumentType) getDataType(referenceDataType.getTargetType().getId()));
367        }
368        // TODO should we recursively invoke replaceTemporaryTypes for the target type? It should only ever be a doc type
369    }
370
371    private void replaceTemporaryTypesInCollection(CollectionDataType collectionDataType, List<DataType> seenStructs) {
372        if (collectionDataType.getNestedType() instanceof TemporaryDataType) {
373            collectionDataType.setNestedType(getDataType(collectionDataType.getNestedType().getCode(), ""));
374        } else {
375            replaceTemporaryTypes(collectionDataType.getNestedType(), seenStructs);
376        }
377    }
378
379    private void replaceTemporaryTypesInMap(MapDataType mapDataType, List<DataType> seenStructs) {
380        if (mapDataType.getValueType() instanceof TemporaryDataType) {
381            mapDataType.setValueType(getDataType(mapDataType.getValueType().getCode(), ""));
382        } else {
383            replaceTemporaryTypes(mapDataType.getValueType(), seenStructs);
384        }
385
386        if (mapDataType.getKeyType() instanceof TemporaryDataType) {
387            mapDataType.setKeyType(getDataType(mapDataType.getKeyType().getCode(), ""));
388        } else {
389            replaceTemporaryTypes(mapDataType.getKeyType(), seenStructs);
390        }
391    }
392
393    private void replaceTemporaryTypesInWeightedSet(WeightedSetDataType weightedSetDataType, List<DataType> seenStructs) {
394        if (weightedSetDataType.getNestedType() instanceof TemporaryDataType) {
395            weightedSetDataType.setNestedType(getDataType(weightedSetDataType.getNestedType().getCode(), ""));
396        } else {
397            replaceTemporaryTypes(weightedSetDataType.getNestedType(), seenStructs);
398        }
399    }
