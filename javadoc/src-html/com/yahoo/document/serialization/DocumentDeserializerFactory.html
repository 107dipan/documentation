002package com.yahoo.document.serialization;
003
004import com.yahoo.document.DocumentTypeManager;
005import com.yahoo.io.GrowableByteBuffer;
006
007/**
008 * Factory for creating document de-serializers tied to a document format.
009 *
010 * @author geirst
011 */
012public class DocumentDeserializerFactory {
013
014    /**
015     * Creates a de-serializer for the current head document format.
016     * This format is an extension of the 4.2 format.
017     */
018    public static DocumentDeserializer createHead(DocumentTypeManager manager, GrowableByteBuffer buf) {
019        return new VespaDocumentDeserializerHead(manager, buf);
020    }
021
022    /**
023     * Creates a de-serializer for the document format that was created on Vespa 4.2.
024     */
025    public static DocumentDeserializer create42(DocumentTypeManager manager, GrowableByteBuffer buf) {
026        return new VespaDocumentDeserializer42(manager, buf);
027    }
028
029    /**
030     * Creates a de-serializer for the document format that was created on Vespa 4.2.
031     */
032    public static DocumentDeserializer create42(DocumentTypeManager manager, GrowableByteBuffer buf, GrowableByteBuffer body) {
033        return new VespaDocumentDeserializer42(manager, buf, body);
034    }
035}
