002package com.yahoo.document.serialization;
003
004import com.yahoo.document.DataType;
005import com.yahoo.document.DocumentUpdate;
006import com.yahoo.document.update.AddValueUpdate;
007import com.yahoo.document.update.ArithmeticValueUpdate;
008import com.yahoo.document.update.AssignValueUpdate;
009import com.yahoo.document.update.ClearValueUpdate;
010import com.yahoo.document.update.FieldUpdate;

014/**
015 * Interface for writing document updates in custom serializers.
016 *
017 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
018 * @since 5.1.27
019 */
020public interface DocumentUpdateWriter {
021    public void write(DocumentUpdate update);
022    public void write(FieldUpdate update);
023    public void write(AddValueUpdate update, DataType superType);
024    public void write(MapValueUpdate update, DataType superType);
025    public void write(ArithmeticValueUpdate update);
026    public void write(AssignValueUpdate update, DataType superType);
027    public void write(RemoveValueUpdate update, DataType superType);
028    public void write(ClearValueUpdate clearValueUpdate, DataType superType);
029}
