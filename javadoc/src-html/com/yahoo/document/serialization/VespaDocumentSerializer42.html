066    public void write(FieldBase field, Document doc) {
067        //save the starting position in the buffer
068        int startPos = buf.position();
069
070        buf.putShort(Document.SERIALIZED_VERSION);
071
072        // Temporary length, fill in after serialization is done.
073        buf.putInt(0);
074
075        doc.getId().serialize(this);
076
077        byte contents = 0x01; // Indicating we have document type which we always have
078        if (doc.getHeader().getFieldCount() > 0) {
079            contents |= 0x2; // Indicate we have header
080        }
081        if (!headerOnly && doc.getBody().getFieldCount() > 0) {
082            contents |= 0x4; // Indicate we have a body
083        }
084        buf.put(contents);
085
086        doc.getDataType().serialize(this);
087
088        if (doc.getHeader().getFieldCount() > 0) {
089            doc.getHeader().serialize(doc.getDataType().getField("header"), this);
090        }
091
092        if (!headerOnly && doc.getBody().getFieldCount() > 0) {
093            doc.getBody().serialize(doc.getDataType().getField("body"), this);
094        }
095
096        int finalPos = buf.position();
097
098        buf.position(startPos + 2);
099        buf.putInt(finalPos - startPos - 2 - 4); // Don't include the length itself or the version
100        buf.position(finalPos);
101
102    }

221     * Write out the value of string field
222     *
223     * @param field - field description (name and data type)
224     * @param value - field value
225     */
226    public void write(FieldBase field, StringFieldValue value) {
227        byte[] stringBytes = createUTF8CharArray(value.getString());
228
229        byte coding = 0;
230        //Use bit 6 of "coding" to say whether span tree is available or not
231        if (!value.getSpanTrees().isEmpty()) {
232            coding |= 64;
233        }
234        buf.put(coding);
235        buf.putInt1_4Bytes(stringBytes.length + 1);
236
237        buf.put(stringBytes);
238        buf.put(((byte) 0));
239
240        Map<String, SpanTree> trees = value.getSpanTreeMap();
241        if ((trees != null) && !trees.isEmpty()) {
242            try {
243                //we don't support serialization of nested span trees, so this is safe:
244                bytePositions = calculateBytePositions(value.getString());
245                //total length. record position and go back here if necessary:
246                int posBeforeSize = buf.position();
247                buf.putInt(0);
248                buf.putInt1_2_4Bytes(trees.size());
249
250                for (SpanTree tree : trees.values()) {
251                    try {
252                        write(tree);
253                    } catch (SerializationException e) {
254                        throw e;
255                    } catch (RuntimeException e) {
256                        throw new SerializationException("Exception thrown while serializing span tree '" +
257                                                         tree.getName() + "'; string='" + value.getString() + "'", e);
258                    }
259                }
260                int endPos = buf.position();
261                buf.position(posBeforeSize);
262                buf.putInt(endPos - posBeforeSize - 4); //length shall exclude itself
263                buf.position(endPos);
264            } finally {
265                bytePositions = null;
266            }
267        }
268    }

293    /**
294     * Write out the value of struct field
295     *
296     * @param field - field description (name and data type)
297     * @param s     - field value
298     */
299    public void write(FieldBase field, Struct s) {
300        // Serialize all parts first.. As we need to know length before starting
301        // Serialize all the fields.
302
303        //keep the buffer we're serializing everything into:
304        GrowableByteBuffer bigBuffer = buf;
305
306        //create a new buffer and serialize into that for a while:
307        GrowableByteBuffer buffer = new GrowableByteBuffer(4096, 2.0f);
308        buf = buffer;
309
310        List<Integer> fieldIds = new LinkedList<>();
311        List<java.lang.Integer> fieldLengths = new LinkedList<>();
312
313        for (Map.Entry<Field, FieldValue> value : s.getFields()) {
314
315            int startPos = buffer.position();
316            value.getValue().serialize(value.getKey(), this);
317
318            fieldLengths.add(buffer.position() - startPos);
319            fieldIds.add(value.getKey().getId(s.getVersion()));
320        }
321
322        // Switch buffers again:
323        buffer.flip();
324        buf = bigBuffer;
325
326        int uncompressedSize = buffer.remaining();
327        Compressor.Compression compression =
328            s.getDataType().getCompressor().compress(buffer.getByteBuffer().array(), buffer.remaining());
329
330        // Actual serialization starts here.
331        int lenPos = buf.position();
332        putInt(null, 0); // Move back to this after compression is done.
333        buf.put(compression.type().getCode());
334
335        if (compression.data() != null && compression.type().isCompressed()) {
336            buf.putInt2_4_8Bytes(uncompressedSize);
337        }
338
339        buf.putInt1_4Bytes(s.getFieldCount());
340
341        for (int i = 0; i < s.getFieldCount(); ++i) {
342            putInt1_4Bytes(null, fieldIds.get(i));
343            putInt2_4_8Bytes(null, fieldLengths.get(i));
344        }
345
346        int pos = buf.position();
347        if (compression.data() != null) {
348            put(null, compression.data());
349        } else {
350            put(null, buffer.getByteBuffer());
351        }
352        int dataLength = buf.position() - pos;
353
354        int posNow = buf.position();
355        buf.position(lenPos);
356        putInt(null, dataLength);
357        buf.position(posNow);
358    }

371     * Write out the value of weighted set field
372     *
373     * @param field - field description (name and data type)
374     * @param ws    - field value
375     */
376    public <T extends FieldValue> void write(FieldBase field, WeightedSet<T> ws) {
377        WeightedSetDataType type = ws.getDataType();
378        putInt(null, type.getNestedType().getId());
379        putInt(null, ws.size());
380
381        Iterator<T> it = ws.fieldValueIterator();
382        while (it.hasNext()) {
383            FieldValue key = it.next();
384            java.lang.Integer value = ws.get(key);
385            int sizePos = buf.position();
386            putInt(null, 0);
387            int startPos = buf.position();
388            key.serialize(this);
389            putInt(null, value);
390            int finalPos = buf.position();
391            int size = finalPos - startPos;
392            buf.position(sizePos);
393            putInt(null, size);
394            buf.position(finalPos);
395        }
396
397    }

428    public void write(Annotation annotation) {
429        buf.putInt(annotation.getType().getId());  //name hash
430
431        byte features = 0;
432        if (annotation.isSpanNodeValid()) {
433            features |= ((byte) 1);
434        }
435        if (annotation.hasFieldValue()) {
436            features |= ((byte) 2);
437        }
438        buf.put(features);
439
440        int posBeforeSize = buf.position();
441        buf.putInt1_2_4BytesAs4(0);
442
443        //write ID of span node:
444        if (annotation.isSpanNodeValid()) {
445            int spanNodeId = annotation.getSpanNode().getScratchId();
446            if (spanNodeId >= 0) {
447                buf.putInt1_2_4Bytes(spanNodeId);
448            } else {
449                throw new SerializationException("Could not serialize annotation, associated SpanNode not found (" + annotation + ")");
450            }
451        }
452
453        //write annotation value:
454        if (annotation.hasFieldValue()) {
455            buf.putInt(annotation.getType().getDataType().getId());
456            annotation.getFieldValue().serialize(this);
457        }
458
459        int end = buf.position();
460        buf.position(posBeforeSize);
461        buf.putInt1_2_4BytesAs4(end - posBeforeSize - 4);
462        buf.position(end);
463    }

465    public void write(SpanTree tree) {
466        //we don't support serialization of nested span trees:
467        if (spanNodeCounter >= 0) {
468            throw new SerializationException("Serialization of nested SpanTrees is not supported.");
469        }
470
471        //we're going to write a new SpanTree, create a new Map for nodes:
472        spanNodeCounter = 0;
473
474        //make sure tree is consistent before continuing:
475        tree.cleanup();
476
477        try {
478            new StringFieldValue(tree.getName()).serialize(this);
479
480            write(tree.getRoot());
481            {
482                //add all annotations to temporary list and sort it, to get predictable serialization
483                List<Annotation> tmpAnnotationList = new ArrayList<Annotation>(tree.numAnnotations());
484                for (Annotation annotation : tree) {
485                    tmpAnnotationList.add(annotation);
486                }
487                Collections.sort(tmpAnnotationList);
488
489                int annotationCounter = 0;
490                //add all annotations to map here, in case of back-references:
491                for (Annotation annotation : tmpAnnotationList) {
492                    annotation.setScratchId(annotationCounter++);
493                }
494
495                buf.putInt1_2_4Bytes(tmpAnnotationList.size());
496                for (Annotation annotation : tmpAnnotationList) {
497                    write(annotation);
498                }
499            }
500        } finally {
501            //we're done, let's set these to null to save memory and prevent madness:
502            spanNodeCounter = -1;
503        }
504    }

506    public void write(SpanNode spanNode) {
507        if (spanNodeCounter >= 0) {
508            spanNode.setScratchId(spanNodeCounter++);
509        }
510        if (spanNode instanceof Span) {
511            write((Span) spanNode);
512        } else if (spanNode instanceof AlternateSpanList) {
513            write((AlternateSpanList) spanNode);
514        } else if (spanNode instanceof SpanList) {
515            write((SpanList) spanNode);
516        } else {
517            throw new IllegalStateException("BUG!! Unable to serialize " + spanNode);
518        }
519    }

521    public void write(Span span) {
522        buf.put(Span.ID);
523
524        if (bytePositions != null) {
525            int byteFrom = bytePositions[span.getFrom()];
526            int byteLength = bytePositions[span.getFrom() + span.getLength()] - byteFrom;
527
528            buf.putInt1_2_4Bytes(byteFrom);
529            buf.putInt1_2_4Bytes(byteLength);
530        } else {
531            throw new SerializationException("Cannot serialize Span " + span + ", no access to parent StringFieldValue.");
532        }
533    }

535    public void write(SpanList spanList) {
536        buf.put(SpanList.ID);
537        buf.putInt1_2_4Bytes(spanList.numChildren());
538        Iterator<SpanNode> children = spanList.childIterator();
539        while (children.hasNext()) {
540            write(children.next());
541        }
542    }

544    public void write(AlternateSpanList altSpanList) {
545        buf.put(AlternateSpanList.ID);
546        buf.putInt1_2_4Bytes(altSpanList.getNumSubTrees());
547        for (int i = 0; i < altSpanList.getNumSubTrees(); i++) {
548            buf.putDouble(altSpanList.getProbability(i));
549            buf.putInt1_2_4Bytes(altSpanList.numChildren(i));
550            Iterator<SpanNode> children = altSpanList.childIterator(i);
551            while (children.hasNext()) {
552                write(children.next());
553            }
554        }
555    }

591        if (superType instanceof ArrayDataType) {
592            CollectionDataType type = (CollectionDataType) superType;
593            IntegerFieldValue index = (IntegerFieldValue) update.getValue();
594            index.serialize(this);
595            putInt(null, update.getUpdate().getValueUpdateClassID().id);
596            update.getUpdate().serialize(this, type.getNestedType());
597        } else if (superType instanceof WeightedSetDataType) {
598            writeValue(this, ((CollectionDataType)superType).getNestedType(), update.getValue());
599            putInt(null, update.getUpdate().getValueUpdateClassID().id);
600            update.getUpdate().serialize(this, DataType.INT);
601        } else {
602            throw new SerializationException("MapValueUpdate only works for arrays and weighted sets");
603        }
604    }

632    /**
633     * Returns the serialized size of the given {@link Document}. Please note that this method performs actual
634     * serialization of the document, but simply return the size of the final {@link GrowableByteBuffer}. If you need
635     * the buffer itself, do NOT use this method.
636     *
637     * @param doc The Document whose size to calculate.
638     * @return The size in bytes.
639     */
640    public static long getSerializedSize(Document doc) {
641        DocumentSerializer serializer = new VespaDocumentSerializerHead(new GrowableByteBuffer());
642        serializer.write(doc);
643        return serializer.getBuf().position();
644    }

646    private static void writeValue(VespaDocumentSerializer42 serializer, DataType dataType, Object value) {
647        FieldValue fieldValue;
648        if (value instanceof FieldValue) {
649            fieldValue = (FieldValue)value;
650        } else {
651            fieldValue = dataType.createFieldValue(value);
652        }
653        fieldValue.serialize(serializer);
