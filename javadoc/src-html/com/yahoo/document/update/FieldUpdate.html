019/**
020 * <p>A field update holds a list of value updates that will be applied atomically to a field in a document.</p>
021 * <p>To create a field update that contains only a single value update, use the static factory methods provided by this
022 * class.</p> <p>Example:</p>
023 * <pre>
024 * FieldUpdate clearFieldUpdate = FieldUpdate.createClearField(field);
025 * </pre>
026 * <p>It is also possible to create a field update that holds more than one value update.</p>
027 * <p>Example:</p>
028 * <pre>
029 * FieldUpdate fieldUpdate = FieldUpdate.create(field);
030 * ValueUpdate incrementValue = ValueUpdate.createIncrement("foo", 130d);
031 * ValueUpdate addValue = ValueUpdate.createAdd("bar", 100);
032 * fieldUpdate.addValueUpdate(incrementValue);
033 * fieldUpdate.addValueUpdate(addValue);</pre>
034 * <p>Note that the addValueUpdate() method returns a reference to itself to support chaining, so the last two
035 * lines could be written as one:</p>
036 * <pre>
037 * fieldUpdate.addValueUpdate(incrementValue).addValueUpdate(addValue);
038 * </pre>
039 * <p>Note also that the second example above is equivalent to:</p>
040 * <pre>
041 * FieldUpdate fieldUpdate = FieldUpdate.createIncrement(field, "foo", 130d);
042 * ValueUpdate addValue = ValueUpdate.createAdd("bar", 100);
043 * fieldUpdate.addValueUpdate(addValue);
044 * </pre>
045 * <p>Note that even though updates take fields as arguments, those fields are not necessarily a field of a document
046 * type - any name/value pair which existing in an updatable structure can be addressed by creating the Fields as
047 * needed. For example:
048 * <pre>
049 * FieldUpdate field=FieldUpdate.createIncrement(new Field("myattribute",DataType.INT),130);
050 * </pre>
051 *
052 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
053 * @see com.yahoo.document.update.ValueUpdate
054 * @see com.yahoo.document.DocumentUpdate
055 */
056public class FieldUpdate {

107    /**
108     * Applies this field update.
109     *
110     * @param doc the document to apply the update to
111     * @return a reference to itself
112     */
113    public FieldUpdate applyTo(Document doc) {
114        for (ValueUpdate vupd : valueUpdates) {
115            DataType dataType = field.getDataType();
116            FieldValue oldValue = doc.getFieldValue(field);
117            boolean existed = (oldValue != null);
118
119            if (!existed) {
120                oldValue = dataType.createFieldValue();
121            }
122
123            FieldValue newValue = vupd.applyTo(oldValue);
124
125            if (newValue == null) {
126                if (existed) {
127                    doc.removeFieldValue(field);
128                }
129            } else {
130                doc.setFieldValue(field, newValue);
131            }
132        }
133        return this;
134    }

136    /**
137     * Adds a value update to the list of value updates.
138     *
139     * @param valueUpdate the ValueUpdate to add
140     * @return a reference to itself
141     * @throws IllegalArgumentException if the data type of the value update is not equal to the data type of this field
142     */
143    public FieldUpdate addValueUpdate(ValueUpdate valueUpdate) {
144        valueUpdate.checkCompatibility(field.getDataType());  //will throw exception
145        valueUpdates.add(valueUpdate);
146        return this;
147    }

149    /**
150     * Adds a value update to the list of value updates.
151     *
152     * @param index the index where this value update should be added
153     * @param valueUpdate the ValueUpdate to add
154     * @return a reference to itself
155     * @throws IllegalArgumentException if the data type of the value update is not equal to the data type of this field
156     */
157    public FieldUpdate addValueUpdate(int index, ValueUpdate valueUpdate) {
158        valueUpdate.checkCompatibility(field.getDataType());  //will throw exception
159        valueUpdates.add(index, valueUpdate);
160        return this;
161    }

163    /**
164     * Adds a list of value updates to the list of value updates.
165     *
166     * @param valueUpdates a list containing the value updates to add
167     * @return a reference to itself
168     * @throws IllegalArgumentException if the data type of the value update is not equal to the data type of this field
169     */
170    public FieldUpdate addValueUpdates(List<ValueUpdate> valueUpdates) {
171        for (ValueUpdate vupd : valueUpdates) {
172            addValueUpdate(vupd);
173        }
174        return this;
175    }

235    /**
236     * Get value updates with the specified valueUpdateClassID. The caller gets ownership of the returned list, and
237     * subsequent modifications to the list does not change the state of this object.
238     *
239     * @param classID the classID of ValueUpdates to return
240     * @return a List of ValueUpdates of the specified classID (possibly empty, but not null)
241     */
242    public List<ValueUpdate> getValueUpdates(ValueUpdate.ValueUpdateClassID classID) {
243        List<ValueUpdate> updateList = new ArrayList<>();
244        for (ValueUpdate vupd : valueUpdates) {
245            if (vupd.getValueUpdateClassID() == classID) {
246                updateList.add(vupd);
247            }
248        }
249        return updateList;
250    }
251
252    /**
253     * Returns whether this field update contains (at least) one update of the given type
254     *
255     * @param classID the classID of ValueUpdates to check for
256     * @return true if there is at least one value update of the given type in this
257     */
258    public boolean hasValueUpdate(ValueUpdate.ValueUpdateClassID classID) {
259        for (ValueUpdate vupd : valueUpdates) {
260            if (vupd.getValueUpdateClassID() == classID) {
261                return true;
262            }
263        }
264        return false;
265    }

268     * Returns whether or not this field update contains any value updates.

272    public boolean isEmpty() {
273        return valueUpdates.isEmpty();
274    }

276    /**
277     * Adds all the {@link ValueUpdate}s of the given FieldUpdate to this. If the given FieldUpdate refers to a
278     * different {@link Field} than this, this method throws an exception.
279     *
280     * @param update The update whose content to add to this.
281     * @throws IllegalArgumentException If the {@link Field} of the given FieldUpdate does not match this.
282     */
283    public void addAll(FieldUpdate update) {
284        if (update == null) {
285            return;
286        }
287        if (!field.equals(update.field)) {
288            throw new IllegalArgumentException("Expected " + field + ", got " + update.field + ".");
289        }
290        addValueUpdates(update.valueUpdates);
291    }
