002package com.yahoo.document;
003
004import com.yahoo.document.idstring.*;
005
006/**
007 * A bucket id contains bit used for various purposes. In most use cases, these can use the default
008 * settings, but the number of bits used for the different purposes is configurable, to allow for
009 * special uses.
010 *
011 * Because of this, bucket ids cannot be generated without knowing how the bucket id is configured to
012 * be put together, so all bucket ids must be generated by this factory class.
013 *
014 * For more information about what the sub parts of a bucket id actually is, read the bucket splitting documentation.
015 *
016 * @author <a href="mailto:humbe@yahoo-inc.com">H&aring;kon Humberset</a>
017 */
018public class BucketIdFactory {
019
020    private final int gidBits;
021    private final int locationBits;
022    private final int countBits;
023
024    private final long initialCount;
025    private final long locationMask;
026    private final long gidMask;
027
028    /** Create a factory, using the default configuration. */
029    public BucketIdFactory() {
030        this(32, 26, 6);
031    }
032
033    /**
034     * Create a factory, using the provided configuration.
035     * @param gidBits How many bits that are used to specify gidbits.
036     */
037    public BucketIdFactory(int locationBits, int gidBits, int countBits) {
038        this.locationBits = locationBits;
039        this.gidBits = gidBits;
040        this.countBits = countBits;
041        initialCount = 58l << (64 - countBits);
042        locationMask = 0xFFFFFFFFFFFFFFFFl >>> (64 - getLocationBitCount());
043        gidMask = ((0xFFFFFFFFFFFFFFFFl >>> getLocationBitCount()) << (getLocationBitCount() + BucketId.COUNT_BITS)) >>> BucketId.COUNT_BITS;
044
045    }
046
047    /**
048     * Create a factory, with parameters gotten from configuration.
049     * TODO: Not implemented yet
050     * @param configId The config id from where to get config.
051     */
052    public BucketIdFactory(String configId) {
053        this(32, 26, 6);
054    }
055
056    /** @return Get number of bits used for storing of LSB part of location.*/
057    public int getLocationBitCount() { return locationBits; }
058
059    /** @return Get number of bits used to specify gid. */
060    public int getGidBitCount() { return gidBits; }
061
062    /** @return Get number of bits used to store bit count used. */
063    public int getCountBitCount() { return countBits; }
064
065    /**
066     * Get the gid bit contribution in the bucket id, shifted to the correct
067     * position in the id.
068     *
069     * @param gid The gid we need to calculate contribution from.
070     * @return A mask to or with the bucket id to get the bit set.
071     */
072    private long getGidContribution(byte[] gid) {
073        long gidbits = 0;
074        for (int i=4; i<12; ++i) {
075            gidbits <<= 8;
076            long tall = gid[15 - i] & 0xFFl;
077            assert(tall >= 0 && tall <= 255);
078            gidbits |= tall;
079        }
080        return gidbits & gidMask;
081    }
082
083    /**
084     * Get the bucket id for a given document.
085     *
086     * @param doc The doc.
087     * @return The bucket id.
088     */
089    public BucketId getBucketId(DocumentId doc) {
090        long location = doc.getScheme().getLocation();
091        byte[] gid = doc.getGlobalId();
092
093        long gidContribution = getGidContribution(gid);
094
095        IdString.GidModifier gm = doc.getScheme().getGidModifier();
096        if (gm != null && gm.usedBits != 0) {
097            gidContribution &= (0xFFFFFFFFFFFFFFFFl << (gm.usedBits + getLocationBitCount()));
098            gidContribution |= (gm.value << getLocationBitCount());
099        }
100
101        return new BucketId(64 - BucketId.COUNT_BITS, initialCount | (gidMask & gidContribution) | (locationMask & location));
102    }
103
104}
