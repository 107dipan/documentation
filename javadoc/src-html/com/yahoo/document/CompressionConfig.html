002package com.yahoo.document;
003
004import com.yahoo.compress.CompressionType;
005
006import java.io.Serializable;
007
008public class CompressionConfig implements Serializable {
009
010    public CompressionConfig(CompressionType type,
011                             int level,
012                             float threshold,
013                             long minSize)
014    {
015        this.type = type;
016        this.compressionLevel = level;
017        this.threshold = threshold;
018        this.minsize = minSize;
019    }
020
021    public CompressionConfig() {
022        this(CompressionType.NONE, 9, 95, 0);
023    }
024
025    public CompressionConfig(CompressionType type) {
026        this(type, 9, 95, 0);
027    }
028
029    public CompressionConfig(CompressionType type, int level, float threshold) {
030        this(type, level, threshold, 0);
031    }
032
033    public final CompressionType type;
034    public int compressionLevel;
035    public float threshold;
036    public final long minsize;
037
038    /** get a multiplier for comparing compressed and original size */
039    public float thresholdFactor() { return 0.01f * threshold; }
040}
