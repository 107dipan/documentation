057    @Override
058    public DocumentType clone() {
059        DocumentType type = (DocumentType) super.clone();
060        type.headerType = headerType.clone();
061        type.bodyType = bodyType.clone();
062        type.inherits = new ArrayList<>(inherits.size());
063        for (DocumentType inherited : inherits) {
064            type.inherits.add(inherited);
065        }
066        return type;
067    }

100    @Override
101    protected void register(DocumentTypeManager manager, List<DataType> seenTypes) {
102        seenTypes.add(this);
103        for (DocumentType type : getInheritedTypes()) {
104            if (!seenTypes.contains(type)) {
105                type.register(manager, seenTypes);
106            }
107        }
108        // Get parent fields into fields specified in this type
109        StructDataType header = headerType.clone();
110        StructDataType body = bodyType.clone();
111
112        header.clearFields();
113        body.clearFields();
114
115        for (Field field : fieldSet()) {
116            (field.isHeader() ? header : body).addField(field);
117        }
118        headerType.assign(header);
119        bodyType.assign(body);
120
121        if (!seenTypes.contains(headerType)) {
122            headerType.register(manager, seenTypes);
123        }
124        if (!seenTypes.contains(bodyType)) {
125            bodyType.register(manager, seenTypes);
126        }
127        manager.registerSingleType(this);
128    }

130    /**
131     * Check if this document type has the given name,
132     * or inherits from a type with that name.
133     */
134    public boolean isA(String docTypeName) {
135        if (getName().equalsIgnoreCase(docTypeName)) {
136            return true;
137        }
138        for (DocumentType parent : inherits) {
139            if (parent.isA(docTypeName)) {
140                return true;
141            }
142        }
143        return false;
144    }

193    /**
194     * Adds a document to the inherited document types of this.
195     * If this type is already directly inherited, nothing is done
196     *
197     * @param type An already DocumentType object.
198     */
199    public void inherit(DocumentType type) {
200        //TODO: There is also a check like the following in SDDocumentType addField(), try to move that to this class' addField() to get it proper,
201        // as this method is called only when the doc types are exported.
202        verifyTypeConsistency(type);
203        if (isRegistered()) {
204            throw new IllegalStateException("You cannot add inheritance to a document type that is already registered.");
205        }
206        if (type == null) {
207            throw new IllegalArgumentException("The document type cannot be null in inherit()");
208        }
209
210        // If it inherits the exact same type
211        if (inherits.contains(type)) return;
212
213        // If we inherit a type, don't inherit the supertype
214        if (inherits.size() == 1 && inherits.get(0).getDataTypeName().equals(new DataTypeName("document"))) {
215            inherits.clear();
216        }
217
218        inherits.add(type);
219    }

221    /**
222     * Fail if the subtype changes the type of any equally named field.
223     *
224     * @param superType The supertype to verify against
225     *                  TODO Add strict type checking no duplicate fields are allowed
226     */
227    private void verifyTypeConsistency(DocumentType superType) {
228        for (Field f : fieldSet()) {
229            Field supField = superType.getField(f.getName());
230            if (supField != null) {
231                if (!f.getDataType().equals(supField.getDataType())) {
232                    throw new IllegalArgumentException("Inheritance type mismatch: field \"" + f.getName() +
233                                                       "\" in datatype \"" + getName() + "\"" +
234                                                       " must have same datatype as in parent document type \"" + superType.getName() + "\"");
235                }
236            }
237        }
238    }

261    /**
262     * Return whether this document type inherits the given document type.
263     *
264     * @param superType The documenttype to check if it inherits.
265     * @return true if it inherits the superType, false if not
266     */
267    public boolean inherits(DocumentType superType) {
268        if (equals(superType)) return true;
269        for (DocumentType type : inherits) {
270            if (type.inherits(superType)) return true;
271        }
272        return false;
273    }

276     * Gets the field matching a given name.
277     *
278     * @param name The name of a field.
279     * @return Returns the matching field, or null if not found.
280     */
281    public Field getField(String name) {
282        Field field = headerType.getField(name);
283        if (field == null) {
284            field = bodyType.getField(name);
285        }
286        if (field == null && !isRegistered()) {
287            for (DocumentType inheritedType : inherits) {
288                field = inheritedType.getField(name);
289                if (field != null) break;
290            }
291        }
292        return field;
293    }

349     * Removes an field from the DocumentType.
350     *
351     * @param name The name of the field.
352     * @return The field that was removed or null if it did not exist.
353     */
354    public Field removeField(String name) {
355        if (isRegistered()) {
356            throw new IllegalStateException("You cannot remove fields from a document type that is already registered.");
357        }
358        Field field = headerType.removeField(name);
359        if (field == null) {
360            field = bodyType.removeField(name);
361        }
362        if (field == null) {
363            for (DocumentType inheritedType : inherits) {
364                field = inheritedType.removeField(name);
365                if (field != null) break;
366            }
367        }
368        return field;
369    }

371    public Collection<Field> getFields() {
372        Collection<Field> collection = new LinkedList<>();
373
374        for (DocumentType type : inherits) {
375            collection.addAll(type.getFields());
376        }
377
378        collection.addAll(headerType.getFields());
379        collection.addAll(bodyType.getFields());
380        return ImmutableList.copyOf(collection);
381    }

383    /**
384     * <p>Returns an ordered set snapshot of all fields of this documenttype,
385     * <i>except the fields of Document</i>.
386     * Only the overridden version will be returned for overridden fields.</p>
387     *
388     * <p>The fields of a document type has a well-defined order which is
389     * exhibited in this set:
390     * - Fields come in the order defined in the document type definition.
391     * - The fields defined in inherited types come before those in
392     * the document type itself.
393     * - When a field in an inherited type is overridden, the value is overridden,
394     * but not the ordering.
395     * </p>
396     *
397     * @return an unmodifiable snapshot of the fields in this type
398     */
399    public Set<Field> fieldSet() {
400        Map<String, Field> map = new LinkedHashMap<String, Field>();
401        for (Field field : getFields()) { // Uniqify on field name
402            map.put(field.getName(), field);
403        }
404        return ImmutableSet.copyOf(map.values());
405    }

407    /**
408     * Returns an iterator over all fields in this documenttype
409     *
410     * @return An iterator for iterating the fields in this documenttype.
411     */
412    public Iterator<Field> fieldIteratorThisTypeOnly() {
413        return new Iterator<Field>() {
414            Iterator<Field> headerIt = headerType.getFields().iterator();
415            Iterator<Field> bodyIt = bodyType.getFields().iterator();
416
417            public boolean hasNext() {
418                if (headerIt != null) {
419                    if (headerIt.hasNext()) return true;
420                    headerIt = null;
421                }
422                return bodyIt.hasNext();
423            }
424
425            public Field next() {
426                return (headerIt != null ? headerIt.next() : bodyIt.next());
427            }
428
429
430            public void remove() {
431                if (headerIt != null) {
432                    headerIt.remove();
433                } else {
434                    bodyIt.remove();
435                }
436            }
437        };
438    }

440    public boolean equals(Object o) {
441        if (!(o instanceof DocumentType)) return false;
442        DocumentType other = (DocumentType) o;
443        // Ignore whether one of them have added inheritance to super Document.0 type
444        if (super.equals(o) && headerType.equals(other.headerType) &&
445            bodyType.equals(other.bodyType)) {
446            if ((inherits.size() > 1 || other.inherits.size() > 1) ||
447                (inherits.size() == 1 && other.inherits.size() == 1)) {
448                return inherits.equals(other.inherits);
449            }
450            return !(((inherits.size() == 1) && !inherits.get(0).getDataTypeName().equals(new DataTypeName("document")))
451                     || ((other.inherits.size() == 1) && !other.inherits.get(0).getDataTypeName().equals(new DataTypeName("document"))));
452        }
453        return false;
454    }
