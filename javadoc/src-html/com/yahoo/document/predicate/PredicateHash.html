002package com.yahoo.document.predicate;

004
005import com.yahoo.text.Utf8;
006
007/**
008 * @author <a href="mailto:magnarn@yahoo-inc.com">Magnar Nedland</a>
009 * This hash function must match the corresponding C++ hash function in 'ConvertClass',
010 * currently located at 'rise_platform/src/interface/cpp/api/ConvertClass.h'
011 */
012public class PredicateHash {
013    @SuppressWarnings("fallthrough")
014    public static long hash64(String s) {
015        byte[] bytes = Utf8.toBytes(s);
016        long a=0L, b=0L;
017        long c=0x9e3779b97f4a7c13L;
018        int len = bytes.length;
019        int offset = 0;
020        while (len >= 24) {
021            c += ((0xffL & (long)bytes[offset + 23]) << 56) +
022                    ((0xffL & (long)bytes[offset + 22]) << 48) +
023                    ((0xffL & (long)bytes[offset + 21]) << 40) +
024                    ((0xffL & (long)bytes[offset + 20]) << 32) +
025                    ((0xffL & (long)bytes[offset + 19]) << 24) +
026                    ((0xffL & (long)bytes[offset + 18]) << 16) +
027                    ((0xffL & (long)bytes[offset + 17]) << 8) +
028                    ((0xffL & (long)bytes[offset + 16]));
029
030            b += ((0xffL & (long)bytes[offset + 15]) << 56) +
031                    ((0xffL & (long)bytes[offset + 14]) << 48) +
032                    ((0xffL & (long)bytes[offset + 13]) << 40) +
033                    ((0xffL & (long)bytes[offset + 12]) << 32) +
034                    ((0xffL & (long)bytes[offset + 11]) << 24) +
035                    ((0xffL & (long)bytes[offset + 10]) << 16) +
036                    ((0xffL & (long)bytes[offset + 9]) << 8) +
037                    ((0xffL & (long)bytes[offset + 8]));
038
039            a += ((0xffL & (long)bytes[offset + 7]) << 56) +
040                    ((0xffL & (long)bytes[offset + 6]) << 48) +
041                    ((0xffL & (long)bytes[offset + 5]) << 40) +
042                    ((0xffL & (long)bytes[offset + 4]) << 32) +
043                    ((0xffL & (long)bytes[offset + 3]) << 24) +
044                    ((0xffL & (long)bytes[offset + 2]) << 16) +
045                    ((0xffL & (long)bytes[offset + 1]) << 8) +
046                    ((0xffL & (long)bytes[offset + 0]));
047
048            // Mix.  This arithmetic must match the mix below.
049            a -= b; a -= c; a ^= (c>>>43);
050            b -= c; b -= a; b ^= (a<<9);
051            c -= a; c -= b; c ^= (b>>>8);
052            a -= b; a -= c; a ^= (c>>>38);
053            b -= c; b -= a; b ^= (a<<23);
054            c -= a; c -= b; c ^= (b>>>5);
055            a -= b; a -= c; a ^= (c>>>35);
056            b -= c; b -= a; b ^= (a<<49);
057            c -= a; c -= b; c ^= (b>>>11);
058            a -= b; a -= c; a ^= (c>>>12);
059            b -= c; b -= a; b ^= (a<<18);
060            c -= a; c -= b; c ^= (b>>>22);
061            // End mix.
062
063            offset += 24;
064            len -= 24;
065        }
066
067        c += bytes.length;
068        switch (len) {
069            case 23: c+= (0xffL & (long)bytes[offset + 22]) << 56;
070            case 22: c+= (0xffL & (long)bytes[offset + 21]) << 48;
071            case 21: c+= (0xffL & (long)bytes[offset + 20]) << 40;
072            case 20: c+= (0xffL & (long)bytes[offset + 19]) << 32;
073            case 19: c+= (0xffL & (long)bytes[offset + 18]) << 24;
074            case 18: c+= (0xffL & (long)bytes[offset + 17]) << 16;
075            case 17: c+= (0xffL & (long)bytes[offset + 16]) << 8;
076                // The first byte of c is reserved for the length
077            case 16: b+= (0xffL & (long)bytes[offset + 15]) << 56;
078            case 15: b+= (0xffL & (long)bytes[offset + 14]) << 48;
079            case 14: b+= (0xffL & (long)bytes[offset + 13]) << 40;
080            case 13: b+= (0xffL & (long)bytes[offset + 12]) << 32;
081            case 12: b+= (0xffL & (long)bytes[offset + 11]) << 24;
082            case 11: b+= (0xffL & (long)bytes[offset + 10]) << 16;
083            case 10: b+= (0xffL & (long)bytes[offset + 9]) << 8;
084            case 9:  b+= (0xffL & (long)bytes[offset + 8]);
085            case 8:  a+= (0xffL & (long)bytes[offset + 7]) << 56;
086            case 7:  a+= (0xffL & (long)bytes[offset + 6]) << 48;
087            case 6:  a+= (0xffL & (long)bytes[offset + 5]) << 40;
088            case 5:  a+= (0xffL & (long)bytes[offset + 4]) << 32;
089            case 4:  a+= (0xffL & (long)bytes[offset + 3]) << 24;
090            case 3:  a+= (0xffL & (long)bytes[offset + 2]) << 16;
091            case 2:  a+= (0xffL & (long)bytes[offset + 1]) << 8;
092            case 1:  a+= (0xffL & (long)bytes[offset + 0]);
093        }
094        // Mix.  This arithmetic must match the mix above.
095        a -= b; a -= c; a ^= (c>>>43);
096        b -= c; b -= a; b ^= (a<<9);
097        c -= a; c -= b; c ^= (b>>>8);
098        a -= b; a -= c; a ^= (c>>>38);
099        b -= c; b -= a; b ^= (a<<23);
100        c -= a; c -= b; c ^= (b>>>5);
101        a -= b; a -= c; a ^= (c>>>35);
102        b -= c; b -= a; b ^= (a<<49);
103        c -= a; c -= b; c ^= (b>>>11);
104        a -= b; a -= c; a ^= (c>>>12);
105        b -= c; b -= a; b ^= (a<<18);
106        c -= a; c -= b; c ^= (b>>>22);
107        // End mix.
108
109        return c;
110    }
111}
