002package com.yahoo.document.datatypes;
003
004import com.yahoo.document.DataType;
005
006public abstract class CompositeFieldValue extends FieldValue {
007    private DataType dataType;
008
009    public CompositeFieldValue(DataType dataType) {
010        this.dataType = dataType;
011    }
012
013    @Override
014    public DataType getDataType() {
015        return dataType;
016    }
017
018    public void setDataType(DataType dataType) {
019        this.dataType = dataType;
020    }
021
022    @Override
023    public boolean equals(Object o) {
024        if (this == o) return true;
025        if (!(o instanceof CompositeFieldValue)) return false;
026        if (!super.equals(o)) return false;
027
028        CompositeFieldValue that = (CompositeFieldValue) o;
029        if (dataType != null ? !dataType.equals(that.dataType) : that.dataType != null) return false;
030        return true;
031    }
032
033    @Override
034    public int hashCode() {
035        int result = super.hashCode();
036        result = 31 * result + (dataType != null ? dataType.hashCode() : 0);
037        return result;
038    }
039}
