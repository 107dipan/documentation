056        }
057        for (int i = 0x20; i < 0x7F; i++) {

079    private void setValue(String value) {
080        for (int i = 0; i < value.length(); i++) {
081            char theChar = value.charAt(i);
082            int codePoint = value.codePointAt(i);
083            if (Character.isHighSurrogate(theChar)) {
084                //skip one char ahead, since codePointAt() consumes one more char in this case
085                ++i;
086            }
087
088            //See http://www.w3.org/TR/2006/REC-xml11-20060816/#charsets
089
090            if (codePoint < 0x80) {  //ascii
091                if (allowedAsciiChars[codePoint]) {
092                    continue;
093                } else {
094                    throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
095                }
096            }
097
098            //source cited above notes that 0x7F-0x84 and 0x86-0x9F are discouraged, but they are still allowed.
099            //see http://www.w3.org/International/questions/qa-controls
100
101            if (codePoint < 0xFDD0) {
102                continue;
103            }
104            if (codePoint <= 0xFDDF) {
105                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
106            }
107
108            if (codePoint < 0x1FFFE) {
109                continue;
110            }
111            if (codePoint <= 0x1FFFF) {
112                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
113            }
114            if (codePoint < 0x2FFFE) {
115                continue;
116            }
117            if (codePoint <= 0x2FFFF) {
118                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
119            }
120            if (codePoint < 0x3FFFE) {
121                continue;
122            }
123            if (codePoint <= 0x3FFFF) {
124                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
125            }
126            if (codePoint < 0x4FFFE) {
127                continue;
128            }
129            if (codePoint <= 0x4FFFF) {
130                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
131            }
132            if (codePoint < 0x5FFFE) {
133                continue;
134            }
135            if (codePoint <= 0x5FFFF) {
136                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
137            }
138            if (codePoint < 0x6FFFE) {
139                continue;
140            }
141            if (codePoint <= 0x6FFFF) {
142                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
143            }
144            if (codePoint < 0x7FFFE) {
145                continue;
146            }
147            if (codePoint <= 0x7FFFF) {
148                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
149            }
150            if (codePoint < 0x8FFFE) {
151                continue;
152            }
153            if (codePoint <= 0x8FFFF) {
154                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
155            }
156            if (codePoint < 0x9FFFE) {
157                continue;
158            }
159            if (codePoint <= 0x9FFFF) {
160                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
161            }
162            if (codePoint < 0xAFFFE) {
163                continue;
164            }
165            if (codePoint <= 0xAFFFF) {
166                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
167            }
168            if (codePoint < 0xBFFFE) {
169                continue;
170            }
171            if (codePoint <= 0xBFFFF) {
172                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
173            }
174            if (codePoint < 0xCFFFE) {
175                continue;
176            }
177            if (codePoint <= 0xCFFFF) {
178                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
179            }
180            if (codePoint < 0xDFFFE) {
181                continue;
182            }
183            if (codePoint <= 0xDFFFF) {
184                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
185            }
186            if (codePoint < 0xEFFFE) {
187                continue;
188            }
189            if (codePoint <= 0xEFFFF) {
190                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
191            }
192            if (codePoint < 0xFFFFE) {
193                continue;
194            }
195            if (codePoint <= 0xFFFFF) {
196                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
197            }
198            if (codePoint < 0x10FFFE) {
199                continue;
200            }
201            if (codePoint <= 0x10FFFF) {
202                throw new IllegalArgumentException("StringFieldValue cannot contain code point 0x" + Integer.toHexString(codePoint).toUpperCase());
203            }
204        }
205        this.value = value;
206    }

224    @Override
225    public StringFieldValue clone() {
226        StringFieldValue strfval = (StringFieldValue) super.clone();
227        if (spanTrees != null) {
228            strfval.spanTrees = new HashMap<String, SpanTree>(spanTrees.size());
229            for (Map.Entry<String, SpanTree> entry : spanTrees.entrySet()) {
230                strfval.spanTrees.put(entry.getKey(), new SpanTree(entry.getValue()));
231            }
232        }
233        return strfval;
234    }

246    /**
247     * Sets a new value for this StringFieldValue.&nbsp;NOTE that doing so will clear all span trees from this value,
248     * since they most certainly will not make sense for a new string value.
249     *
250     * @param o the new String to assign to this. An argument of null is equal to calling clear().
251     */
252    @Override
253    public void assign(Object o) {
254        if (spanTrees != null) {
255            spanTrees.clear();
256            spanTrees = null;
257        }
258
259        if (!checkAssign(o)) {
260            return;
261        }
262        if (o instanceof StringFieldValue) {
263            spanTrees=((StringFieldValue)o).spanTrees;
264        }
265        if (o instanceof String) {
266            setValue((String) o);
267        } else if (o instanceof StringFieldValue || o instanceof NumericFieldValue) {
268            setValue(o.toString());
269        } else {
270            throw new IllegalArgumentException("Class " + o.getClass() + " not applicable to an " + this.getClass() + " instance.");
271        }
272    }

307    /**
308     * Sets the span tree with annotations over this String.
309     *
310     * @param spanTree the span tree with annotations over this String
311     * @return the input spanTree for chaining
312     * @throws IllegalArgumentException if a span tree with the given name already exists.
313     */
314    public SpanTree setSpanTree(SpanTree spanTree) {
315        if (spanTrees == null) {
316            spanTrees = new HashMap(1);
317        }
318        if (spanTrees.containsKey(spanTree.getName())) {
319            throw new IllegalArgumentException("Span tree " + spanTree.getName() + " already exists.");
320        }
321        spanTrees.put(spanTree.getName(), spanTree);
322        spanTree.setStringFieldValue(this);
323        return spanTree;
324    }
325
326    /**
327     * Removes the span tree associated with the given name.
328     *
329     * @param name the name of the span tree to remove
330     * @return the span tree previously associated with the given name, or null if it did not exist.
331     */
332    public SpanTree removeSpanTree(String name) {
333        if (spanTrees == null) {
334            return null;
335        }
336        SpanTree tree = spanTrees.remove(name);
337        if (tree != null) {
338            tree.setStringFieldValue(null);
339        }
340        return tree;
341    }

409    @Override
410    public int compareTo(FieldValue fieldValue) {
411        int comp = super.compareTo(fieldValue);
412
413        if (comp != 0) {
414            return comp;
415        }
416
417        //types are equal, this must be of this type
418        StringFieldValue otherValue = (StringFieldValue) fieldValue;
419        comp = value.compareTo(otherValue.value);
420
421        if (comp != 0) {
422            return comp;
423        }
424
425        if (spanTrees == null) {
426            comp = (otherValue.spanTrees == null) ? 0 : -1;
427        } else {
428            if (otherValue.spanTrees == null) {
429                comp = 1;
430            } else {
431                comp = CollectionComparator.compare(spanTrees.keySet(), otherValue.spanTrees.keySet());
432                if (comp != 0) {
433                    return comp;
434                }
435                comp = CollectionComparator.compare(spanTrees.values(), otherValue.spanTrees.values());
436            }
437        }
438        return comp;
439    }
