002package com.yahoo.document.datatypes;
003
004import com.yahoo.document.DataType;
005import com.yahoo.document.Field;
006import com.yahoo.document.PrimitiveDataType;
007import com.yahoo.document.serialization.FieldReader;
008import com.yahoo.net.Url;
009
010import java.net.URI;
011
012/**
013 * Created with IntelliJ IDEA.
014 * User: magnarn
015 * Date: 11/2/12
016 * Time: 2:37 PM
017 */
018public class UriFieldValue extends StringFieldValue {
019    public static class Factory extends PrimitiveDataType.Factory {
020        public FieldValue create() {
021            return new UriFieldValue();
022        }
023    }
024    public UriFieldValue() { super(); }
025
026    public UriFieldValue(String value) {
027        super(value);
028        Url.fromString(value);  // Throws if value is invalid.
029    }
030
031    @Override
032    public void assign(Object obj) {
033        if (obj instanceof URI) {
034            obj = obj.toString();
035        }
036        super.assign(obj);
037    }
038
039    @Override
040    public DataType getDataType() {
041        return DataType.URI;
042    }
043
044    @Override
045    public void deserialize(Field field, FieldReader reader) {
046        super.deserialize(field, reader);
047        Url.fromString(toString());  // Throws if value is invalid.
048    }
049}
