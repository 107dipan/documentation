002package com.yahoo.document.datatypes;
003
004import com.yahoo.document.DataType;
005import com.yahoo.document.Field;
006import com.yahoo.document.PrimitiveDataType;
007import com.yahoo.document.serialization.FieldReader;
008import com.yahoo.document.serialization.FieldWriter;
009import com.yahoo.document.serialization.XmlSerializationHelper;
010import com.yahoo.document.serialization.XmlStream;
011import com.yahoo.vespa.objects.Ids;
012
013/**
014 * FieldValue which encapsulates an int.
015 *
016 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
017 */
018public final class IntegerFieldValue extends NumericFieldValue {
019
020    private static class Factory extends PrimitiveDataType.Factory {
021        public FieldValue create() {
022            return new IntegerFieldValue();
023        }
024    }
025
026    public static PrimitiveDataType.Factory getFactory() { return new Factory(); }
027    public static final int classId = registerClass(Ids.document + 11, IntegerFieldValue.class);
028    private int value;
029
030    public IntegerFieldValue() {
031        this(0);
032    }
033
034    public IntegerFieldValue(int value) {
035        this.value = value;
036    }
037
038    public IntegerFieldValue(Integer value) {
039        this.value = value;
040    }
041
042    public IntegerFieldValue(String s) {
043        value = Integer.parseInt(s);
044    }
045
046    @Override
047    public IntegerFieldValue clone() {
048        IntegerFieldValue val = (IntegerFieldValue) super.clone();
049        val.value = value;
050        return val;
051    }
052
053    @Override
054    public Number getNumber() {
055        return value;
056    }
057
058    @Override
059    public void clear() {
060        value = 0;
061    }
062
063    @Override
064    public void assign(Object obj) {
065        if (!checkAssign(obj)) {
066            return;
067        }
068        if (obj instanceof Number) {
069            value = ((Number) obj).intValue();
070        } else if (obj instanceof NumericFieldValue) {
071            value = (((NumericFieldValue) obj).getNumber().intValue());
072        } else if (obj instanceof String || obj instanceof StringFieldValue) {
073            value = Integer.parseInt(obj.toString());
074        } else {
075            throw new IllegalArgumentException("Class " + obj.getClass() + " not applicable to an " + this.getClass() + " instance.");
076        }
077    }
078
079    public int getInteger() {
080        return value;
081    }
082
083    @Override
084    public Object getWrappedValue() {
085        return value;
086    }
087
088    @Override
089    public void printXml(XmlStream xml) {
090        XmlSerializationHelper.printIntegerXml(this, xml);
091    }
092
093    @Override
094    public String toString() {
095        return "" + value;
096    }
097
098    @Override
099    public int hashCode() {
100        int result = super.hashCode();
101        result = 31 * result + value;
102        return result;
103    }
104
105    @Override
106    public boolean equals(Object o) {
107        if (this == o) return true;
108        if (!(o instanceof IntegerFieldValue)) return false;
109        if (!super.equals(o)) return false;
110
111        IntegerFieldValue that = (IntegerFieldValue) o;
112        return (value == that.value);
113    }
114
115    @Override
116    public void serialize(Field field, FieldWriter writer) {
117        writer.write(field, this);
118    }
119
120    /* (non-Javadoc)
121      * @see com.yahoo.document.datatypes.FieldValue#deserialize(com.yahoo.document.Field, com.yahoo.document.serialization.FieldReader)
122      */
123
124    @Override
125    public void deserialize(Field field, FieldReader reader) {
126        reader.read(field, this);
127    }
128
129    @Override
130    public DataType getDataType() {
131        return DataType.INT;
132    }
133
134    @Override
135    public int compareTo(FieldValue fieldValue) {
136        int comp = super.compareTo(fieldValue);
137
138        if (comp != 0) {
139            return comp;
140        }
141
142        //types are equal, this must be of this type
143        IntegerFieldValue otherValue = (IntegerFieldValue) fieldValue;
144        if (value < otherValue.value) {
145            return -1;
146        } else if (value > otherValue.value) {
147            return 1;
148        } else {
149            return 0;
150        }
151    }
152
153}
