002package com.yahoo.document;
003
004/**
005 * Base class for "document operations".
006 * These include "put" (DocumentPut), "update" (DocumentUpdate), and "remove" (DocumentRemove).
007 * Historically, put operations were represented by the Document class alone,
008 * but since it doesn't make much sense to put a *test and set* condition in Document,
009 * a more uniform interface for document operations was needed.
010 *
011 * @author Vegard Sjonfjell
012 */
013public abstract class DocumentOperation {
014
015    private TestAndSetCondition condition = TestAndSetCondition.NOT_PRESENT_CONDITION;
016
017    public abstract DocumentId getId();
018
019    public void setCondition(TestAndSetCondition condition) {
020        this.condition = condition;
021    }
022
023    public TestAndSetCondition getCondition() {
024        return condition;
025    }
026
027    protected DocumentOperation() {}
028
029    /**
030     * Copy constructor
031     * @param other DocumentOperation to copy
032     */
033    protected DocumentOperation(DocumentOperation other) {
034        this.condition = other.condition;
035    }
036
037}
