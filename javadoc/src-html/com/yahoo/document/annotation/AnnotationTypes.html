006import java.util.Arrays;

009/**
010 * This is a container for all {@link Annotation}s constants used by built-in Vespa features. These must be in sync with
011 * the corresponding class in the C++ file 'document/datatype/annotationtype.h'.
012 *
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
014 */
015@SuppressWarnings({ "UnusedDeclaration" })
016public final class AnnotationTypes {
017
018    private AnnotationTypes() {
019        // unreachable
020    }
021
022    public static final AnnotationType TERM = new AnnotationType("term", DataType.STRING, 1);
023    public static final AnnotationType TOKEN_TYPE = new AnnotationType("token_type", DataType.INT, 2);
024    public static final AnnotationType CANONICAL = new AnnotationType("canonical", DataType.STRING, 3);
025    public static final AnnotationType NORMALIZED = new AnnotationType("normalized", DataType.STRING, 4);
026    public static final AnnotationType READING = new AnnotationType("reading", DataType.STRING, 5);
027    public static final AnnotationType STEM = new AnnotationType("stem", DataType.STRING, 6);
028    public static final AnnotationType TRANSFORMED = new AnnotationType("transformed", DataType.STRING, 7);
029    public static final AnnotationType PROXIMITY_BREAK = new AnnotationType("proximity_break", DataType.DOUBLE, 8);
030    public static final AnnotationType SPECIAL_TOKEN = new AnnotationType("special_token", 9);
031
032    public static final List<AnnotationType> ALL_TYPES = Arrays.asList(TERM, TOKEN_TYPE, CANONICAL, NORMALIZED, READING,
033                                                                       STEM, TRANSFORMED, PROXIMITY_BREAK,
034                                                                       SPECIAL_TOKEN);
