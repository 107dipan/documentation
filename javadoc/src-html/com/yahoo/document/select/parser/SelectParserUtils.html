002package com.yahoo.document.select.parser;
003
004import com.yahoo.javacc.UnicodeUtilities;
005
006import java.math.BigInteger;
007
008/**
009 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
010 */
011public class SelectParserUtils {
012
013    public static long decodeLong(String str) {
014        if (str.startsWith("0x") || str.startsWith("0X")) {
015            str = Long.toString(new BigInteger(str.substring(2), 16).longValue());
016        }
017        return Long.decode(str);
018    }
019
020    public static String quote(String str, char quote) {
021        return UnicodeUtilities.quote(str, quote);
022    }
023
024    public static String unquote(String str) {
025        return UnicodeUtilities.unquote(str);
026    }
027}
