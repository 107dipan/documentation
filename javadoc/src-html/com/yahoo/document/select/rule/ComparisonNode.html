099    public OrderingSpecification getOrdering(IdNode lhs, LiteralNode rhs, String operator, int order) {
100        if (lhs.getWidthBits() == -1 || lhs.getDivisionBits() == -1 || !(rhs.getValue() instanceof Long)) {
101            return null;
102        }
103
104        if (operator.equals("==") || operator.equals("=")) {
105            return new OrderingSpecification(order, (Long)rhs.getValue(), lhs.getWidthBits(), lhs.getDivisionBits());
106        } 
107
108        if (order == OrderingSpecification.ASCENDING) {
109            if ((operator.equals("<") || operator.equals("<="))) {
110                return new OrderingSpecification(order, 0, lhs.getWidthBits(), lhs.getDivisionBits());
111            } 
112            if (operator.equals(">")) {
113                return new OrderingSpecification(order, (Long)rhs.getValue() + 1, lhs.getWidthBits(), lhs.getDivisionBits());
114            }
115            if (operator.equals(">=")) {
116                return new OrderingSpecification(order, (Long)rhs.getValue(), lhs.getWidthBits(), lhs.getDivisionBits());
117            }
118        } else {
119            if (operator.equals("<")) {
120                return new OrderingSpecification(order, (Long)rhs.getValue() - 1, lhs.getWidthBits(), lhs.getDivisionBits());
121            }
122            if (operator.equals("<=")) {
123                return new OrderingSpecification(order, (Long)rhs.getValue(), lhs.getWidthBits(), lhs.getDivisionBits());
124            }
125        }
126        return null;
127    }
128
129    public OrderingSpecification getOrdering(int order) {
130        if (lhs instanceof IdNode && rhs instanceof LiteralNode) {
131            return getOrdering((IdNode)lhs, (LiteralNode)rhs, operator, order);
132        } else if (rhs instanceof IdNode && lhs instanceof LiteralNode) {
133            return getOrdering((IdNode)rhs, (LiteralNode)rhs, operator, order);
134        }
135
136        return null;
137    }

140    public BucketSet getBucketSet(BucketIdFactory factory) {
141        if (operator.equals("==") || operator.equals("=")) {
142            if (lhs instanceof IdNode && rhs instanceof LiteralNode) {
143                return compare(factory, (IdNode)lhs, (LiteralNode)rhs, operator);
144            } else if (rhs instanceof IdNode && lhs instanceof LiteralNode) {
145                return compare(factory, (IdNode)rhs, (LiteralNode)lhs, operator);
146            } else if (lhs instanceof SearchColumnNode && rhs instanceof LiteralNode) {
147                return compare(factory, (SearchColumnNode)lhs, (LiteralNode)rhs);
148            } else if (rhs instanceof SearchColumnNode && lhs instanceof LiteralNode) {
149                return compare(factory, (SearchColumnNode)rhs, (LiteralNode)lhs);
150            }
151        }
152        return null;
153    }

155    /**
156     * Compares a search column node with a literal node.
157     *
158     * @param factory The bucket id factory used.
159     * @param node The search column node.
160     * @param literal The literal node to compare to.
161     * @return The bucket set containing the buckets covered.
162     */
163    private BucketSet compare(BucketIdFactory factory, SearchColumnNode node, LiteralNode literal) {
164        Object value = literal.getValue();
165        int bucketCount = (int) Math.pow(2, 16);
166        if (value instanceof Long) {
167            BucketSet ret = new BucketSet();
168            for (int i = 0; i < bucketCount; i++) {
169                BucketId id = new BucketId(16, i);
170                if ((Long)value == node.getDistribution().getColumn(id)) {
171                    ret.add(new BucketId(16, i));
172                }
173            }
174            return ret;
175        }
176        return null;
177    }

179    private BucketSet compare(BucketIdFactory factory, IdNode id, LiteralNode literal, String operator) {
180        String field = id.getField();
181        Object value = literal.getValue();
182        if (field == null) {
183            if (value instanceof String) {
184                String name = (String)value;
185                if ((operator.equals("=") && name.contains("*")) ||
186                    (operator.equals("=~") && ((name.contains("*") || name.contains("?")))))
187                {
188                    return null; // no idea
189                }
190                return new BucketSet(factory.getBucketId(new DocumentId(name)));
191            }
192        } else if (field.equalsIgnoreCase("user")) {
193            if (value instanceof Long) {
194                return new BucketSet(new BucketId(factory.getLocationBitCount(), (Long)value));
195            }
196        } else if (field.equalsIgnoreCase("group")) {
197            if (value instanceof String) {
198                String name = (String)value;
199                if ((operator.equals("=") && name.contains("*")) ||
200                        (operator.equals("=~") && ((name.contains("*") || name.contains("?")))))
201                {
202                    return null; // no idea
203                }
204                return new BucketSet(new BucketId(factory.getLocationBitCount(), new GroupDocIdString("", name, "").getLocation()));
205            }
206        } else if (field.equalsIgnoreCase("bucket")) {
207            if (value instanceof Long) {
208                return new BucketSet(new BucketId((Long)value));
209            }
210        }
211        return null;
212    }

214    // Inherit doc from Node.
215    public Object evaluate(Context context) {
216        Object oLeft = lhs.evaluate(context);
217        Object oRight = rhs.evaluate(context);
218        if (oLeft == null && oRight == null) {
219            return new ResultList(Result.TRUE);
220        }
221        if (oLeft == Result.INVALID || oRight == Result.INVALID) {
222            return new ResultList(Result.INVALID);
223        }
224        if (oLeft instanceof AttributeNode.VariableValueList && oRight instanceof AttributeNode.VariableValueList) {
225            if (operator.equals("==")) {
226                return evaluateListsTrue((AttributeNode.VariableValueList)oLeft, (AttributeNode.VariableValueList)oRight);
227            } else if (operator.equals("!=")) {
228                return evaluateListsFalse((AttributeNode.VariableValueList)oLeft, (AttributeNode.VariableValueList)oRight);
229            } else {
230                return new ResultList(Result.INVALID);
231            }
232        } else if (oLeft instanceof AttributeNode.VariableValueList) {
233            return evaluateListAndSingle((AttributeNode.VariableValueList)oLeft, oRight);
234        } else if (oRight instanceof AttributeNode.VariableValueList) {
235            return evaluateListAndSingle((AttributeNode.VariableValueList)oRight, oLeft);
236        }
237        return new ResultList(evaluateBool(oLeft, oRight));
238    }
239
240    public ResultList evaluateListsTrue(AttributeNode.VariableValueList lhs, AttributeNode.VariableValueList rhs) {
241        if (lhs.size() != rhs.size()) {
242            return new ResultList(Result.FALSE);
243        }
244
245        for (int i = 0; i < lhs.size(); i++) {
246            if (!lhs.get(i).getVariables().equals(rhs.get(i).getVariables())) {
247                return new ResultList(Result.FALSE);
248            }
249
250            if (evaluateEquals(lhs.get(i).getValue(), rhs.get(i).getValue()) == Result.FALSE) {
251                return new ResultList(Result.FALSE);
252            }
253        }
254
255        return new ResultList(Result.TRUE);
256    }
257
258    public ResultList evaluateListsFalse(AttributeNode.VariableValueList lhs, AttributeNode.VariableValueList rhs) {
259        ResultList lst = evaluateListsTrue(lhs, rhs);
260        if (lst.toResult() == Result.TRUE) {
261            return new ResultList(Result.FALSE);
262        } else if (lst.toResult() == Result.FALSE) {
263            return new ResultList(Result.TRUE);
264        } else {
265            return lst;
266        }
267    }
268
269    public ResultList evaluateListAndSingle(AttributeNode.VariableValueList lhs, Object rhs) {
270        if (rhs == null && lhs == null) {
271            return new ResultList(Result.TRUE);
272        }
273
274        if (rhs == null || lhs == null) {
275            return new ResultList(Result.FALSE);
276        }
277
278        ResultList retVal = new ResultList();
279        for (int i = 0; i < lhs.size(); i++) {
280                Result result = evaluateBool(lhs.get(i).getValue(), rhs);
281                retVal.add((FieldPathIteratorHandler.VariableMap)lhs.get(i).getVariables().clone(), result);
282        }
283
284        return retVal;
285    }

287    /**
288     * Evaluate this expression on two operands, given that they are not invalid.
289     *
290     * @param lhs Left hand side of operation.
291     * @param rhs Right hand side of operation.
292     * @return The evaluation result.
293     */
294    private Result evaluateBool(Object lhs, Object rhs) {
295        if (operator.equals("==")) {
296            return evaluateEquals(lhs, rhs);
297        } else if (operator.equals("!=")) {
298            return Result.invert(evaluateEquals(lhs, rhs));
299        } else if (operator.equals("<") || operator.equals("<=") ||
300                   operator.equals(">") || operator.equals(">=")) {
301            return evaluateNumber(lhs, rhs);
302        } else if (operator.equals("=~") || operator.equals("=")) {
303            return evaluateString(lhs, rhs);
304        }
305        throw new IllegalStateException("Comparison operator '" + operator + "' is not supported.");
306    }

308    /**
309     * Compare two operands for equality.
310     *
311     * @param lhs Left hand side of operation.
312     * @param rhs Right hand side of operation.
313     * @return Wether or not the two operands are equal.
314     */
315    private Result evaluateEquals(Object lhs, Object rhs) {
316        if (lhs == null || rhs == null) {
317            return Result.toResult(lhs == rhs);
318        }
319
320        double a = getAsNumber(lhs);
321        double b = getAsNumber(rhs);
322        if (Double.isNaN(a) || Double.isNaN(b)) {
323                return Result.toResult(lhs.toString().equals(rhs.toString()));
324        }
325        return Result.toResult(a == b); // Ugh, comparing doubles? Should be converted to long value perhaps...
326    }
327
328    private double getAsNumber(Object value) {
329        if (value instanceof Number) {
330            return ((Number)value).doubleValue();
331        } else if (value instanceof NumericFieldValue) {
332            return getAsNumber(((NumericFieldValue)value).getNumber());
333        } else {
334            return Double.NaN; //new IllegalStateException("Term '" + value + "' (" + value.getClass() + ") does not evaluate to a number.");
335        }
336    }

338    /**
339     * Evalutes the value of this term over a document, given that both operands must be numbers.
340     *
341     * @param lhs Left hand side of operation.
342     * @param rhs Right hand side of operation.
343     * @return The evaluation result.
344     */
345    private Result evaluateNumber(Object lhs, Object rhs) {
346        double a = getAsNumber(lhs);
347        double b = getAsNumber(rhs);
348        if (Double.isNaN(a) || Double.isNaN(b)) {
349                return Result.INVALID;
350        }
351        if (operator.equals("<")) {
352            return Result.toResult(a < b);
353        } else if (operator.equals("<=")) {
354            return Result.toResult(a <= b);
355        } else if (operator.equals(">")) {
356            return Result.toResult(a > b);
357        } else {
358            return Result.toResult(a >= b);
359        }
360    }
361
362    /**
363     * Evalutes the value of this term over a document, given that both operands must be strings.
364     *
365     * @param lhs Left hand side of operation.
366     * @param rhs Right hand side of operation.
367     * @return The evaluation result.
368     */
369    private Result evaluateString(Object lhs, Object rhs) {
370        String left = "" + lhs; // Allows null objects to evaluate to string.
371        String right = "" + rhs;
372        if (operator.equals("=~")) {
373            return Result.toResult(Pattern.compile(right).matcher(left).find());
374        } else {
375            return Result.toResult(Pattern.compile(globToRegex(right)).matcher(left).find());
376        }
377    }

379    /**
380     * Converts a glob pattern to a corresponding regular expression string.
381     *
382     * @param glob The glob pattern.
383     * @return The regex string.
384     */
385    private String globToRegex(String glob) {
386        StringBuilder ret = new StringBuilder();
387        ret.append("^");
388        for (int i = 0; i < glob.length(); i++) {
389            ret.append(globToRegex(glob.charAt(i)));
390        }
391        ret.append("$");
392
393        return ret.toString();
394    }
395
396    /**
397     * Converts a single character in a glob expression to the corresponding regular expression string.
398     *
399     * @param glob The glob character.
400     * @return The regex string.
401     */
402    private String globToRegex(char glob) {
403        switch (glob) {
404            case'*':
405                return ".*";
406            case'?':
407                return ".";
408            case'^':
409            case'$':
410            case'|':
411            case'{':
412            case'}':
413            case'(':
414            case')':
415            case'[':
416            case']':
417            case'\\':
418            case'+':
419            case'.':
420                return "\\" + glob;
421            default:
422                return "" + glob;
423        }
424    }
