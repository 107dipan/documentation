002package com.yahoo.document.select;
003
004import com.yahoo.document.select.rule.AttributeNode;
005
006/**
007 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
008 */
009public enum Result {
010
011    /**
012     * Defines all enumeration constants.
013     */
014    TRUE,
015    FALSE,
016    INVALID;
017
018    // Inherit doc from Object.
019    public String toString() {
020        return name().toLowerCase();
021    }
022    
023    /**
024     * Inverts the result value to the appropriate value. True → False, False → True and Invalid → Invalid.
025     * @return inverted result
026     */
027    public static Result invert(Result result) {
028        if (result == Result.TRUE) return Result.FALSE;
029        if (result == Result.FALSE) return Result.TRUE;
030        return Result.INVALID;
031    }
032
033    /**
034     * Converts the given object value into an instance of this Result enumeration.
035     *
036     * @param value The object to convert.
037     * @return The corresponding result value.
038     */
039    public static Result toResult(Object value) {
040        if (value == null || value == Result.FALSE || value == Boolean.FALSE ||
041            (Number.class.isInstance(value) && ((Number)value).doubleValue() == 0)) {
042            return Result.FALSE;
043        } else if (value == INVALID) {
044            return Result.INVALID;
045        } else if (value instanceof AttributeNode.VariableValueList) {
046            return ((AttributeNode.VariableValueList)value).isEmpty() ? Result.FALSE : Result.TRUE;
047        } else if (value instanceof ResultList) {
048            return ((ResultList)value).toResult();
049        } else {
050            return Result.TRUE;
051        }
052    }
053}
