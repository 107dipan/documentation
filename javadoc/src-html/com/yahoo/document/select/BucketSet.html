002package com.yahoo.document.select;
003
004import com.yahoo.document.BucketId;
005
006import java.util.HashSet;
007
008/**
009 * A set of bucket ids covered by a document selector.
010 *
011 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
012 */
013public class BucketSet extends HashSet<BucketId> {
014
015    /**
016     * Constructs a new bucket set that contains no ids.
017     */
018    public BucketSet() {
019        // empty
020    }
021
022    /**
023     * Constructs a new bucket set that contains a single id.
024     *
025     * @param id The id to add to this as initial value.
026     */
027    public BucketSet(BucketId id) {
028        add(id);
029    }
030
031    /**
032     * Constructs a new bucket set that is a copy of another.
033     *
034     * @param set The set to copy.
035     */
036    public BucketSet(BucketSet set) {
037        this.addAll(set);
038    }
039
040    /**
041     * Returns the intersection between this bucket set and another.
042     *
043     * @param rhs The set to form an intersection with.
044     * @return The intersection.
045     */
046    public BucketSet intersection(BucketSet rhs) {
047        if (rhs == null) {
048            return new BucketSet(this); // The other has all buckets marked, this is the smaller.
049        } else {
050            BucketSet ret = new BucketSet(this);
051            ret.retainAll(rhs);
052            return ret;
053        }
054    }
055
056    /**
057     * Returns the union between this bucket set and another.
058     *
059     * @param rhs The set to form a union with.
060     * @return The union.
061     */
062    public BucketSet union(BucketSet rhs) {
063        if (rhs == null) {
064            return null;
065        } else {
066            BucketSet ret = new BucketSet(this);
067            ret.addAll(rhs);
068            return ret;
069        }
070    }
071
072}
