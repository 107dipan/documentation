002package com.yahoo.document.select;
003
004public class OrderingSpecification {
005    public static int ASCENDING = 0;
006    public static int DESCENDING = 1;
007
008    public int order;
009    public long orderingStart;
010    public short widthBits;
011    public short divisionBits;
012
013    public OrderingSpecification() {
014        this(ASCENDING, (long)0, (short)0, (short)0);
015    }
016
017    public OrderingSpecification(int order) {
018        this(order, (long)0, (short)0, (short)0);
019    }
020
021    public OrderingSpecification(int order, long orderingStart, short widthBits, short divisionBits) {
022        this.order = order;
023        this.orderingStart = orderingStart;
024        this.widthBits = widthBits;
025        this.divisionBits = divisionBits;
026    }
027
028    public int getOrder() { return order; }
029    public long getOrderingStart() { return orderingStart; }
030    public short getWidthBits() { return widthBits; }
031    public short getDivisionBits() { return divisionBits; }
032
033    public boolean equals(Object other) {
034        OrderingSpecification o = (OrderingSpecification)other;
035        if (o == null) return false;
036
037        return (order == o.order && orderingStart == o.orderingStart && widthBits == o.widthBits && divisionBits == o.divisionBits);
038    }
039
040    public String toString() {
041        return "O: " + order + " S:" + orderingStart + " W:" + widthBits + " D:" + divisionBits;
042    }
043}
