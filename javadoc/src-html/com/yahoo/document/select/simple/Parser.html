004/**
005 * @author baldersheim
006 */
007public abstract class Parser {
008    public abstract boolean parse(CharSequence s);
009    public CharSequence getRemaining() { return remaining; }
010    protected void setRemaining(CharSequence r) { remaining = r; }
011    private CharSequence remaining;
012    protected int eatWhite(CharSequence s) {
013        int pos = 0;
014        for (;pos < s.length() && Character.isSpaceChar(s.charAt(pos)); pos++);
015        return pos;
016    }
017    protected boolean icmp(char c, char l) {
018        return Character.toLowerCase(c) == l;
019    }
020}
