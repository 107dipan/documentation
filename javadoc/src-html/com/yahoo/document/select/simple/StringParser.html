002package com.yahoo.document.select.simple;
003
004import com.yahoo.document.select.rule.LiteralNode;
005
006/**
007 * @author baldersheim
008 */
009public class StringParser extends Parser {
010    private LiteralNode value;
011    public LiteralNode getValue() { return value; }
012    public boolean parse(CharSequence s) {
013        boolean retval = false;
014        int pos = eatWhite(s);
015        if (pos + 1 < s.length()) {
016        if (s.charAt(pos++) == '"') {
017            StringBuffer str = new StringBuffer("");
018            for(; (pos < s.length()) && (s.charAt(pos) != '"');pos++) {
019                if ((pos < s.length()) && (s.charAt(pos) == '\\')) {
020                    pos++;
021                }
022                str.append(s.charAt(pos));
023            }
024            if (s.charAt(pos) == '"') {
025                pos++;
026                retval = true;
027                value = new LiteralNode(str.toString());
028            }
029        }
030
031        setRemaining(s.subSequence(pos, s.length()));
032    }
033        return retval;
034    }
035}
