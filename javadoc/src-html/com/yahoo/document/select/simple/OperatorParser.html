004/**
005 * @author baldersheim
006 */
007public class OperatorParser extends Parser {
008    private String operator;
009    public String getOperator() { return operator; }
010    public boolean parse(CharSequence s) {
011        boolean retval = false;
012        int pos = eatWhite(s);
013
014        if (pos+1 < s.length()) {
015            retval = true;
016            int startPos = pos;
017            if (s.charAt(pos) == '=') {
018                pos++;
019                if ((s.charAt(pos) == '=') || (s.charAt(pos) == '~')) {
020                    pos++;
021                }
022            } else if (s.charAt(pos) == '>') {
023                pos++;
024                if (s.charAt(pos) == '=') {
025                    pos++;
026                }
027            } else if (s.charAt(pos) == '<') {
028                pos++;
029                if (s.charAt(pos) == '=') {
030                    pos++;
031                }
032            } else if ((s.charAt(pos) == '!') && (s.charAt(pos) == '=')) {
033                pos += 2;
034            } else {
035                retval = false;
036            }
037            if (retval) {
038                operator = s.subSequence(startPos, pos).toString();
039            }
040        }
041        setRemaining(s.subSequence(pos, s.length()));
042
043        return retval;
044    }
045}
