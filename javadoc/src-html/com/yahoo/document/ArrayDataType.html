002package com.yahoo.document;
003
004import com.yahoo.document.datatypes.Array;
005import com.yahoo.vespa.objects.Ids;
006
007import java.util.ArrayList;
008import java.util.List;
009
010/**
011 * @author Einar M R Rosenvinge
012 */
013public class ArrayDataType extends CollectionDataType {
014
015    // The global class identifier shared with C++.
016    public static int classId = registerClass(Ids.document + 54, ArrayDataType.class);
017
018    public ArrayDataType(DataType nestedType) {
019        super("Array<"+nestedType.getName()+">", 0, nestedType);
020        setId(getName().toLowerCase().hashCode());
021    }
022
023    public ArrayDataType(DataType nestedType, int code) {
024        super("Array<"+nestedType.getName()+">", code, nestedType);
025    }
026
027    public ArrayDataType clone() {
028        return (ArrayDataType) super.clone();
029    }
030
031    public Array createFieldValue() {
032        return new Array(this);
033    }
034
035    @Override
036    public Class getValueClass() {
037        return Array.class;
038    }
039
040    @Override
041    public FieldPath buildFieldPath(String remainFieldName)
042    {
043        if (remainFieldName.length() > 0 && remainFieldName.charAt(0) == '[') {
044            int endPos = remainFieldName.indexOf(']');
045            if (endPos == -1) {
046                throw new IllegalArgumentException("Array subscript must be closed with ]");
047            } else {
048                FieldPath path = getNestedType().buildFieldPath(skipDotInString(remainFieldName, endPos));
049                List<FieldPathEntry> tmpPath = new ArrayList<FieldPathEntry>(path.getList());
050                if (remainFieldName.charAt(1) == '$') {
051                    tmpPath.add(0, FieldPathEntry.newVariableLookupEntry(remainFieldName.substring(2, endPos), getNestedType()));
052                } else {
053                    tmpPath.add(0, FieldPathEntry.newArrayLookupEntry(Integer.parseInt(remainFieldName.substring(1, endPos)), getNestedType()));
054                }
055
056                return new FieldPath(tmpPath);
057            }
058        }
059
060        return getNestedType().buildFieldPath(remainFieldName);
061    }
062
063}
