017/**
018 * <p>Specifies one or more field updates to a document.</p> <p>A document update contains a list of {@link
019 * com.yahoo.document.update.FieldUpdate field updates} for fields to be updated by this update. Each field update is
020 * applied atomically, but the entire document update is not. A document update can only contain one field update per
021 * field. To make multiple updates to the same field in the same document update, add multiple {@link
022 * com.yahoo.document.update.ValueUpdate value updates} to the same field update.</p> <p>To update a document and
023 * set a string field to a new value:</p>
024 * <pre>
025 * DocumentType musicType = DocumentTypeManager.getInstance().getDocumentType("music", 0);
026 * DocumentUpdate docUpdate = new DocumentUpdate(musicType,
027 *         new DocumentId("doc:test:http://music.yahoo.com/"));
028 * FieldUpdate update = FieldUpdate.createAssign(musicType.getField("artist"), "lillbabs");
029 * docUpdate.addFieldUpdate(update);
030 * </pre>
031 *
032 * @author Einar M R Rosenvinge
033 * @see com.yahoo.document.update.FieldUpdate
034 * @see com.yahoo.document.update.ValueUpdate
035 */
036public class DocumentUpdate extends DocumentOperation implements Iterable<FieldPathUpdate> {
037
038    //see src/vespa/document/util/identifiableid.h
039    public static final int CLASSID = 0x1000 + 6;
040
041    private DocumentId docId;
042    private List<FieldUpdate> fieldUpdates;
043    private List<FieldPathUpdate> fieldPathUpdates;
044    private DocumentType documentType;
045    private Optional<Boolean> createIfNonExistent = Optional.empty();
046
047    /**
048     * Creates a DocumentUpdate.
049     *
050     * @param docId   the ID of the update
051     * @param docType the document type that this update is valid for
052     */
053    public DocumentUpdate(DocumentType docType, DocumentId docId) {
054        this(docType, docId, new ArrayList<FieldUpdate>());
055    }
056
057    /**
058     * Creates a new document update using a reader
059     */
060    public DocumentUpdate(DocumentUpdateReader reader) {
061        docId = null;
062        documentType = null;
063        fieldUpdates = new ArrayList<>();
064        fieldPathUpdates = new ArrayList<>();
065        reader.read(this);
066    }
067
068    /**
069     * Creates a DocumentUpdate.
070     *
071     * @param docId   the ID of the update
072     * @param docType the document type that this update is valid for
073     */
074    public DocumentUpdate(DocumentType docType, String docId) {
075        this(docType, new DocumentId(docId));
076    }
077
078    private DocumentUpdate(DocumentType docType, DocumentId docId, List<FieldUpdate> fieldUpdates) {
079        this.docId = docId;
080        this.documentType = docType;
081        this.fieldUpdates = fieldUpdates;
082        this.fieldPathUpdates = new ArrayList<>();
083    }
084
085    public DocumentId getId() {
086        return docId;
087    }
088
089    /**
090     * Sets the document id of the document to update.
091     * Use only while deserializing - changing the document id after creation has undefined behaviour.
092     */
093    public void setId(DocumentId id) {
094        docId = id;
095    }
096
097    /**
098     * Applies this document update.
099     *
100     * @param doc the document to apply the update to
101     * @return a reference to itself
102     * @throws IllegalArgumentException if the document does not have the same document type as this update
103     */
104    public DocumentUpdate applyTo(Document doc) {
105        if (!documentType.equals(doc.getDataType())) {
106            throw new IllegalArgumentException(
107                    "Document " + doc + " must have same type as update, which is type " + documentType);
108        }
109
110        for (FieldUpdate fieldUpdate : fieldUpdates) {
111            fieldUpdate.applyTo(doc);
112        }
113        for (FieldPathUpdate fieldPathUpdate : fieldPathUpdates) {
114            fieldPathUpdate.applyTo(doc);
115        }
116        return this;
117    }
118

120     * Get an unmodifiable list of all field updates that this document update specifies.

165     * Replaces the field update at the specified index in the list of field updates.

176    /**
177     * Returns the update for a field
178     *
179     * @param field the field to return the update of
180     * @return the update for the field, or null if that field has no update in this

182    public FieldUpdate getFieldUpdate(Field field) {
183        return getFieldUpdate(field.getName());
184    }
185
186    /** Removes all field updates from the list for field updates. */
187    public void clearFieldUpdates() {
188        fieldUpdates.clear();
189    }
190
191    /**
192     * Returns the update for a field name
193     *
194     * @param fieldName the field name to return the update of
195     * @return the update for the field, or null if that field has no update in this
196     */
197    public FieldUpdate getFieldUpdate(String fieldName) {
198        for (FieldUpdate fieldUpdate : fieldUpdates) {
199            if (fieldUpdate.getField().getName().equals(fieldName)) {
200                return fieldUpdate;
201            }
202        }
203        return null;
204    }

206    /**
207     * Assigns the field updates of this document update.
208     * This document update receives ownership of the list - it can not be subsequently used
209     * by the caller. The list may not be unmodifiable.
210     *
211     * @param fieldUpdates the new list of updates of this
212     * @throws NullPointerException if the argument passed is null
213     */
214    public void setFieldUpdates(List<FieldUpdate> fieldUpdates) {
215        if (fieldUpdates == null) {
216            throw new NullPointerException("The field updates of a document update can not be null");
217        }
218        this.fieldUpdates = fieldUpdates;
219    }
220
221    /**
222     * Get the number of field updates in this document update.
223     *
224     * @return the size of the List of FieldUpdates
225     */
226    public int size() {
227        return fieldUpdates.size();
228    }
229
230    /**
231     * Adds the given {@link FieldUpdate} to this DocumentUpdate. If this DocumentUpdate already contains a FieldUpdate
232     * for the named field, the content of the given FieldUpdate is added to the existing one.
233     *
234     * @param update The FieldUpdate to add to this DocumentUpdate.
235     * @return This, to allow chaining.
236     * @throws IllegalArgumentException If the {@link DocumentType} of this DocumentUpdate does not have a corresponding
237     *                                  field.
238     */
239    public DocumentUpdate addFieldUpdate(FieldUpdate update) {
240        String fieldName = update.getField().getName();
241        if (!documentType.hasField(fieldName)) {
242            throw new IllegalArgumentException("Document type '" + documentType.getName() + "' does not have field '" +
243                                               fieldName + "'.");
244        }
245        FieldUpdate prevUpdate = getFieldUpdate(fieldName);
246        if (prevUpdate != update) {
247            if (prevUpdate != null) {
248                prevUpdate.addAll(update);
249            } else {
250                fieldUpdates.add(update);
251            }
252        }
253        return this;
254    }
255
256    /**
257     * Adds a field path update to perform on the document.
258     *
259     * @return a reference to itself.
260     */
261    public DocumentUpdate addFieldPathUpdate(FieldPathUpdate fieldPathUpdate) {
262        fieldPathUpdates.add(fieldPathUpdate);
263        return this;
264    }
265
266    // TODO: Remove this when we figure out correct behaviour.
267
268    public void addFieldUpdateNoCheck(FieldUpdate fieldUpdate) {
269        fieldUpdates.add(fieldUpdate);
270    }
271
272    /**
273     * Adds all the field- and field path updates of the given document update to this. If the given update refers to a
274     * different document or document type than this, this method throws an exception.
275     *
276     * @param update The update whose content to add to this.
277     * @throws IllegalArgumentException If the {@link DocumentId} or {@link DocumentType} of the given DocumentUpdate
278     *                                  does not match the content of this.
279     */
280    public void addAll(DocumentUpdate update) {
281        if (update == null) {
282            return;
283        }
284        if (!docId.equals(update.docId)) {
285            throw new IllegalArgumentException("Expected " + docId + ", got " + update.docId + ".");
286        }
287        if (!documentType.equals(update.documentType)) {
288            throw new IllegalArgumentException("Expected " + documentType + ", got " + update.documentType + ".");
289        }
290        for (FieldUpdate fieldUpd : update.fieldUpdates) {
291            addFieldUpdate(fieldUpd);
292        }
293        for (FieldPathUpdate pathUpd : update.fieldPathUpdates) {
294            addFieldPathUpdate(pathUpd);
295        }
296    }

303     * @throws IndexOutOfBoundsException if index is out of range
304     */

306        return fieldUpdates.remove(index);

326    @Override
327    public boolean equals(Object o) {
328        if (this == o) return true;
329        if (!(o instanceof DocumentUpdate)) return false;
330
331        DocumentUpdate that = (DocumentUpdate) o;
332
333        if (docId != null ? !docId.equals(that.docId) : that.docId != null) return false;
334        if (documentType != null ? !documentType.equals(that.documentType) : that.documentType != null) return false;
335        if (fieldPathUpdates != null ? !fieldPathUpdates.equals(that.fieldPathUpdates) : that.fieldPathUpdates != null)
336            return false;
337        if (fieldUpdates != null ? !fieldUpdates.equals(that.fieldUpdates) : that.fieldUpdates != null) return false;
338        if (this.getCreateIfNonExistent() != ((DocumentUpdate) o).getCreateIfNonExistent()) return false;
339
340        return true;
341    }
342
343    @Override
344    public int hashCode() {
345        int result = docId != null ? docId.hashCode() : 0;
346        result = 31 * result + (fieldUpdates != null ? fieldUpdates.hashCode() : 0);
347        result = 31 * result + (fieldPathUpdates != null ? fieldPathUpdates.hashCode() : 0);
348        result = 31 * result + (documentType != null ? documentType.hashCode() : 0);
349        return result;
350    }
351
352    @Override
353    public String toString() {
354        StringBuilder string = new StringBuilder();
355        string.append("update of document '");
356        string.append(docId);
357        string.append("': ");
358        string.append("create-if-non-existent=");
359        string.append(createIfNonExistent.orElse(false));
360        string.append(": ");
361        string.append("[");
362
363        for (Iterator<FieldUpdate> i = fieldUpdates.iterator(); i.hasNext();) {
364            FieldUpdate fieldUpdate = i.next();
365            string.append(fieldUpdate);
366            if (i.hasNext()) {
367                string.append(", ");
368            }
369        }
370        string.append("]");
371
372        if (fieldPathUpdates.size() > 0) {
373            string.append(" [ ");
374            for (FieldPathUpdate up : fieldPathUpdates) {
375                string.append(up.toString() + " ");
376            }
377            string.append(" ]");
378        }
379
380        return string.toString();
381    }
