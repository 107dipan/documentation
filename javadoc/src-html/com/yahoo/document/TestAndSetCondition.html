002package com.yahoo.document;
003
004import com.google.common.annotations.Beta;
005
006import java.util.Optional;
007
008/**
009 * The TestAndSetCondition class represents a test and set condition.
010 * A test and set condition is an (optional) string representing a
011 * document selection (cf. document selection language), which is used
012 * to match a document for test and set. If #isPresent evaluates to false,
013 * the condition is not present and matches any document.
014 *
015 * @author Vegard Sjonfjell
016 */
017@Beta
018public class TestAndSetCondition {
019    public static final TestAndSetCondition NOT_PRESENT_CONDITION = new TestAndSetCondition();
020
021    private final String conditionStr;
022
023    public TestAndSetCondition() {
024        this("");
025    }
026
027    public TestAndSetCondition(String conditionStr) {
028        this.conditionStr = conditionStr;
029    }
030
031    public String getSelection() { return conditionStr; }
032
033    public boolean isPresent() { return !conditionStr.isEmpty(); }
034
035    /**
036     * Maps and optional test and set conditiong string to a TestAndSetCondition.
037     * If the condition string is not present, a "not present" condition is returned
038     * @param conditionString test and set conditiong string (document selection)
039     * @return a TestAndSetCondition representing the condition string or a "not present" condition
040     */
041    public static TestAndSetCondition fromConditionString(Optional<String> conditionString) {
042        return conditionString
043                .map(TestAndSetCondition::new)
044                .orElse(TestAndSetCondition.NOT_PRESENT_CONDITION);
045    }
046}
