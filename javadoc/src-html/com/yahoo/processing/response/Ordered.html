002package com.yahoo.processing.response;
003
004/**
005 * This is an <i>optional marker interface</i>.
006 * DataLists may implement this to return false to indicate that the order of the elements of
007 * the list is insignificant. The usage of this is to allow the content of a list to be rendered in the order
008 * in which it completes rather than in the order in which it is added to the list.
009 *
010 * @author  bratseth
011 * @since   5.1.19
012 */
013public interface Ordered {
014
015    /** Returns false if the data in this list can be returned in any order. Default: true, meaning the order matters */
016    public boolean isOrdered();
017
018}
