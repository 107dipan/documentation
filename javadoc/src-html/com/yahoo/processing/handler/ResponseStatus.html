002package com.yahoo.processing.handler;
003
004import com.yahoo.processing.Request;
005import com.yahoo.processing.response.AbstractData;
006
007/**
008 * <p>A data item holding a response HTTP status code.
009 * If this is present in a Response it will determine the HTTP status of the response (when returned over HTTP),
010 * regardless of any errors present in the result which might otherwise determine the response status.</p>
011 *
012 * <p>If several ResponseStatus instances are present, the first one encountered by a depth-first search through
013 * the data composite tree will be used.</p>
014 *
015 * <p>Note that this must be added to the response before any response data is writable to take effect.</p>
016 *
017 * @author bratseth
018 */
019public class ResponseStatus extends AbstractData {
020
021    /** A http status code */
022    private final int code;
023
024    public ResponseStatus(int code, Request request) {
025        super(request);
026        this.code = code;
027    }
028
029    /** Returns the code of this */
030    public int code() { return code; }
031
032    @Override
033    public String toString() {
034        return "HTTP response " + code;
035    }
036
037}
