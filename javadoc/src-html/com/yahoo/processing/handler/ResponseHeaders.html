002package com.yahoo.processing.handler;
003
004import com.google.common.collect.ImmutableMap;
005import com.yahoo.processing.Request;
006import com.yahoo.processing.response.AbstractData;
007
008import java.util.Collections;
009import java.util.List;
010import java.util.Map;
011
012/**
013 * Holds a set of headers which will be added to the Processing response.
014 * A Response may contain multiple such data objects, and all of them will be added to the response.
015 *
016 * @author bratseth
017 * @since  5.1.23
018 */
019public class ResponseHeaders extends AbstractData {
020
021    private final Map<String,List<String>> headers;
022
023    /**
024     * Creates a response headers object with a set of headers.
025     *
026     * @param headers the headers to copy into this object
027     */
028    public ResponseHeaders(Map<String,List<String>> headers, Request request) {
029        super(request);
030        this.headers = ImmutableMap.copyOf(headers);
031    }
032
033    /** Returns an unmodifiable map of the response headers of this */
034    public Map<String,List<String>> headers() { return headers; }
035
036}
