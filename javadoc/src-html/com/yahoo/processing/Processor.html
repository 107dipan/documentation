002package com.yahoo.processing;
003
004import com.yahoo.component.chain.ChainedComponent;
005import com.yahoo.processing.execution.Execution;
006
007/**
008 * Superclass of chainable components processing Requests to create Responses.
009 * <p>
010 * Processors typically changes the Request and/or the Response. It may also make multiple
011 * forward requests, in series or parallel, or manufacture the response content itself or by calling
012 * an external service.
013 * <p>
014 * Typical usage:
015 * <code>
016 * public class MyProcessor extends Processor {
017 *
018 *     &#64;Override
019 *     public Response process(Request request, Execution execution) {
020 *         // process the request here
021 *         Response response = execution.process(request); // Pass along to get the Response
022 *         // process (or fill in) Data/DataList items on the response here
023 *         return response;
024 *     }
025 *
026 * }
027 * </code>
028 *
029 * @author bratseth
030 */
031public abstract class Processor extends ChainedComponent {
032
033    /**
034     * Performs a processing request and returns the response
035     *
036     * @return a Response instance - never null - containing the data produced by this processor
037     *         and those it forwards the request to
038     */
039    public abstract Response process(Request request, Execution execution);
040
041
042}
