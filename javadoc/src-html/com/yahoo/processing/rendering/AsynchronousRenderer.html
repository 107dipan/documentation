002package com.yahoo.processing.rendering;
003
004import com.yahoo.jdisc.handler.CompletionHandler;
005import com.yahoo.jdisc.handler.ContentChannel;
006import com.yahoo.processing.Response;
007
008import java.util.concurrent.Executor;
009
010/**
011 * Superclass of all asynchronous renderers.
012 * Subclasses this to receive an executor and the network wiring necessary to respectively
013 * run callback listeners and close up the channel when the response is complete.
014 *
015 * @author bratseth
016 */
017public abstract class AsynchronousRenderer <RESPONSE extends Response> extends Renderer<RESPONSE> {
018
019    /**
020     * Exposes JDisc wiring to ensure asynchronous cleanup.
021     *
022     * @param channel the channel to the client receiving the response
023     * @param completionHandler the JDisc completion handler which will be invoked at the end
024     *        of the rendering
025     * @throws IllegalStateException if attempted invoked more than once
026     */
027    public abstract void setNetworkWiring(ContentChannel channel, CompletionHandler completionHandler);
028
029}
