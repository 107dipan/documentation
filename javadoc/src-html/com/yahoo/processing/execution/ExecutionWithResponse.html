002package com.yahoo.processing.execution;
003
004import com.yahoo.component.chain.Chain;
005import com.yahoo.processing.Processor;
006import com.yahoo.processing.Request;
007import com.yahoo.processing.Response;
008
009/**
010 * An execution which has a response which is returned when this gets to the end of the chain.
011 * This is useful to run processing chains where a response exists up front, typically for on completion listeners.
012 *
013 * @author bratseth
014 */
015public class ExecutionWithResponse extends Execution {
016
017    private Response response;
018
019    /**
020     * Creates an execution which will return a given response at the end of the chain.
021     *
022     * @param chain     the chain to execute in this
023     * @param response  the response this will return from {@link #process} then the end of this chain is reached
024     * @param execution the the parent of this execution
025     */
026    public ExecutionWithResponse(Chain<? extends Processor> chain, Response response, Execution execution) {
027        super(chain, execution);
028        this.response = response;
029    }
030
031    @Override
032    protected Response defaultResponse(Request request) {
033        return response;
034    }
035
036}
