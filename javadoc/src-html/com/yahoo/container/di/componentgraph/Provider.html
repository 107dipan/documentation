004/**
005 * <p>Provides a component of the parameter type T.
006 * If (and only if) dependency injection does not have a component of type T,
007 * it will request one from the Provider providing type T.</p>
008 *
009 * <p>Providers are useful in these situations:</p>
010 * <ul>
011 *     <li>Some code is needed to create the component instance in question.</li>
012 *     <li>The component creates resources that must be deconstructed.</li>
013 *     <li>A fallback component should be provided in case the application (or system)
014 *     does not provide a component instance.</li>
015 * </ul>
016 *
017 * @author tonytv
018 * @author gjoranv
019 */
020public interface Provider<T> {
021
022    T get();
023    void deconstruct();
024
025}
