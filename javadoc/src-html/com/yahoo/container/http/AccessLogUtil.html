002package com.yahoo.container.http;
003
004import com.yahoo.jdisc.http.HttpRequest;
005
006import java.net.InetSocketAddress;
007import java.net.URI;
008import java.util.List;
009
010/**
011 * @author bakksjo
012 */
013public class AccessLogUtil {
014    public static String getHttpMethod(final HttpRequest httpRequest) {
015        return httpRequest.getMethod().toString();
016    }
017
018    public static URI getUri(final HttpRequest httpRequest) {
019        return httpRequest.getUri();
020    }
021
022    public static String getHttpVersion(final HttpRequest httpRequest) {
023        return httpRequest.getVersion().toString();
024    }
025
026    public static String getReferrerHeader(final HttpRequest httpRequest) {
027        // Yes, the header name is misspelled in the standard
028        return getFirstHeaderValue(httpRequest, "Referer");
029    }
030
031    public static String getUserAgentHeader(final HttpRequest httpRequest) {
032        return getFirstHeaderValue(httpRequest, "User-Agent");
033    }
034
035    public static InetSocketAddress getRemoteAddress(final HttpRequest httpRequest) {
036        return (InetSocketAddress) httpRequest.getRemoteAddress();
037    }
038
039    private static String getFirstHeaderValue(final HttpRequest httpRequest, final String headerName) {
040        final List<String> headerValues = httpRequest.headers().get(headerName);
041        return (headerValues == null || headerValues.isEmpty()) ? "" : headerValues.get(0);
042    }
043}
