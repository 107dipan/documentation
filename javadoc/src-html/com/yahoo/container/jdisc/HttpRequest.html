029/**
030 * Wraps a JDisc HTTP request for a synchronous API.
031 * <p>
032 * The properties of this request represents what was received in the request
033 * and are thus immutable. If you need mutable abstractions, use a higher level
034 * framework, e.g. Processing.
035 *
036 * @author hmusum
037 * @author Steinar Knutsen
038 */
039public class HttpRequest {

055        InputStream requestData = null;

058        private String nag = " must be set before the attempted operation.";
059        SocketAddress remoteAddress;

061        private void boom(Object ref, String what) {
062            if (ref == null) {
063                throw new IllegalStateException(what + nag);
064            }
065        }
066
067        private void requireUri() {
068            boom(uri, "An URI");
069        }
070
071        private void requireContainer() {
072            boom(container, "A CurrentContainer instance");
073        }

075        private void ensureJdiscParent() {
076            if (jdiscRequest == null) {
077                if (parent == null) {
078                    throw new IllegalStateException("Neither another HttpRequest nor JDisc request available.");
079                } else {
080                    jdiscRequest = parent.getJDiscRequest();
081                }
082            }
083        }
084
085        private void ensureRequestData() {
086            if (requestData == null) {
087                if (parent == null) {
088                    throw new IllegalStateException(
089                            "Neither another HttpRequest nor request data input stream available.");
090                } else {
091                    requestData = parent.getData();
092                }
093            }
094        }

123        private Builder(HttpRequest parent, com.yahoo.jdisc.http.HttpRequest jdiscRequest) {
124            this.parent = parent;

130        private void populateProperties() {

209        /**
210         * Create a new HTTP request without creating a new JDisc request. This
211         * is for scenarios where another HTTP request handler is invoked
212         * directly without dispatching through JDisc. The parent JDisc request
213         * for the original HttpRequest will be passed on the new HttpRequest
214         * instance's JDisc request, but no properties will be propagated into
215         * the original JDisc request.
216         *
217         * @return a new HttpRequest instance reflecting the given request data
218         *         and parameters
219         */
220        public HttpRequest createDirectRequest() {
221            ensureRequestData();
222            ensureJdiscParent();
223            return new HttpRequest(jdiscRequest, requestData, properties);
224        }

226        /**
227         * Start of API for synchronous HTTP request dispatch. Not yet ready for
228         * use.
229         *
230         * @return a new client request
231         */
232        public HttpRequest createClientRequest() {
233            ensureJdiscParent();
234            requireUri();
235            com.yahoo.jdisc.http.HttpRequest clientRequest;
236            if (method == null) {
237                clientRequest = com.yahoo.jdisc.http.HttpRequest
238                        .newClientRequest(jdiscRequest, uri);
239            } else {
240                if (version == null) {
241                    clientRequest = com.yahoo.jdisc.http.HttpRequest
242                            .newClientRequest(jdiscRequest, uri, method);
243                } else {
244                    clientRequest = com.yahoo.jdisc.http.HttpRequest
245                            .newClientRequest(jdiscRequest, uri, method,
246                                    version);
247                }
248            }
249            setParameters(clientRequest);
250            // TODO set requestData sanely
251            return new HttpRequest(clientRequest, requestData, properties);
252        }
253
254        /**
255         * Start of API for synchronous HTTP request dispatch. Not yet ready for
256         * use.
257         *
258         * @return a new server request
259         */
260        public HttpRequest createServerRequest() {
261            requireUri();
262            requireContainer();
263            com.yahoo.jdisc.http.HttpRequest serverRequest;
264            if (method == null) {
265                serverRequest = com.yahoo.jdisc.http.HttpRequest
266                        .newServerRequest(container, uri);
267            } else {
268                if (version == null) {
269                    serverRequest = com.yahoo.jdisc.http.HttpRequest
270                            .newServerRequest(container, uri, method);
271                } else {
272                    if (remoteAddress == null) {
273                        serverRequest = com.yahoo.jdisc.http.HttpRequest
274                                .newServerRequest(container, uri, method,
275                                        version);
276                    } else {
277                        serverRequest = com.yahoo.jdisc.http.HttpRequest
278                                .newServerRequest(container, uri, method,
279                                        version, remoteAddress);
280                    }
281                }
282            }
283            setParameters(serverRequest);
284            // TODO IO wiring
285            return new HttpRequest(serverRequest, requestData, properties);
286        }

288        private void setParameters(
289                com.yahoo.jdisc.http.HttpRequest request) {
290            for (Map.Entry<String, String> entry : properties.entrySet()) {
291                request.parameters().put(entry.getKey(), wrap(entry.getValue()));
292            }
293        }

310    /**
311     * Wrap a JDisc HTTP request in a synchronous API. The properties from the
312     * JDisc request will be copied into the HTTP request. The mappings in
313     * propertyOverrides will mask the settings in the JDisc request. The
314     * content of propertyOverrides will be copied, so it is safe to re-use and
315     * changes in propertyOverrides after constructing the HttpRequest instance
316     * will obviously not be reflected by the request. The same applies for
317     * JDisc parameters.
318     *
319     * @param jdiscHttpRequest
320     *            the JDisc request
321     * @param requestData
322     *            the associated input stream, e.g. with POST request
323     * @param propertyOverrides
324     *            properties which should not have the same settings as in the
325     *            parent JDisc request, may be null
326     */
327    public HttpRequest(com.yahoo.jdisc.http.HttpRequest jdiscHttpRequest,
328                       InputStream requestData, Map<String, String> propertyOverrides) {
329        parentRequest = jdiscHttpRequest;
330        this.requestData = requestData;
331        properties = copyProperties(jdiscHttpRequest.parameters(), propertyOverrides);
332    }

348    /**
349     * Create a new HTTP request from an URI.
350     *
351     * @param container the current container instance
352     * @param uri the request parameters
353     * @param method GET, POST, etc
354     * @param requestData the associated data stream, may be null
355     * @param properties a set of properties to set in the request in addition to the implicit ones from the URI
356     * @return a new HTTP request
357     */
358    public static HttpRequest createRequest(CurrentContainer container,
359                                            URI uri, Method method, InputStream requestData,
360                                            Map<String, String> properties) {
361        com.yahoo.jdisc.http.HttpRequest clientRequest = 
362                com.yahoo.jdisc.http.HttpRequest.newClientRequest(new Request(container, uri), uri, method);
363        setProperties(clientRequest, properties);
364        return new HttpRequest(clientRequest, requestData);
365    }
366
367    private static void setProperties(com.yahoo.jdisc.http.HttpRequest clientRequest, Map<String, String> properties) {
368        if (properties == null) return;
369
370        for (Map.Entry<String, String> entry : properties.entrySet()) {
371            clientRequest.parameters().put(entry.getKey(), wrap(entry.getValue()));
372        }
373    }

375    // conservative code in case anything else depends on modifying these lists
376    private static List<String> wrap(String value) {
377        List<String> l = new ArrayList<>(4);
378        l.add(value);
379        return l;
380    }

430    private static Map<String, String> copyProperties(Map<String, List<String>> parameters, Map<String, String> parameterMask) {
431        Map<String, String> mask;
432        Map<String, String> view;
433
434        if (parameterMask != null) {
435            mask = parameterMask;
436        } else {
437            mask = Collections.emptyMap();
438        }
439        view = new HashMap<>(parameters.size() + mask.size());
440        for (Map.Entry<String, List<String>> parameter : parameters.entrySet()) {
441            if (existsAsOriginalParameter(parameter.getValue())) {
442                List<String> values = parameter.getValue();
443                view.put(parameter.getKey(), values.get(values.size() - 1)); // prefer the last value
444            }
445        }
446        view.putAll(mask);
447        return Collections.unmodifiableMap(view);
448    }
449
450    private static boolean existsAsOriginalParameter(List<String> value) {
451        return value != null && value.size() > 0 && value.get(0) != null;
452    }

502    /**
503     * Helper method to parse boolean request flags, using
504     * Boolean.parseBoolean(String). Unset values are regarded as false.
505     *
506     * @param name
507     *            the name of a request property
508     * @return whether the property has been explicitly set to true
509     */
510    public boolean getBooleanProperty(String name) {
511        if (getProperty(name) == null) {
512            return false;
513        }
514        return Boolean.parseBoolean(getProperty(name));
515    }

518     * Check whether a property exists.
519     *
520     * @param name the name of a request property
521     * @return true if the property has a value
522     */
523    public boolean hasProperty(String name) {
524        return properties.containsKey(name);
525    }

527    /**
528     * Access an HTTP header in the request. Multi-value headers are not
529     * supported.
530     *
531     * @param name
532     *            the name of an HTTP header
533     * @return the first pertinent value
534     */
535    public String getHeader(String name) {
536        if (parentRequest.headers().get(name) == null)
537            return null;
538        return parentRequest.headers().get(name).get(0);
539    }
