002package com.yahoo.container.jdisc;
003
004import com.yahoo.jdisc.handler.CompletionHandler;
005
006/**
007 * A completion handler which does access logging.
008 *
009 * @see ThreadedHttpRequestHandler#createLoggingCompletionHandler(long, long, HttpResponse, HttpRequest, ContentChannelOutputStream)
010 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
011 */
012public interface LoggingCompletionHandler extends CompletionHandler {
013    /**
014     * Set the commit start time to the current time. Commit start is only well
015     * defined for synchronous renderers, it is the point in time when rendering
016     * has finished, but there may still be I/O operations to transfer the data
017     * to the client pending.
018     */
019    public void markCommitStart();
020}
