004/**
005 * Helper class to account for measuring how many queries times outs.
006 *
007 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
008 * @deprecated this is not in use and will be removed in the next major release
009 */
010// TODO: Remove on Vespa 7
011@Deprecated
012public final class TimeoutRate {
013
014    private int timeouts = 0;
015    private int total = 0;
016
017    public void addQuery(Boolean timeout) {
018        if (timeout) {
019            timeouts += 1;
020        }
021        total += 1;
022    }
023
024    public void merge(TimeoutRate other) {
025        timeouts += other.timeouts;
026        total += other.total;
027    }
028
029    public double timeoutFraction() {
030        if (total == 0) {
031            return 0.0d;
032        } else {
033            return ((double) timeouts) / ((double) total);
034        }
035    }
036
037    public int getTotal() {
038        return total;
039    }
040
041}
