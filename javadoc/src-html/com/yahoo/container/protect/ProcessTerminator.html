002package com.yahoo.container.protect;
003
004import com.yahoo.protect.Process;
005
006/**
007 * An injectable terminator of the Java vm.
008 * Components that encounters conditions where the vm should be terminator should
009 * request an instance of this injected. That makes termination testable
010 * as tests can create subclasses of this which register the termination request
011 * rather than terminating.
012 *
013 * @author bratseth
014 */
015public class ProcessTerminator {
016
017    /** Logs and dies without taking a thread dump */
018    public void logAndDie(String message) {
019        logAndDie(message, false);
020    }
021
022    /**
023     * Logs and dies
024     *
025     * @param dumpThreads if true the stack trace of all threads is dumped to the
026     *                   log with level info before shutting down
027     */
028    public void logAndDie(String message, boolean dumpThreads) {
029        Process.logAndDie(message, dumpThreads);
030    }
031
032}
