004/**
005 * Error codes to use in ErrorMessage instances for container applications.
006 *
007 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
008 */
009public enum Error {
010
011    NO_BACKENDS_IN_SERVICE(0),
012    NULL_QUERY(1),
013    REQUEST_TOO_LARGE(2),
014    ILLEGAL_QUERY(3),
015    INVALID_QUERY_PARAMETER(4),
016    UNSPECIFIED(5),
017    ERROR_IN_PLUGIN(6),
018    INVALID_QUERY_TRANSFORMATION(7),
019    RESULT_HAS_ERRORS(8),
020    SERVER_IS_MISCONFIGURED(9),
021    BACKEND_COMMUNICATION_ERROR(10),
022    NO_ANSWER_WHEN_PINGING_NODE(11),
023    TIMEOUT(12),
024    EMPTY_DOCUMENTS(13),
025    UNAUTHORIZED(14),
026    FORBIDDEN(15),
027    NOT_FOUND(16),
028    BAD_REQUEST(17),
029    INTERNAL_SERVER_ERROR(18);
030
031    public final int code;
032
033    Error(int code) {
034        this.code = code;
035    }
036
037}
