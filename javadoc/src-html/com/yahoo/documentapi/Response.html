002package com.yahoo.documentapi;
003
004/**
005 * <p>An asynchronous response from the document api.
006 * Subclasses of this provide additional response information for particular operations.</p>
007 *
008 * <p>This is a <i>value object</i>.</p>
009 *
010 * @author bratseth
011 */
012public class Response {
013
014    private long requestId;
015    private String textMessage = null;
016    private boolean success = true;
017
018    /** Creates a successful response containing no information */
019    public Response(long requestId) {
020        this(requestId, null, true);
021    }
022
023    /**
024     * Creates a successful response containing a textual message
025     *
026     * @param textMessage the message to encapsulate in the Response
027     */
028    public Response(long requestId, String textMessage) {
029        this(requestId, textMessage, true);
030    }
031
032    /**
033     * Creates a response containing a textual message
034     *
035     * @param textMessage the message to encapsulate in the Response
036     * @param success     true if the response represents a successful call
037     */
038    public Response(long requestId, String textMessage, boolean success) {
039        this.requestId = requestId;
040        this.textMessage = textMessage;
041        this.success = success;
042    }
043
044    /**
045     * Returns the text message of this response or null if there is none
046     *
047     * @return the message, or null
048     */
049    public String getTextMessage() { return textMessage; }
050
051    /**
052     * Returns whether this response encodes a success or a failure
053     *
054     * @return true if success
055     */
056    public boolean isSuccess() { return success; }
057
058    public long getRequestId() { return requestId; }
059
060    public int hashCode() {
061        return (new Long(requestId).hashCode()) + (textMessage == null ? 0 : textMessage.hashCode()) +
062                (success ? 1 : 0);
063    }
064
065    public boolean equals(Object o) {
066        if (!(o instanceof Response)) {
067            return false;
068        }
069        Response other = (Response) o;
070
071        return requestId == other.requestId && success == other.success && (
072                textMessage == null && other.textMessage == null ||
073                        textMessage != null && other.textMessage != null && textMessage.equals(other.textMessage));
074    }
075
076    public String toString() {
077        return "Response " + requestId + (textMessage == null ? "" : textMessage) +
078                (success ? " SUCCESSFUL" : " UNSUCCESSFUL");
079    }
080
081}
