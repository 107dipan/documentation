015/**
016 * <p>Enables transparent iteration of super/sub-buckets</p>
017 *
018 * <p>Thread safety: safe for threads to hold their own iterators (no shared state),
019 * as long as they also hold the ProgressToken object associated with it. No two
020 * VisitorIterator instances may share the same progress token instance at the
021 * same time.
022 * Concurrent access to a single VisitorIterator instance is not safe and must
023 * be handled atomically by the caller.</p>
024 *
025 * @author <a href="mailto:vekterli@yahoo-inc.com">Tor Brede Vekterli</a>
026 */
027public class VisitorIterator {
028    private ProgressToken progressToken;
029    private BucketSource bucketSource;
030    private int distributionBitCount;
031
032    private static final Logger log = Logger.getLogger(VisitorIterator.class.getName());

052    /**
053     * Provides an abstract interface to <code>VisitorIterator</code> for
054     * how pending buckets are acquired, decoupling this from the iteration
055     * itself.
056     *
057     * <em>Important</em>: it is the responsibility of the {@link BucketSource} implementation
058     * to ensure that progress information is honored for (partially) finished buckets.
059     * From the point of view of the iterator itself, it should not have to deal with
060     * filtering away already finished buckets, as this is a detail best left to
061     * bucket sources.
062     */
063    protected static interface BucketSource {
064        public boolean hasNext();
065        public boolean shouldYield();
066        public boolean visitsAllBuckets();
067        public BucketProgress getNext();
068        public long getTotalBucketCount();
069        public int getDistributionBitCount();
070        public void setDistributionBitCount(int distributionBitCount,
071                                            ProgressToken progress);
072        public void update(BucketId superbucket, BucketId progress,
073                           ProgressToken token);
074    }

076    /**
077     * Provides a bucket source that encompasses the entire range available
078     * through a given value of distribution bits
079     */
080    protected static class DistributionRangeBucketSource implements BucketSource {
081        private boolean flushActive = false;
082        private int distributionBitCount;
083        // Wouldn't need this if this were a non-static class, but do it for
084        // the sake of keeping things identical in Java and C++
085        private ProgressToken progressToken;
086
087        public DistributionRangeBucketSource(int distributionBitCount,
088                                             ProgressToken progress) {
089            progressToken = progress;
090
091            // New progress token (could also be empty, in which this is a
092            // no-op anyway)
093            if (progressToken.getTotalBucketCount() == 0) {
094                assert(progressToken.isEmpty()) : "inconsistent progress state";
095                progressToken.setTotalBucketCount(1L << distributionBitCount);
096                progressToken.setDistributionBitCount(distributionBitCount);
097                progressToken.setBucketCursor(0);
098                progressToken.setFinishedBucketCount(0);
099                this.distributionBitCount = distributionBitCount;
100            }
101            else {
102                this.distributionBitCount = progressToken.getDistributionBitCount();
103                // Quick consistency check to ensure the user isn't trying to eg.
104                // pass a progress token for an explicit document selection
105                if (progressToken.getTotalBucketCount() != (1L << progressToken.getDistributionBitCount())) {
106                    throw new IllegalArgumentException("Total bucket count in existing progress is not "
107                            + "consistent with that of the current document selection");
108                }
109            }
110
111            if (!progress.isFinished()) {
112                if (log.isLoggable(LogLevel.DEBUG)) {
113                    log.log(LogLevel.DEBUG, "Importing unfinished progress token with " +
114                            "bits: " + progressToken.getDistributionBitCount() +
115                            ", active: " + progressToken.getActiveBucketCount() +
116                            ", pending: " + progressToken.getPendingBucketCount() +
117                            ", cursor: " + progressToken.getBucketCursor() +
118                            ", finished: " + progressToken.getFinishedBucketCount() +
119                            ", total: " + progressToken.getTotalBucketCount());
120                }
121                if (!progress.isEmpty()) {
122                    // Lower all active to pending
123                    if (progressToken.getActiveBucketCount() > 0) {
124                        if (log.isLoggable(LogLevel.DEBUG)) {
125                            log.log(LogLevel.DEBUG, "Progress token had active buckets upon range " +
126                                    "construction. Setting these as pending");
127                        }
128                        progressToken.setAllBucketsToState(ProgressToken.BucketState.BUCKET_PENDING);
129                    }
130                    // Fixup for any buckets that were active when progress was written
131                    // but are now pending and with wrong dist bits (used-bits). Buckets
132                    // split here may very well be split/merged again if we set a new dist
133                    // bit count, but that is the desired process
134                    correctInconsistentPending(progressToken.getDistributionBitCount());
135                    // Fixup for bucket cursor in case of bucket space downscaling
136                    correctTruncatedBucketCursor();
137
138                    if (log.isLoggable(LogLevel.DEBUG)) {
139                        log.log(LogLevel.DEBUG, "Partial bucket space progress; continuing "+
140                                "from position " + progressToken.getBucketCursor());
141                    }
142                }
143                progressToken.setFinishedBucketCount(progressToken.getBucketCursor() -
144                        progressToken.getPendingBucketCount());
145            } else {
146                assert(progressToken.getBucketCursor() == progressToken.getTotalBucketCount());
147            }
148            // Should be all fixed up and good to go
149            progressToken.setInconsistentState(false);
150        }
151
152        protected boolean isLosslessResetPossible() {
153            // #pending must be equal to cursor, i.e. all buckets ever fetched
154            // must be located in the set of pending
155            if (progressToken.getPendingBucketCount() != progressToken.getBucketCursor()) {
156                return false;
157            }
158            // Check if all pending buckets have a progress of 0
159            for (Map.Entry<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry> entry
160                        : progressToken.getBuckets().entrySet()) {
161                if (entry.getValue().getState() != ProgressToken.BucketState.BUCKET_PENDING) {
162                    return false;
163                }
164                if (entry.getValue().getProgress().getId() != 0) {
165                    return false;
166                }
167            }
168            return true;
169        }
170
171        /**
172         * Ensure that a given <code>ProgressToken</code> instance only has
173         * buckets pending that have a used-bits count of that of the
174         * <code>targetDistCits</code>. This is done by splitting or merging
175         * all inconsistent buckets until the desired state is reached.
176         *
177         * Time complexity is approx <i>O(4bn)</i> where <i>b</i> is the maximum
178         * delta of bits to change anywhere in the set of pending and <i>n</i>
179         * is the number of pending. This includes the time spent making shallow
180         * map copies.
181         *
182         * @param targetDistBits The desired distribution bit count of the buckets
183         */
184        private void correctInconsistentPending(int targetDistBits) {
185            boolean maybeInconsistent = true;
186            long bucketsSplit = 0, bucketsMerged = 0;
187            long pendingBefore = progressToken.getPendingBucketCount();
188            ProgressToken p = progressToken;
189
190            // Optimization: before doing any splitting/merging at all, we check
191            // to see if we can't simply just reset the entire internal state
192            // with the new distribution bit count. This ensures that if we go
193            // from eg. 1 bit to 20 bits, we won't have to perform a grueling
194            // half a million splits to cover the same bucket space as that 1
195            // single-bit bucket once did
196            if (isLosslessResetPossible()) {
197                if (log.isLoggable(LogLevel.DEBUG)) {
198                    log.log(LogLevel.DEBUG, "At start of bucket space and all " +
199                            "buckets have no progress; doing a lossless reset " +
200                            "instead of splitting/merging");
201                }
202                assert(p.getActiveBucketCount() == 0);
203                p.clearAllBuckets();
204                p.setBucketCursor(0);
205                return;
206            }
207
208            while (maybeInconsistent) {
209                BucketId lastMergedBucket = null;
210                maybeInconsistent = false;
211                // Make a shallow working copy of the bucket map. BucketKeyWrapper
212                // keys are considered immutable, and should thus not be at risk
213                // for being changed during the inner loop
214                // Do separate passes for splitting and merging just to make
215                // absolutely sure that the two ops won't step on each others'
216                // toes. This isn't wildly efficient, but the data sets in question
217                // are presumed to be low in size and this is presumed to be a very
218                // infrequent operation
219                TreeMap<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry> buckets
220                        = new TreeMap<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry>(p.getBuckets());
221                for (Map.Entry<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry> entry
222                        : buckets.entrySet()) {
223                    assert(entry.getValue().getState() == ProgressToken.BucketState.BUCKET_PENDING);
224                    BucketId pending = new BucketId(ProgressToken.keyToBucketId(entry.getKey().getKey()));
225                    if (pending.getUsedBits() < targetDistBits) {
226                        if (pending.getUsedBits() + 1 < targetDistBits) {
227                            maybeInconsistent = true; // Do another pass
228                        }
229                        p.splitPendingBucket(pending);
230                        ++bucketsSplit;
231                    }
232                }
233
234                // Make new map copy with potentially split buckets
235                buckets = new TreeMap<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry>(p.getBuckets());
236                for (Map.Entry<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry> entry
237                        : buckets.entrySet()) {
238                    assert(entry.getValue().getState() == ProgressToken.BucketState.BUCKET_PENDING);
239                    BucketId pending = new BucketId(ProgressToken.keyToBucketId(entry.getKey().getKey()));
240                    if (pending.getUsedBits() > targetDistBits) {
241                        // If this is the right sibling of an already merged left sibling,
242                        // it's already been merged away, so we should skip it
243                        if (lastMergedBucket != null) {
244                            BucketId rightCheck = new BucketId(lastMergedBucket.getUsedBits(),
245                                    lastMergedBucket.getId() | (1L << (lastMergedBucket.getUsedBits() - 1)));
246                            if (pending.equals(rightCheck)) {
247                                if (log.isLoggable(LogLevel.SPAM)) {
248                                    log.log(LogLevel.SPAM, "Skipped " + pending +
249                                            ", as it was right sibling of " + lastMergedBucket);
250                                }
251                                continue;
252                            }
253                        }
254                        if (pending.getUsedBits() - 1 > targetDistBits) {
255                            maybeInconsistent = true; // Do another pass
256                        }
257                        p.mergePendingBucket(pending);
258                        ++bucketsMerged;
259
260                        lastMergedBucket = pending;
261                    }
262                }
263            }
264            if ((bucketsSplit > 0 || bucketsMerged > 0) && log.isLoggable(LogLevel.DEBUG)) {
265                log.log(LogLevel.DEBUG, "Existing progress' pending buckets had inconsistent " +
266                        "distribution bits; performed " + bucketsSplit + " split ops and " +
267                        bucketsMerged + " merge ops. Pending: " + pendingBefore + " -> " +
268                        p.getPendingBucketCount());
269            }
270        }
271
272        private void correctTruncatedBucketCursor() {
273            // We've truncated the bucket cursor, but in doing so we might
274            // have moved back beyond where there are pending buckets. Consider
275            // having a cursor value of 3 at 31 bits and then moving to 11 bits.
276            // With 1 pending we'll normally reach a cursor of 0, even though it
277            // should be 1
278            for (ProgressToken.BucketKeyWrapper bucketKey
279                    : progressToken.getBuckets().keySet()) {
280                BucketId bid = bucketKey.toBucketId();
281                long idx = bucketKey.getKey() >>> (64 - bid.getUsedBits());
282                if (bid.getUsedBits() == distributionBitCount
283                        && idx >= progressToken.getBucketCursor()) {
284                    progressToken.setBucketCursor(idx + 1);
285                }
286            }
287            if (log.isLoggable(LogLevel.SPAM)) {
288                log.log(LogLevel.SPAM, "New range bucket cursor is " +
289                        progressToken.getBucketCursor());
290            }
291        }

294            return progressToken.getBucketCursor() < (1L << distributionBitCount);
295        }

297        public boolean shouldYield() {
298            // If we need to flush all active buckets, stall the iteration until

304            return true;

308            return 1L << distributionBitCount;

311        public BucketProgress getNext() {
312            assert(hasNext()) : "getNext() called with hasNext() == false";
313            long currentPosition = progressToken.getBucketCursor();
314            long key = ProgressToken.makeNthBucketKey(currentPosition, distributionBitCount);
315            ++currentPosition;
316            progressToken.setBucketCursor(currentPosition);
317            return new BucketProgress(
318                    new BucketId(ProgressToken.keyToBucketId(key)),
319                    new BucketId());
320        }

326        public void setDistributionBitCount(int distributionBitCount,
327                                            ProgressToken progress)
328        {
329            this.distributionBitCount = distributionBitCount;
330
331            // There might be a case where we're waiting for active buckets
332            // already when a new distribution bit change comes in. If so,
333            // don't do anything at all yet with the set of pending
334            if (progressToken.getActiveBucketCount() > 0) {
335                flushActive = true;
336                if (log.isLoggable(LogLevel.DEBUG)) {
337                    log.log(LogLevel.DEBUG, "Holding off new/pending buckets and consistency " +
338                            "correction until all " + progress.getActiveBucketCount() +
339                            " active buckets have been updated");
340                }
341                progressToken.setInconsistentState(true);
342            } else {
343                // Only perform the actual distribution bit bucket ops if we've
344                // got no pending buckets
345                int delta = distributionBitCount - progressToken.getDistributionBitCount();
346
347                // Must do this before setting the bucket cursor to allow
348                // reset-checking to be performed
349                correctInconsistentPending(distributionBitCount);
350                if (delta > 0) {
351                    if (log.isLoggable(LogLevel.DEBUG)) {
352                        log.log(LogLevel.DEBUG, "Increasing distribution bits for full bucket " +
353                                "space range source from " + progressToken.getDistributionBitCount() + " to " +
354                                distributionBitCount);
355                    }
356                    progressToken.setFinishedBucketCount(progressToken.getFinishedBucketCount() << delta);
357                    // By n-doubling the position, the bucket key ordering ensures
358                    // we go from eg. 3:0x02 to 4:0x02 to 5:02 etc.
359                    progressToken.setBucketCursor(progressToken.getBucketCursor() << delta);
360                } else if (delta < 0) {
361                    if (log.isLoggable(LogLevel.DEBUG)) {
362                        log.log(LogLevel.DEBUG, "Decreasing distribution bits for full bucket " +
363                                "space range source from " + progressToken.getDistributionBitCount() +
364                                " to " + distributionBitCount + " bits");
365                    }
366                    // Scale down bucket space and cursor
367                    progressToken.setBucketCursor(progressToken.getBucketCursor() >>> -delta);
368                    progressToken.setFinishedBucketCount(progressToken.getFinishedBucketCount() >>> -delta);
369                }
370
371                progressToken.setTotalBucketCount(1L << distributionBitCount);
372                progressToken.setDistributionBitCount(distributionBitCount);
373
374                correctTruncatedBucketCursor();
375                progressToken.setInconsistentState(false);
376            }
377        }

379        public void update(BucketId superbucket, BucketId progress,
380                           ProgressToken token) {
381            progressToken.updateProgress(superbucket, progress);
382
383            if (superbucket.getUsedBits() != distributionBitCount) {
384                if (!progress.equals(ProgressToken.FINISHED_BUCKET)) {
385                    // We should now always flush active buckets before doing a
386                    // consistency fix. This simplifies things greatly
387                    assert(flushActive);
388                    if (log.isLoggable(LogLevel.DEBUG)) {
389                        log.log(LogLevel.DEBUG, "Received non-finished bucket " +
390                                superbucket + " with wrong distribution bit count (" +
391                                superbucket.getUsedBits() + "). Waiting to correct " +
392                                "until all active are done");
393                    }
394                } else {
395                    if (log.isLoggable(LogLevel.DEBUG)) {
396                        log.log(LogLevel.DEBUG, "Received finished bucket " +
397                                superbucket + " with wrong distribution bit count (" +
398                                superbucket.getUsedBits() + "). Waiting to correct " +
399                                "until all active are done");
400                    }
401                }
402            }
403
404            if (progressToken.getActiveBucketCount() == 0) {
405                if (flushActive) {
406                    if (log.isLoggable(LogLevel.DEBUG)) {
407                        log.log(LogLevel.DEBUG, "All active buckets flushed, " +
408                                "correcting progress token and continuing normal operation");
409                    }
410                    // Trigger the actual bucket state change this time
411                    setDistributionBitCount(distributionBitCount, progressToken);
412                    assert(progressToken.getDistributionBitCount() == distributionBitCount);
413                }
414                flushActive = false;
415                // Update #finished since we might have had inconsistent active
416                // buckets that have prevent us from getting a correct value. At
417                // this point, however, all pending buckets should presumably be
418                // at the same, correct dist bit count, so we can safely compute
419                // a new count
420                // TODO: ensure this is consistent
421                if (progressToken.getPendingBucketCount() <= progressToken.getBucketCursor()) {
422                    progressToken.setFinishedBucketCount(progressToken.getBucketCursor() -
423                            progressToken.getPendingBucketCount());
424                }
425            }
426        }

429    /**
430     * Provides an explicit set of bucket IDs to iterate over. Will immediately
431     * set these as pending in the {@link ProgressToken}, as it is presumed this set is
432     * rather small. Changing the distribution bit count for this source is
433     * effectively a no-op, as explicit bucket IDs should not be implicitly
434     * changed.
435     */
436    protected static class ExplicitBucketSource implements BucketSource {
437        private int distributionBitCount;
438        private long totalBucketCount = 0;
439
440        public ExplicitBucketSource(Set<BucketId> superbuckets,
441                                    int distributionBitCount,
442                                    ProgressToken progress) {
443            this.distributionBitCount = progress.getDistributionBitCount();
444            this.totalBucketCount = superbuckets.size();
445
446            // New progress token?
447            if (progress.getTotalBucketCount() == 0) {
448                progress.setTotalBucketCount(this.totalBucketCount);
449                progress.setDistributionBitCount(distributionBitCount);
450                this.distributionBitCount = distributionBitCount;
451            }
452            else {
453                // Quick consistency check to ensure the user isn't trying to eg.
454                // pass a progress token for another document selection
455                if (progress.getTotalBucketCount() != totalBucketCount
456                        || (progress.getFinishedBucketCount() + progress.getPendingBucketCount()
457                            + progress.getActiveBucketCount() != totalBucketCount)) {
458                    throw new IllegalArgumentException("Total bucket count in existing progress is not " +
459                            "consistent with that of the current document selection");
460                }
461                if (progress.getBucketCursor() != 0) {
462                    // Trying to use a range source progress file
463                    throw new IllegalArgumentException("Cannot use given progress file with the "+
464                            "current document selection");
465                }
466                this.distributionBitCount = progress.getDistributionBitCount();
467            }
468
469            if (progress.isFinished() || !progress.isEmpty()) return;
470
471            for (BucketId id : superbuckets) {
472                // Add all superbuckets with zero sub-bucket progress and pending
473                progress.addBucket(id, new BucketId(), ProgressToken.BucketState.BUCKET_PENDING);
474            }
475        }

503        public void setDistributionBitCount(int distributionBitCount,
504                                            ProgressToken progress)
505        {
506            // Setting distribution bits for explicit bucket source is essentially
507            // a no-op, since its buckets already are fixed at 32 used bits.
508            progress.setDistributionBitCount(distributionBitCount);
509            this.distributionBitCount = distributionBitCount;
510            if (log.isLoggable(LogLevel.DEBUG)) {
511                log.log(LogLevel.DEBUG, "Set distribution bit count to "
512                        + distributionBitCount + " for explicit bucket source (no-op)");
513            }
514        }

516        public void update(BucketId superbucket, BucketId progress,
517                           ProgressToken token) {
518            // Simply delegate to ProgressToken, as it maintains all progress state
519            token.updateProgress(superbucket, progress);
520        }

524     * @param bucketSource An instance of {@link BucketSource}, providing the working set for
525     * the iterator
526     * @param progressToken A {@link ProgressToken} instance, allowing the progress of
527     * finished or partially finished buckets to be tracked
528     *
529     * @see BucketSource
530     * @see ProgressToken
531     */
532    private VisitorIterator(ProgressToken progressToken,
533                            BucketSource bucketSource) {
534        assert(progressToken.getDistributionBitCount() == bucketSource.getDistributionBitCount())
535                : "inconsistent distribution bit counts";
536        this.distributionBitCount = progressToken.getDistributionBitCount();
537        this.progressToken = progressToken;
538        this.bucketSource = bucketSource;
539    }

542    /**
543     * @return The pair [superbucket, progress] that specifies the next iterable
544     * bucket. When a superbucket is initially returned, the pair is equal to
545     * that of [superbucket, 0], as there has been no progress into its sub-buckets
546     * yet (if they exist).
547     *
548     * Precondition: <code>hasNext() == true</code>
549     */
550    public BucketProgress getNext() {
551        assert(progressToken.getDistributionBitCount() == bucketSource.getDistributionBitCount())
552                : "inconsistent distribution bit counts for progress and source";
553        assert(hasNext());
554        // We prioritize returning buckets in the pending map over those
555        // that may be in the bucket source, since we want to avoid growing
556        // the map too much
557        if (progressToken.hasPending()) {
558            // Find first pending bucket in token
559            TreeMap<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry> buckets = progressToken.getBuckets();
560            ProgressToken.BucketEntry pending = null;
561            BucketId superbucket = null;
562            for (Map.Entry<ProgressToken.BucketKeyWrapper, ProgressToken.BucketEntry> entry : buckets.entrySet()) {
563                if (entry.getValue().getState() == ProgressToken.BucketState.BUCKET_PENDING) {
564                    pending = entry.getValue();
565                    superbucket = new BucketId(ProgressToken.keyToBucketId(entry.getKey().getKey()));
566                    break;
567                }
568            }
569            assert(pending != null) : "getNext() called with inconsistent state";
570
571            // Set bucket to active, since it's not awaiting an update
572            pending.setState(ProgressToken.BucketState.BUCKET_ACTIVE);
573
574            progressToken.setActiveBucketCount(progressToken.getActiveBucketCount() + 1);
575            progressToken.setPendingBucketCount(progressToken.getPendingBucketCount() - 1);
576
577            return new BucketProgress(superbucket, pending.getProgress());
578        } else {
579            BucketProgress ret = bucketSource.getNext();
580            progressToken.addBucket(ret.getSuperbucket(), ret.getProgress(),
581                    ProgressToken.BucketState.BUCKET_ACTIVE);
582            return ret;
583        }
584    }

586    /**
587     * <p>Check whether or not it is valid to call {@link #getNext()} with the current
588     * iterator state.</p>
589     *
590     * <p>There exists a case wherein <code>hasNext</code> may return false before {@link #update} is
591     * called, but true afterwards. This happens when the set of pending buckets is
592     * empty, the bucket source is empty <em>but</em> the set of active buckets is
593     * not. A future progress update on any of the buckets in the active set may
594     * or may not make that bucket available to the pending set again.
595     * This must be handled explicitly by the caller by checking {@link #isDone()}
596     * and ensuring that {@link #update} is called before retrying <code>hasNext</code>.</p>
597     *
598     * <p>This method will also return false if the number of distribution bits have
599     * changed and there are active buckets needing to be flushed before the
600     * iterator will allow new buckets to be handed out.</p>
601     *
602     * @return Whether or not it is valid to call {@link #getNext()} with the current
603     * iterator state.
604     */
605    public boolean hasNext() {
606        return (progressToken.hasPending() || bucketSource.hasNext()) && !bucketSource.shouldYield();
607    }
608
609    /**
610     * Check if the iterator is actually done
611     *
612     * @see #hasNext()
613     *
614     * @return <code>true</code> <em>iff</em> the bucket source is empty and
615     * there are no pending or active buckets in the progress token.
616     */
617    public boolean isDone() {
618        return !(hasNext() || progressToken.hasActive());
619    }

621    /**
622     * <p>Tell the iterator that we've finished processing up to <i>and
623     * including</i> <code>progress</code>. <code>progress</code> may be a sub-bucket <i>or</i>
624     * the invalid 0-bucket (in case the caller fails to process the bucket and
625     * must return it to the set of pending) <em>or</em> the special case <code>BucketId(Integer.MAX_VALUE)</code>,
626     * the latter indicating to the iterator that traversal is complete for
627     * <code>superbucket</code>'s tree. The null bucket should only be used if no
628     * non-null updates have yet been given for the superbucket.</p>
629     *
630     * <p>It is a requirement that each superbucket returned by {@link #getNext()} must
631     * eventually result in 1-n update operations, where the last update operation
632     * has the special progress==super case.</p>
633     *
634     * <p>If the document selection used to create the iterator is unknown and there
635     * were active buckets at the time of a distribution bit state change, such
636     * a bucket passed to <code>update()</code> will be in an inconsistent state
637     * with regards to the number of bits it uses. For unfinished buckets, this
638     * is handled by splitting or merging it until it's consistent, depending on
639     * whether or not it had a lower or higher distribution bit count than that of
640     * the current system state. For finished buckets of a lower dist bit count,
641     * the amount of finished buckets in the ProgressToken is adjusted upwards
642     * to compensate for the fact that a bucket using fewer distribution bits
643     * actually covers more of the bucket space than the ones that are currently
644     * in use. For finished buckets of a higher dist bit count, the number of
645     * finished buckets is <em>not</em> increased at that point in time, since
646     * such a bucket doesn't actually cover an entire bucket with the current state.</p>
647     *
648     * <p>All this is done automatically and transparently to the caller once all
649     * active buckets have been updated.</p>
650     *
651     * @param superbucket A valid bucket ID that has been retrieved earlier through
652     * {@link #getNext()}
653     * @param progress A bucket logically contained within <code>super</code>. Subsequent
654     * updates for the same superbucket must have <code>progress</code> be in an increasing
655     * order, where order is defined as the in-order traversal of the bucket split
656     * tree. May also be the null bucket if the superbucket has not seen any "proper"
657     * progress updates yet or the special case Integer.MAX_VALUE. Note that inconsistent
658     * splitting might actually see <code>progress</code> as containing <code>super</code>
659     * rather than vice versa, so this is explicitly allowed to pass by the code.
660     */
661    public void update(BucketId superbucket, BucketId progress) {
662        // Delegate to bucket source, as it knows how to deal with buckets
663        // that are in an inconsistent state wrt distribution bit count
664        bucketSource.update(superbucket, progress, progressToken);
665    }

692    /**
693     * <p>Set the distribution bit count for the iterator and the buckets it
694     * currently maintains and will return in the future.</p>
695     *
696     * <p>For document selections that result in a explicit set of buckets, this
697     * is essentially a no-op, so in such a case, disregard the rest of this text.</p>
698     *
699     * <p>Changing the number of distribution bits for an unknown document
700     * selection will effectively scale the bucket space that will be visited;
701     * each bit increase or decrease doubling or halving its size, respectively.
702     * When increasing, any pending buckets will be split to ensure the total
703     * bucket space covered remains the same. Correspondingly, when decreasing,
704     * any pending buckets will be merged appropriately.</p>
705     *
706     * <p>If there are buckets active at the time of the change, the actual
707     * bucket splitting/merging operations are kept on hold until all active
708     * buckets have been updated, at which point they will be automatically
709     * performed. The iterator will force such an update by not giving out
710     * any new or pending buckets until that happens.</p>
711     *
712     * <p><em>Note:</em> when decreasing the number of distribution bits,
713     * there is a chance of losing superbucket progress in a bucket that
714     * is merged with another bucket, leading to potential duplicate
715     * results.</p>
716     *
717     * @param distBits New system state distribution bit count
718     */
719    public void setDistributionBitCount(int distBits) {
720        if (distributionBitCount != distBits) {
721            bucketSource.setDistributionBitCount(distBits, progressToken);
722            distributionBitCount = distBits;
723            if (log.isLoggable(LogLevel.DEBUG)) {
724                log.log(LogLevel.DEBUG, "Set visitor iterator distribution bit count to "
725                        + distBits);
726            }
727        }
728    }

734    /**
735     * Create a new <code>VisitorIterator</code> instance based on the given document
736     * selection string.
737     *
738     * @param documentSelection Document selection string used to create the
739     * <code>VisitorIterator</code> instance. Depending on the characteristics of the
740     * selection, the iterator may iterate over only a small subset of the buckets or
741     * every bucket in the system. Both cases will be handled efficiently.
742     * @param idFactory {@link BucketId} factory specifying the number of distribution bits
743     * to use et al.
744     * @param progress A unique {@link ProgressToken} instance which is used for maintaining the state
745     * of the iterator. Can <em>not</em> be shared with other iterator instances at the same time.
746     * If <code>progress</code> contains work done in an earlier iteration run, the iterator will pick
747     * up from where it left off
748     * @return A new <code>VisitorIterator</code> instance
749     * @throws ParseException if <code>documentSelection</code> fails to properly parse
750     */
751    public static VisitorIterator createFromDocumentSelection(
752            String documentSelection,
753            BucketIdFactory idFactory,
754            int distributionBitCount,
755            ProgressToken progress) throws ParseException {
756        BucketSelector bucketSel = new BucketSelector(idFactory);
757        Set<BucketId> rawBuckets = bucketSel.getBucketList(documentSelection);
758        BucketSource src;
759
760        // Depending on whether the expression yielded an unknown number of
761        // buckets, we create either an explicit bucket source or a distribution
762        // bit-based range source
763        if (rawBuckets == null) {
764            // Range source
765            src = new DistributionRangeBucketSource(distributionBitCount, progress);
766        } else {
767            // Explicit source
768            src = new ExplicitBucketSource(rawBuckets, distributionBitCount, progress);
769        }
770
771        return new VisitorIterator(progress, src);
772    }

774    /**
775     * Create a new <code>VisitorIterator</code> instance based on the given
776     * set of buckets. This is supported for internal use only, and is required
777     * by Synchronization. Use {@link #createFromDocumentSelection} instead for
778     * all normal purposes.
779     *
780     * @param bucketsToVisit The set of buckets that will be visited
781     * @param distributionBitCount Number of distribution bits to use
782     * @param progress A unique ProgressToken instance which is used for maintaining the state
783     * of the iterator. Can <em>not</em> be shared with other iterator instances at the same time.
784     * If <code>progress</code> contains work done in an earlier iteration run, the iterator will pick
785     * up from where it left off
786     * @return A new <code>VisitorIterator</code> instance
787     */
788    public static VisitorIterator createFromExplicitBucketSet(
789            Set<BucketId> bucketsToVisit,
790            int distributionBitCount,
791            ProgressToken progress) {
792        // For obvious reasons, always create an explicit source here
793        BucketSource src = new ExplicitBucketSource(bucketsToVisit,
794                distributionBitCount, progress);
795        return new VisitorIterator(progress, src);
796    }
