002package com.yahoo.documentapi;
003
004import com.yahoo.document.BucketId;
005import com.yahoo.documentapi.messagebus.protocol.DocumentListEntry;
006
007import java.util.List;
008
009/**
010 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
011 */
012public class BucketListVisitorResponse extends VisitorResponse {
013    private BucketId bucketId;
014    private List<DocumentListEntry> documents;
015
016    public BucketListVisitorResponse(BucketId bucketId, List<DocumentListEntry> documents, AckToken token) {
017        super(token);
018        this.bucketId = bucketId;
019        this.documents = documents;
020    }
021
022    public BucketId getBucketId() {
023        return bucketId;
024    }
025
026    public List<DocumentListEntry> getDocuments() {
027        return documents;
028    }
029}
