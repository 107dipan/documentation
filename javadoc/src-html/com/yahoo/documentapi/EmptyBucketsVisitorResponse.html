002package com.yahoo.documentapi;
003
004import com.yahoo.document.BucketId;
005
006import java.util.List;
007
008/**
009 * Response containing list of empty buckets.
010 *
011 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
012 */
013public class EmptyBucketsVisitorResponse extends VisitorResponse {
014    private List<BucketId> bucketIds;
015    /**
016     * Creates visitor response containing an ack token.
017     *
018     * @param bucketIds the empty buckets
019     * @param token the ack token
020     */
021    public EmptyBucketsVisitorResponse(List<BucketId> bucketIds, AckToken token) {
022        super(token);
023        this.bucketIds = bucketIds;
024    }
025
026    public List<BucketId> getBucketIds() { return bucketIds; }
027}
