002package com.yahoo.documentapi;
003
004/**
005 * Superclass of all document api sessions. A session provides a handle through
006 * which an application can work with a document repository. There are various
007 * session subclasses for various types of interaction with the repository.
008 * <p>
009 * Each session can be used by multiple client application threads, i.e they are
010 * multithread safe.
011 *
012 * @author bratseth
013 */
014public interface Session {
015
016    /**
017     * Returns the next response of this session. This method returns immediately.
018     *
019     * @return the next response, or null if no response is ready at this time
020     */
021    public Response getNext();
022
023    /**
024     * Returns the next response of this session. This will block until a response is ready
025     * or until the given timeout is reached
026     *
027     * @param timeoutMilliseconds the max time to wait for a response.
028     * @return the next response, or null if no response becomes ready before the timeout expires
029     * @throws InterruptedException if this thread is interrupted while waiting
030     */
031    public Response getNext(int timeoutMilliseconds) throws InterruptedException;
032
033    /**
034     * Destroys this session and frees up any resources it has held. Making further calls on a destroyed
035     * session causes a runtime exception.
036     */
037    public void destroy();
038
039}
