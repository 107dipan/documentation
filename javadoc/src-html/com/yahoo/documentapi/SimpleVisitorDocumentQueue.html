002package com.yahoo.documentapi;
003
004import com.yahoo.document.Document;
005import com.yahoo.document.DocumentId;
006import com.yahoo.vdslib.DocumentList;
007
008import java.util.LinkedList;
009import java.util.List;
010
011/**
012 * A simple document queue that queues up all results and automatically acks
013 * them.
014 * <p>
015 * Retrieving the list is not thread safe, so wait until visitor is done. This
016 * is a simple class merely meant for testing.
017 *
018 * @author <a href="mailto:humbe@yahoo-inc.com">H&aring;kon Humberset</a>
019 */
020public class SimpleVisitorDocumentQueue extends DumpVisitorDataHandler {
021    private final List<Document> documents = new LinkedList<Document>();
022
023    // Inherit doc from VisitorDataHandler
024    public void reset() {
025        super.reset();
026        documents.clear();
027    }
028
029    @Override
030    public void onDocument(Document doc, long timestamp) {
031        documents.add(doc);
032    }
033
034    public void onRemove(DocumentId docId) {}
035
036    /** @return a list of all documents retrieved so far */
037    public List<Document> getDocuments() {
038        return documents;
039    }
040
041}
