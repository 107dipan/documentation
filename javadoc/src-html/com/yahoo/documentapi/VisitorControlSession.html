002package com.yahoo.documentapi;
003
004/**
005 * Superclass for document <i>visiting</i> functionality - accessing
006 * documents in an order decided by the document repository. This allows much
007 * higher read throughput than random access.
008 * <p>
009 * The class supplies an interface for functions that are common for different
010 * kinds of visitor sessions, such as acking visitor data and aborting the
011 * session.
012 *
013 * @author <a href="mailto:humbe@yahoo-inc.com">H&aring;kon Humberset</a>
014 */
015public interface VisitorControlSession {
016    /**
017     * Acknowledges a response previously retrieved by the <code>getNext</code>
018     * method.
019     *
020     * @param token The ack token. You must get this from the visitor response
021     *              returned by the <code>getNext</code> method.
022     */
023    public void ack(AckToken token);
024
025    /**
026     * Aborts the session.
027     */
028    public void abort();
029
030    /**
031     * Returns the next response of this session. This method returns immediately.
032     *
033     * @return the next response, or null if no response is ready at this time
034     */
035    public VisitorResponse getNext();
036
037    /**
038     * Returns the next response of this session. This will block until a response is ready
039     * or until the given timeout is reached
040     *
041     * @param timeoutMilliseconds the max time to wait for a response. If the number is 0, this will block
042     *                            without any timeout limit
043     * @return the next response, or null if no response becomes ready before the timeout expires
044     * @throws InterruptedException if this thread is interrupted while waiting
045     */
046    public VisitorResponse getNext(int timeoutMilliseconds) throws InterruptedException;
047
048    /**
049     * Destroys this session and frees up any resources it has held.
050     */
051    public void destroy();
052
053}
