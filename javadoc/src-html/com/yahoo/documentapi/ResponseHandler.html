003
004/**
005 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
006 */
007public interface ResponseHandler {
008
009    /**
010     * This method is called once for each document api operation invoked on a {@link AsyncSession}. There is no
011     * guarantee as to which thread calls this, so any implementation must be thread-safe.
012     *
013     * @param response The response to process.
014     */
015    public void handleResponse(Response response);
016}
