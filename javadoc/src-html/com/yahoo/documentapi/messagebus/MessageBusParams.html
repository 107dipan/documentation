002package com.yahoo.documentapi.messagebus;
003
004import com.yahoo.documentapi.DocumentAccessParams;
005import com.yahoo.documentapi.messagebus.loadtypes.LoadTypeSet;
006import com.yahoo.messagebus.SourceSessionParams;
007import com.yahoo.messagebus.network.rpc.RPCNetworkParams;
008
009/**
010 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
011 */
012public class MessageBusParams extends DocumentAccessParams {
013
014    private String routingConfigId = null;
015    private String protocolConfigId = null;
016    private String route = "route:default";
017    private int traceLevel = 0;
018    private RPCNetworkParams rpcNetworkParams = new RPCNetworkParams();
019    private com.yahoo.messagebus.MessageBusParams mbusParams = new com.yahoo.messagebus.MessageBusParams();
020    private SourceSessionParams sourceSessionParams = new SourceSessionParams();
021    private LoadTypeSet loadTypes;
022
023    public MessageBusParams() {
024        this(new LoadTypeSet());
025    }
026
027    public MessageBusParams(LoadTypeSet loadTypes) {
028        this.loadTypes = loadTypes;
029    }
030
031    /**
032     *
033     * @return Returns the set of load types accepted by this Vespa installation
034     */
035    public LoadTypeSet getLoadTypes() {
036        return loadTypes;
037    }
038
039    /**
040     * Returns the id to resolve to routing config.
041     *
042     * @return The config id.
043     */
044    public String getRoutingConfigId() {
045        return routingConfigId;
046    }
047
048    /**
049     * Sets the id to resolve to routing config. This has a proper default value that holds for Vespa applications, and
050     * can therefore be left unset.
051     *
052     * @param configId The config id.
053     * @return This object for chaining.
054     */
055    public MessageBusParams setRoutingConfigId(String configId) {
056        routingConfigId = configId;
057        return this;
058    }
059
060    /**
061     * Returns the id to resolve to protocol config.
062     *
063     * @return The config id.
064     */
065    public String getProtocolConfigId() {
066        return protocolConfigId;
067    }
068
069    /**
070     * Sets the id to resolve to protocol config. This has a proper default value that holds for Vespa applications,
071     * and can therefore be left usnet.
072     *
073     * @param configId The config id.
074     * @return This, to allow chaining.
075     */
076    public MessageBusParams setProtocolConfigId(String configId) {
077        protocolConfigId = configId;
078        return this;
079    }
080
081    /**
082     * Sets the name of the route to send appropriate requests to. This is a convenience method for prefixing a route
083     * with "route:", and using {@link #setRoute} instead.
084     *
085     * @param routeName The route name.
086     * @return This object for chaining.
087     */
088    public MessageBusParams setRouteName(String routeName) {
089        return setRoute("route:" + routeName);
090    }
091
092    /**
093     * Sets the route string to send all requests to. This string will be parsed as a route string, so setting a route
094     * name directly will not necessarily have the intended consequences. Use "route:&lt;routename&gt;" syntax for route
095     * names, or the convenience method {@link #setRouteName} for this.
096     *
097     * @param route The route string.
098     * @return This object for chaining.
099     */
100    public MessageBusParams setRoute(String route) {
101        this.route = route;
102        return this;
103    }
104
105    /**
106     * Returns the route string that all requests will be sent to.
107     *
108     * @return The route string.
109     */
110    public String getRoute() {
111        return route;
112    }
113
114    /**
115     * Returns the trace level to use when sending.
116     *
117     * @return The trace level.
118     */
119    public int getTraceLevel() {
120        return traceLevel;
121    }
122
123    /**
124     * Sets the trace level to use when sending.
125     *
126     * @param traceLevel The trace level.
127     * @return This object for chaining.
128     */
129    public MessageBusParams setTraceLevel(int traceLevel) {
130        this.traceLevel = traceLevel;
131        return this;
132    }
133
134    /**
135     * Returns the params object used to instantiate the rpc network layer for message bus.
136     *
137     * @return The params object.
138     */
139    public RPCNetworkParams getRPCNetworkParams() {
140        return rpcNetworkParams;
141    }
142
143    /**
144     * Sets the params object used to instantiate the rpc network layer for message bus.
145     *
146     * @param params The params object.
147     * @return This object for chaining.
148     */
149    public MessageBusParams setRPCNetworkParams(RPCNetworkParams params) {
150        rpcNetworkParams = new RPCNetworkParams(params);
151        return this;
152    }
153
154    /**
155     * Returns the params object used to instantiate the message bus.
156     *
157     * @return The params object.
158     */
159    public com.yahoo.messagebus.MessageBusParams getMessageBusParams() {
160        return mbusParams;
161    }
162
163    /**
164     * Sets the params object used to instantiate the message bus.
165     *
166     * @param params The params object.
167     * @return This object for chaining.
168     */
169    public MessageBusParams setMessageBusParams(com.yahoo.messagebus.MessageBusParams params) {
170        mbusParams = new com.yahoo.messagebus.MessageBusParams(params);
171        return this;
172    }
173
174    /**
175     * Returns a reference to the extended source session params object.
176     *
177     * @return The params object.
178     */
179    public SourceSessionParams getSourceSessionParams() {
180        return sourceSessionParams;
181    }
182
183    /**
184     * Sets the extended source session params.
185     *
186     * @param params The params object.
187     * @return This object for chaining.
188     */
189    public MessageBusParams setSourceSessionParams(SourceSessionParams params) {
190        sourceSessionParams = new SourceSessionParams(params);
191        return this;
192    }
193}
