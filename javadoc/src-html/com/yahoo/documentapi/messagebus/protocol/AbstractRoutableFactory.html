002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.text.Utf8;
005import com.yahoo.vespa.objects.Deserializer;
006import com.yahoo.vespa.objects.Serializer;
007
008/**
009 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
010 */
011public abstract class AbstractRoutableFactory implements RoutableFactory {
012
013    /**
014     * Reads a string from the given buffer that was previously written by {@link #encodeString(String,
015     * com.yahoo.vespa.objects.Serializer)}.
016     *
017     * @param in The byte buffer to read from.
018     * @return The decoded string.
019     */
020    public static String decodeString(Deserializer in) {
021        int length = in.getInt(null);
022        if (length == 0) {
023            return "";
024        }
025        return Utf8.toString(in.getBytes(null, length));
026    }
027
028    /**
029     * Writes the given string to the given byte buffer in such a way that it can be decoded using {@link
030     * #decodeString(com.yahoo.vespa.objects.Deserializer)}.
031     *
032     * @param str The string to encode.
033     * @param out The byte buffer to write to.
034     */
035    public static void encodeString(String str, Serializer out) {
036        if (str == null || str.isEmpty()) {
037            out.putInt(null, 0);
038        } else {
039            byte[] nameBytes = Utf8.toBytes(str);
040            out.putInt(null, nameBytes.length);
041            out.put(null, nameBytes);
042        }
043    }
044}
