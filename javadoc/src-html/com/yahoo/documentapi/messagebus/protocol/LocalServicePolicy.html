002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.jrt.slobrok.api.Mirror;
005import com.yahoo.messagebus.metrics.MetricSet;
006import com.yahoo.messagebus.routing.*;
007
008import java.util.ArrayList;
009import java.util.HashMap;
010import java.util.List;
011import java.util.Map;
012
013/**
014 * This policy implements the logic to prefer local services that matches a slobrok pattern.
015 *
016 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
017 */
018public class LocalServicePolicy implements DocumentProtocolRoutingPolicy {
019
020    private final String localAddress;
021    private Map<String, CacheEntry> cache = new HashMap<String, CacheEntry>();
022
023    /**
024     * Constructs a policy that will choose local services that match the slobrok pattern in which this policy occured.
025     * If no local service can be found, this policy simply returns the asterisk to allow the network to choose any.
026     *
027     * @param param The address to use for this, if empty this will resolve to hostname.
028     */
029    public LocalServicePolicy(String param) {
030        localAddress = (param != null && param.length() > 0) ? param : null;
031    }
032
033    // Inherit doc from RoutingPolicy.
034    public void select(RoutingContext ctx) {
035        Route route = new Route(ctx.getRoute());
036        route.setHop(0, getRecipient(ctx));
037        ctx.addChild(route);
038    }
039
040    // Inherit doc from RoutingPolicy.
041    public void merge(RoutingContext ctx) {
042        DocumentProtocol.merge(ctx);
043    }
044
045    /**
046     * Returns the appropriate recipient hop for the given routing context. This method provides synchronized access to
047     * the internal cache.
048     *
049     * @param ctx The routing context.
050     * @return The recipient hop to use.
051     */
052    private synchronized Hop getRecipient(RoutingContext ctx) {
053        CacheEntry entry = update(ctx);
054        if (entry.recipients.isEmpty()) {
055            Hop hop = new Hop(ctx.getRoute().getHop(0));
056            hop.setDirective(ctx.getDirectiveIndex(), new VerbatimDirective("*"));
057            return hop;
058        }
059        if (++entry.offset >= entry.recipients.size()) {
060            entry.offset = 0;
061        }
062        return new Hop(entry.recipients.get(entry.offset));
063    }
064
065    /**
066     * Updates and returns the cache entry for the given routing context. This method assumes that synchronization is
067     * handled outside of it.
068     *
069     * @param ctx The routing context.
070     * @return The updated cache entry.
071     */
072    private CacheEntry update(RoutingContext ctx) {
073        String key = getCacheKey(ctx);
074        CacheEntry entry = cache.get(key);
075        if (entry == null) {
076            entry = new CacheEntry();
077            cache.put(key, entry);
078        }
079        int upd = ctx.getMirror().updates();
080        if (entry.generation != upd) {
081            entry.generation = upd;
082            entry.recipients.clear();
083
084            Mirror.Entry[] arr = ctx.getMirror().lookup(ctx.getHopPrefix() + "*" + ctx.getHopSuffix());
085            String self = localAddress != null ? localAddress : toAddress(ctx.getMessageBus().getConnectionSpec());
086            for (Mirror.Entry item : arr) {
087                if (self.equals(toAddress(item.getSpec()))) {
088                    entry.recipients.add(Hop.parse(item.getName()));
089                }
090            }
091        }
092        return entry;
093    }
094
095    /**
096     * Returns a cache key for this instance of the policy. Because behaviour is based on the hop in which the policy
097     * occurs, the cache key is the hop string itself.
098     *
099     * @param ctx The routing context.
100     * @return The cache key.
101     */
102    private String getCacheKey(RoutingContext ctx) {
103        return ctx.getRoute().getHop(0).toString();
104    }
105
106    /**
107     * Defines the necessary cache data.
108     */
109    private class CacheEntry {
110        private final List<Hop> recipients = new ArrayList<Hop>();
111        private int generation = 0;
112        private int offset = 0;
113    }
114
115    /**
116     * Searches the given connection spec for a hostname or IP address. If an address is not found, this method returns
117     * null.
118     *
119     * @param connection The connection spec to search.
120     * @return The address, may be null.
121     */
122    private static String toAddress(String connection) {
123        if (connection.startsWith("tcp/")) {
124            int pos = connection.indexOf(':');
125            if (pos > 4) {
126                return connection.substring(4, pos);
127            }
128        }
129        return null;
130    }
131
132    public void destroy() {
133    }
134
135    public MetricSet getMetrics() {
136        return null;
137    }
138}
