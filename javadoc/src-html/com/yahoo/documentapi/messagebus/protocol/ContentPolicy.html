002package com.yahoo.documentapi.messagebus.protocol;
003
004import java.util.Map;
005
006/**
007 * Policy to talk to content clusters.
008 */
009public class ContentPolicy extends StoragePolicy {
010
011    public static class ContentParameters extends Parameters {
012
013        public ContentParameters(Map<String, String> parameters) {
014            super(parameters);
015        }
016
017        @Override
018        public String getDistributionConfigId() {
019            if (distributionConfigId != null) {
020                return distributionConfigId;
021            }
022            return clusterName;
023        }
024
025        @Override
026        public SlobrokHostPatternGenerator createPatternGenerator() {
027            return new SlobrokHostPatternGenerator(getClusterName()) {
028                public String getDistributorHostPattern(Integer distributor) {
029                    return "storage/cluster." + getClusterName() + "/distributor/" + (distributor == null ? "*" : distributor) + "/default";
030                }
031            };
032        }
033    }
034
035    public ContentPolicy(Map<String, String> params) {
036        super(new ContentParameters(params), params);
037    }
038
039    public ContentPolicy(String parameters) {
040        this(parse(parameters));
041    }
042
043}
