205    /**
206     * <p>Define the different priorities allowed for document api messages. Most user traffic should be fit into the
207     * NORMAL categories. Traffic in the HIGH end will be usually be prioritized over important maintenance operations.
208     * Traffic in the LOW end will be prioritized after these operations.</p>
209     */
210    public static enum Priority {
211        HIGHEST(0),
212        VERY_HIGH(1),
213        HIGH_1(2),
214        HIGH_2(3),
215        HIGH_3(4),
216        NORMAL_1(5),
217        NORMAL_2(6),
218        NORMAL_3(7),
219        NORMAL_4(8),
220        NORMAL_5(9),
221        NORMAL_6(10),
222        LOW_1(11),
223        LOW_2(12),
224        LOW_3(13),
225        VERY_LOW(14),
226        LOWEST(15);
227
228        private final int val;
229
230        private Priority(int val) {
231            this.val = val;
232        }
233
234        public int getValue() {
235            return val;
236        }
237    }

239    /**
240     * Get a priority enum instance by its value.
241     *
242     * @param val The value of the priority to return.
243     * @return The priority enum instance.
244     * @throws IllegalArgumentException If priority value is unknown.
245     */
246    public static Priority getPriority(int val) {
247        for (Priority pri : Priority.values()) {
248            if (val == pri.val) {
249                return pri;
250            }
251        }
252        throw new IllegalArgumentException("Unknown priority: " + val);
253    }

480    /**
481     * This method implements the common way to merge document replies for whatever routing policy. In case of an error
482     * in any of the replies, it will prepare an EmptyReply() and add all errors to it. If there are no errors, this
483     * method will use the first reply in the list and transfer whatever feed answers might exist in the replies to it.
484     *
485     * @param ctx  The context whose children to merge.
486     * @param mask The indexes of the children to skip.
487     */
488    public static void merge(RoutingContext ctx, Set<Integer> mask) {
489        List<Reply> replies = new LinkedList<>();
490        for (RoutingNodeIterator it = ctx.getChildIterator();
491             it.isValid(); it.next()) {
492            Reply ref = it.getReplyRef();
493            replies.add(ref);
494        }
495        Tuple2<Integer, Reply> tuple = merge(replies, mask);
496        if (tuple.first != null) {
497            ctx.getChildIterator().skip(tuple.first).removeReply();
498        }
499        ctx.setReply(tuple.second);
500    }
501
502    private static Tuple2<Integer, Reply> merge(List<Reply> replies, Set<Integer> mask) {
503        ReplyMerger rm = new ReplyMerger();
504        for (int i = 0; i < replies.size(); ++i) {
505            if (mask.contains(i)) {
506                continue;
507            }
508            rm.merge(i, replies.get(i));
509        }
510        return rm.mergedReply();
511    }

515     * in any of the replies, it will prepare an EmptyReply() and add all errors to it. If there are no errors, this

531     * @return Whether or not the reply has only the given error code.
532     */
533    public static boolean hasOnlyErrorsOfType(Reply reply, int errCode) {
534        if (!reply.hasErrors()) {
535            return false;
536        }
537        for (int i = 0; i < reply.getNumErrors(); ++i) {
538            if (reply.getError(i).getCode() != errCode) {
539                return false;
540            }
541        }
542        return true;
543    }
