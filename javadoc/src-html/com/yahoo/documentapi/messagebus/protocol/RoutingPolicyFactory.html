002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.messagebus.routing.RoutingPolicy;
005
006/**
007 * This interface defines the necessary methods of a routing policy factory that can be plugged into a {@link
008 * DocumentProtocol} using the {@link DocumentProtocol#putRoutingPolicyFactory(String, RoutingPolicyFactory)} method.
009 *
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
011 */
012public interface RoutingPolicyFactory {
013
014    /**
015     * This method creates and returns a routing policy that corresponds to the implementing class, using the given
016     * parameter string. There is only ever one instance of a routing policy for a given name and parameter combination,
017     * and because of this the policies must be state-less beyond what can be derived from the parameter string. Because
018     * there is only a single thread running route resolution within message bus, it is not necessary to make policies
019     * thread-safe. For more information see {@link RoutingPolicy}.
020     *
021     * Do NOT throw exceptions out of this method because that will cause the running thread to die, just return null to
022     * signal failure instead.
023     *
024     * @param param The parameter to use when creating the policy.
025     * @return The created routing policy.
026     */
027    public DocumentProtocolRoutingPolicy createPolicy(String param);
028
029    /**
030     * Destroys this factory and frees up any resources it has held. Making further calls on a destroyed
031     * factory causes a runtime exception.
032     */
033    public void destroy();
034}
