002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.document.serialization.DocumentDeserializer;
005import com.yahoo.document.serialization.DocumentSerializer;
006import com.yahoo.documentapi.messagebus.loadtypes.LoadTypeSet;
007import com.yahoo.messagebus.Routable;
008
009/**
010 * <p>This interface defines the necessary methods of a routable factory that can be plugged into a {@link
011 * DocumentProtocol} using the {@link DocumentProtocol#putRoutableFactory(int, RoutableFactory,
012 * com.yahoo.component.VersionSpecification)} method. </p>
013 *
014 * <p>Notice that no routable type is passed to the
015 * {@link #decode(DocumentDeserializer, com.yahoo.documentapi.messagebus.loadtypes.LoadTypeSet)} method, so
016 * you may NOT share a factory across multiple routable types. To share serialization logic between factory use a common
017 * superclass or composition with a common serialization utility.</p>
018 *
019 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
020 */
021public interface RoutableFactory {
022
023    /**
024     * <p>This method encodes the content of the given routable into a byte buffer that can later be decoded using the
025     * {@link #decode(DocumentDeserializer, com.yahoo.documentapi.messagebus.loadtypes.LoadTypeSet)} method.</p> <p>Return false to signal failure.</p>
026     * <p>This method is NOT exception safe.</p>
027     *
028     * @param obj The routable to encode.
029     * @param out The buffer to write into.
030     * @return True if the routable could be encoded.
031     */
032    boolean encode(Routable obj, DocumentSerializer out);
033
034    /**
035     * <p>This method decodes the given byte bufer to a routable.</p> <p>Return false to signal failure.</p> <p>This
036     * method is NOT exception safe.</p>
037     *
038     * @param in        The buffer to read from.
039     * @param loadTypes The LoadTypeSet to inject into the Routable.
040     * @return The decoded routable.
041     */
042    Routable decode(DocumentDeserializer in, LoadTypeSet loadTypes);
043
044}
