002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.vdslib.DocumentSummary;
005
006public class DocumentSummaryMessage extends VisitorMessage {
007
008    private DocumentSummary documentSummary = null;
009
010    public void setDocumentSummary(DocumentSummary summary) {
011        documentSummary = summary;
012    }
013
014    public DocumentSummary getResult() {
015        return documentSummary;
016    }
017
018    @Override
019    public DocumentReply createReply() {
020        return new VisitorReply(DocumentProtocol.REPLY_DOCUMENTSUMMARY);
021    }
022
023    @Override
024    public int getType() {
025        return DocumentProtocol.MESSAGE_DOCUMENTSUMMARY;
026    }
027}
