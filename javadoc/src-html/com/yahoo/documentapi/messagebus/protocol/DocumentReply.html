002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.messagebus.Reply;
005import com.yahoo.text.Utf8String;
006
007/**
008 * This class implements a generic document protocol reply that can be reused by document messages that require no
009 * special reply implementation while still allowing applications to distinguish between types.
010 *
011 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
012 */
013public class DocumentReply extends Reply {
014
015    private final int type;
016    private DocumentProtocol.Priority priority = DocumentProtocol.Priority.NORMAL_3;
017
018    /**
019     * Constructs a new reply of given type.
020     *
021     * @param type The type code to assign to this.
022     */
023    public DocumentReply(int type) {
024        this.type = type;
025    }
026
027    /**
028     * Returns the priority tag for this message.
029     * @return The priority.
030     */
031    public DocumentProtocol.Priority getPriority() {
032        return priority;
033    }
034
035    /**
036     * Sets the priority tag for this message.
037     *
038     * @param priority The priority to set.
039     */
040    public void setPriority(DocumentProtocol.Priority priority) {
041        this.priority = priority;
042    }
043
044    @Override
045    public Utf8String getProtocol() {
046        return DocumentProtocol.NAME;
047    }
048
049    @Override
050    public final int getType() {
051        return type;
052    }
053}
