002package com.yahoo.documentapi;
003
004/**
005 * This response is provided for successful document update operations. Use the
006 * wasFound() method to check whether or not the document was actually found.
007 *
008 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
009 */
010public class UpdateResponse extends Response {
011
012    private final boolean wasFound;
013
014    public UpdateResponse(long requestId, boolean wasFound) {
015        super(requestId);
016        this.wasFound = wasFound;
017    }
018
019    public boolean wasFound() {
020        return wasFound;
021    }
022
023    @Override
024    public int hashCode() {
025        return super.hashCode() + Boolean.valueOf(wasFound).hashCode();
026    }
027
028    @Override
029    public boolean equals(Object obj) {
030        if (!(obj instanceof UpdateResponse)) {
031            return false;
032        }
033        if (!super.equals(obj)) {
034            return false;
035        }
036        UpdateResponse rhs = (UpdateResponse)obj;
037        if (wasFound != rhs.wasFound) {
038            return false;
039        }
040        return true;
041    }
042
043    @Override
044    public String toString() {
045        return "Update" + super.toString() + " " + wasFound;
046    }
047}
