002package com.yahoo.config;
003
004import java.util.List;
005
006/**
007 * @author gjoranv
008 * @since 5.1.4
009 */
010public class InnerNodeVector<NODE extends InnerNode> extends NodeVector<NODE> {
011
012    NODE defaultNode;
013
014    /**
015     * Creates a new vector with the given default node.
016     */
017    // TODO: remove this ctor when the library uses reflection via builders, and resizing won't be necessary
018    public InnerNodeVector(NODE defaultNode) {
019        assert (defaultNode != null) : "The default node cannot be null";
020
021        this.defaultNode = defaultNode;
022        if (createNew() == null) {
023            throw new NullPointerException("Unable to duplicate the default node.");
024        }
025    }
026
027    public InnerNodeVector(List<NODE> nodes, NODE defaultNode) {
028        this(defaultNode);
029        for (NODE node : nodes) {
030            vector.add(node);
031        }
032    }
033
034    /**
035     * Creates a new Node by creating a new instance with the 0-argument constructor
036     */
037    // TODO: remove when the library uses reflection via builders
038    @SuppressWarnings("unchecked")
039    protected NODE createNew() {
040        try {
041            return (NODE) defaultNode.getClass().newInstance();
042        } catch (IllegalAccessException | InstantiationException ex) {
043            throw new ConfigurationRuntimeException(ex);
044        }
045    }
046
047}
