019/**
020 * Used for subscribing to one or more configs. Can optionally be given a {@link ConfigSource} for the configs
021 * that will be used when {@link #subscribe(Class, String)} is called.
022 *
023 * {@link #subscribe(Class, String)} on the configs needed, call {@link #nextConfig(long)} and get the config from the
024 * {@link ConfigHandle} which {@link #subscribe(Class, String)} returned.
025 *
026 * @author vegardh
027 * @since 5.1
028 */
029public class ConfigSubscriber {

032    private State state = State.OPEN;
033    protected List<ConfigHandle<? extends ConfigInstance>> subscriptionHandles = new ArrayList<>();
034    private final ConfigSource source;
035    private long generation = -1;

067    /**
068     * Subscribes on the given type of {@link ConfigInstance} with the given config id.
069     *
070     * The method blocks until the first config is ready to be fetched with {@link #nextConfig()}.
071     *
072     * @param configClass The class, typically generated from a def-file using config-class-plugin
073     * @param configId Identifies the service in vespa-services.xml, or null if you are using a local {@link ConfigSource} which does not use config id.
074     * Also supported: raw:, file:, dir: or jar: config id which addresses config locally in the same way.
075     *
076     * @return a ConfigHandle
077     */
078    public <T extends ConfigInstance> ConfigHandle<T> subscribe(Class<T> configClass, String configId) {
079        return subscribe(configClass, configId, source, new TimingValues());
080    }

082    /**
083     * Subscribes on the given type of {@link ConfigInstance} with the given config id and subscribe timeout.
084     *
085     * The method blocks until the first config is ready to be fetched with {@link #nextConfig()}.
086     *
087     * @param configClass The class, typically generated from a def-file using config-class-plugin
088     * @param configId    Identifies the service in vespa-services.xml, or possibly raw:, file:, dir: or jar: type config which addresses config locally.
089     * @param timeoutMillis The time to wait for a config to become available, in milliseconds
090     * @return a ConfigHandle
091     */
092    public <T extends ConfigInstance> ConfigHandle<T> subscribe(Class<T> configClass, String configId, long timeoutMillis) {
093        return subscribe(configClass, configId, source, new TimingValues().setSubscribeTimeout(timeoutMillis));
094    }

096    // for testing
097    <T extends ConfigInstance> ConfigHandle<T> subscribe(Class<T> configClass, String configId, ConfigSource source, TimingValues timingValues) {
098        checkStateBeforeSubscribe();
099        final ConfigKey<T> configKey = new ConfigKey<>(configClass, configId);
100        ConfigSubscription<T> sub = ConfigSubscription.get(configKey, this, source, timingValues);
101        ConfigHandle<T> handle = new ConfigHandle<>(sub);
102        subscribeAndHandleErrors(sub, configKey, handle, timingValues);
103        return handle;
104    }
105
106    protected void checkStateBeforeSubscribe() {
107        if (state != State.OPEN)
108            throw new IllegalStateException("Adding subscription after calling nextConfig() is not allowed");
109    }
110
111    protected void subscribeAndHandleErrors(ConfigSubscription<?> sub, ConfigKey<?> configKey, ConfigHandle<?> handle, TimingValues timingValues) {
112        subscriptionHandles.add(handle);
113        // Must block here until something available from the subscription, so we know that it offers something when the user calls nextConfig
114        boolean subOk = sub.subscribe(timingValues.getSubscribeTimeout());
115        throwIfExceptionSet(sub);
116        if (!subOk) {
117            //sub.close();
118            //subscriptionHandles.remove(handle);
119            throw new ConfigurationRuntimeException("Subscribe for '" + configKey + "' timed out (timeout was " + timingValues.getSubscribeTimeout() + " ms): " + sub);
120        }
121    }

123    /**
124     * Use this for waiting for a new config that has changed.
125     *
126     * Returns true if:
127     *
128     * It is the first time nextConfig() is called on this subscriber, and the framework has fetched config for all subscriptions. (Typically a first time config.)
129     *
130     * or
131     *
132     * All configs for the subscriber have a new generation since the last time nextConfig() was called, AND they have the same generation AND there is a change in config for at least one
133     * of the configs. (Typically calls for a reconfig.)
134     *
135     * You can check which configs are changed by calling {@link ConfigHandle#isChanged()} on the handle you got from {@link #subscribe(Class, String)}.
136     *
137     * If the call times out (timeout 1000 ms), no handle will have the changed flag set. You should not configure anything then.
138     *
139     * @return true if a config/reconfig of your system should happen
140     * @throws ConfigInterruptedException if thread performing this call interrupted.
141     */
142    public boolean nextConfig() {
143        return nextConfig(TimingValues.defaultNextConfigTimeout);
144    }

147     * Use this for waiting for a new config that has changed, with the given timeout.

149     * Returns true if:
150     *
151     * It is the first time nextConfig() is called on this subscriber, and the framework has fetched config for all subscriptions. (Typically a first time config.)
152     *
153     * or
154     *
155     * All configs for the subscriber have a new generation since the last time nextConfig() was called, AND they have the same generation AND there is a change in config for at least one
156     * of the configs. (Typically calls for a reconfig.)
157     *
158     * You can check which configs are changed by calling {@link ConfigHandle#isChanged()} on the handle you got from {@link #subscribe(Class, String)}.

195     * Use this for waiting for a new config generation, with the given timeout

197     * Returns true if:
198     *
199     * It is the first time nextGeneration() is called on this subscriber, and the framework has fetched config for all subscriptions. (Typically a first time config.)
200     *
201     * or

204     * none of the configs have to be changed, but they might be.
205     *

207     *
208     * If the call times out (timeout 1000 ms), no handle will have the changed flag set. You should not configure anything then.
209     *
210     * @param timeoutMillis timeout in milliseconds
211     * @return true if generations for all configs have been updated.
212     * @throws ConfigInterruptedException if thread performing this call interrupted.
213     */

215        return acquireSnapshot(timeoutMillis, false);

218    /**
219     * Acquire a snapshot of all configs with the same generation within a timeout.
220     * @param timeoutInMillis timeout to wait in milliseconds
221     * @param requireChange if set, at least one config have to change
222     * @return true, if a new config generation has been found for all configs (additionally requires
223     * that at lest one of them has changed if <code>requireChange</code> is true), false otherwise
224     */
225    private boolean acquireSnapshot(long timeoutInMillis, boolean requireChange) {
226        if (state == State.CLOSED) return false;
227        long started = System.currentTimeMillis();
228        long timeLeftMillis = timeoutInMillis;
229        state = State.FROZEN;
230        boolean anyConfigChanged = false;
231        boolean allGenerationsChanged = true;
232        boolean allGenerationsTheSame = true;
233        Long currentGenChecker = null;
234        for (ConfigHandle<? extends ConfigInstance> h : subscriptionHandles) {
235            h.setChanged(false); // Reset this flag, if it was set, the user should have acted on it the last time this method returned true.
236        }
237        boolean reconfigDue;
238        do {
239            // Keep on polling the subscriptions until we have a new generation across the board, or it times out
240            for (ConfigHandle<? extends ConfigInstance> h : subscriptionHandles) {
241                ConfigSubscription<? extends ConfigInstance> subscription = h.subscription();
242                if (!subscription.nextConfig(timeLeftMillis)) {
243                    // This subscriber has no new state and we know it has exhausted all time
244                    return false;
245                }
246                throwIfExceptionSet(subscription);
247                if (currentGenChecker == null) currentGenChecker = subscription.getGeneration();
248                if (!currentGenChecker.equals(subscription.getGeneration())) allGenerationsTheSame = false;
249                allGenerationsChanged = allGenerationsChanged && subscription.isGenerationChanged();
250                if (subscription.isConfigChanged()) anyConfigChanged = true;
251                timeLeftMillis = timeLeftMillis - (System.currentTimeMillis() - started);
252            }
253            reconfigDue = (anyConfigChanged || !requireChange) && allGenerationsChanged && allGenerationsTheSame;
254            if (!reconfigDue && timeLeftMillis > 0) {
255                sleep();
256            }
257        } while (!reconfigDue && timeLeftMillis > 0);
258        if (reconfigDue) {
259            // This indicates the clients will possibly reconfigure their services, so "reset" changed-logic in subscriptions.
260            // Also if appropriate update the changed flag on the handler, which clients use.
261            markSubsChangedSeen();
262            generation = subscriptionHandles.get(0).subscription().getGeneration();
263        }
264        return reconfigDue;
265    }
266
267    private void sleep() {
268        try {
269            Thread.sleep(10);
270        } catch (InterruptedException e) {
271            throw new ConfigInterruptedException(e);
272        }
273    }

275    /**
276     * If a {@link ConfigSubscription} has its exception set, reset that field and throw it
277     *
278     * @param sub {@link ConfigSubscription}
279     */
280    protected void throwIfExceptionSet(ConfigSubscription<? extends ConfigInstance> sub) {
281        RuntimeException subThrowable = sub.getException();
282        if (subThrowable != null) {
283            sub.setException(null);
284            throw subThrowable;
285        }
286    }
287
288    private void markSubsChangedSeen() {
289        for (ConfigHandle<? extends ConfigInstance> h : subscriptionHandles) {
290            ConfigSubscription<? extends ConfigInstance> sub = h.subscription();
291            h.setChanged(sub.isConfigChanged());
292            sub.resetChangedFlags();
293        }
294    }

296    /**
297     * Closes all open {@link ConfigSubscription}s
298     */
299    public void close() {
300        state = State.CLOSED;
301        for (ConfigHandle<? extends ConfigInstance> h : subscriptionHandles) {
302            h.subscription().close();
303        }
304        closeRequesters();
305        log.log(LogLevel.DEBUG, "Config subscriber has been closed.");
306    }
307
308    /**
309     * Closes all open requesters
310     */
311    protected void closeRequesters() {
312        for (JRTConfigRequester requester : requesters.values()) {
313            requester.close();
314        }
315    }

317    @Override
318    public String toString() {
319        String ret = "Subscriber state:" + state;
320        for (ConfigHandle<?> h : subscriptionHandles) {
321            ret = ret + "\n" + h.toString();
322        }
323        return ret;
324    }

326    /**
327     * Convenience method to start a daemon thread called "Vespa config thread" with the given runnable. If you want the runnable to
328     * handle a {@link ConfigSubscriber} or {@link ConfigHandle} you have declared locally outside, declare them as final to make it work.
329     *
330     * @param runnable a class implementing {@link java.lang.Runnable}
331     * @return the newly started thread
332     */
333    public Thread startConfigThread(Runnable runnable) {
334        Thread t = new Thread(runnable);
335        t.setDaemon(true);
336        t.setName("Vespa config thread");
337        t.start();
338        return t;
339    }

378    /**
379     * Use this convenience method if you only want to subscribe on <em>one</em> config, and want generic error handling.
380     * Implement {@link SingleSubscriber} and pass to this method.
381     * You will get initial config, and a config thread will be started. The method will throw in your thread if initial
382     * configuration fails, and the config thread will print a generic error message (but continue) if it fails thereafter. The config
383     * thread will stop if you {@link #close()} this {@link ConfigSubscriber}.
384     *
385     * @param <T> ConfigInstance type
386     * @param singleSubscriber The object to receive config
387     * @param configClass      The class, typically generated from a def-file using config-class-plugin
388     * @param configId         Identifies the service in vespa-services.xml
389     * @return The handle of the config
390     * @see #startConfigThread(Runnable)
391     */
392    public <T extends ConfigInstance> ConfigHandle<T> subscribe(final SingleSubscriber<T> singleSubscriber, Class<T> configClass, String configId) {
393        if (!subscriptionHandles.isEmpty())
394            throw new IllegalStateException("Can not start single-subscription because subscriptions were previously opened on this.");
395        final ConfigHandle<T> handle = subscribe(configClass, configId);
396        if (!nextConfig())
397            throw new ConfigurationRuntimeException("Initial config of " + configClass.getName() + " failed.");
398        singleSubscriber.configure(handle.getConfig());
399        startConfigThread(new Runnable() {
400            @Override
401            public void run() {
402                while (!isClosed()) {
403                    try {
404                        if (nextConfig()) {
405                            if (handle.isChanged()) singleSubscriber.configure(handle.getConfig());
406                        }
407                    } catch (Exception e) {
408                        log.log(LogLevel.ERROR, "Exception from config system, continuing config thread: " + Exceptions.toMessageString(e));
409                    }
410                }
411            }
412        });
413        return handle;
414    }

434    /**
435     * Finalizer to ensure that we do not leak resources on reconfig. Though finalizers are bad,
436     * this is not a performance critical object as it will be deconstructed typically container reconfig.
437     */
438    @Override
439    protected void finalize() throws Throwable {
440        try {
441            if (!isClosed()) {
442                close();
443            }
444        } finally {
445            super.finalize();
446        }
