002package com.yahoo.config.provision;
003
004
005import java.util.Objects;
006
007/**
008 * A Brooklyn rotation, e.g. rotation-042.vespa.a02.yahoodns.net.
009 */
010public class Rotation {
011
012    private final String id;
013
014    public Rotation(String id) {
015        this.id = Objects.requireNonNull(id, "Rotation id cannot be null");
016    }
017
018    public String getId() {
019        return id;
020    }
021
022    @Override
023    public boolean equals(Object o) {
024        if (this == o) {
025            return true;
026        }
027        if (!(o instanceof Rotation)) {
028            return false;
029        }
030        final Rotation that = (Rotation) o;
031        return (this.id.equals(that.id));
032    }
033
034    @Override
035    public int hashCode() {
036        return id.hashCode();
037    }
038
039}
