002package com.yahoo.config.provision;
003
004import com.yahoo.transaction.NestedTransaction;
005
006import java.util.Collection;
007import java.util.List;
008
009/**
010 * Interface used by the config system to acquire hosts.
011 *
012 * @author lulf
013 * @since 5.11
014 */
015public interface Provisioner {
016
017    /**
018     * Prepares allocation of a set of hosts with a given type, common id and the amount.
019     *
020     * @param applicationId the application requesting hosts
021     * @param cluster the specification of the cluster to allocate nodes for
022     * @param capacity the capacity requested
023     * @param groups the number of node groups to divide the requested capacity into
024     * @param logger a logger which receives messages which are returned to the requestor
025     * @return the specification of the hosts allocated
026     */
027    List<HostSpec> prepare(ApplicationId applicationId, ClusterSpec cluster, Capacity capacity, int groups, ProvisionLogger logger);
028
029    /**
030     * Activates the allocation of nodes to this application captured in the hosts argument.
031     *
032     * @param transaction Transaction with operations to commit together with any operations done within the provisioner.
033     * @param application The {@link ApplicationId} that was activated.
034     * @param hosts a set of {@link HostSpec}.
035     */
036    void activate(NestedTransaction transaction, ApplicationId application, Collection<HostSpec> hosts);
037
038    /**
039     * Transactionally remove this application.
040     *
041     * @param transaction Transaction with operations to commit together with any operations done within the provisioner.
042     * @param application the application to remove
043     */
044    @SuppressWarnings("deprecation")
045    void remove(NestedTransaction transaction, ApplicationId application);
046
047    /**
048     * Requests a restart of the services of the given application
049     *
050     * @param application the application to restart
051     * @param filter a filter which matches the application nodes to restart
052     */
053    void restart(ApplicationId application, HostFilter filter);
054
055}
