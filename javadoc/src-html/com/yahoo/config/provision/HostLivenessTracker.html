001package com.yahoo.config.provision;
002
003import java.time.Instant;
004import java.util.Optional;
005
006/**
007 * Instances of this are used to keep track of (notify and query)
008 * which hosts are currently connected to the config system.
009 * 
010 * @author bratseth
011 */
012public interface HostLivenessTracker {
013
014    /** Called each time a config request is received from a client */
015    void receivedRequestFrom(String hostname);
016
017    /** 
018     * Returns the epoch timestamp of the last request received from the given hostname,
019     * or empty if there is no memory of this host making a request
020     */
021    Optional<Instant> lastRequestFrom(String hostname);
022
023}
