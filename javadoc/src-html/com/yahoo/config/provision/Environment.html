002package com.yahoo.config.provision;
003
004/**
005 * Environments in hosted Vespa.
006 *
007 * @author bratseth
008 * @since 5.11
009 */
010public enum Environment {
011
012    /** The environment in which any external or internal applications serve actual requests */
013    prod,
014
015    /** Production-like environment which runs staging tests before an app is deployed to production */
016    staging,
017
018    /** Environment for running system tests before an app is deployed to staging */
019    test,
020
021    /** Environment used by individual developers to experiment */
022    dev,
023
024    /** Environment used to run performance and stability experiments */
025    perf;
026
027    /** Returns whether deployments to this environment are done manually */
028    public boolean isManuallyDeployed() { return this == dev || this == perf; }
029
030    /** Returns the prod environment. This is useful for non-hosted properties where we just need any consistent value */
031    public static Environment defaultEnvironment() { return prod; }
032
033    /** Returns the environment name from the string value returned by value() */
034    public static Environment from(String value) {
035        switch(value) {
036            case "prod" : return prod;
037            case "staging" : return staging;
038            case "test" : return test;
039            case "dev" : return dev;
040            case "perf" : return perf;
041            default : throw new IllegalStateException("'" + value + "' is not a valid environment identifier");
042        }
043    }
044
045    /** Returns a name of this which is used in external API's and stored in persistent stores */
046    public String value() {
047        switch(this) {
048            case prod : return "prod";
049            case staging : return "staging";
050            case test : return "test";
051            case dev : return "dev";
052            case perf : return "perf";
053            default : throw new IllegalStateException();
054        }
055    }
056
057}
