002package com.yahoo.config.provision;
003
004import java.time.Duration;
005import java.util.Optional;
006
007/**
008 * A deployer is used to deploy applications.
009 *
010 * @author bratseth
011 */
012public interface Deployer {
013
014    /**
015     * Creates a new deployment from the active application, if available.
016     *
017     * @param application the active application to be redeployed
018     * @param timeout the timeout to use for each individual deployment operation
019     * @return a new deployment from the local active, or empty if a local active application
020     *         was not present for this id (meaning it either is not active or active on another
021     *         node in the config server cluster)
022     */
023    Optional<Deployment> deployFromLocalActive(ApplicationId application, Duration timeout);
024
025}
