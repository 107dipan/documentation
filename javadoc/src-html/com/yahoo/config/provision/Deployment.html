002package com.yahoo.config.provision;
003
004/**
005 * A deployment of an application
006 *
007 * @author bratseth
008 */
009public interface Deployment {
010
011    /**
012     * Prepares activation of this deployment.
013     * This will do all validation and preparatory steps in the system such that a subsequent activation should
014     * be fast and error free.
015     */
016    void prepare();
017
018    /** Activates this deployment. This will prepare it if necessary. */
019    void activate();
020
021    /**
022     * Request a restart of services of this application on hosts matching the filter.
023     * This is sometimes needed after activation, but can also be requested without
024     * doing prepare and activate in the same session.
025     */
026    void restart(HostFilter filter);
027
028}
