002package com.yahoo.config.provision;
003
004import java.util.Objects;
005
006/**
007 * Represents an applications instance name, which may be any kind of string or default. This type is defined
008 * in order to provide a type safe API for defining environments.
009 *
010 * @author lulf
011 * @since 5.25
012 */
013public class InstanceName implements Comparable<InstanceName> {
014
015    private final String instanceName;
016
017    private InstanceName(String instanceName) {
018        this.instanceName = instanceName;
019    }
020
021    @Override
022    public int hashCode() {
023        return instanceName.hashCode();
024    }
025
026    @Override
027    public boolean equals(Object obj) {
028        if (!(obj instanceof InstanceName)) return false;
029        return Objects.equals(((InstanceName) obj).instanceName, instanceName);
030    }
031
032    @Override
033    public String toString() {
034        return instanceName;
035    }
036
037    public static InstanceName from(String name) {
038        return new InstanceName(name);
039    }
040
041    public static InstanceName defaultName() {
042        return new InstanceName("default");
043    }
044
045    public boolean isDefault() {
046        return equals(InstanceName.defaultName());
047    }
048
049    public String value() { return instanceName; }
050
051    @Override
052    public int compareTo(InstanceName instance) {
053        return instanceName.compareTo(instance.instanceName);
054    }
055
056}
