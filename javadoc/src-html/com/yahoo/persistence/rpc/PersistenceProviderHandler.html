032    public void initialize(PersistenceProvider provider, DocumentTypeManager manager) {
033        this.provider = provider;
034        this.docTypeManager = manager;
035
036        if (!started) {
037            addMethod(new Method("vespa.persistence.connect", "s", "", this, "RPC_connect")
038                    .paramDesc(0, "buildId", "Id to make sure client and server come from the same build"));
039            addMethod(new PersistenceProviderMethod("initialize", this));
040            addMethod(new PersistenceProviderMethod("getPartitionStates", this, "", "IS"));
041            addMethod(new PersistenceProviderMethod("listBuckets", this, "l", "L")
042                    .paramDesc("partitionId", "The partition to list buckets for")
043                    .returnDesc("bucketIds", "An array of bucketids"));
044            addMethod(new PersistenceProviderMethod("getModifiedBuckets", this, "", "L")
045                    .returnDesc("bucketIds", "An array of bucketids"));
046            addMethod(new PersistenceProviderMethod("setClusterState", this, "x")
047                    .paramDesc("clusterState", "The updated cluster state"));
048            addMethod(new BucketProviderMethod("setActiveState", this, "b")
049                    .paramDesc("bucketState", "The new state (active/not active)"));
050            addMethod(new BucketProviderMethod("getBucketInfo", this, "", "iiiiibb")
051                    .returnDesc("checksum", "The bucket checksum")
052                    .returnDesc("documentCount", "The number of unique documents stored in the bucket")
053                    .returnDesc("documentSize", "The size of the unique documents")
054                    .returnDesc("entryCount", "The number of entries (inserts/removes) in the bucket")
055                    .returnDesc("usedSize", "The number of bytes used by the bucket in total")
056                    .returnDesc("ready", "Whether the bucket is \"ready\" for external reads or not")
057                    .returnDesc("active", "Whether the bucket has been activated for external reads or not"));
058            addMethod(new TimestampedProviderMethod("put", this, "x")
059                    .paramDesc("document", "The serialized document"));
060            addMethod(new TimestampedProviderMethod("removeById", this, "s", "b")
061                    .paramDesc("documentId", "The ID of the document to remove")
062                    .returnDesc("existed", "Whether or not the document existed"));
063            addMethod(new TimestampedProviderMethod("removeIfFound", this, "s", "b")
064                    .paramDesc("documentId", "The ID of the document to remove")
065                    .returnDesc("existed", "Whether or not the document existed"));
066            addMethod(new TimestampedProviderMethod("update", this, "x", "l")
067                    .paramDesc("update", "The document update to apply")
068                    .returnDesc("existingTimestamp", "The timestamp of the document that the update was applied to, or 0 if it didn't exist"));
069            addMethod(new BucketProviderMethod("flush", this));
070            addMethod(new BucketProviderMethod("get", this, "ss", "lx")
071                    .paramDesc("fieldSet", "A set of fields to return")
072                    .paramDesc("documentId", "The document ID to fetch")
073                    .returnDesc("timestamp", "The timestamp of the document fetched")
074                    .returnDesc("document", "A serialized document"));
075            addMethod(new BucketProviderMethod("createIterator", this, "ssllLb", "l")
076                    .paramDesc("fieldSet", "A set of fields to return")
077                    .paramDesc("documentSelectionString", "Document selection to match with")
078                    .paramDesc("timestampFrom", "lowest timestamp to include")
079                    .paramDesc("timestampTo", "Highest timestamp to include")
080                    .paramDesc("timestampSubset", "Array of timestamps to include")
081                    .paramDesc("includedVersions", "Document versions to include")
082                    .returnDesc("iteratorId", "An iterator id to use for further calls to iterate and destroyIterator"));
083            addMethod(new PersistenceProviderMethod("iterate", this, "ll", "LISXb")
084                    .paramDesc("iteratorId", "An iterator id previously returned by createIterator")
085                    .paramDesc("maxByteSize", "The maximum number of bytes to return in this call (approximate)")
086                    .returnDesc("timestampArray", "Array of timestamps for DocEntries")
087                    .returnDesc("flagArray", "Array of flags for DocEntries")
088                    .returnDesc("docIdArray", "Array of document ids for DocEntries")
089                    .returnDesc("docArray", "Array of documents for DocEntries")
090                    .returnDesc("completed", "Whether or not iteration completed"));
091            addMethod(new PersistenceProviderMethod("destroyIterator", this, "l")
092                    .paramDesc("iteratorId", "An iterator id previously returned by createIterator"));
093            addMethod(new BucketProviderMethod("createBucket", this));
094            addMethod(new BucketProviderMethod("deleteBucket", this));
095            addMethod(new BucketProviderMethod("split", this, "llll")
096                    .paramDesc("target1Bucket", "Bucket id of first split target")
097                    .paramDesc("target1Partition", "Partition id of first split target")
098                    .paramDesc("target2Bucket", "Bucket id of second split target")
099                    .paramDesc("target2Partition", "Partition id of second split target"));
100            addMethod(new PersistenceProviderMethod("join", this, "llllll")
101                    .paramDesc("source1Bucket", "Bucket id of first source bucket")
102                    .paramDesc("source1Partition", "Partition id of first source bucket")
103                    .paramDesc("source1Bucket", "Bucket id of second source bucket")
104                    .paramDesc("source1Partition", "Partition id of second source bucket")
105                    .paramDesc("source1Bucket", "Bucket id of target bucket")
106                    .paramDesc("source1Partition", "Partition id of target bucket"));
107            addMethod(new BucketProviderMethod("move", this, "l")
108                    .paramDesc("partitionId", "The partition to move the bucket to"));
109            addMethod(new BucketProviderMethod("maintain", this, "b")
110                    .paramDesc("maintenanceLevel", "LOW or HIGH maintenance"));
111            addMethod(new TimestampedProviderMethod("removeEntry", this));
112
113            start();
114            started = false;
115        }
116    }

121    public void addResult(Result result, Request req) {
122        req.returnValues().add(new Int8Value((byte) result.getErrorType().ordinal()));
123        req.returnValues().add(new StringValue(result.getErrorMessage()));
124    }

130    public void RPC_getPartitionStates(Request req) {
131        PartitionStateListResult result = provider.getPartitionStates();
132        addResult(result, req);
133
134        int[] states = new int[result.getPartitionStates().size()];
135        String[] reasons = new String[result.getPartitionStates().size()];
136
137        for (int i = 0; i < states.length; ++i) {
138            states[i] = result.getPartitionStates().get(i).getState().ordinal();
139            reasons[i] = result.getPartitionStates().get(i).getReason();
140        }
141
142        req.returnValues().add(new Int32Array(states));
143        req.returnValues().add(new StringArray(reasons));
144    }

146    void addBucketIdListResult(BucketIdListResult result, Request req) {
147        addResult(result, req);
148
149        long[] retVal = new long[result.getBuckets().size()];
150        for (int i = 0; i < retVal.length; ++i) {
151            retVal[i] = result.getBuckets().get(i).getRawId();
152        }
153
154        req.returnValues().add(new Int64Array(retVal));
155    }

257    byte[] serializeDocument(Document doc) {
258        if (doc != null) {
259            GrowableByteBuffer buf = new GrowableByteBuffer();
260            DocumentSerializer serializer = DocumentSerializerFactory.create42(buf);
261            doc.serialize(serializer);
262            buf.flip();
263            return buf.array();
264        } else {
265            return new byte[0];
266        }
267    }

278    public void RPC_createIterator(Request req) {
279        try {
280            TreeSet<Long> timestampSet = new TreeSet<Long>();
281            long[] timestamps = req.parameters().get(6).asInt64Array();
282            for (long l : timestamps) {
283                timestampSet.add(l);
284            }
285
286            Selection selection;
287            if (timestamps.length > 0) {
288                selection = new Selection(timestampSet);
289            } else {
290                selection = new Selection(
291                        req.parameters().get(3).asString(),
292                        req.parameters().get(4).asInt64(),
293                        req.parameters().get(5).asInt64());
294            }
295
296            CreateIteratorResult result = provider.createIterator(
297                    getBucket(req, 0),
298                    getFieldSet(req, 2),
299                    selection,
300                    PersistenceProvider.IncludedVersions.values()[req.parameters().get(7).asInt8()]);
301
302            addResult(result, req);
303            req.returnValues().add(new Int64Value(result.getIteratorId()));
304        } catch (ParseException e) {
305            addResult(new Result(Result.ErrorType.PERMANENT_ERROR, "Unparseable document selection expression"), req);
306            req.returnValues().add(new Int64Value(0));
307        } catch (Exception e) {
308            e.printStackTrace();
309            throw new RuntimeException(e);
310        }
311    }

313    public void RPC_iterate(Request req) {
314        try {
315            long iteratorId = req.parameters().get(0).asInt64();
316            long maxByteSize = req.parameters().get(1).asInt64();
317
318            IterateResult result = provider.iterate(iteratorId, maxByteSize);
319
320            addResult(result, req);
321
322            int count = result.getEntries() != null ? result.getEntries().size() : 0;
323            long[] timestamps = new long[count];
324            int[] flags = new int[count];
325            String[] docIds = new String[count];
326            byte[][] documents = new byte[count][];
327
328            for (int i = 0; i < count; ++i) {
329                DocEntry entry = result.getEntries().get(i);
330                timestamps[i] = entry.getTimestamp();
331                flags[i] = entry.getType().ordinal();
332
333                if (entry.getDocumentId() != null) {
334                    docIds[i] = entry.getDocumentId().toString();
335                } else {
336                    docIds[i] = "";
337                }
338
339                if (entry.getDocument() != null) {
340                    documents[i] = serializeDocument(entry.getDocument());
341                } else {
342                    documents[i] = (new byte[0]);
343                }
344            }
345
346            req.returnValues().add(new Int64Array(timestamps));
347            req.returnValues().add(new Int32Array(flags));
348            req.returnValues().add(new StringArray(docIds));
349            req.returnValues().add(new DataArray(documents));
350            req.returnValues().add(new Int8Value(result.isCompleted() ? (byte)1 : (byte)0));
351        } catch (Exception e) {
352            e.printStackTrace();
353            throw new RuntimeException(e);
354        }
355    }
