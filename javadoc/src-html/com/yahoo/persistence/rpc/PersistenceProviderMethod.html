002package com.yahoo.persistence.rpc;
003
004import com.yahoo.jrt.*;
005
006/**
007 * Class to represent a JRT method used by PersistenceProviderHandler.
008 */
009public class PersistenceProviderMethod extends Method {
010    int nextReturnDesc = 0;
011    int nextParamDesc;
012
013    PersistenceProviderMethod returnDesc(String code, String text) {
014        returnDesc(nextReturnDesc, code, text);
015        ++nextReturnDesc;
016        return this;
017    }
018
019    PersistenceProviderMethod paramDesc(String code, String text) {
020        paramDesc(nextParamDesc, code, text);
021        ++nextParamDesc;
022        return this;
023    }
024
025    public PersistenceProviderMethod(String name, PersistenceProviderHandler owner, String paramTypes) {
026        this(name, owner, paramTypes, "");
027    }
028
029    public PersistenceProviderMethod(String name, PersistenceProviderHandler owner) {
030        this(name, owner, "", "");
031    }
032
033    public PersistenceProviderMethod(String name, PersistenceProviderHandler owner, String paramTypes, String returnTypes) {
034        super("vespa.persistence." + name, paramTypes, "bs" + returnTypes, owner, "RPC_" + name);
035        returnDesc("code", "Error code, or 0 if successful");
036        returnDesc("message", "Error message");
037    }
038
039}
