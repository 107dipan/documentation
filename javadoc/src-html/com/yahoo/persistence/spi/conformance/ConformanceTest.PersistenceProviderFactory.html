077    List<DocEntry> iterateBucket(PersistenceProvider spi, Bucket bucket, PersistenceProvider.IncludedVersions versions) throws Exception {
078        List<DocEntry> ret = new ArrayList<DocEntry>();
079
080        CreateIteratorResult iter = spi.createIterator(
081                bucket,
082                new AllFields(),
083                new Selection("", 0, Long.MAX_VALUE),
084                versions);
085
086        assertFalse(iter.hasError());
087
088        while (true) {
089            IterateResult result = spi.iterate(iter.getIteratorId(), Long.MAX_VALUE);
090            assertFalse(result.hasError());
091
092            ret.addAll(result.getEntries());
093
094            if (result.isCompleted()) {
095                break;
096            }
097        }
098
099        Collections.sort(ret);
100
101        return ret;
102    }

104    void testBasicsIteration(PersistenceProvider provider, Bucket bucket, Document doc1, Document doc2, boolean includeRemoves) throws Exception {
105        Selection selection = new Selection("true", 0, Long.MAX_VALUE);
106
107        CreateIteratorResult iter = provider.createIterator(
108                bucket,
109                new AllFields(),
110                selection,
111                includeRemoves ? PersistenceProvider.IncludedVersions.NEWEST_DOCUMENT_OR_REMOVE : PersistenceProvider.IncludedVersions.NEWEST_DOCUMENT_ONLY);
112
113        assertTrue(!iter.hasError());
114
115        IterateResult result = provider.iterate(iter.getIteratorId(), Long.MAX_VALUE);
116        assertTrue(!result.hasError());
117        assertTrue(result.isCompleted());
118        assertEquals(new Result(), provider.destroyIterator(iter.getIteratorId()));
119
120        long timeRemoveDoc1 = 0;
121        long timeDoc1 = 0;
122        long timeDoc2 = 0;
123
124        for (DocEntry entry : result.getEntries()) {
125            assertNotNull(entry.getDocumentId());
126
127            if (entry.getDocumentId().equals(doc1.getId())) {
128                assertTrue("Got removed document 1 when iterating without removes", includeRemoves);
129
130                if (entry.getType() == DocEntry.Type.REMOVE_ENTRY) {
131                    timeRemoveDoc1 = entry.getTimestamp();
132                } else {
133                    timeDoc1 = entry.getTimestamp();
134                }
135            } else if (entry.getDocumentId().equals(doc2.getId())) {
136                assertEquals(DocEntry.Type.PUT_ENTRY, entry.getType());
137                timeDoc2 = entry.getTimestamp();
138            } else {
139                assertFalse("Unknown document " + entry.getDocumentId(), false);
140            }
141        }
142
143        assertEquals(2, timeDoc2);
144        assertTrue(timeDoc1 == 0 || timeRemoveDoc1 != 0);
145    }

694    /**
695     * Do a full bucket iteration, returning a vector of DocEntry chunks.
696     */
697    List<IterateResult> doIterate(PersistenceProvider spi,
698                                  long id,
699                                  long maxByteSize,
700                                  int maxChunks)
701    {
702        List<IterateResult> chunks = new ArrayList<IterateResult>();
703
704        while (true) {
705            IterateResult result = spi.iterate(id, maxByteSize);
706            assertFalse(result.hasError());
707
708            assertTrue(result.getEntries().size() > 0);
709            chunks.add(result);
710
711            if (result.isCompleted()
712                    || (maxChunks != 0 && chunks.size() >= maxChunks))
713            {
714                break;
715            }
716        }
717        return chunks;
718    }

722            for (DocEntry e : i.getEntries()) {
723                if (e.getType() == DocEntry.Type.PUT_ENTRY && e.getDocument() != null && e.getDocument().equals(doc)) {
724                    return true;
725                }
726            }
727        }

729        return false;

738            }

741        return false;

744    void verifyDocs(List<DocEntry> docs, List<IterateResult> chunks, List<String> removes) {
745        int docCount = 0;
746        int removeCount = 0;
747        for (IterateResult result : chunks) {
748            for (DocEntry e : result.getEntries()) {
749                if (e.getType() == DocEntry.Type.PUT_ENTRY) {
750                    ++docCount;
751                } else {
752                    ++removeCount;
753                }
754            }
755        }
756
757        assertEquals(docs.size(), docCount);
758
759        for (DocEntry e : docs) {
760            assertTrue(e.getDocument().toString(), containsDocument(chunks, e.getDocument()));
761        }
762
763        if (removes != null) {
764            assertEquals(removes.size(), removeCount);
765
766            for (String docId : removes) {
767                assertTrue(docId, containsRemove(chunks, docId));
768            }
769        }
770    }
