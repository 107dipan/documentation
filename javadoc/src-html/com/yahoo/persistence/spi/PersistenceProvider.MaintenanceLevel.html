002package com.yahoo.persistence.spi;
003
004import com.yahoo.document.Document;
005import com.yahoo.document.DocumentId;
006import com.yahoo.document.DocumentUpdate;
007import com.yahoo.document.fieldset.FieldSet;
008import com.yahoo.persistence.spi.result.*;
009
010/**
011 * <p>
012 * This interface is the basis for a persistence provider in Vespa.
013 * A persistence provider is used by Vespa Storage to provide an elastic stateful system.
014 * </p>
015 * <p>
016 * The Vespa distribution mechanisms are based on distributing "buckets"
017 * between the nodes in the system. A bucket is an abstract concept that
018 * groups a set of documents. The persistence provider can choose freely
019 * how to implement a bucket, but it needs to be able to access a bucket as
020 * a unit. The placement of these units is controlled by the distributors.
021 * </p>
022 * <p>
023 * A persistence provider may support multiple "partitions". One example of
024 * a partition is a physical disk, but the exact meaning of "partitions"
025 * is left to the provider. It must be able to report to the service layer though.
026 * </p>
027 * <p>
028 * All operations return a Result object. The base Result class only encapsulates
029 * potential errors, which can be <i>transient</i>, <i>permanent</i> or <i>fatal</i>.
030 * Transient errors are errors where it's conceivable that retrying the operation
031 * would lead to success, either on this data copy or on others. Permanent errors
032 * are errors where the request itself is faulty. Fatal errors are transient errors
033 * that have uncovered a problem with this instance of the provider (such as a failing disk),
034 * and where the provider wants the process to be shut down.
035 * </p>
036 * <p>
037 * All write operations have a timestamp. This timestamp is generated
038 * by the distributor, and is guaranteed to be unique for the bucket we're
039 * writing to. A persistence provider is required to store "entries" for each of
040 * these operations, and associate the timestamp with that entry.
041 * Iteration code can retrieve these entries, including entries
042 * for remove operations. The provider is not required to keep any history beyond
043 * the last operation that was performed on a given document.
044 * </p>
045 * <p>
046 * The contract for all write operations is that after returning from the function,
047 * provider read methods (get, iterate) should reflect the modified state.
048 * </p>
049 */
050public interface PersistenceProvider
051{
052    /**
053     * The different types of entries that can be returned
054     * from an iterator.
055     */
056    public enum IncludedVersions {
057        NEWEST_DOCUMENT_ONLY,
058        NEWEST_DOCUMENT_OR_REMOVE,
059        ALL_VERSIONS
060    }
061
062    /**
063     * The different kinds of maintenance we can do.
064     * LOW maintenance may be run more often than HIGH.
065     */
066    public enum MaintenanceLevel {
067        LOW,
068        HIGH
069    }
070
071    /**
072     * Initializes the persistence provider. This function is called exactly once when
073     * the persistence provider starts. If any error is returned here, the service layer
074     * will shut down.
075     */
076    Result initialize();
077
078    /**
079     * Returns a list of the partitions available,
080     * and which are up and down.
081     */
082    PartitionStateListResult getPartitionStates();
083
084    /**
085     * Return list of buckets that provider has stored on the given partition.
086     */
087    BucketIdListResult listBuckets(short partition);
088
089    /**
090     * Updates the persistence provider with the last cluster state.
091     * Only cluster states that are relevant for the provider are supplied (changes
092     * that relate to the distributor will not cause an update here).
093     */
094    Result setClusterState(ClusterState state);
095
096    /**
097     * Sets the bucket state to active or inactive. After this returns,
098     * other buckets may be deactivated, so the node must be able to serve
099     * the data from its secondary index or get reduced coverage.
100     */
101    Result setActiveState(Bucket bucket, BucketInfo.ActiveState active);
102
103    /**
104     * If the bucket doesn't exist, return empty bucket info.
105     */
106    BucketInfoResult getBucketInfo(Bucket bucket);
107
108    /**
109     * Stores the given document.
110     *
111     * @param timestamp The timestamp for the new bucket entry.
112     */
113    Result put(Bucket bucket, long timestamp, Document doc);
114
115    /**
116     * <p>
117     * Removes the document referenced by the document id.
118     * It is strongly recommended to keep entries for the removes for
119     * some period of time. For recovery to work properly, a node that
120     * has been down for a longer period of time than that should be totally
121     * erased. If not, documents that have been removed but have documents
122     * on nodes that have been down will be reinserted.
123     * </p>
124     * <p>
125     * Postconditions:
126     * A successful invocation of this function must add the remove to the
127     * bucket regardless of whether the document existed. More specifically,
128     * iterating over the bucket while including removes after this call
129     * shall yield a remove-entry at the given timestamp for the given
130     * document identifier as part of its result set. The remove entry
131     * shall be added even if there exist removes for the same document id
132     * at other timestamps in the bucket.
133     * </p>
134     * <p>
135     * Also, if the given timestamp is higher to or equal than any
136     * existing put entry, those entries should not be returned in subsequent
137     * get calls. If the timestamp is lower than an existing put entry,
138     * those entries should still be available.
139     * </p>
140     * @param timestamp The timestamp for the new bucket entry.
141     * @param id The ID to remove
142     */
143    RemoveResult remove(Bucket bucket, long timestamp, DocumentId id);
144    /**
145     * <p>
146     * See remove()
147     * </p>
148     * <p>
149     * Used for external remove operations. removeIfFound() has no extra
150     * postconditions than remove, but it may choose to <i>not</i> include
151     * a remove entry if there didn't already exist a put entry for the given
152     * entry. It is recommended, but not required, to not insert entries in this
153     * case, though if remove entries are considered critical it might be better
154     * to insert them in both cases.
155     * </p>
156     * @param timestamp The timestamp for the new bucket entry.
157     * @param id The ID to remove
158     */
159    RemoveResult removeIfFound(Bucket bucket, long timestamp, DocumentId id);
160
161    /**
162     * Removes the entry with the given timestamp. This is usually used to revert
163     * previously performed operations. This operation should be
164     * successful even if there doesn't exist such an entry.
165     */
166    Result removeEntry(Bucket bucket, long timestampToRemove);
167
168    /**
169     * Partially modifies a document referenced by the document update.
170     *
171     * @param timestamp The timestamp to use for the new update entry.
172     * @param update The document update to apply to the stored document.
173     */
174    UpdateResult update(Bucket bucket, long timestamp, DocumentUpdate update);
175
176    /**
177     * <p>
178     * For providers that store data persistently on disk, the contract of
179     * flush is that data has been stored persistently so that if the node should
180     * restart, the data will be available.
181     * </p>
182     * <p>
183     * The service layer may choose to batch certain commands. This means
184     * that the service layer will lock the bucket only once, then perform several
185     * commands, and finally get the bucket info from the bucket, and then flush it.
186     * This can be used to improve performance by caching the modifications, and
187     * persisting them to disk only when flush is called. The service layer guarantees
188     * that after one of these operations, flush() is called, regardless of whether
189     * the operation succeeded or not, before another bucket is processed in the same
190     * worker thread. The following operations can be batched and have the guarantees
191     * above:
192     * - put
193     * - get
194     * - remove
195     * - removeIfFound
196     * - update
197     * - removeEntry
198     * </p>
199     */
200    Result flush(Bucket bucket);
201
202    /**
203     * Retrieves the latest version of the document specified by the
204     * document id. If no versions were found, or the document was removed,
205     * the result should be successful, but contain no document (see GetResult).
206     *
207     * @param fieldSet A set of fields that should be retrieved.
208     * @param id The document id to retrieve.
209     */
210    GetResult get(Bucket bucket, FieldSet fieldSet, DocumentId id);
211
212    /**
213     * Create an iterator for a given bucket and selection criteria, returning
214     * a unique, non-zero iterator identifier that can be used by the caller as
215     * an argument to iterate and destroyIterator.
216     *
217     * Each successful invocation of createIterator shall be paired with
218     * a later invocation of destroyIterator by the caller to ensure
219     * resources are freed up. NOTE: this may not apply in a shutdown
220     * situation due to service layer communication channels closing down.
221     *
222     * It is assumed that a successful invocation of this function will result
223     * in some state being established in the persistence provider, holding
224     * the information required to match iterator ids up to their current
225     * iteration progress and selection criteria. destroyIterator will NOT
226     * be called when createIterator returns an error.
227     *
228     * @param selection Selection criteria used to limit the subset of
229     *   the bucket's documents that will be returned by the iterator. The
230     *   provider implementation may use these criteria to optimize its
231     *   operation as it sees fit, as long as doing so does not violate
232     *   selection correctness.
233     * @return A process-globally unique iterator identifier iff the result
234     *   is successful and internal state has been created, otherwise an
235     *   error. Identifier must be non-zero, as zero is used internally to
236     *   signify an invalid iterator ID.
237     */
238    CreateIteratorResult createIterator(Bucket bucket,
239                                        FieldSet fieldSet,
240                                        Selection selection,
241                                        IncludedVersions versions);
242
243    /**
244     * Iterate over a bucket's document space using a valid iterator id
245     * received from createIterator. Each invocation of iterate upon an
246     * iterator that has not yet fully exhausted its document space shall
247     * return a minimum of 1 document entry per IterateResult to ensure progress.
248     * An implementation shall limit the result set per invocation to document
249     * entries whose combined in-memory/serialized size is a "soft" maximum of
250     * maxByteSize. More specifically, the sum of getSize() over all returned
251     * DocEntry instances should be &lt;= (maxByteSize + the size of the last
252     * document in the result set). This special case allows for limiting the
253     * result set both by observing "before the fact" that the next potential
254     * document to include would exceed the max size and by observing "after
255     * the fact" that the document that was just added caused the max size to
256     * be exceeded.
257     * However, if a document exceeds maxByteSize and not including it implies
258     * the result set would be empty, it must be included in the result anyway
259     * in order to not violate the progress requirement.
260     *
261     * The caller shall not make any assumptions on whether or not documents
262     * that arrive to--or are removed from--the bucket in the time between
263     * separate invocations of iterate for the same iterator id will show up
264     * in the results, assuming that these documents do not violate the
265     * selection criteria. This means that there is no requirement for
266     * maintaining a "snapshot" view of the bucket's state as it existed upon
267     * the initial createIterator call. Neither shall the caller make any
268     * assumptions on the ordering of the returned documents.
269     *
270     * The IterateResult shall--for each document entry that matches the
271     * selection criteria and falls within the maxByteSize limit mentioned
272     * above--return the following information in its result:
273     *
274     *  -- For non-removed entries: A DocEntry where getDocumentOperation() will
275     *     return a valid DocumentPut instance and getSize() will return the
276     *     serialized size of the document.
277     *  -- For removed entries: A DocEntry where getDocumentId() will
278     *     return a valid document identifier. Remove entries shall not
279     *     contain document instances.
280     *  -- For meta entries: A DocEntry that shall not contain a document
281     *     instance nor should it include a document id instance (if
282     *     included, would be ignored by the service layer in any context
283     *     where metadata-only is requested).
284     *
285     * The service layer shall guarantee that no two invocations of iterate
286     * will happen simultaneously/concurrently for the same iterator id.
287     *
288     * Upon a successful invocation of iterate, the persistence provider shall
289     * update its internal state to account for the progress made so that new
290     * invocations will cover a new subset of the document space. When an
291     * IterateResult contains the final documents for the iteration, i.e. the
292     * iterator has reached its end, setCompleted() must be set on the result
293     * to indicate this to the caller. Calling iterate on an already completed
294     * iterator must only set this flag on the result and return without any
295     * documents.
296     *
297     * @param iteratorId An iterator ID returned by a previous call to createIterator
298     * @param maxByteSize An indication of the maximum number of bytes that should be returned.
299     */
300    IterateResult iterate(long iteratorId, long maxByteSize);
301
302    /**
303     * <p>
304     * Destroys the iterator specified by the given id.
305     * </p>
306     * <p>
307     * IMPORTANT: this method has different invocation semantics than
308     * the other provider methods! It may be called from the context of
309     * ANY service layer thread, NOT just from the thread in which
310     * createIterator was invoked! The reason for this is because internal
311     * iterator destroy messages aren't mapped to partition threads in the
312     * way other messages are due to their need for guaranteed execution.
313     * </p>
314     * <p>
315     * This in turn implies that iterator states must be shared between
316     * partitions (and thus protected against cross-partition concurrent
317     * access).
318     * </p>
319     * @param iteratorId The iterator id previously returned by createIterator.
320     */
321    Result destroyIterator(long iteratorId);
322
323    /**
324     * Tells the provider that the given bucket has been created in the
325     * service layer. There is no requirement to do anything here.
326     */
327    Result createBucket(Bucket bucket);
328
329    /**
330     * Deletes the given bucket and all entries contained in that bucket.
331     * After this operation has succeeded, a restart of the provider should
332     * not yield the bucket in getBucketList().
333     */
334    Result deleteBucket(Bucket bucket);
335
336    /**
337     * This function is called continuously by the service layer. It allows
338     * the provider to signify whether it has done any out-of-band changes to
339     * buckets that need to be recognized by the rest of the system. The service
340     * layer will proceed to call getBucketInfo() on each of the returned buckets.
341     * After a call to getModifiedBuckets(), the provider should clear it's list
342     * of modified buckets, so that the next call does not return the same buckets.
343     */
344    BucketIdListResult getModifiedBuckets();
345
346    /**
347     * Allows the provider to do periodic maintenance and verification.
348     *
349     * @param level The level of maintenance to do. LOW maintenance is scheduled more
350     * often than HIGH maintenance, so should be cheaper.
351     */
352    Result maintain(Bucket bucket, MaintenanceLevel level);
353
354    /**
355     * <p>
356     * Splits the source bucket into the two target buckets.
357     * After the split, all documents belonging to target1 should be
358     * in that bucket, and all documents belonging to target2 should be
359     * there. The information in SplitResult should reflect
360     * this.
361     * </p>
362     * <p>
363     * Before calling this function, the service layer will iterate the bucket
364     * to figure out which buckets the source should be split into. This may
365     * result in splitting more than one bucket bit at a time.
366     * </p>
367     */
368    Result split(Bucket source, Bucket target1, Bucket target2);
369
370    /**
371     * Joins two buckets into one. After the join, all documents from
372     * source1 and source2 should be stored in the target bucket.
373     */
374    Result join(Bucket source1, Bucket source2, Bucket target);
375
376    /**
377     * Moves a bucket from one partition to another.
378     *
379     * @param partitionId The partition to move to.
380     */
381    Result move(Bucket bucket, short partitionId);
382}
