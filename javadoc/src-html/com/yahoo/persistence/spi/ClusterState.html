002package com.yahoo.persistence.spi;
003
004/**
005 * Class that allows a provider to figure out if the node is currently up, the cluster is up and/or a
006 * given bucket should be "ready" given the state.
007 */
008public interface ClusterState {
009    /**
010     * Returns true if the system has been set up to have
011     * "ready" nodes, and the given bucket is in the ideal state
012     * for readiness.
013     *
014     * @param bucket The bucket to check.
015     * @return Returns true if the bucket should be set to "ready".
016     */
017    public boolean shouldBeReady(Bucket bucket);
018
019    /**
020     * @return Returns false if the cluster has been deemed down. This can happen
021     * if the fleet controller has detected that too many nodes are down
022     * compared to the complete list of nodes, and deigns the system to be
023     * unusable.
024     */
025    public boolean clusterUp();
026
027    /**
028     * @return Returns false if this node has been set in a state where it should not
029     * receive external load.
030     */
031    public boolean nodeUp();
032}
