002package com.yahoo.persistence.spi.result;
003
004import com.yahoo.persistence.spi.DocEntry;
005
006import java.util.List;
007
008/**
009 * Result class for iterate requests
010 */
011public class IterateResult extends Result {
012    List<DocEntry> entries = null;
013    boolean isCompleted = false;
014
015    /**
016     * Creates a result with an error.
017     *
018     * @param type The type of error
019     * @param message A human-readable error message to further detail the error.
020     */
021    public IterateResult(Result.ErrorType type, String message) {
022        super(type, message);
023    }
024
025    /**
026     * Creates a successful result.
027     *
028     * @param entries The next chunk of entries that were found during iteration.
029     * @param isCompleted Set to true if there are no more entries to iterate through.
030     */
031    public IterateResult(List<DocEntry> entries, boolean isCompleted) {
032        this.entries = entries;
033        this.isCompleted = isCompleted;
034    }
035
036    public List<DocEntry> getEntries() {
037        return entries;
038    }
039
040    public boolean isCompleted() {
041        return isCompleted;
042    }
043}
