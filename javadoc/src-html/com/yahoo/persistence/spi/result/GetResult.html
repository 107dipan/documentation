002package com.yahoo.persistence.spi.result;
003
004import com.yahoo.document.Document;
005
006/**
007 * Result class for Get operations
008 */
009public class GetResult extends Result {
010    Document doc;
011    long lastModifiedTimestamp = 0;
012
013    /**
014     * Constructor to use when there was an error retrieving the document.
015     * Not finding the document is not an error in this context.
016     *
017     * @param type The type of error.
018     * @param message A human readable message further detailing the error.
019     */
020    GetResult(ErrorType type, String message) {
021        super(type, message);
022    }
023
024    /**
025     * Constructor to use when we didn't find the document in question.
026     */
027    public GetResult() {}
028
029    /**
030     * Constructor to use when we found the document asked for.
031     *
032     * @param doc The document we found
033     * @param lastModifiedTimestamp The timestamp with which the document was stored.
034     */
035    public GetResult(Document doc, long lastModifiedTimestamp) {
036        this.doc = doc;
037        this.lastModifiedTimestamp = lastModifiedTimestamp;
038    }
039
040    /**
041     * @return Returns the timestamp at which the document was last modified, or 0 if
042     * no document was found.
043     */
044    public long getLastModifiedTimestamp() { return lastModifiedTimestamp;}
045
046    /**
047     * @return Returns true if the document was found.
048     */
049    public boolean wasFound() {
050        return doc != null;
051    }
052
053    public boolean hasDocument() {
054        return doc != null;
055    }
056
057    public Document getDocument() {
058        return doc;
059    }
060}
