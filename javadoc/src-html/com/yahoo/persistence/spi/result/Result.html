002package com.yahoo.persistence.spi.result;
003
004/**
005 * Represents a result from an SPI method invocation.
006 */
007public class Result {
008
009    /**
010     * Enumeration of the various categories of errors that can be returned
011     * in a result.
012     *
013     * The categories are:
014     *
015     * TRANSIENT_ERROR: Operation failed, but may succeed if attempted again or on other data copies
016     * PERMANENT_ERROR: Operation failed because it was somehow malformed or the operation parameters were wrong. Operation won't succeed
017     * on other data copies either.
018     * FATAL_ERROR: Operation failed in such a way that this node should be stopped (for instance, a disk failure). Operation will be retried
019     * on other data copies.
020     */
021    public enum ErrorType {
022        NONE,
023        TRANSIENT_ERROR,
024        PERMANENT_ERROR,
025        UNUSED_ID,
026        FATAL_ERROR
027    }
028
029    /**
030     * Constructor to use for a result where there is no error.
031     */
032    public Result() {
033    }
034
035    /**
036     * Creates a result with an error.
037     *
038     * @param type The type of error
039     * @param message A human-readable error message to further detail the error.
040     */
041    public Result(ErrorType type, String message) {
042        this.type = type;
043        this.message = message;
044    }
045
046    public boolean equals(Result other) {
047        return type.equals(other.type) &&
048                message.equals(other.message);
049    }
050
051    @Override
052    public boolean equals(Object otherResult) {
053        if (otherResult instanceof Result) {
054            return equals((Result)otherResult);
055        }
056
057        return false;
058    }
059
060    public boolean hasError() {
061        return type != ErrorType.NONE;
062    }
063
064    public ErrorType getErrorType() {
065        return type;
066    }
067
068    public String getErrorMessage() {
069        return message;
070    }
071
072    @Override
073    public String toString() {
074        if (type == null) {
075            return "Result(OK)";
076        }
077
078        return "Result(" + type.toString() + ", " + message + ")";
079    }
080
081    ErrorType type = ErrorType.NONE;
082    String message = "";
083}
