002package com.yahoo.persistence.spi.result;
003
004import com.yahoo.persistence.spi.BucketInfo;
005
006/**
007 * Result class for the getBucketInfo() function.
008 */
009public class BucketInfoResult extends Result {
010    BucketInfo bucketInfo = null;
011
012    /**
013     * Constructor to use for a result where an error has been detected.
014     * The service layer will not update the bucket information in this case,
015     * so it should not be returned either.
016     *
017     * @param type The type of error.
018     * @param message A human readable message further detailing the error.
019     */
020    public BucketInfoResult(ErrorType type, String message) {
021        super(type, message);
022    }
023
024    /**
025     * Constructor to use when the write operation was successful,
026     * and the bucket info was modified.
027     *
028     * @param info Returns the information about the bucket.
029     */
030    public BucketInfoResult(BucketInfo info) {
031        this.bucketInfo = info;
032    }
033
034    public BucketInfo getBucketInfo() {
035        return bucketInfo;
036    }
037}
