002package com.yahoo.persistence.spi.result;
003
004import com.yahoo.document.BucketId;
005
006import java.util.List;
007
008/**
009 * Result class used for bucket id list requests.
010 */
011public class BucketIdListResult extends Result {
012    List<BucketId> buckets;
013
014    /**
015     * Creates a result with an error.
016     *
017     * @param type The type of error
018     * @param message A human-readable error message to further detail the error.
019     */
020    public BucketIdListResult(ErrorType type, String message) {
021        super(type, message);
022    }
023
024    /**
025     * Creates a result containing a list of all the buckets the requested partition has.
026     *
027     * @param buckets The list of buckets.
028     */
029    public BucketIdListResult(List<BucketId> buckets) {
030        this.buckets = buckets;
031    }
032
033    public List<BucketId> getBuckets() {
034        return buckets;
035    }
036}
