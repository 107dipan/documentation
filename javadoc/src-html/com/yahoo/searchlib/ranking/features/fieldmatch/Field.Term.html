002package com.yahoo.searchlib.ranking.features.fieldmatch;
003
004import com.google.common.collect.ImmutableList;
005
006import java.util.List;
007
008/**
009 * Represents a document field which can be matched and ranked against.
010 *
011 * @author bratseth
012 */
013public class Field {
014
015    private final ImmutableList<Term> terms;
016
017    /** Creates a field from a space-separated string */
018    public Field(String fieldString) {
019        ImmutableList.Builder<Term> list = new ImmutableList.Builder<>();
020        for (String term : fieldString.split(" "))
021            list.add(new Term(term));
022        this.terms = list.build();
023    }
024
025    /** Creates a field from a list of terms */
026    public Field(List<Term> terms) {
027        this.terms = ImmutableList.copyOf(terms);
028    }
029
030    /** Returns an immutable list of the terms in this */
031    public List<Term> terms() { return terms; }
032
033    /** A term in a field */
034    public static class Term {
035
036        private final String value;
037        private final float exactness;
038
039        /** Creates a term with the given value and full exactness (1.0) */
040        public Term(String value) {
041            this(value, 1.0f);
042        }
043
044        public Term(String value, float exactness) {
045            this.value = value;
046            this.exactness = exactness;
047        }
048
049        /** Returns the string value of this term */
050        public String value() { return value; }
051
052        /**
053         * Returns the degree to which this term is exactly what was in the document (1.0),
054         * or some stemmed form (closer to 0)
055         */
056        public float exactness() { return exactness; }
057
058    }
059
060}
