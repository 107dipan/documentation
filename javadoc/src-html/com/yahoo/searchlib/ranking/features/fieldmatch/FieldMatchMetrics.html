089    /**
090     * Returns a metric by name
091     *
092     * @throws IllegalArgumentException if the metric name (case sensitive) is not present
093     */
094    public float get(String name) {
095        try {
096            Method getter=getClass().getMethod("get" + name.substring(0,1).toUpperCase() + name.substring(1));
097            return ((Number)getter.invoke(this)).floatValue();
098        }
099        catch (NoSuchMethodException e) {
100            throw new IllegalArgumentException("No metric named '" + name + "' is known");
101        }
102        catch (Exception e) {
103            throw new RuntimeException("Error getting metric '" + name + "'",e);
104        }
105    }

124    /** Returns the number of tokens in the field preceding the start of the first matched segment */

130    /** Returns the number of query terms which was matched in this field */

134    public int getPairs() { return pairs; }

136    /**
137     * Returns the normalized proximity of the matched terms, weighted by the connectedness of the query terms.
138     * This number is 0.1 if all the matched terms are and have default or lower connectedness, close to 1 if they
139     * are following in sequence and have a high connectedness, and close to 0 if they are far from each other in the
140     * segment or out of order
141     */
142    public float getAbsoluteProximity() {
143        if (pairs <1) return 0.1f;
144
145        return proximity/pairs;
146    }
147
148    /**
149     * Returns the normalized proximity of the matched terms, not taking term connectedness into account.
150     * This number is close to 1 if all the matched terms are
151     * following each other in sequence, and close to 0 if they are far from each other or out of order
152     */
153    public float getUnweightedProximity() {
154        if (pairs <1) return 1f;
155        return unweightedProximity/pairs;
156    }
157
158    /**
159     * Returns the sum of the distance between all segments making up a match to the query, measured
160     * as the sum of the number of token positions separating the <i>start</i> of each field adjacent segment.
161     */
162    public float getSegmentDistance() { return segmentDistance; }
163
164    /**
165     * <p>Returns the normalized weight of this match relative to the whole query:
166     * The sum of the weights of all <i>matched</i> terms/the sum of the weights of all <i>query</i> terms
167     * If all the query terms were matched, this is 1. If no terms were matched, or these matches has weight zero,
168     * this is 0.</p>
169     *
170     * <p>As the sum of this number over all the terms of the query is always 1, sums over all fields of
171     * normalized rank features for each field multiplied by this number for the same field will produce a
172     * normalized number.</p>
173     *
174     * <p>Note that this scales with the number of matched query terms in the field. If you want a component which does
175     * not, divide by matches.</p>
176     */
177    public float getWeight() { return weight; }
178
179    /**
180     * <p>Returns the normalized term significance (1-frequency) of the terms of this match relative to the whole query:
181     * The sum of the significance of all <i>matched</i> terms/the sum of the significance of all <i>query</i> terms
182     * If all the query terms were matched, this is 1. If no terms were matched, or if the significance of all the matched terms
183     * is zero (they are present in all (possible) documents), this number is zero.</p>
184     *
185     * <p>As the sum of this number over all the terms of the query is always 1, sums over all fields of
186     * normalized rank features for each field multiplied by this number for the same field will produce a
187     * normalized number.</p>
188     *
189     * <p>Note that this scales with the number of matched query terms in the field. If you want a component which does
190     * not, divide by matches.</p>
191     */
192    public float getSignificance() { return significance; }
193
194    /**
195     * <p>Returns a normalized measure of the number of occurrence of the terms of the query.
196     * This number is 1 if there are many occurrences of the query terms <i>in absolute terms,
197     * or relative to the total content of the field</i>, and 0 if there are none.</p>
198     *
199     * <p>This is suitable for occurrence in fields containing regular text.</p>
200     */
201    public float getOccurrence() { return occurrence; }
202
203    /**
204     * <p>Returns a normalized measure of the number of occurrence of the terms of the query:
205     *
206     * <code>sum over all query terms(min(number of occurrences of the term,maxOccurrences))/(query term count*100)</code>
207     *
208     * <p>This number is 1 if there are many occurrences of the query terms, and 0 if there are none.
209     * This number does not take the actual length of the field into account, so it is suitable for uses of occurrence
210     * to denote importance across multiple terms.</p>
211     */
212    public float getAbsoluteOccurrence() { return absoluteOccurrence; }
213
214    /**
215     * <p>Returns a normalized measure of the number of occurrence of the terms of the query, weighted by term weight.
216     * This number is close to 1 if there are many occurrences of highly weighted query terms,
217     * in absolute terms, or relative to the total content of the field, and 0 if there are none.</p>
218     */
219    public float getWeightedOccurrence() { return weightedOccurrence; }
220
221    /**
222     * <p>Returns a normalized measure of the number of occurrence of the terms of the query, taking weights
223     * into account so that occurrences of higher weighted query terms has more impact than lower weighted terms.</p>
224     *
225     * <p>This number is 1 if there are many occurrences of the highly weighted terms, and 0 if there are none.
226     * This number does not take the actual length of the field into account, so it is suitable for uses of occurrence
227     * to denote importance across multiple terms.</p>
228     */
229    public float getWeightedAbsoluteOccurrence() { return weightedAbsoluteOccurrence; }
230
231    /**
232     * <p>Returns a normalized measure of the number of occurrence of the terms of the query
233     * <i>in absolute terms,
234     * or relative to the total content of the field</i>, weighted by term significance.
235     *
236     * <p>This number is 1 if there are many occurrences of the highly significant terms, and 0 if there are none.</p>
237     */
238    public float getSignificantOccurrence() { return significantOccurrence; }
239
240    /**
241     * <p>Returns the degree to which the query terms submitted matched exactly terms contained in the document.
242     * This is 1 if all the terms matched exactly, and closer to 0 as more of the terms was matched only as stem forms.
243     * </p>
244     *
245     * <p>This is the query term weighted average of the exactness of each match, where the exactness of a match is
246     * the product of the exactness of the matching query term and the matching field term:
247     * <code>
248     * sum over matching query terms(query term weight * query term exactness * field term exactness) /
249     * sum over matching query terms(query term weight)
250     * </code>
251     */
252    public float getExactness() {
253        if (matches == 0) return 0;
254        return weightedExactnessSum / weightSum;
255    }

259    /** The ratio of query tokens which was matched in the field: <code>matches/queryLength</code> */
260    public float getQueryCompleteness() {
261        return (float)matches/source.getQuery().getTerms().length;
262    }
263
264    /** The ratio of query tokens which was matched in the field: <code>matches/fieldLength</code> */
265    public float getFieldCompleteness() {
266        return (float)matches/source.getField().terms().size();
267    }
268
269    /**
270     * Total completeness, where field completeness is more important:
271     * <code>queryCompleteness * ( 1 - fieldCompletenessImportance) + fieldCompletenessImportance * fieldCompleteness</code>
272     */
273    public float getCompleteness() {
274        float fieldCompletenessImportance=source.getParameters().getFieldCompletenessImportance();
275        return getQueryCompleteness() * ( 1 - fieldCompletenessImportance) + fieldCompletenessImportance*getFieldCompleteness();
276    }
277
278    /** Returns how well the order of the terms agreed in segments: <code>1-outOfOrder/pairs</code> */
279    public float getOrderness() {
280        if (pairs ==0) return 1f;
281        return 1-(float)outOfOrder/pairs;
282    }
283
284    /** Returns the degree to which different terms are related (occurring in the same segment): <code>1-segments/(matches-1)</code> */
285    public float getRelatedness() {
286        if (matches==0) return 0;
287        if (matches==1) return 1;
288        return 1-(float)(segments-1)/(matches-1);
289    }
290
291    /** Returns <code>longestSequence/matches</code> */
292    public float getLongestSequenceRatio() {
293        if (matches==0) return 0;
294        return (float)longestSequence/matches;
295    }
296
297    /** Returns the closeness of the segments in the field: <code>1-segmentDistance/fieldLength</code> */
298    public float getSegmentProximity() {
299        if (matches==0) return 0;
300        return 1-(float)segmentDistance/source.getField().terms().size();
301    }
302
303    /**
304     * Returns a value which is close to 1 when matched terms are close and close to zero when they are far apart
305     * in the segment. Relatively more connected terms influence this value more.
306     * This is absoluteProximity/average connectedness.
307     */
308    public float getProximity() {
309        float totalConnectedness=0;
310        for (int i=1; i<queryLength; i++) {
311            totalConnectedness+=Math.max(0.1,source.getQuery().getTerms()[i].getConnectedness());
312        }
313        float averageConnectedness=0.1f;
314        if (queryLength>1)
315            averageConnectedness=totalConnectedness/(queryLength-1);
316        return getAbsoluteProximity()/averageConnectedness;
317    }
318
319    /**
320     * <p>Returns the average of significance and weight.</p>
321     *
322     * <p>As the sum of this number over all the terms of the query is always 1, sums over all fields of
323     * normalized rank features for each field multiplied by this number for the same field will produce a
324     * normalized number.</p>
325     *
326     * <p>Note that this scales with the number of matched query terms in the field. If you want a component which does
327     * not, divide by matches.</p>
328     */
329    public float getImportance() {
330        return (getSignificance() + getWeight()) / 2;
331    }
332
333    /** A normalized measure of how early the first segment occurs in this field: <code>1-head/(max(6,field.length)-1)</code> */
334    public float getEarliness() {
335        if (matches == 0) return 0; // Covers field.length==0 too
336        if (source.getField().terms().size() == 1) return 1;
337        return 1 - (float)head/(max(6, source.getField().terms().size()) - 1);
338    }
339
340    /**
341     * <p>A ready-to-use aggregate match score. Use this if you don't have time to find a better application specific
342     * aggregate score of the fine grained match metrics.</p>
343     *
344     * <p>The current formula is
345     *
346     * <code>
347     * ( proximityCompletenessImportance * (1-relatednessImportance + relatednessImportance*relatedness)
348     *               proximity * exactness * completeness^2 + earlinessImportance * earliness + segmentProximityImportance * segmentProximity )
349     * / (proximityCompletenessImportance + earlinessImportance + relatednessImportance)</code>
350     *
351     * but this is subject to change (i.e improvement) at any time.
352     * </p>
353     *
354     *
355     * <p>Weight and significance are not taken into account because this is mean to capture tha quality of the
356     * match in this field, while those measures relate this match to matches in other fields. This number
357     * can be multiplied with those values when combining with other field match scores.</p>
358     */
359    public float getMatch() {
360        float proximityCompletenessImportance = source.getParameters().getProximityCompletenessImportance();
361        float earlinessImportance = source.getParameters().getEarlinessImportance();
362        float relatednessImportance = source.getParameters().getRelatednessImportance();
363        float segmentProximityImportance = source.getParameters().getSegmentProximityImportance();
364        float occurrenceImportance = source.getParameters().getOccurrenceImportance();
365        float scaledRelatedness = 1 - relatednessImportance + relatednessImportance*getRelatedness();
366
367        return ( proximityCompletenessImportance * scaledRelatedness * getProximity() * getExactness() * getCompleteness() * getCompleteness()
368                 + earlinessImportance * getEarliness()
369                 + segmentProximityImportance * getSegmentProximity()
370                 + occurrenceImportance * getOccurrence())
371               / (proximityCompletenessImportance + earlinessImportance + segmentProximityImportance + occurrenceImportance);
372    }
373
374    /**
375     * <p>The metric use to select the best segments during execution of the string match metric algorithm.</p>
376     *
377     * <p>This metric, and any metric it depends on, must be correct each time a segment is completed,
378     * not only when the metrics are complete, because this metric is used to choose segments during calculation.</p>
379     */
380    float getSegmentationScore() {
381        if (segments==0) return 0;
382        return getAbsoluteProximity() * getExactness() / (segments * segments);
383    }

385    // Events emitted from the computer while matching strings  ----------------------------------------------------
386    // Note that one move in the computer may cause multiple events

388    // Events on single positions ----------
389
390    /** Called once for every match */
391    void onMatch(int i, int j) {
392        if (matches>=source.getField().terms().size()) return;
393        matches++;
394        weight += (float)source.getQuery().getTerms()[i].getWeight() / source.getQuery().getTotalTermWeight();
395        significance += source.getQuery().getTerms()[i].getSignificance() / source.getQuery().getTotalSignificance();
396        int queryTermWeight = source.getQuery().getTerms()[i].getWeight();
397        weightedExactnessSum += queryTermWeight * source.getQuery().getTerms()[i].getExactness() * source.getField().terms().get(j).exactness();
398        weightSum += queryTermWeight;
399    }
400
401    /** Called once per sequence, when the sequence starts */
402    void onSequenceStart(int j) {
403        if (head==-1 || j<head)
404            head=j;
405
406        currentSequence=1;
407    }
408
409    /** Called once per sequence when the sequence ends */
410    void onSequenceEnd(int j) {
411        int sequenceTail = source.getField().terms().size() - j - 1;
412        if (tail ==-1 || sequenceTail < tail)
413            tail = sequenceTail;
414
415        if (currentSequence > longestSequence)
416            longestSequence = currentSequence;
417        currentSequence = 0;
418    }

435    /** Called once when matching is complete */
436    void onComplete() {
437        // segment distance - calculated from sorted segment starts
438        if (segmentStarts.size()<=1) {
439            segmentDistance=0;
440        }
441        else {
442            Collections.sort(segmentStarts);
443            for (int i=1; i<segmentStarts.size(); i++) {
444                segmentDistance+=segmentStarts.get(i)-segmentStarts.get(i-1)+1;
445            }
446        }
447
448        if (head==-1) head=0;
449        if (tail==-1) tail=0;
450    }

452    // Events on pairs ----------
453
454    /** Called when <i>any</i> pair is encountered */
455    void onPair(int i, int j, int previousJ) {
456        int distance = j-previousJ-1;
457        if (distance < 0) distance++; // Discontinuity where the two terms are in the same position
458        if (abs(distance) > source.getParameters().getProximityLimit()) return; // Contribution=0
459
460        // We have an in-segment pair
461        float pairProximity = source.getParameters().getProximity(distance + source.getParameters().getProximityLimit());
462
463        unweightedProximity += pairProximity;
464
465        float connectedness = source.getQuery().getTerms()[i].getConnectedness();
466        proximity += pow(pairProximity, connectedness/0.1) * max(0.1, connectedness);
467
468        pairs++;
469    }
470
471    /** Called when an in-sequence pair is encountered */
472    void onInSequence(int i, int j, int previousJ) {
473        currentSequence++;
474    }
475
476    /** Called when a gap (within a sequence) is encountered */
477    void onInSegmentGap(int i, int j, int previousJ) {
478        gaps++;
479        if (j>previousJ) {
480            gapLength+=abs(j-previousJ)-1; // gap length may be 0 if the gap was in the query
481        }
482        else {
483            outOfOrder++;
484            gapLength+=abs(j-previousJ);
485        }
486    }
487
488    /**
489     * Called when a new segment is started
490     *
491     * @param previousJ the end of the previous segment, or -1 if this is the first segment
492     * */
493    void onNewSegment(int i, int j, int previousJ) {
494        segments++;
495        segmentStarts.add(j);
496    }

506            throw new RuntimeException("Programming error",e);
507        }
508    }

515    public String toStringDump() {
516        try {
517            StringBuilder b=new StringBuilder();
518            for (Method m : this.getClass().getDeclaredMethods()) {
519                if ( ! m.getName().startsWith("get")) continue;
520                if (m.getReturnType()!=Integer.TYPE && m.getReturnType()!=Float.TYPE) continue;
521                if ( m.getParameterTypes().length!=0 ) continue;
522
523                Object value=m.invoke(this,new Object[0]);
524                b.append(m.getName().substring(3,4).toLowerCase() + m.getName().substring(4) + ": " + value + "\n");
525            }
526            return b.toString();
527        }
528        catch (Exception e) {
529            throw new RuntimeException("Programming error",e);
530        }
531    }
