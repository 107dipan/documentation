009/**
010 * <p>Calculates a set of metrics capturing information about the degree of agreement between a query
011 * and a field string. This algorithm attempts to capture the property of text that very close tokens
012 * are usually part of the same semantic structure, while tokens farther apart are much more loosely related.
013 * The algorithm will locate alternative such regions containing multiple query tokens (segments), do a more
014 * detailed analysis of these segments and choose the ones producing the best overall set of match metrics
015 * (subject to certain resource constraints).</p>
016 *
017 * <p>Such segments are found by looking at query terms in sequence from
018 * left to right and finding matches in the field. All alternative segment start points are explored, and the
019 * segmentation achieving the best overall string match metric score is preferred. Dynamic programming
020 * is used to avoid redoing work on segmentations.</p>
021 *
022 * <p>When a segment start point is found, subsequent tokens from the query are searched in the field
023 * from this starting point in "semantic order". This search order can be defined independently of the
024 * algorithm. The current order searches <i>proximityLimit</i> tokens ahead first, then the same distance backwards
025 * (so if you need to go two steps backwards in the field from the segment starting point, the real distance is -2,
026 * but the "semantic distance" is proximityLimit+2).</p>
027 *
028 * <p>The actual metrics are calculated during execution of this algorithm by the {@link FieldMatchMetrics} class,
029 * by receiving events emitted from the algorithm. Any set of metrics derivable from these events are computable using
030 * this algorithm.</p>
031 *
032 * <p>Terminology:
033 * <ul>
034 * <li><b>Sequence</b> - A set of adjacent matched tokens in the field
035 * <li><b>Segment</b> - A field area containing matches to a continuous section of the query
036 * <li><b>Gap</b> - A chunk of adjacent tokens <i>inside a segment</i> separating two matched characters
037 * <li><b>Semantic distance</b> - A non-continuous distance between tokens in j, where the non-continuousness is
038 * mean to capture the semantic similarity between the query and those tokens.
039 * </ul>
040 *
041 * <p>Notation: A position index in the query is denoted <code>i</code>. A position index in the field is
042 * denoted <code>j</code>.</p>
043 *
044 * <p>This class is not multithread safe, but is reusable across queries for a single thread.</p>
045 *
046 * @author  bratseth
047 */
048public final class FieldMatchMetricsComputer {
049
050    private Query query;
051
052    private Field field;
053
054    private final FieldMatchMetricsParameters parameters;
055
056    /** The metrics of the currently explored segmentation */
057    private FieldMatchMetrics metrics;
058
059    /**
060     * Known segment starting points. The array is 0..i, one element per starting point query item i,
061     * and a last element representing the entire query.
062     */
063    private List<SegmentStartPoint> segmentStartPoints=new ArrayList<>();
064
065    /** True to collect trace */
066    private boolean collectTrace;
067
068    private int alternativeSegmentationsTried=0;

091        return compute(query,fieldString,false);

094    /**
095     * Computes the string match metrics from a query and field string.
096     *
097     * @param query the query to compute over
098     * @param fieldString the field value to compute over - tokenized by splitting on space
099     * @param collectTrace true to accumulate trace information in the trace returned with the metrics
100     */
101    public FieldMatchMetrics compute(Query query, String fieldString, boolean collectTrace) {
102        return compute(query, new Field(fieldString), collectTrace);
103    }
104
105    /**
106     * Computes the string match metrics from a query and field.
107     *
108     * @param query the query to compute over
109     * @param field the field value to compute over
110     * @param collectTrace true to accumulate trace information in the trace returned with the metrics
111     */
112    public FieldMatchMetrics compute(Query query, Field field, boolean collectTrace) {
113        // 1. Reset state
114        this.collectTrace = collectTrace;
115        this.query = query;
116        this.field = field;
117        segmentStartPoints.clear();
118        for (int i = 0; i <= query.getTerms().length; i++)
119            segmentStartPoints.add(null);
120        alternativeSegmentationsTried = 0;
121        metrics = new FieldMatchMetrics(this);
122
123        // 2. Compute
124        exploreSegments();
125        return metrics;
126    }
127
128    /** Finds segment candidates and explores them until we have the best segmentation history of the entire query */
129    private void exploreSegments() {
130        if (collectTrace)
131            metrics.trace().add("Calculating matches for\n    " + query + "\n    " + field + "\n");
132
133        // Create an initial start point
134        SegmentStartPoint segmentStartPoint=new SegmentStartPoint(metrics,this);
135        segmentStartPoints.set(0,segmentStartPoint);
136
137        // Explore segmentations
138        while (segmentStartPoint!=null) {
139            metrics =segmentStartPoint.getMetrics().clone();
140            if (collectTrace)
141                metrics.trace().add("\nLooking for segment from " + segmentStartPoint + "..." + "\n");
142            boolean found=findAlternativeSegmentFrom(segmentStartPoint);
143            if (collectTrace)
144                metrics.trace().add(found ? "...found segment: " + metrics.getSegmentStarts() + " score: " +
145                        metrics.getSegmentationScore() : "...no complete and improved segment existed" + "\n");
146            if (!found)
147                segmentStartPoint.setOpen(false);
148            segmentStartPoint=findOpenSegment(segmentStartPoint.getI());
149        }
150
151        metrics=findLastStartPoint().getMetrics(); // these metrics are the final set
152        setOccurrenceCounts(metrics);
153        metrics.onComplete();
154        metrics.setComplete(true);
155    }
156
157    /**
158     * Find correspondences from a segment starting point
159     *
160     * @return true if a segment was found, false if none could be found
161     */
162    private boolean findAlternativeSegmentFrom(SegmentStartPoint segmentStartPoint) {
163        // i: index into the query
164        // j: index into the field
165        int semanticDistanceExplored=segmentStartPoint.getSemanticDistanceExplored();
166        int previousI=-1;
167        int previousJ=segmentStartPoint.getPreviousJ();
168        boolean hasOpenSequence=false;
169        boolean isFirst=true;
170
171        for (int i=segmentStartPoint.getStartI(); i<query.getTerms().length; i++) {
172            int semanticDistance=findClosestInFieldBySemanticDistance(i,previousJ,semanticDistanceExplored);
173            int j=semanticDistanceToFieldIndex(semanticDistance,previousJ);
174
175            if (j==-1 && semanticDistanceExplored>0 && isFirst) {
176                return false; // Segment explored before, and no more matches found
177            }
178
179            if ( hasOpenSequence && ( j==-1 || j!=previousJ+1 ) ) {
180                metrics.onSequenceEnd(previousJ);
181                hasOpenSequence=false;
182            }
183
184            if (isFirst) {
185                if (j!=-1) {
186                    segmentStart(i,j,isFirst ? -1 : previousJ);
187                    segmentStartPoint.exploredTo(j);
188                    isFirst=false;
189                }
190                else {
191                    segmentStartPoint.incrementStartI(); // Remember that there are no matches for this i
192                }
193            }
194            else {
195                if (Math.abs(j-previousJ) >= parameters.getProximityLimit()) {
196                    segmentEnd(i-1,previousJ);
197                    return true;
198                }
199                else if (j!=-1) {
200                    inSegment(i,j,previousJ,previousI);
201                }
202            }
203
204            if (j!=-1)
205                metrics.onMatch(i,j);
206
207            if (j!=-1 && !hasOpenSequence) {
208                metrics.onSequenceStart(j);
209                hasOpenSequence=true;
210            }
211
212            if (j!=-1)
213                semanticDistanceExplored=1; // Skip the current match when looking for the next
214            else
215                semanticDistanceExplored=0;
216
217            if (j>=0) {
218                previousI=i;
219                previousJ=j;
220            }
221        }
222
223        if (hasOpenSequence)
224            metrics.onSequenceEnd(previousJ);
225
226        if (!isFirst) {
227            segmentEnd(query.getTerms().length-1,previousJ);
228            return true;
229        }
230        else {
231            return false;
232        }
233    }
234
235    /**
236     * Implements the preferred search order for finding a match to a query item - first
237     * looking close in the right order, then close in the reverse order, then far in the right order
238     * and lastly far in the reverse order.
239     *
240     * @param startSemanticDistance is the semantic distance we must be larger than or equal to
241     * @return the semantic distance of the next mathing j larger than startSemanticDistance, or -1 if
242     *         there are no matches larger than startSemanticDistance
243     */
244    private int findClosestInFieldBySemanticDistance(int i,int previousJ,int startSemanticDistance) {
245        String term=query.getTerms()[i].getTerm();
246        for (int distance=startSemanticDistance; distance<field.terms().size(); distance++) {
247            int j=semanticDistanceToFieldIndex(distance,previousJ);
248            if (term.equals(field.terms().get(j).value()))
249                return distance;
250        }
251        return -1;
252    }
253
254    /**
255     * Returns the field index (j) from a starting point zeroJ and the distance form zeroJ in the
256     * semantic distance space
257     *
258     * @return the field index, or -1 (undefined) if the semanticDistance is -1
259     */
260    int semanticDistanceToFieldIndex(int semanticDistance,int zeroJ) {
261        if (semanticDistance == -1) return -1;
262        int firstSegmentLength = Math.min(parameters.getProximityLimit(),field.terms().size() - zeroJ);
263        int secondSegmentLength = Math.min(parameters.getProximityLimit(), zeroJ);
264        if (semanticDistance < firstSegmentLength)
265            return zeroJ + semanticDistance;
266        else if (semanticDistance < firstSegmentLength+secondSegmentLength)
267            return zeroJ - semanticDistance - 1 + firstSegmentLength;
268        else if (semanticDistance < field.terms().size() - zeroJ+secondSegmentLength)
269            return zeroJ + semanticDistance - secondSegmentLength;
270        else
271            return field.terms().size() - semanticDistance - 1;
272    }
273
274    /**
275     * Returns the semantic distance from a starting point zeroJ to a field index j
276     *
277     * @return the semantic distance, or -1 (undefined) if j is -1
278     */
279    int fieldIndexToSemanticDistance(int j,int zeroJ) {
280        if (j == -1) return -1;
281        int firstSegmentLength = Math.min(parameters.getProximityLimit(), field.terms().size() - zeroJ);
282        int secondSegmentLength = Math.min(parameters.getProximityLimit(), zeroJ);
283        if (j >= zeroJ) {
284            if ( (j - zeroJ) < firstSegmentLength )
285                return j - zeroJ; // 0..limit
286            else
287                return j - zeroJ+secondSegmentLength; // limit*2..field.length-zeroJ
288        }
289        else {
290            if ( (zeroJ - j - 1) < secondSegmentLength )
291                return zeroJ - j + firstSegmentLength-1; // limit..limit*2
292            else
293                return (zeroJ - j - 1) + field.terms().size() - zeroJ; // field.length-zeroJ..
294        }
295
296    }
297
298    private void inSegment(int i, int j, int previousJ, int previousI) {
299        metrics.onPair(i, j, previousJ);
300        if (j==previousJ+1 && i==previousI+1) {
301            metrics.onInSequence(i, j, previousJ);
302        }
303        else {
304            metrics.onInSegmentGap(i, j, previousJ);
305            if (collectTrace)
306                metrics.trace().add("      in segment gap: " + i + "->" + j + " (" + query.getTerms()[i] + ")" + "\n");
307        }
308    }
309
310    /** Returns whether this segment was accepted as a starting point */
311    private boolean segmentStart(int i,int j,int previousJ) {
312        metrics.onNewSegment(i, j, previousJ);
313
314        if (previousJ>=0)
315            metrics.onPair(i,j,previousJ);
316
317        if (collectTrace)
318            metrics.trace().add("    new segment at:   " + i + "->" + j + " (" + query.getTerms()[i] + ")" + "\n");
319        return true;
320    }
321
322    /**
323     * Registers an end of a segment
324     *
325     * @param i the i at which this segment ends
326     * @param j the j at which this segment ends
327     */
328    private void segmentEnd(int i,int j) {
329        if (collectTrace)
330            metrics.trace().add("    segment ended at: " + i + "->" + j + " (" + query.getTerms()[i] + ")" + "\n");
331        SegmentStartPoint startOfNext=segmentStartPoints.get(i + 1);
332        if (startOfNext==null)
333            segmentStartPoints.set(i+1,new SegmentStartPoint(i+1,j, metrics,this));
334        else
335            startOfNext.offerHistory(j, metrics, collectTrace);
336    }
337
338    /** Returns the next open segment to explore, or null if no more segments exists or should be explored */
339    private SegmentStartPoint findOpenSegment(int startI) {
340        for (int i=startI; i<segmentStartPoints.size(); i++) {
341            SegmentStartPoint startPoint=segmentStartPoints.get(i);
342            if (startPoint==null || !startPoint.isOpen()) continue;
343
344            if (startPoint.getSemanticDistanceExplored()==0) return startPoint; // First attempt
345
346            if (alternativeSegmentationsTried>=parameters.getMaxAlternativeSegmentations()) continue;
347            alternativeSegmentationsTried++;
348            return startPoint;
349        }
350
351        return null;
352    }
353
354    private SegmentStartPoint findLastStartPoint() {
355        for (int i=segmentStartPoints.size()-1; i>=0; i--) {
356            SegmentStartPoint startPoint=segmentStartPoints.get(i);
357            if (startPoint!=null)
358                return startPoint;
359        }
360        return null; // Impossible
361    }

363    /** Counts all occurrences of terms of the query in the field and set those metrics */
364    private void setOccurrenceCounts(FieldMatchMetrics metrics) {
365        Set<QueryTerm> uniqueQueryTerms=new HashSet<>();
366        for (QueryTerm queryTerm : query.getTerms())
367            uniqueQueryTerms.add(queryTerm);
368
369        List<Float> weightedOccurrences=new ArrayList<Float>();
370        List<Float> significantOccurrences=new ArrayList<Float>();
371
372        int divider = Math.min(field.terms().size(),parameters.getMaxOccurrences()*uniqueQueryTerms.size());
373        int maxOccurence = Math.min(field.terms().size(),parameters.getMaxOccurrences());
374
375        float occurrence=0;
376        float absoluteOccurrence=0;
377        float weightedAbsoluteOccurrence=0;
378        int totalWeight=0;
379        float totalWeightedOccurrences=0;
380        float totalSignificantOccurrences=0;
381
382        for (QueryTerm queryTerm : uniqueQueryTerms) {
383            int termOccurrences=0;
384            for (Field.Term fieldTerm : field.terms()) {
385                if (fieldTerm.value().equals(queryTerm.getTerm()))
386                    termOccurrences++;
387                if (termOccurrences == parameters.getMaxOccurrences()) break;
388            }
389            occurrence+=(float)termOccurrences/divider;
390
391            absoluteOccurrence+=(float)termOccurrences/(parameters.getMaxOccurrences()*uniqueQueryTerms.size());
392
393            weightedAbsoluteOccurrence+=(float)termOccurrences*queryTerm.getWeight()/parameters.getMaxOccurrences();
394            totalWeight+=queryTerm.getWeight();
395
396            totalWeightedOccurrences+=(float)maxOccurence*queryTerm.getWeight()/divider;
397            weightedOccurrences.add((float)termOccurrences*queryTerm.getWeight()/divider);
398
399            totalSignificantOccurrences+=(float)maxOccurence*queryTerm.getSignificance()/divider;
400            significantOccurrences.add((float)termOccurrences*queryTerm.getSignificance()/divider);
401        }
402
403        float weightedOccurrenceSum=0;
404        for (float weightedOccurence : weightedOccurrences)
405            weightedOccurrenceSum+=weightedOccurence/totalWeightedOccurrences;
406
407        float significantOccurrenceSum=0;
408        for (float significantOccurence : significantOccurrences)
409            significantOccurrenceSum+=significantOccurence/totalSignificantOccurrences;
410
411        if (totalWeight>0)
412            weightedAbsoluteOccurrence=weightedAbsoluteOccurrence/totalWeight;
413
414        metrics.setOccurrence(occurrence);
415        metrics.setAbsoluteOccurrence(absoluteOccurrence);
416        metrics.setWeightedOccurrence(weightedOccurrenceSum);
417        metrics.setWeightedAbsoluteOccurrence(weightedAbsoluteOccurrence);
418        metrics.setSignificantOccurrence(significantOccurrenceSum);
419    }

421    /** Returns the parameter settings of this */
422    public FieldMatchMetricsParameters getParameters() { return parameters; }
423
424    Query getQuery() { return query; }
425
426    Field getField() { return field; }

428    @Override
429    public String toString() {
430        return query + "\n" + field + "\n" + metrics + "\n";
431    }
