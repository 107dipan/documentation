002package com.yahoo.searchlib.rankingexpression.rule;
003
004import com.yahoo.searchlib.rankingexpression.evaluation.Context;
005import com.yahoo.searchlib.rankingexpression.evaluation.Value;
006
007import java.util.Deque;
008
009/**
010 * A node which holds a constant (frozen) value.
011 *
012 * @author Simon Thoresen
013 */
014public final class ConstantNode extends ExpressionNode {
015
016    private final String sourceImage;
017
018    private final Value value;
019
020    public ConstantNode(Value value) {
021        this(value,null);
022    }
023
024    /**
025     * Creates a constant value
026     *
027     * @param value the value. Ownership of this value is transferred to this.
028     * @param sourceImage the source string image producing this value
029     */
030    public ConstantNode(Value value, String sourceImage) {
031        value.freeze();
032        this.value = value;
033        this.sourceImage = sourceImage;
034    }
035
036    public Value getValue() { return value; }
037
038    @Override
039    public String toString(SerializationContext context, Deque<String> path, CompositeNode parent) {
040        return sourceString();
041    }
042
043    /** Returns the string which created this, or the value.toString() if not known */
044    public String sourceString() {
045        if (sourceImage != null) return sourceImage;
046        return value.toString();
047    }
048
049    @Override
050    public Value evaluate(Context context) {
051        return value;
052    }
053
054}
