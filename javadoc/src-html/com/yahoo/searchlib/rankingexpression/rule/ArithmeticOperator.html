002package com.yahoo.searchlib.rankingexpression.rule;
003
004import com.yahoo.searchlib.rankingexpression.evaluation.Value;
005
006import java.util.ArrayList;
007import java.util.Collections;
008import java.util.List;
009
010/**
011 * A mathematical operator
012 *
013 * @author bratseth
014 */
015public enum ArithmeticOperator {
016
017    PLUS(0, "+") { public Value evaluate(Value x, Value y) {
018        return x.add(y);
019    }},
020    MINUS(1, "-") { public Value evaluate(Value x, Value y) {
021        return x.subtract(y);
022    }},
023    MULTIPLY(2, "*") { public Value evaluate(Value x, Value y) {
024        return x.multiply(y);
025    }},
026    DIVIDE(3, "/") { public Value evaluate(Value x, Value y) {
027        return x.divide(y);
028    }};
029
030    /** A list of all the operators in this in order of decreasing precedence */
031    public static final List<ArithmeticOperator> operatorsByPrecedence = operatorsByPrecedence();
032
033    private final int precedence;
034    private final String image;
035
036    private ArithmeticOperator(int precedence, String image) {
037        this.precedence = precedence;
038        this.image = image;
039    }
040
041    /** Returns true if this operator has precedence over the given operator */
042    public boolean hasPrecedenceOver(ArithmeticOperator op) {
043        return precedence > op.precedence;
044    }
045
046    public abstract Value evaluate(Value x, Value y);
047
048    @Override
049    public String toString() {
050        return image;
051    }
052
053    private static List<ArithmeticOperator> operatorsByPrecedence() {
054        List<ArithmeticOperator> operators = new ArrayList<>();
055        operators.add(DIVIDE);
056        operators.add(MULTIPLY);
057        operators.add(MINUS);
058        operators.add(PLUS);
059        return Collections.unmodifiableList(operators);
060    }
061
062}
