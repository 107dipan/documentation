002package com.yahoo.searchlib.rankingexpression.rule;
003
004import com.yahoo.searchlib.rankingexpression.evaluation.Context;
005import com.yahoo.searchlib.rankingexpression.evaluation.Value;
006
007import java.util.Collections;
008import java.util.Deque;
009import java.util.List;
010
011/**
012 * A node which flips the sign of the value produced from the nested expression
013 *
014 * @author bratseth
015 */
016public class NegativeNode extends CompositeNode {
017
018    private final ExpressionNode value;
019
020    /** Constructs a new negative node */
021    public NegativeNode(ExpressionNode value) {
022        this.value = value;
023    }
024
025    /** Returns the node creating the value negated by this */
026    public ExpressionNode getValue() { return value; }
027
028    @Override
029    public List<ExpressionNode> children() {
030        return Collections.singletonList(value);
031    }
032
033    @Override
034    public String toString(SerializationContext context, Deque<String> path, CompositeNode parent) {
035        return "-" + value.toString(context, path, parent);
036    }
037
038    @Override
039    public Value evaluate(Context context) {
040        return value.evaluate(context).negate();
041    }
042
043    @Override
044    public NegativeNode setChildren(List<ExpressionNode> children) {
045        if (children.size() != 1) throw new IllegalArgumentException("Expected 1 children but got " + children.size());
046        return new NegativeNode(children.get(0));
047    }
048
049}
