002package com.yahoo.searchlib.rankingexpression.rule;
003
004import java.util.List;
005
006/**
007 * The parent of all node types which contains child nodes.
008 *
009 * @author bratseth
010 */
011public abstract class CompositeNode extends ExpressionNode {
012
013    /**
014     * <p>Returns a read-only list containing the immediate children of this composite</p>
015     *
016     * @return The children of this.
017     */
018    public abstract List<ExpressionNode> children();
019
020    /**
021     * Returns a copy of this where the children is replaced by the given children.
022     *
023     * @throws IllegalArgumentException if the given list of children has different size than children()
024     */
025    public abstract CompositeNode setChildren(List<ExpressionNode> children);
026
027}
