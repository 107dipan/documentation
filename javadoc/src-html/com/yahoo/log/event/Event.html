015/**
016 * The Event class is the base class for all VESPA defined events.
017 * All specific Event classes extend this abstract class.  An event
018 * is more or less a type and a set of properties.  For convenience
019 * we use the logging system to transport Event instances, so the
020 * typical use is to serialize events into the payload of log
021 * messages.
022 *
023 * <P>
024 * Note that the static methods started(), stopped() etc are for use
025 * with this class so using them in the subclasses isn't really
026 * sanctioned.  These methods are what the user wants to use for
027 * logging events, rather than making events him/herself and stuffing
028 * them through the logging API.
029 *
030 * @author  <a href="mailto:borud@yahoo-inc.com">Bjorn Borud</a>
031 */
032public abstract class Event implements Serializable {
033    private static Logger log = Logger.getLogger(Event.class.getName());

036    private static final int INITIAL     = 0;
037    private static final int IN_NAME     = 1;
038    private static final int IN_UNQUOTED = 2;
039    private static final int IN_QUOTED   = 3;
040    private static final int EQUALS      = 4;

110    /**
111     * Convenience method which returns a property.  If the
112     * property contains whitespace the property will be
113     * enclosed in quotes.
114     *
115     * FIXME: quotes inside the value are not quoted
116     *
117     */
118    public String getValuePossiblyQuote (String name) {
119        String tmp =  values.get(name);
120        if (tmp == null) {
121            return "";
122        }
123
124        Matcher m = whitespace.matcher(tmp);
125        if (m.find()) {
126            return new StringBuffer(tmp.length() + 2)
127                .append("\"")
128                .append(tmp)
129                .append("\"")
130                .toString();
131        }
132
133        return tmp;
134    }
135
136    /**
137     * Get the name of the event instance.
138     *
139     * @return the name of the event instance.
140     */
141    public String getName() {
142        String tmp = this.getClass().getName();
143        int last = tmp.lastIndexOf(".");
144        if (last == -1) {
145            return tmp.toLowerCase();
146        }
147
148        return tmp.substring(last+1).toLowerCase();
149    }
150
151    /**
152     * This method returns the string representation of the
153     * event and must return something that can be parsed
154     * by the parse method.
155     */
156    public String toString () {
157        StringBuilder buff = new StringBuilder(128)
158            .append(getName())
159            .append("/")
160            .append(version);
161
162        for (String name : values.keySet()) {
163            buff.append(" ")
164            .append(name)
165            .append("=")
166            .append(getValuePossiblyQuote(name));
167        }
168
169        return buff.toString();
170    }

173    /**
174     * Okay, so I am good at making state machine based parsers,
175     * but every time I do it it feels uncomfortable because it
176     * isn't the sort of thing one ought to do by freehand :-).
177     *
178     * <P>
179     * Enjoy the graphic
180     *
181     * <PRE>
182     *    ___
183     *    ',_`""\        .---,
184     *       \   :~""``/`    |
185     *        `;'     //`\   /
186     *        /   __     |   ('.
187     *       |_ ./O)\     \  `) \
188     *      _/-.    `      `"`  |`-.
189     *  .-=: `                  /   `-.
190     * /o o \   ,_,           .        '.
191     * L._._;_.-'           .            `'-.
192     *   `'-.`             '                 `'-.
193     *       `.         '                        `-._
194     *         '-._. -'                              '.
195     *            \                                    `\
196     *
197     *
198     *
199     * </PRE>
200     */
201    private static void parseValuePairs (String s, Event event) {
202        int state = INITIAL;
203        int i     = 0;
204        int mark  = 0;
205        String name = null;
206
207        while (i < s.length()) {
208            switch(s.charAt(i)) {
209
210                case ' ':
211                    if (state == IN_UNQUOTED) {
212                        state = INITIAL;
213                        event.setValue(name, s.substring(mark, i));
214                    }
215
216                    if (state == INITIAL) {
217                        mark = -1;
218                        break;
219                    }
220
221                    if (state == IN_QUOTED) {
222                        break;
223                    }
224
225                    throw new IllegalStateException("space not allowed at " + i);
226
227                case '=':
228                    if (state == IN_NAME) {
229                        name = s.substring(mark, i);
230                        state = EQUALS;
231                        break;
232                    }
233                    if (state == IN_QUOTED) {
234                        break;
235                    }
236
237                    throw new IllegalStateException("'=' not allowed at " + i);
238
239                case '"':
240                    if (state == EQUALS) {
241                        state = IN_QUOTED;
242                        mark = i;
243                        break;
244                    }
245
246                    if (state == IN_QUOTED) {
247                        // skip escaped
248                        if (s.charAt(i-1) == '\\') {
249                            break;
250                        }
251                        event.setValue(name, s.substring(mark+1, i));
252                        state = INITIAL;
253                        break;
254                    }
255
256                    throw new IllegalStateException("'\"' not allowed at " + i);
257
258                // ordinary characters
259                default:
260                    if (state == INITIAL) {
261                        state = IN_NAME;
262                        mark = i;
263                        break;
264                    }
265
266                    if (state == EQUALS) {
267                        state = IN_UNQUOTED;
268                        mark = i;
269                        break;
270                    }
271            }
272            i++;
273        }
274
275        // mopping up.  when there is no more input to be processed
276        // we need to take action if we are in one of the below states
277        switch (state) {
278            case IN_UNQUOTED:
279                event.setValue(name, s.substring(mark, i));
280                break;
281
282            case IN_QUOTED:
283                event.setValue(name, s.substring(mark+1, i));
284                break;
285
286            case IN_NAME:
287                throw new IllegalStateException("ended in name");
288
289            case EQUALS:
290                event.setValue(name, null);
291                break;
292        }
293    }

296    /**
297     * Parse string representation of Event and emit correct Event
298     * subtype.
299     *
300     * @param s A string containing an event
301     * @return Event represented by <code>s</code>.
302     * @throws MalformedEventException if unable to deciper Event
303     *           from string.
304     */
305    public static Event parse (String s) throws MalformedEventException {
306        Matcher m1 = eventFmt.matcher(s);
307        if (! m1.matches()) {
308            throw new MalformedEventException(s);
309        }
310        String eventName = m1.group(1);
311        String eventVersion = m1.group(2);
312        String rest = m1.group(3);
313
314        String className = new StringBuffer(eventName.length()
315                                            + packageName.length()
316                                            + 1)
317            .append(packageName)
318            .append(".")
319            .append(eventName.substring(0,1).toUpperCase())
320            .append(eventName.substring(1).toLowerCase())
321            .toString();
322
323        Event event;
324        try {
325            event = (Event) Class.forName(className).newInstance();
326        }
327        catch (ClassNotFoundException e) {
328            event = new Unknown().setName(eventName);
329        }
330        catch (Exception e) {
331            log.log(Level.WARNING, "Event instantiation problem", e);
332            return null;
333        }
334
335        event.setVersion(Integer.parseInt(eventVersion));
336
337        try {
338            parseValuePairs(rest, event);
339        }
340        catch (IllegalStateException | NumberFormatException e) {
341            throw new MalformedEventException(e);
342        }
343
344        return event;
345    }

347    /**
348     * Find the stack frame of the last called before we entered
349     * the Event class and get the Logger belonging to that class.
350     * If for some reason we fail to do this, just return the
351     * default Logger.
352     *
353     * <P>
354     * Beware, if you come here for cleverness and enlightenment
355     * then know that this technique might not be exceptionally fast
356     * so don't abuse this mechanism blindly.  (Do what I do:  abuse
357     * it with both eyes open :-).
358     *
359     */
360    private static final Logger getCallerLogger() {
361        StackTraceElement stack[] = (new Throwable()).getStackTrace();
362        int i = 0;
363        while (i < stack.length) {
364            StackTraceElement frame = stack[i];
365            String cname = frame.getClassName();
366            if (cname.equals("com.yahoo.log.event.Event")) {
367                break;
368            }
369            i++;
370        }
371
372        while (i < stack.length) {
373            StackTraceElement frame = stack[i];
374            String cname = frame.getClassName();
375            if (!cname.equals("com.yahoo.log.event.Event")) {
376                return Logger.getLogger(cname);
377            }
378            i++;
379        }
380
381        return Logger.getLogger("");
382    }

384    /**
385     * Internal method which prepares Event log messages.  Not
386     * the prettiest way to do it...
387     */
388    private static final void log(Logger logger, Object param) {
389        LogRecord r = new LogRecord(LogLevel.EVENT, null);
390        r.setParameters(new Object[] {param});
391        r.setLoggerName(logger.getName());
392        logger.log(r);
393    }
