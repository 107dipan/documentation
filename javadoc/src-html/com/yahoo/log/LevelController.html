002/**
003 * This is the interface for controlling the log level of a
004 * component logger.  This hides the actual controlling
005 * mechanism.
006 *
007 * @author arnej27959
008 *
009 */
010
011/**
012 * @author arnej27959
013 **/
014package com.yahoo.log;
015
016import java.util.logging.Level;
017
018public interface LevelController {
019
020    /**
021     * should we actually publish a log message with the given Level now?
022     **/
023    public boolean shouldLog(Level level);
024
025    /**
026     * return a string suitable for printing in a logctl file.
027     * the string must be be 4 * 8 characters, where each group
028     * of 4 characters is either "  ON" or " OFF".
029     **/
030    public String getOnOffString();
031
032    /**
033     * check the current state of logging and reflect it into the
034     * associated Logger instance, if available.
035     **/
036    public void checkBack();
037    public Level getLevelLimit();
038}
