002package com.yahoo.log;
003
004import java.io.Serializable;
005import java.util.Comparator;
006
007/**
008 * Order LogMessage instances based on timestamp. The default ordering is ascending.
009 * This may be reversed by the constructor argument.
010 *
011 * Note: this comparator imposes orderings that are inconsistent with equals.
012 * This is due to only looking at the timestamp, so two different messages with
013 * the same timestamp would appear "equal" to this comparator.
014 *
015 * @author vlarsen
016 *
017 */
018public class LogMessageTimeComparator implements Comparator<LogMessage>, Serializable {
019        private static final long serialVersionUID = 1L;
020
021        /** Indicate the ordering of the timecomparison */
022        private boolean ascending = true;
023
024        /**
025         * Create a Time comparator for logmessages. Order is ascending.
026         *
027         */
028        public LogMessageTimeComparator() {}
029
030        /**
031         * Create a Time comparator for logmessages. The chronological order is dependent
032         * on the argument to the constructor.
033         *
034         * @param ascending true if you want LogMessages ordered ascending according to timestamp.
035         */
036        public LogMessageTimeComparator(boolean ascending) {
037                this.ascending = ascending;
038        }
039
040        public int compare(LogMessage message1, LogMessage message2) {
041                return ascending ?
042                                Long.valueOf(message1.getTime()).compareTo(Long.valueOf(message2.getTime()))
043                                : Long.valueOf(message2.getTime()).compareTo(Long.valueOf(message1.getTime()));
044        }
045}
