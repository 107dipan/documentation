002package com.yahoo.yolean.function;
003
004import java.util.Objects;
005
006/**
007 * Functional interface that mirrors the Consumer interface, but allows for an
008 * exception to be thrown.
009 *
010 * @author oyving
011 */
012@FunctionalInterface
013public interface ThrowingConsumer<T, E extends Throwable> {
014    void accept(T input) throws E;
015
016    default ThrowingConsumer<T, E> andThen(ThrowingConsumer<? super T, ? extends E> after) {
017        Objects.requireNonNull(after);
018        return (T t) -> { accept(t); after.accept(t); };
019    }
020}
