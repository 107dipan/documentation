002package com.yahoo.yolean.trace;
003
004/**
005 * <p>This class is an abstract visitor of {@link TraceNode}. See {@link TraceNode#accept(TraceVisitor)}.</p>
006 *
007 * @author bratseth
008 * @since 5.1.15
009 */
010public abstract class TraceVisitor {
011
012    /**
013     * <p>Visits a {@link TraceNode}. Called by {@link TraceNode#accept(TraceVisitor)}, before visiting its
014     * children.</p>
015     *
016     * @param node the <tt>TraceNode</tt> being visited
017     * @see TraceNode#accept(TraceVisitor)
018     */
019    public abstract void visit(TraceNode node);
020
021    /**
022     * <p>Enters a {@link TraceNode}. This method is called after {@link #visit(TraceNode)}, but before visiting its
023     * children. Note that this method is NOT called if a <tt>TraceNode</tt> has zero children.</p>
024     * <p>The default implementation of this method does nothing.</p>
025     *
026     * @param node the <tt>TraceNode</tt> being entered
027     * @see #leaving(TraceNode)
028     */
029    @SuppressWarnings("UnusedParameters")
030    public void entering(TraceNode node) {
031        // empty
032    }
033
034    /**
035     * <p>Leaves a {@link TraceNode}. This method is called after {@link #entering(TraceNode)}, and after visiting its
036     * children. Note that this method is NOT called if a <tt>TraceNode</tt> has zero children.</p>
037     * <p>The default implementation of this method does nothing.</p>
038     *
039     * @param node the <tt>TraceNode</tt> being left
040     */
041    @SuppressWarnings("UnusedParameters")
042    public void leaving(TraceNode node) {
043        // empty
044    }
045}
