002package com.yahoo.application.container.handler;
003
004import com.google.common.annotations.Beta;
005import com.yahoo.jdisc.HeaderFields;
006
007import javax.annotation.concurrent.NotThreadSafe;
008import java.util.Collection;
009import java.util.List;
010import java.util.Map;
011import java.util.Set;
012
013/**
014 * A multi-map for Request and Response header fields.
015 *
016 * @see Request
017 * @see Response
018 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
019 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
020 */
021@NotThreadSafe
022@Beta
023public class Headers implements Map<String, List<String>> {
024    private final HeaderFields h = new HeaderFields();
025
026    @Override
027    public int size() {
028        return h.size();
029    }
030
031    @Override
032    public boolean isEmpty() {
033        return h.isEmpty();
034    }
035
036    @Override
037    public boolean containsKey(Object key) {
038        return h.containsKey(key);
039    }
040
041    @Override
042    public boolean containsValue(Object value) {
043        return h.containsValue(value);
044    }
045
046    @Override
047    public List<String> get(Object key) {
048        return h.get(key);
049    }
050
051    @Override
052    public List<String> put(String key, List<String> value) {
053        return h.put(key, value);
054    }
055
056    @Override
057    public List<String> remove(Object key) {
058        return h.remove(key);
059    }
060
061    @Override
062    public void putAll(Map<? extends String, ? extends List<String>> m) {
063        h.putAll(m);
064    }
065
066    @Override
067    public void clear() {
068        h.clear();
069    }
070
071    @Override
072    public Set<String> keySet() {
073        return h.keySet();
074    }
075
076    @Override
077    public Collection<List<String>> values() {
078        return h.values();
079    }
080
081    @Override
082    public Set<Entry<String, List<String>>> entrySet() {
083        return h.entrySet();
084    }
085
086    @Override
087    public String toString() {
088        return h.toString();
089    }
090
091    @Override
092    public boolean equals(Object obj) {
093        return obj instanceof Headers && h.equals(((Headers)obj).h);
094    }
095
096    @Override
097    public int hashCode() {
098        return h.hashCode();
099    }
100
101
102    /*
103    CONVENIENCE METHODS:
104     */
105
106    /**
107     * <p>Convenience method for checking whether or not a named header contains a specific value. If the named header
108     * is not set, or if the given value is not contained within that header's value list, this method returns
109     * <em>false</em>.</p>
110     *
111     * <p><em>NOTE:</em> This method is case-SENSITIVE.</p>
112     *
113     * @param key   The key whose values to search in.
114     * @param value The values to search for.
115     * @return True if the given value was found in the named header.
116     * @see #containsIgnoreCase
117     */
118    public boolean contains(String key, String value) {
119        return h.contains(key, value);
120    }
121
122    /**
123     * <p>Convenience method for checking whether or not a named header contains a specific value, regardless of case.
124     * If the named header is not set, or if the given value is not contained within that header's value list, this
125     * method returns <em>false</em>.</p>
126     *
127     * <p><em>NOTE:</em> This method is case-INSENSITIVE.</p>
128     *
129     * @param key   The key whose values to search in.
130     * @param value The values to search for, ignoring case.
131     * @return True if the given value was found in the named header.
132     * @see #contains
133     */
134    public boolean containsIgnoreCase(String key, String value) {
135        return h.containsIgnoreCase(key, value);
136    }
137
138    /**
139     * <p>Adds the given value to the entry of the specified key. If no entry exists for the given key, a new one is
140     * created containing only the given value.</p>
141     *
142     * @param key   The key with which the specified value is to be associated.
143     * @param value The value to be added to the list associated with the specified key.
144     */
145    public void add(String key, String value) {
146        h.add(key, value);
147    }
148
149    /**
150     * <p>Adds the given values to the entry of the specified key. If no entry exists for the given key, a new one is
151     * created containing only the given values.</p>
152     *
153     * @param key    The key with which the specified value is to be associated.
154     * @param values The values to be added to the list associated with the specified key.
155     */
156    public void add(String key, List<String> values) {
157        h.add(key, values);
158    }
159
160    /**
161     * <p>Adds all the entries of the given map to this. This is the same as calling {@link #add(String, List)} for each
162     * entry in <tt>values</tt>.</p>
163     *
164     * @param values The values to be added to this.
165     */
166    public void addAll(Map<? extends String, ? extends List<String>> values) {
167        h.addAll(values);
168    }
169
170    /**
171     * <p>Convenience method to call {@link #put(String, List)} with a singleton list that contains the specified
172     * value.</p>
173     *
174     * @param key   The key of the entry to put.
175     * @param value The value to put.
176     * @return The previous value associated with <tt>key</tt>, or <tt>null</tt> if there was no mapping for
177     *         <tt>key</tt>.
178     */
179    public List<String> put(String key, String value) {
180        return h.put(key, value);
181    }
182
183    /**
184     * <p>Removes the given value from the entry of the specified key.</p>
185     *
186     * @param key   The key of the entry to remove from.
187     * @param value The value to remove from the entry.
188     * @return True if the value was removed.
189     */
190    public boolean remove(String key, String value) {
191        return h.remove(key, value);
192    }
193
194    /**
195     * <p>Convenience method for retrieving the first value of a named header field. If the header is not set, or if the
196     * value list is empty, this method returns null.</p>
197     *
198     * @param key The key whose first value to return.
199     * @return The first value of the named header, or null.
200     */
201    public String getFirst(String key) {
202        return h.getFirst(key);
203    }
204
205    /**
206     * <p>Convenience method for checking whether or not a named header field is <em>true</em>. To satisfy this, the
207     * header field needs to have at least 1 entry, and Boolean.valueOf() of all its values must parse as
208     * <em>true</em>.</p>
209     *
210     * @param key The key whose values to parse as a boolean.
211     * @return The boolean value of the named header.
212     */
213    public boolean isTrue(String key) {
214        return h.isTrue(key);
215    }
216
217    /**
218     * <p>Returns an unmodifiable list of all key-value pairs of this. This provides a flattened view on the content of
219     * this map.</p>
220     *
221     * @return The collection of entries.
222     */
223    public List<Entry<String, String>> entries() {
224        return h.entries();
225    }
226}
