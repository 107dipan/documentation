002package com.yahoo.io.reader;
003
004import com.google.common.annotations.Beta;
005
006import java.io.IOException;
007import java.io.Reader;
008import java.util.List;
009
010/**
011 * A reader with a name. All reader methods are delegated to the wrapped reader.
012 *
013 * @author bratseth
014 */
015@Beta
016public class NamedReader extends Reader {
017
018    private final String name;
019    private final Reader reader;
020
021    public NamedReader(String name, Reader reader) {
022        this.name = name;
023        this.reader = reader;
024    }
025
026    public String getName() { return name; }
027
028    public Reader getReader() { return reader; }
029
030    /** Returns the name */
031    public @Override String toString() {
032        return name;
033    }
034
035    // The rest is reader method implementations which delegates to the wrapped reader
036    public @Override int read(java.nio.CharBuffer charBuffer) throws java.io.IOException { return reader.read(charBuffer); }
037    public @Override int read() throws java.io.IOException { return reader.read(); }
038    public @Override int read(char[] chars) throws java.io.IOException { return reader.read(chars); }
039    public @Override int read(char[] chars, int i, int i1) throws java.io.IOException { return reader.read(chars,i,i1); }
040    public @Override long skip(long l) throws java.io.IOException { return reader.skip(l); }
041    public @Override boolean ready() throws java.io.IOException { return reader.ready(); }
042    public @Override boolean markSupported() { return reader.markSupported(); }
043    public @Override void mark(int i) throws java.io.IOException { reader.mark(i); }
044    public @Override void reset() throws java.io.IOException { reader.reset(); }
045    public @Override void close() throws java.io.IOException { reader.close(); }
046
047    /** Convenience method for closing a list of readers. Does nothing if the given reader list is null. */
048    public static void closeAll(List<NamedReader> readers) {
049        if (readers==null) return;
050        for (Reader reader : readers) {
051            try {
052                reader.close();
053            }
054            catch (IOException e) {
055                // Nothing to do about it
056            }
057        }
058    }
059
060}
