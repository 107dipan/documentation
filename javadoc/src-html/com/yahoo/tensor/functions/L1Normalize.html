001package com.yahoo.tensor.functions;
002
003import com.google.common.annotations.Beta;
004
005import java.util.Collections;
006import java.util.List;
007
008/**
009 * @author bratseth
010 */
011@Beta
012public class L1Normalize extends CompositeTensorFunction {
013
014    private final TensorFunction argument;
015    private final String dimension;
016    
017    public L1Normalize(TensorFunction argument, String dimension) {
018        this.argument = argument;
019        this.dimension = dimension;
020    }
021
022    @Override
023    public List<TensorFunction> functionArguments() { return Collections.singletonList(argument); }
024
025    @Override
026    public TensorFunction replaceArguments(List<TensorFunction> arguments) {
027        if ( arguments.size() != 1)
028            throw new IllegalArgumentException("L1Normalize must have 1 argument, got " + arguments.size());
029        return new L1Normalize(arguments.get(0), dimension);
030    }
031
032    @Override
033    public PrimitiveTensorFunction toPrimitive() {
034        TensorFunction primitiveArgument = argument.toPrimitive();
035        // join(x, reduce(x, "avg", "dimension"), f(x,y) (x / y))
036        return new Join(primitiveArgument,
037                        new Reduce(primitiveArgument, Reduce.Aggregator.sum, dimension),
038                        ScalarFunctions.divide());
039    }
040    
041    @Override
042    public String toString(ToStringContext context) {
043        return "l1_normalize(" + argument.toString(context) + ", " + dimension + ")";
044    }
045
046}
