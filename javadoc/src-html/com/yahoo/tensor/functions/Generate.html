001package com.yahoo.tensor.functions;
002
003import com.google.common.annotations.Beta;
004import com.yahoo.tensor.DimensionSizes;
005import com.yahoo.tensor.IndexedTensor;
006import com.yahoo.tensor.Tensor;
007import com.yahoo.tensor.TensorType;
008import com.yahoo.tensor.evaluation.EvaluationContext;
009
010import java.util.Collections;
011import java.util.List;
012import java.util.Objects;
013import java.util.function.Function;
014
015/**
016 * An indexed tensor whose values are generated by a function
017 * 
018 * @author bratseth
019 */
020@Beta
021public class Generate extends PrimitiveTensorFunction {
022
023    private final TensorType type;
024    private final Function<List<Integer>, Double> generator;
025
026    /**
027     * Creates a generated tensor
028     * 
029     * @param type the type of the tensor
030     * @param generator the function generating values from a list of ints specifying the indexes of the
031     *                  tensor cell which will receive the value
032     * @throws IllegalArgumentException if any of the tensor dimensions are not indexed bound
033     */
034    public Generate(TensorType type, Function<List<Integer>, Double> generator) {
035        Objects.requireNonNull(type, "The argument tensor type cannot be null");
036        Objects.requireNonNull(generator, "The argument function cannot be null");
037        validateType(type);
038        this.type = type;
039        this.generator = generator;
040    }
041    
042    private void validateType(TensorType type) {
043        for (TensorType.Dimension dimension : type.dimensions())
044            if (dimension.type() != TensorType.Dimension.Type.indexedBound)
045                throw new IllegalArgumentException("A generated tensor can only have indexed bound dimensions");
046    }
047
048    @Override
049    public List<TensorFunction> functionArguments() { return Collections.emptyList(); }
050
051    @Override
052    public TensorFunction replaceArguments(List<TensorFunction> arguments) {
053        if ( arguments.size() != 1)
054            throw new IllegalArgumentException("Generate must have 0 arguments, got " + arguments.size());
055        return this;
056    }
057
058    @Override
059    public PrimitiveTensorFunction toPrimitive() { return this; }
060    
061    @Override
062    public Tensor evaluate(EvaluationContext context) {
063        Tensor.Builder builder = Tensor.Builder.of(type);
064        IndexedTensor.Indexes indexes = IndexedTensor.Indexes.of(dimensionSizes(type));
065        for (int i = 0; i < indexes.size(); i++) {
066            indexes.next();
067            builder.cell(generator.apply(indexes.toList()), indexes.indexesForReading());
068        }
069        return builder.build();
070    }
071    
072    private DimensionSizes dimensionSizes(TensorType type) {
073        DimensionSizes.Builder b = new DimensionSizes.Builder(type.dimensions().size());
074        for (int i = 0; i < b.dimensions(); i++)
075            b.set(i, type.dimensions().get(i).size().get());
076        return b.build();
077    }
078    
079    @Override
080    public String toString(ToStringContext context) { return type + "(" + generator + ")"; }
081
082}
