033/**
034 * A multidimensional array which can be used in computations.
035 * <p>
036 * A tensor consists of a set of <i>dimension</i> names and a set of <i>cells</i> containing scalar <i>values</i>.
037 * Each cell is is identified by its <i>address</i>, which consists of a set of dimension-label pairs which defines
038 * the location of that cell. Both dimensions and labels are string on the form of an identifier or integer.
039 * <p>
040 * The size of the set of dimensions of a tensor is called its <i>order</i>.
041 * <p>
042 * In contrast to regular mathematical formulations of tensors, this definition of a tensor allows <i>sparseness</i>
043 * as there is no built-in notion of a contiguous space, and even in cases where a space is implied (such as when
044 * address labels are integers), there is no requirement that every implied cell has a defined value.
045 * Undefined values have no define representation as they are never observed.
046 * <p>
047 * Tensors can be read and serialized to and from a string form documented in the {@link #toString} method.
048 *
049 * @author bratseth
050 */
051@Beta
052public interface Tensor {

085        if (type().dimensions().size() > 0)
086            throw new IllegalStateException("This tensor is not dimensionless. Dimensions: " + type().dimensions().size());
087        if (size() == 0) return Double.NaN;
088        return valueIterator().next();
089    }

186    /**
187     * Returns this tensor on the form
188     * <code>{address1:value1,address2:value2,...}</code>
189     * where each address is on the form <code>{dimension1:label1,dimension2:label2,...}</code>,
190     * and values are numbers.
191     * <p>
192     * Cells are listed in the natural order of tensor addresses: Increasing size primarily
193     * and by element lexical order secondarily.
194     * <p>
195     * Note that while this is suggestive of JSON, it is not JSON.
196     */
197    @Override
198    String toString();
199
200    /**
201     * Call this from toString in implementations to return the standard string format.
202     * (toString cannot be a default method because default methods cannot override super methods).
203     *
204     * @param tensor the tensor to return the standard string format of
205     * @return the tensor on the standard string format
206     */
207    static String toStandardString(Tensor tensor) {
208        if (tensor.isEmpty() && ! tensor.type().dimensions().isEmpty()) // explicitly output type TODO: Never do that?
209            return tensor.type() + ":" + contentToString(tensor);
210        else
211            return contentToString(tensor);
212    }
213
214    static String contentToString(Tensor tensor) {
215        List<java.util.Map.Entry<TensorAddress, Double>> cellEntries = new ArrayList<>(tensor.cells().entrySet());
216        if (tensor.type().dimensions().isEmpty()) {
217            if (cellEntries.isEmpty()) return "{}";
218            return "{" + cellEntries.get(0).getValue() +"}";
219        }
220        
221        Collections.sort(cellEntries, java.util.Map.Entry.<TensorAddress, Double>comparingByKey());
222
223        StringBuilder b = new StringBuilder("{");
224        for (java.util.Map.Entry<TensorAddress, Double> cell : cellEntries) {
225            b.append(cell.getKey().toString(tensor.type())).append(":").append(cell.getValue());
226            b.append(",");
227        }
228        if (b.length() > 1)
229            b.setLength(b.length() - 1);
230        b.append("}");
231        return b.toString();
232    }

236    /**
237     * Returns whether this tensor and the given tensor is mathematically equal:
238     * That they have the same dimension *names* and the same content.
239     */
240    boolean equals(Object o);
241
242    /** 
243     * Implement here to make this work across implementations.
244     * Implementations must override equals and call this because this is an interface and cannot override equals.
245     */
246    static boolean equals(Tensor a, Tensor b) {
247        if (a == b) return true;
248        if ( ! a.type().mathematicallyEquals(b.type())) return false;
249        if ( a.size() != b.size()) return false;
250        for (Iterator<Cell> aIterator = a.cellIterator(); aIterator.hasNext(); ) {
251            Cell aCell = aIterator.next();
252            if ( ! aCell.getValue().equals(b.get(aCell.getKey()))) return false;
253        }
254        return true;
255    }

288    class Cell implements Map.Entry<TensorAddress, Double> {
289
290        private final TensorAddress address;
291        private final Double value;
292
293        Cell(TensorAddress address, Double value) {
294            this.address = address;
295            this.value = value;
296        }
297
298        @Override
299        public TensorAddress getKey() { return address; }
300
301        /** 
302         * Returns the direct index which can be used to locate this cell, or -1 if not available.
303         * This is for optimizations mapping between tensors where this is possible without creating a
304         * TensorAddress.
305         */
306        int getDirectIndex() { return -1; }
307        
308        @Override
309        public Double getValue() { return value; }
310
311        @Override
312        public Double setValue(Double value) {
313            throw new UnsupportedOperationException("A tensor cannot be modified");
314        }
315
316        @Override
317        public boolean equals(Object o) {
318            if (o == this) return true;
319            if ( ! ( o instanceof Map.Entry)) return false;
320            Map.Entry<?,?> other = (Map.Entry)o;
321            if ( ! this.getValue().equals(other.getValue())) return false;
322            if ( ! this.getKey().equals(other.getKey())) return false;
323            return true;
324        }
325
326        @Override
327        public int hashCode() {
328            return getKey().hashCode() ^ getValue().hashCode(); // by Map.Entry spec
329        }
330
331    }

333    interface Builder {
334
335        /** Creates a suitable builder for the given type */
336        static Builder of(TensorType type) {
337            boolean containsIndexed = type.dimensions().stream().anyMatch(d -> d.isIndexed());
338            boolean containsMapped = type.dimensions().stream().anyMatch( d ->  ! d.isIndexed());
339            if (containsIndexed && containsMapped)
340                throw new IllegalArgumentException("Combining indexed and mapped dimensions is not supported yet");
341            if (containsMapped)
342                return MappedTensor.Builder.of(type);
343            else // indexed or empty
344                return IndexedTensor.Builder.of(type);
345        }
346
347        /** Creates a suitable builder for the given type */
348        static Builder of(TensorType type, DimensionSizes dimensionSizes) {
349            boolean containsIndexed = type.dimensions().stream().anyMatch(d -> d.isIndexed());
350            boolean containsMapped = type.dimensions().stream().anyMatch( d ->  ! d.isIndexed());
351            if (containsIndexed && containsMapped)
352                throw new IllegalArgumentException("Combining indexed and mapped dimensions is not supported yet");
353            if (containsMapped)
354                return MappedTensor.Builder.of(type);
355            else // indexed or empty
356                return IndexedTensor.Builder.of(type, dimensionSizes);
357        }
358
359        /** Returns the type this is building */
360        TensorType type();
361        
362        /** Return a cell builder */
363        CellBuilder cell();
364
365        /** Add a cell */
366        Builder cell(TensorAddress address, double value);
367        
368        /** Add a cell */
369        Builder cell(double value, int ... labels);
370
371        /** 
372         * Add a cell 
373         * 
374         * @param cell a cell providing the location at which to add this cell 
375         * @param value the value to assign to the cell
376         */
377        default Builder cell(Cell cell, double value) {
378            return cell(cell.getKey(), value);
379        }

381        Tensor build();

383        class CellBuilder {
384
385            private final TensorAddress.Builder addressBuilder;
386            private final Tensor.Builder tensorBuilder;
387            
388            CellBuilder(TensorType type, Tensor.Builder tensorBuilder) {
389                addressBuilder = new TensorAddress.Builder(type);
390                this.tensorBuilder = tensorBuilder;
391            }
392
393            public CellBuilder label(String dimension, String label) {
394                addressBuilder.add(dimension, label);
395                return this;
396            }
397
398            public CellBuilder label(String dimension, int label) {
399                return label(dimension, String.valueOf(label));
400            }
401
402            public Builder value(double cellValue) {
403                return tensorBuilder.cell(addressBuilder.build(), cellValue);
404            }
405
406        }
