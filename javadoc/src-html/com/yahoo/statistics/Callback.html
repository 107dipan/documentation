002package com.yahoo.statistics;
003
004/**
005 * Definition of the callback interface for the statistics API. It is a common
006 * use case to need a reference to the Handle (e.g. Value or Counter) which a
007 * callback is related to. Since everything in a Handle since 5.1.4 is fully
008 * initialized from the constructor, it became cumbersome to use Runnable for
009 * the callback and this interface came into use.
010 *
011 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
012 * @since 5.1.4
013 */
014public interface Callback {
015    /**
016     * Invoked each logging cycle right before the events for a Handle are
017     * emitted to the log.
018     *
019     * @param h
020     *            the handle which invoked this callback
021     * @param firstTime
022     *            true the first time the method is invoked from h, false later
023     */
024    public void run(Handle h, boolean firstTime);
025}
