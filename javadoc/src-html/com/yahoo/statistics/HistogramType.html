004
005/**
006 * Enumeration of how a histogram should be represented from
007 * admin server.
008 *
009 * @author  <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
010 */
011public final class HistogramType {
012    private final String representation;
013
014    private HistogramType(String representation) {
015        this.representation = representation;
016    }
017
018    public String toString() { return representation; }
019
020    /**
021     * Basic histograms, each bucket is count representing the bucket's
022     * defined interval.
023     */
024    public static final HistogramType REGULAR = new HistogramType("REGULAR");
025
026    /**
027     * Cumulative histograms, that is, a given bucket contains the count
028     * for values corresponding to "itself" and all preceding buckets.
029     */
030    public static final HistogramType CUMULATIVE =
031        new HistogramType("CUMULATIVE");
032
033
034    /**
035     * Reverse cumulative histograms, that is, a given bucket contains
036     * the count for values corresponding to "itself" and all following
037     * buckets.
038     */
039    public static final HistogramType REVERSE_CUMULATIVE =
040        new HistogramType("REVERSE_CUMULATIVE");
041
042}
