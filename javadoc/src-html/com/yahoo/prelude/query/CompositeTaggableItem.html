002package com.yahoo.prelude.query;
003
004/**
005 * Common implementation for Item classes implementing the TaggableItem interface.
006 * Note that this file exist in 3 copies that should be kept in sync:
007 *
008 * CompositeTaggableItem.java
009 * SimpleTaggableItem.java
010 * TaggableSegmentItem.java
011 *
012 * These should only have trivial differences.
013 * (multiple inheritance or mixins would have been nice).
014 *
015 * @author arnej27959
016 */
017public abstract class CompositeTaggableItem extends CompositeItem implements TaggableItem {
018
019    public int getUniqueID() {
020        return uniqueID;
021    }
022
023    public void setUniqueID(int id) {
024        setHasUniqueID(true);
025        uniqueID = id;
026    }
027
028    /** See {@link TaggableItem#setConnectivity} */
029    public void setConnectivity(Item item, double connectivity) {
030        setHasUniqueID(true);
031        item.setHasUniqueID(true);
032        if (connectedItem != null) {
033            // untangle old connectivity
034            connectedItem.connectedBacklink = null;
035        }
036        this.connectivity = connectivity;
037        connectedItem = item;
038        connectedItem.connectedBacklink = this;
039    }
040
041    public Item getConnectedItem() {
042        return connectedItem;
043    }
044
045    public double getConnectivity() {
046        return connectivity;
047    }
048
049    public void setSignificance(double significance) {
050        setHasUniqueID(true);
051        setExplicitSignificance(true);
052        this.significance = significance;
053    }
054
055    public void setExplicitSignificance(boolean explicitSignificance) {
056        this.explicitSignificance = explicitSignificance;
057    }
058
059    public boolean hasExplicitSignificance() {
060        return explicitSignificance;
061    }
062
063    public double getSignificance() {
064        return significance;
065    }
066
067    //Change access privilege from protected to public.
068    public boolean hasUniqueID() {
069        return super.hasUniqueID();
070    }
071}
