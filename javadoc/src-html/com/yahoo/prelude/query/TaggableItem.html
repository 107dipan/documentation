002package com.yahoo.prelude.query;
003
004
005/**
006 * An interface used for anything which may be addressed using an external,
007 * unique ID in the query tree in the backend.
008 *
009 * @author  <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
010 */
011public interface TaggableItem {
012
013    public int getUniqueID();
014    public void setUniqueID(int id);
015    public boolean hasUniqueID();
016
017    /**
018     * Set the connectivity to another term in the same query tree.
019     * This is used to influence ranking features taking proximity into account: nativeRank and a subset of the
020     * fieldMatch features.
021     * <p>
022     * By default consecutive query terms are 'somewhat' connected, meaning ranking features will be better in documents
023     * where the terms are found close to each other. This effect can be increased or decreased by manipulating the
024     * connectivity value. Typical use is to increase the connectivity between terms in the query that we believe are
025     * semantically connected. E.g in the query 'new york hotel', it is a good idea to increase the connectivity between
026     * "new" and "york" to ensure that a document containing "List of hotels in New York" is ranked above one containing
027     * "List of new hotels in York".
028     *
029     * @param item the item this should be connected to - in practice the next consecutive item in the query
030     * @param connectivity a value between 0 (none) and 1 (maximal), defining the connectivity between this and the
031     *        argument item. The default connectivity is 0.1.
032     */
033    public void setConnectivity(Item item, double connectivity);
034    public Item getConnectedItem();
035    public double getConnectivity();
036
037
038    /**
039     * Used for setting explicit term significance (in the tf/idf sense) to a single term or phrase,
040     * relative to the rest of the query.
041     * This influences ranking features which take term significance into account and overrides the default
042     * partial corpus based term significance computation happening in the backend.
043     */
044    public void setSignificance(double significance);
045    public boolean hasExplicitSignificance();
046    public void setExplicitSignificance(boolean significance);
047    public double getSignificance();
048}
