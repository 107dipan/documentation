002package com.yahoo.prelude.query;
003
004
005/**
006 * An interface used for anything which represents a single block
007 * of query input.
008 *
009 * @author  <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
010 */
011public interface BlockItem extends HasIndexItem {
012
013    /**
014     * The untransformed raw text from the user serving as base for
015     * this item.
016     */
017    String getRawWord();
018
019    /** Returns the substring which is the origin of this item, or null if none */
020    public Substring getOrigin();
021
022    /** Returns the value of this term as a string */
023    public abstract String stringValue();
024
025    /**
026     * Is this block of text conceptually from the user query?
027     */
028    boolean isFromQuery();
029
030    boolean isStemmed();
031
032    /**
033     * Does this item represent "usual words"?
034     */
035    boolean isWords();
036
037    /**
038     * If the block has to be resegmented, what operator should be chosen if it
039     * is necessary to change operator?
040     */
041    SegmentingRule getSegmentingRule();
042}
