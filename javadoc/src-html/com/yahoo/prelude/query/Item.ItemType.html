017/**
018 * <p>A term of the query language. As "term" is also the common term (sorry)
019 * for a literal to be found (or not) in a search index, the term <i>item</i>
020 * is used for <i>query language</i> terms.</p>
021 *
022 * <p>The query is represented as a composite tree of
023 * Item subclasses. This allow arbitrary complex combinations of ands,
024 * nots, phrases and so on.</p>
025 *
026 * <p>Items are in general mutable and not thread safe.</p>
027 *
028 * @author bratseth
029 * @author havardpe
030 */
031public abstract class Item implements Cloneable {

234    public abstract int encode(ByteBuffer buffer);
235
236    protected void encodeThis(ByteBuffer buffer) {
237        int FEAT_SHIFT = 5;
238        int CODE_MASK = 0x1f;
239        int FEAT_MASK = 0xe0;
240        int FEAT_WEIGHT = 0x01;
241        int FEAT_UNIQUEID = 0x02;
242        int FEAT_FLAGS = 0x04;
243
244        int features = 0;
245
246        if (weight != DEFAULT_WEIGHT) {
247            features |= FEAT_WEIGHT;
248        }
249        if (hasUniqueID()) {
250            features |= FEAT_UNIQUEID;
251        }
252        byte flags = getFlagsFeature();
253        if (flags != 0) {
254            features |= FEAT_FLAGS;
255        }
256        byte type = (byte)(((getCode() & CODE_MASK)
257                       | ((features << FEAT_SHIFT) & FEAT_MASK)) & 0xff);
258
259        buffer.put(type);
260        if ((features & FEAT_WEIGHT) != 0) {
261            IntegerCompressor.putCompressedNumber(weight, buffer);
262        }
263        if ((features & FEAT_UNIQUEID) != 0) {
264            IntegerCompressor.putCompressedPositiveNumber(uniqueID, buffer);
265        }
266        if (flags != 0) {
267            buffer.put(flags);
268        }
269    }
270
271    /**
272     * Returns an integer that contains all feature flags for this item. This must be kept in sync with the flags
273     * defined in searchlib/parsequery/parse.h.
274     *
275     * @return The feature flags.
276     */
277    private byte getFlagsFeature() {
278        byte FLAGS_NORANK = 0x01;
279        byte FLAGS_SPECIALTOKEN = 0x02;
280        byte FLAGS_NOPOSITIONDATA = 0x04;
281        byte FLAGS_ISFILTER = 0x08;
282
283        byte ret = 0;
284        if (!isRanked()) {
285            ret |= FLAGS_NORANK;
286        }
287        if (isFromSpecialToken()) {
288            ret |= FLAGS_SPECIALTOKEN;
289        }
290        if (!usePositionData()) {
291            ret |= FLAGS_NOPOSITIONDATA;
292        }
293        if (isFilter()) {
294            ret |= FLAGS_ISFILTER;
295        }
296        return ret;
297    }

314    /**
315     * <p>Returns the canonical query language string of this item.</p>
316     *
317     * <p>The canonical language represent an item by the string
318     * <pre>
319     * ([itemName] [body])
320     * </pre>
321     * where the body may recursively be other items.
322     *
323     * <p>
324     * TODO: Change the output query language into a canonical form of the input
325     *       query language
326     */
327    public String toString() {
328        StringBuilder buffer = new StringBuilder();
329
330        if (shouldParenthize()) {
331            buffer.append("(");
332        }
333        if (isFilter()) {
334            buffer.append("|");
335        }
336        appendHeadingString(buffer);
337        appendBodyString(buffer);
338        if (shouldParenthize()) {
339            buffer.append(")");
340        }
341
342        if (weight != DEFAULT_WEIGHT) {
343            buffer.append("!");
344            buffer.append(weight);
345        }
346
347        return buffer.toString();
348    }

374    /** Returns a deep copy of this item */
375    public Item clone() {
376        try {
377            Item clone = (Item)super.clone();
378            if (this.annotations != null)
379                clone.annotations = this.annotations.clone();
380            // note: connectedItem and connectedBacklink references are corrected in CompositeItem.clone()
381            return clone;
382        } catch (CloneNotSupportedException e) {
383            throw new RuntimeException("Someone made Item unclonable");
384        }
385    }

387    /**
388     * Returns whether this item is of the same class and
389     * contains the same state as the given item
390     */
391    public boolean equals(Object object) {
392        if (object == null) {
393            return false;
394        }
395        if (object.getClass() != this.getClass()) {
396            return false;
397        } // Fails on different c.l.'s
398
399        Item other = (Item) object;
400
401        if (this.creator != other.creator) {
402            return false;
403        }
404        if (this.weight != other.weight) {
405            return false;
406        }
407        // if (this.termIndex!=other.termIndex) return false;
408
409        return true;
410    }

480        discloser.addProperty("usePositionData", usePositionData);
481        discloser.addProperty("significance", significance);
