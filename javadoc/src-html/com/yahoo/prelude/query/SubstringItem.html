002package com.yahoo.prelude.query;

004
005/**
006 * A word that matches substrings of words
007 *
008 * @author banino
009 */
010public class SubstringItem extends WordItem {
011
012    public SubstringItem(String substring) {
013        this(substring, false);
014    }
015
016    public SubstringItem(String substring, boolean isFromQuery) {
017        super(substring, isFromQuery);
018    }
019
020    public ItemType getItemType() {
021        return ItemType.SUBSTRING;
022    }
023
024    public String getName() {
025        return "SUBSTRING";
026    }
027
028    public String stringValue() {
029        return "*" + getWord() + "*";
030    }
031
032}
