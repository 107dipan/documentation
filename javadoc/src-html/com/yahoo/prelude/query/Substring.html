005/**
006 * An substring which also provides access to the full (query) string it is a substring of.
007 * This is immutable.
008 *
009 * @author  bratseth
010 */
011public class Substring {
012
013    /** The start of the substring */
014    public final int start;
015
016    /** The end of the substring */
017    public final int end;
018
019    /** The string this is a substring of */
020    public final String string;
021
022    public Substring(int start, int end,String string) {
023        this.start = start;
024        this.end = end;
025        this.string=string;
026    }
027
028    public String getValue() {
029        return string.substring(start,end);
030    }
031
032    /** Returns the entire string this is a substring of. The start and end offsets are into this string. */
033    public String getSuperstring() { return string; }
034
035    /**
036     * Returns the character n places (0 base) after the end of the value substring into the superstring.
037     * For example charAfter(0) returns the first character after the end of the substring
038     *
039     * @return the char n planes after the end of the substring
040     * @throws IndexOutOfBoundsException if the string is not long enough to have a character at this position
041     */
042    public char charAfter(int n) {
043        return string.charAt(end+n);
044    }
045
046    /**
047     * Returns the character n places (0 base) before the start of the value substring into the superstring.
048     * For example charBefore(0) returns the first character before the start of the substring
049     *
050     * @return the char n planes before the start of the substring
051     * @throws IndexOutOfBoundsException if the string does not have a character at this position
052     */
053    public char charBefore(int n) {
054        return string.charAt(start-1-n);
055    }
056
057    @Override
058    public String toString() {
059        return "(" + start + ' ' + end + ')';
060    }
061
062}
