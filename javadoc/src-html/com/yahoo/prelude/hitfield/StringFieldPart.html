002package com.yahoo.prelude.hitfield;
003
004/**
005 * Represents an element of a hit property which is a possibly
006 * mutable string element
007 *
008 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
009 */
010public class StringFieldPart implements FieldPart {
011    private String content;
012    private final String initContent;
013    // Whether this element represents a (part of) a token or a
014    // delimiter string. When splitting existing parts, the new
015    // parts should inherit this state from the object they were
016    // split from.
017    private boolean tokenOrDelimiter;
018    public StringFieldPart(String content, boolean tokenOrDelimiter) {
019        this.content = content;
020        initContent = content;
021        this.tokenOrDelimiter = tokenOrDelimiter;
022    }
023    public boolean isFinal() { return false; }
024    public boolean isToken() { return tokenOrDelimiter; }
025    public String getContent() { return content; }
026    public void setContent(String content) {
027        this.content = content;
028    }
029    public String getInitContent() { return initContent; }
030    public String toString() { return content; }
031}
