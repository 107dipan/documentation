002package com.yahoo.prelude.templates;
003
004import java.util.Collection;
005import java.util.HashMap;
006import java.util.LinkedHashMap;
007import java.util.Map;
008
009/** A context having a map as secondary storage */
010public class MapContext extends Context {
011
012    private Map<String, Object> map = new LinkedHashMap<>();
013
014    @Override
015    public Object get(String key) {
016        return normalizeValue(map.get(key));
017    }
018
019    public Object put(String name, Object value) {
020        return map.put(name, value);
021    }
022
023    public Object remove(Object name) {
024        return map.remove(name);
025    }
026
027    @Override
028    public Collection<? extends Object> getKeys() {
029        return map.keySet();
030    }
031
032}
