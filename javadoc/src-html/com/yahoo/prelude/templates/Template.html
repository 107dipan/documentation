002package com.yahoo.prelude.templates;
003
004import java.io.Writer;
005
006
007/**
008 * A template turns a template string and some state into
009 * an instantiated string. Add support for a particular
010 * template mechanism by subclassing this.
011 *
012 * @author bratseth
013 */
014public abstract class Template<T extends Writer> {
015
016    /**
017     * Renders this template
018     *
019     * @param context the context to evaluate in
020     * @param writer the writer to render to
021     */
022    public abstract void render(Context context,T writer)
023        throws java.io.IOException;
024
025
026    /**
027     * Get template name
028     *
029     * @return template name
030     */
031    public abstract String getName();
032
033}
