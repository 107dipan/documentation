002package com.yahoo.language.detect;
003
004import java.nio.ByteBuffer;
005
006/**
007 * Abstract superclass of all Detectors used for language and encoding detection.
008 *
009 * @author Einar M R Rosenvinge
010 */
011public interface Detector {
012
013    /**
014     * Detects language and encoding of the supplied byte array, possibly using a language/encoding hint.
015     *
016     * @param input  the buffer that is to be inspected
017     * @param offset the offset to detect from
018     * @param length the size to detect from
019     * @param hint   a hint to the detector, or null for no hint
020     * @return an array of possible language/encoding pairs, sorted by decreasing confidence (possibly empty, but never null)
021     * @throws DetectionException if detection fails
022     */
023    Detection detect(byte[] input, int offset, int length, Hint hint);
024
025    /**
026     * Detects language and encoding of the supplied ByteBuffer, possibly using a language/encoding hint.
027     *
028     * @param input the buffer that is to be inspected, from its current position to its limit
029     * @param hint  a hint to the detector, or null for no hint
030     * @return an array of possible language/encoding pairs, sorted by decreasing confidence (possibly empty, but never null)
031     * @throws DetectionException if detection fails
032     */
033    Detection detect(ByteBuffer input, Hint hint);
034
035    /**
036     * Detects language of the supplied String, possibly using a language hint.
037     *
038     * @param input the string that is to be inspected
039     * @param hint  a hint to the detector, or null for no hint
040     * @return an array of possible language/encoding pairs, sorted by decreasing confidence (possibly empty, but never null)
041     * @throws DetectionException if detection fails
042     */
043    Detection detect(String input, Hint hint);
044
045}
