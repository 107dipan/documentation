002package com.yahoo.language.detect;
003
004import com.yahoo.language.Language;
005
006import java.nio.charset.Charset;
007import java.nio.charset.UnsupportedCharsetException;
008
009/**
010 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
011 */
012public class Detection {
013
014    private final Language language;
015    private final String encodingName;
016    private final boolean local;
017
018    public Detection(Language language, String encodingName, boolean local) {
019        this.language = language;
020        this.encodingName = encodingName;
021        this.local = local;
022    }
023
024    public Language getLanguage() {
025        return language;
026    }
027
028    public Charset getEncoding() {
029        if (encodingName == null) {
030            return null;
031        }
032        try {
033            return Charset.forName(encodingName);
034        } catch (UnsupportedCharsetException e) {
035            // ignore
036        }
037        return null;
038    }
039
040    public String getEncodingName() {
041        return encodingName;
042    }
043
044    public boolean isLocal() {
045        return local;
046    }
047}
