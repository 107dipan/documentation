002package com.yahoo.language.process;
003
004/**
005 * An enumeration of token types.
006 *
007 * @author <a href="mailto:mathiasm@yahoo-inc.com">Mathias MÃ¸lster Lidal</a>
008 */
009public enum TokenType {
010
011    UNKNOWN(0),
012    SPACE(1),
013    PUNCTUATION(2),
014    SYMBOL(3),
015    ALPHABETIC(4),
016    NUMERIC(5),
017    MARKER(255);
018
019    private final int value;
020
021    TokenType(int value) {
022        this.value = value;
023    }
024
025    /** Returns an int code for this type */
026    public int getValue() { return value; }
027
028    /**
029     * Marker for whether this type of token can be indexed for search.
030     * Note that a Token can be excluded from an index, even though the token type marks
031     * it as indexable.
032     *
033     * @see com.yahoo.language.process.Token#isIndexable()
034     * @return whether this type of token can be indexed
035     */
036    public boolean isIndexable() {
037        switch (this) {
038            case ALPHABETIC: case NUMERIC: return true;
039            default: return false;
040        }
041    }
042
043    /** Translates this from the int code representation returned from {@link #getValue} */
044    public static TokenType valueOf(int value) {
045        for (TokenType type : values()) {
046            if (value == type.value) return type;
047        }
048        return UNKNOWN;
049    }
050
051}
