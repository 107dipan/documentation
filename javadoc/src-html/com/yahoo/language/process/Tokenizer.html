002package com.yahoo.language.process;
003
004import com.yahoo.language.Language;
005
006/**
007 * Language-sensitive tokenization of a text string.
008 *
009 * @author <a href="mailto:mathiasm@yahoo-inc.com">Mathias MÃ¸lster Lidal</a>
010 */
011public interface Tokenizer {
012
013    /**
014     * Returns the tokens produced from an input string under the rules of the given Language and additional options
015     *
016     * @param input the string to tokenize. May be arbitrarily large.
017     * @param language the language of the input string.
018     * @param stemMode the stem mode applied on the returned tokens
019     * @param removeAccents if true accents and similar are removed from the returned tokens
020     * @return the tokens of the input String.
021     * @throws ProcessingException If the underlying library throws an Exception.
022     */
023    Iterable<Token> tokenize(String input, Language language, StemMode stemMode, boolean removeAccents);
024
025    /**
026     * Return a replacement for an input token string.
027     * This accepts strings returned by Token.getTokenString
028     * and returns a replacement which will be used as the index token.
029     * The input token string is returned if there is no replacement.
030     * <p>
031     * This default implementation always returns the input token string.
032     *
033     * @param tokenString the token string of the term to lookup a replacement for
034     * @return the replacement, if any, or the argument token string if not
035     */
036    default String getReplacementTerm(String tokenString) { return tokenString; }
037
038}
