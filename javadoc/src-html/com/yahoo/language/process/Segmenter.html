002package com.yahoo.language.process;
003
004import com.yahoo.language.Language;
005
006import java.util.List;
007
008/**
009 * <p>Interface providing segmentation, i.e. splitting of CJK character blocks into separate tokens. This is primarily a
010 * convenience feature for users who don't need full tokenization (or who use a separate tokenizer and only need CJK
011 * processing).</p>
012 *
013 * @author <a href="mailto:mathiasm@yahoo-inc.com">Mathias MÃ¸lster Lidal</a>
014 */
015public interface Segmenter {
016
017    /**
018     * Split input-string into tokens, and returned a list of tokens in unprocessed form (i.e. lowercased, normalized
019     * and stemmed if applicable, see @link{StemMode} for list of stemming options). It is assumed that the input only
020     * contains word-characters, any punctuation and spacing tokens will be removed.
021     *
022     * @param input the text to segment.
023     * @param language language of input text.
024     * @return the list of segments.
025     * @throws ProcessingException if an exception is encountered during processing
026     */
027    List<String> segment(String input, Language language);
028
029}
