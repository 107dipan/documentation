002package com.yahoo.language;
003
004import java.util.Locale;
005
006/**
007 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
008 */
009public final class LocaleFactory {
010
011    private static final Locale UNKNOWN = new Locale("", "", "");
012
013    private LocaleFactory() {
014        // hide
015    }
016
017    /**
018     * <p>Implements a simple parser for RFC5646 language tags. The language tag is parsed into a Locale.</p>
019     *
020     * @param tag The language tag to parse.
021     * @return The corrseponding Locale.
022     */
023    @SuppressWarnings("ConstantConditions")
024    public static Locale fromLanguageTag(String tag) {
025        // TODO: Should be replaced by return Locale.forLanguageTag(tag); ?
026
027        tag.getClass(); // throws NullPointerException
028        tag = tag.trim();
029        if (tag.isEmpty()) {
030            return UNKNOWN;
031        }
032        String language = "";
033        String region = "";
034        String script = "";
035        String[] parts = tag.split("-");
036        for (int partIdx = 0; partIdx < parts.length; ++partIdx) {
037            String part = parts[partIdx];
038            int partLen = part.length();
039            if (partIdx == 0) {
040                if (partLen == 2 || partLen == 3) {
041                    language = part;
042                }
043            } else if (partIdx == 1 || partIdx == 2) {
044                if (partLen == 2 || partLen == 3) {
045                    region = part;
046                } else if (partLen == 4) {
047                    script = part;
048                }
049            }
050        }
051        if (language.isEmpty()) {
052            return UNKNOWN;
053        }
054        return new Locale(language, region, script);
055    }
056
057}
