099    /**
100     * Creates an URI, optionaly keeping the fragment (the part starting by #).
101     * If the uri is hierarchical, it is normalized and incorrect hierarchical uris
102     * which looks like urls are attempted repaired.
103     *
104     * <p>Relative uris are not supported.</p>
105     *
106     * @param uriString the uri string
107     * @param keepFragment true to keep the fragment
108     * @param hierarchicalOnly will force any uri string given to be parsed as
109     *        a hierarchical one, causing the uri to be invalid if it isn't
110     * @throws NullPointerException if the given uriString is null
111     */
112    public URI(String uriString, boolean keepFragment, boolean hierarchicalOnly) {
113        if (uriString == null) {
114            throw new NullPointerException("Can not create an uri from null");
115        }
116
117        if (!keepFragment) {
118            int fragmentIndex = uriString.indexOf("#");
119
120            if (fragmentIndex >= 0) {
121                uriString = uriString.substring(0, fragmentIndex);
122            }
123        }
124
125        try {
126            this.uri = uriString.trim();
127            opaque = isOpaque(uri);
128
129            // No further parsing of opaque uris
130            if (isOpaque() && !hierarchicalOnly) {
131                return;
132            }
133            opaque = false;
134            normalizeHierarchical();
135        } catch (IllegalArgumentException e) {
136            if (e.getMessage() != null) {
137                invalidExplanation = e.getMessage();
138            } else {
139                Throwable t = e.getCause();
140                if (t != null && t.getMessage() != null) {
141                    invalidExplanation = t.getMessage();
142                } else {
143                    invalidExplanation = "Invalid uri: " + e;
144                }
145            }
146        }
147    }

149    /** Creates an url type uri */
150    public URI(String scheme, String host, int port, String rest) {
151        this.scheme = scheme;
152        this.host = host;
153        this.port = port;
154        this.rest = rest;
155        recombine();
156        normalizeHierarchical();
157        opaque = false;
158    }

160    /** Returns whether an url is opaque or hierarchical */
161    private boolean isOpaque(String uri) {
162        int colonIndex = uri.indexOf(":");
163
164        if (colonIndex < 0) {
165            return true;
166        } else {
167            return !(uri.length() > colonIndex + 1
168                    && uri.charAt(colonIndex + 1) == '/');
169        }
170    }

177        return invalidExplanation == null;

180    /**
181     * Normalizes this hierarchical uri according to FRC 2396 and the Overture
182     * standard. Before normalizing, some simple heuritics are use to make
183     * the uri complete if needed. After normalizing, the scheme,
184     * host, port and rest of this uri is set if defined.
185     *
186     * @throws IllegalArgumentException if this uri can not be normalized into a legal uri
187     */
188    private void normalizeHierarchical() {
189        complete();
190        escapeNonAscii();
191        unescapeHtmlEntities();
192        decompose();
193        lowCaseHost();
194        removeDefaultPortNumber();
195        removeTrailingHostDot();
196        makeDoubleSlashesSingle();
197        recombine();
198    }

200    /** Applies simple heuristics to complete this uri if needed */
201    private void complete() {
202        if (uri.startsWith("www.")) {
203            uri = "http://" + uri;
204        } else if (uri.startsWith("WWW")) {
205            uri = "http://" + uri;
206        } else if (uri.startsWith("/http:")) {
207            uri = uri.substring(1);
208        } else if (isFileURIShortHand(uri)) {
209            uri = "file://" + uri;
210        }
211    }
212
213    private boolean isFileURIShortHand(String uri) {
214        if (uri.indexOf(":\\") == 1) {
215            return true;
216        }
217        if (uri.indexOf("c:/") == 0) {
218            return true;
219        }
220        if (uri.indexOf("d:/") == 0) {
221            return true;
222        }
223        return false;
224    }

226    /**
227     * Decomposes this uri into scheme, host, port and rest.
228     */
229    private void decompose() {
230        java.net.URI neturi = java.net.URI.create(uri).normalize();
231
232        scheme = neturi.getScheme();
233
234        host = neturi.getHost();
235        boolean portAlreadyParsed = false;
236
237        // No host if the host contains underscores
238        if (host == null) {
239            host = neturi.getAuthority();
240            if (host != null) {
241                int colonPos = host.lastIndexOf(":");
242                if (!scheme.equals("file") && colonPos > -1) {
243                    //we probably have an (illegal) URI of type http://under_score.com:5000/
244                    try {
245                        port = Integer.parseInt(host.substring(colonPos + 1, host.length()));
246                        host = host.substring(0, colonPos);
247                        portAlreadyParsed = true;
248                    } catch (NumberFormatException nfe) {
249                        //empty
250                    }
251                }
252            }
253        }
254
255        if ("file".equalsIgnoreCase(scheme)) {
256            if (host == null) {
257                host = "localhost";
258            } else {
259                host = repairWindowsDrive(host, uri);
260            }
261        }
262        if (host == null) {
263            throw new IllegalArgumentException(
264                    "A complete uri must specify a host");
265        }
266        if (!portAlreadyParsed) {
267            port = neturi.getPort();
268        }
269        rest = (neturi.getRawPath() != null ? neturi.getRawPath() : "")
270                + (neturi.getRawQuery() != null
271                        ? ("?" + neturi.getRawQuery())
272                        : "")
273                        + (neturi.getRawFragment() != null
274                                ? ("#" + neturi.getRawFragment())
275                                : "");
276    }

278    /** c: turns to c when interpreted by URI. Repair it */
279    private String repairWindowsDrive(String host, String uri) {
280        if (host.length() != 1) {
281            return host;
282        }
283        int driveIndex = uri.indexOf(host + ":");
284
285        if (driveIndex == 5 || driveIndex == 7) { // file:<drive> or file://<drive>
286            return host + ":";
287        } else {
288            return host;
289        }
290    }

292    /** "http://a/\u00E6" → "http://a/%E6;" */
293    private void escapeNonAscii() {
294        char[] uriChars = uri.toCharArray();
295        StringBuilder result = new StringBuilder(uri.length());
296
297        for (char uriChar : uriChars) {
298            if (uriChar >= 0x80 || uriChar == 0x22) {
299                result.append("%");
300                result.append(Integer.toHexString(uriChar));
301                result.append(";");
302            } else {
303                result.append(uriChar);
304            }
305        }
306        uri = result.toString();
307    }
308
309    /** "http://a/&amp;amp;" → "http://a/&amp;" Currently ampersand only */
310    private void unescapeHtmlEntities() {
311        int ampIndex = uri.indexOf("&amp;");
312
313        if (ampIndex < 0) {
314            return;
315        }
316
317        StringBuilder result = new StringBuilder(uri.substring(0, ampIndex));
318
319        while (ampIndex >= 0) {
320            result.append("&");
321            int nextAmpIndex = uri.indexOf("&amp;", ampIndex + 5);
322
323            result.append(
324                    uri.substring(ampIndex + 5,
325                    nextAmpIndex > 0 ? nextAmpIndex : uri.length()));
326            ampIndex = nextAmpIndex;
327        }
328        uri = result.toString();
329    }

336    /** "http://a:80" → "http://a" and "https://a:443" → https//a */
337    private void removeDefaultPortNumber() {
338        if (port == 80 && scheme.equals("http")) {
339            port = -1;
340        } else if (port == 443 && scheme.equals("https")) {
341            port = -1;
342        }
343    }

345    /** "http://a./b" → "http://a/b" */
346    private void removeTrailingHostDot() {
347        if (host.endsWith(".")) {
348            host = host.substring(0, host.length() - 1);
349        }
350    }

352    /** "http://a//b" → "http://a/b" */
353    private void makeDoubleSlashesSingle() {
354        StringBuilder result = new StringBuilder(rest.length());
355        char[] restChars = rest.toCharArray();
356
357        for (int i = 0; i < restChars.length; i++) {
358            if (!(i + 1 < restChars.length && restChars[i] == '/'
359                    && restChars[i + 1] == '/')) {
360                result.append(restChars[i]);
361            }
362        }
363        rest = result.toString();
364    }

366    /** Recombines the uri from the scheme, host, port and rest */
367    private void recombine() {
368        StringBuilder recombined = new StringBuilder(100);
369
370        recombined.append(scheme);
371        recombined.append("://");
372        recombined.append(host);
373        if (port > -1) {
374            recombined.append(":").append(port);
375        }
376        if (rest != null) {
377            if (!rest.startsWith("/")) {
378                recombined.append("/");
379            }
380            recombined.append(rest);
381        } else {
382            recombined.append("/"); // RFC 2396 violation, as required by search
383        }
384        uri = recombined.toString();
385    }

438    public String getDomain() {
439        if (parsedDomain) {
440            return domain;
441        }
442        String host = getHost();
443        if (host == null) return null;
444
445        int firstDotPos = host.indexOf(".");
446        int lastDotPos = host.lastIndexOf(".");
447
448        String domain;
449        if (firstDotPos < 0) {
450            // "." was not found at all
451            domain = host;
452        } else if (firstDotPos == lastDotPos) {
453            //there is only one "." in the host
454            domain = host;
455        } else {
456            //for www.host.com return host.com
457            //TODO: Must be corrected when implementing tldlist
458            domain = host.substring(firstDotPos + 1, host.length());
459        }
460
461        this.parsedDomain = true;
462        this.domain = domain;
463        return domain;
464    }

466    public String getMainTld() {
467        if (parsedMainTld) {
468            return mainTld;
469        }
470        String host = getHost();
471        if (host == null) return null;
472
473        int lastDotPos = host.lastIndexOf(".");
474
475        String mainTld;
476        if (lastDotPos < 0) {
477            //no ".", no TLD
478            mainTld = null;
479        } else if (lastDotPos == host.length() - 1) {
480            //the "." is the last character
481            mainTld = null;
482        } else {
483            //for www.yahoo.co.uk return uk
484            //TODO: Implement list of TLDs from config?
485            mainTld = host.substring(lastDotPos + 1, host.length());
486        }
487        this.parsedMainTld = true;
488        this.mainTld = mainTld;
489        return mainTld;
490    }

492    public String getPath() {
493        if (parsedPath) {
494            return path;
495        }
496        String rest = this.rest;
497        if (rest == null) return null;
498
499        rest = removeFragment(rest);
500
501        int queryPos = rest.lastIndexOf("?");
502        if (queryPos > -1) {
503            rest = rest.substring(0, queryPos);
504        }
505        this.parsedPath = true;
506        this.path = rest;
507        return this.path;
508    }

510    private String removeFragment(String path) {
511        int fragmentPos = path.lastIndexOf("#");
512        return (fragmentPos > -1) ? path.substring(0, fragmentPos) : path;
513    }

515    public String getFilename() {
516        if (parsedFilename) {
517            return filename;
518        }
519        String path = getPath();
520        if (path == null) return null;
521
522        path = removeParams(path);
523
524        int lastSlash = path.lastIndexOf("/");
525
526        String filename;
527        if (lastSlash < 0) {
528            //there is no slash, return the path, excluding params
529            filename = path;
530        } else if (lastSlash == path.length() - 1) {
531            //the slash is the last character, there is no filename here
532            filename = "";
533        } else {
534            filename = path.substring(lastSlash + 1, path.length());
535        }
536        this.parsedFilename = true;
537        this.filename = filename;
538        return filename;
539    }

541    private String removeParams(String filename) {
542        int firstSemicolon = filename.indexOf(";");
543
544        if (firstSemicolon < 0) {
545            //there are no params
546            return filename;
547        }
548        return filename.substring(0, firstSemicolon);
549    }

636    public static String[] tokenize(String item) {
637        return tokenizePattern.split(item);
638    }

640    public List<Token> tokenize() {
641        List<Token> tokens = new ArrayList<>();
642
643        tokens.addAll(tokenize(URLContext.URL_SCHEME, getScheme()));
644        tokens.addAll(tokenize(URLContext.URL_HOST, getHost()));
645        tokens.addAll(tokenize(URLContext.URL_PORT, getPort() > -1 ? "" + getPort() : null));
646        tokens.addAll(tokenize(URLContext.URL_PATH, getPath()));
647        tokens.addAll(tokenize(URLContext.URL_QUERY, getQuery()));
648        tokens.addAll(tokenize(URLContext.URL_FRAGMENT, getFragment()));
649
650        return tokens;
651    }
652
653    private List<Token> tokenize(URLContext context, String item) {
654        if (item == null) {
655            return new ArrayList<>(0);
656        }
657        String[] tokenStrings = tokenize(item);
658        List<Token> tokens = new ArrayList<>(tokenStrings.length);
659        for (String tokenString : tokenStrings) {
660            if (tokenString.length() > 0) {
661                tokens.add(new Token(context, tokenString));
662            }
663        }
664        return tokens;
665    }

741    /**
742     * Returns the depth of this uri.
743     * The depth of an hierarchical uri equals the number of slashes
744     * which are not separating the protocol and the host, and not at the end.
745     *
746     * @return the depth of this uri if it is hierarchical, or 0 if it is opaque
747     */
748    public int getDepth() {
749        int colonIndex = uri.indexOf(':');
750
751        // count number of slashes in the Uri
752        int currentIndex = colonIndex;
753        int depth = 0;
754
755        while (currentIndex != -1) {
756            currentIndex = uri.indexOf('/', currentIndex);
757            if (currentIndex != -1) {
758                depth++;
759                currentIndex++;
760            }
761        }
762
763        if (uri.charAt(colonIndex + 1) == '/') {
764            depth--;
765        }
766        if (uri.charAt(colonIndex + 2) == '/') {
767            depth--;
768        }
769        if ((uri.charAt(uri.length() - 1) == '/')
770                && ((uri.length() - 1) > (colonIndex + 2))) {
771            depth--;
772        }
773        return depth;
774    }
