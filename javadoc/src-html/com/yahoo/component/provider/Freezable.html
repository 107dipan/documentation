002package com.yahoo.component.provider;
003
004/**
005 * A class which may be irreversibly frozen. Any attempt to change the state of this class after
006 * freezing throws an IllegalStateException.
007 *
008 * @author bratseth
009 */
010public interface Freezable {
011
012    /**
013     * Freezes this component to prevent further changes. Override this to freeze internal data
014     * structures and dependent objects. Overrides must call super.
015     * Calling freeze on an already frozen class must have no effect.
016     */
017    public void freeze();
018
019
020    /**
021     * Inspect whether this object can be changed. If the object is immutable
022     * from construction, this should return true, even if freeze() never has
023     * been invoked.
024     *
025     * @return true if this instance is in an immutable state, false otherwise
026     * @since 5.1.4
027     */
028    public boolean isFrozen();
029}
