002package com.yahoo.component.provider;
003
004import com.yahoo.component.AbstractComponent;
005import com.yahoo.component.ComponentId;
006
007/**
008 * Superclass for freezable components
009 *
010 * @author bratseth
011 */
012public class FreezableComponent extends AbstractComponent implements Freezable {
013
014    /** True when this cannot be changed any more */
015    private boolean frozen=false;
016
017    protected FreezableComponent(ComponentId id) {
018        super(id);
019    }
020
021    @SuppressWarnings("unused")
022    protected FreezableComponent() {}
023
024    /**
025     * Freezes this component to prevent further changes. Override this to freeze internal data
026     * structures and dependent objects. Overrides must call super.
027     * Calling freeze on an already frozen registry must have no effect.
028     */
029    public synchronized void freeze() { frozen=true; }
030
031    /** Returns whether this is currently frozen */
032    public final boolean isFrozen() { return frozen; }
033
034    /** Throws an IllegalStateException if this is frozen */
035    protected void ensureNotFrozen() {
036        if (frozen)
037            throw new IllegalStateException(this + " is frozen and cannot be modified");
038    }
039
040    /** Clones this. The clone will <i>not</i> be frozen */
041    public @Override FreezableComponent clone() {
042        FreezableComponent clone=(FreezableComponent)super.clone();
043        clone.frozen = false;
044        return clone;
045    }
046
047}
