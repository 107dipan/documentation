002package com.yahoo.component.provider;
003
004/**
005 * Convenience superclass of non-component freezables
006 *
007 * @author bratseth
008 */
009public class FreezableClass implements Freezable {
010
011    /** True when this cannot be changed any more */
012    private boolean frozen=false;
013
014    /**
015     * Freezes this class to prevent further changes. Override this to freeze internal data
016     * structures and dependent objects. Overrides must call super.
017     * Calling freeze on an already frozen registry must have no effect.
018     */
019    public synchronized void freeze() {
020        frozen=true;
021    }
022
023    /** Returns whether this is currently frozen */
024    public final boolean isFrozen() { return frozen; }
025
026    /** Throws an IllegalStateException if this is frozen */
027    protected void ensureNotFrozen() {
028        if (frozen)
029            throw new IllegalStateException(this + " is frozen and cannot be modified");
030    }
031
032    /** Clones this. The clone is <i>not</i> frozen */
033    public @Override FreezableClass clone() {
034        try {
035            FreezableClass clone=(FreezableClass)super.clone();
036            clone.frozen = false;
037            return clone;
038        }
039        catch (CloneNotSupportedException e) {
040            throw new RuntimeException(e);
041        }
042    }
043
044}
