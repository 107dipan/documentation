006/**
007 * Components or phases providing names contained in this list must be
008 * placed later in the chain than the component that is annotated.
009 * <p>
010 * See {@link com.yahoo.component.chain.dependencies.ordering.ChainBuilder}
011 * for dependency handling information.
012 *
013 * @author  tonytv
014 */
015@Retention(RetentionPolicy.RUNTIME)
016@Target(ElementType.TYPE)
017@Inherited
018public @interface Before {
019    public abstract String[] value() default {};
020}
