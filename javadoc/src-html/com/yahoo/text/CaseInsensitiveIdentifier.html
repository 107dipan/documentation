004/**
005 * @author baldersheim
006 */
007public class CaseInsensitiveIdentifier extends Identifier {
008    private final Identifier original;
009
010    public CaseInsensitiveIdentifier(String s) {
011        this(new Utf8String(s));
012    }
013    public CaseInsensitiveIdentifier(byte [] utf8) {
014        this(new Utf8Array(utf8));
015    }
016    public CaseInsensitiveIdentifier(AbstractUtf8Array utf8) {
017        super(utf8.ascii7BitLowerCase());
018        original = new Identifier(utf8);
019    }
020    public String toString() { return original.toString(); }
021}
