009/**
010 * A stream wrapper which contains utility methods for writing xml.
011 * All methods return this for convenience.
012 * <p>
013 * The methods of this writer can be used in conjunction with writing tags in raw form directly to the writer
014 * if some care is taken to close start tags and insert line breaks explicitly. If all content is written
015 * using these methods, start tags are closed and newlines inserted automatically as appropriate.
016 *
017 * @author bratseth
018 * @author baldersheim
019 */
020public class XMLWriter extends ForwardWriter {
021
022    /** Configuration */
023    private final int maxIndentLevel, maxLineSeparatorLevel;
024
025    /** The current list of parent tags */
026    private final List<Utf8String> openTags = new ArrayList<>();
027    private final List<Utf8String> unmodifiableOpenTags = Collections.unmodifiableList(openTags);
028
029    /** Control state */
030    private boolean inOpenStartTag, currentIsMultiline, isFirstInParent;
031
032    /** Write markup directly to this with no encoding if it is non-null (an optimization) */
033    private final boolean markupIsAscii;
034    static private final Utf8String SPACE = new Utf8String(" ");
035    static private final Utf8String INDENT = new Utf8String("  ");
036    static private final Utf8String ATTRIBUTE_START = new Utf8String("=\"");
037    static private final Utf8String ATTRIBUTE_END = new Utf8String("\"");
038    static private final Utf8String ENCODING_START = new Utf8String("<?xml version=\"1.0\" encoding=\"");
039    static private final Utf8String ENCODING_END = new Utf8String("\" ?>\n");
040    static private final Utf8String LF = new Utf8String("\n");
041    static private final Utf8String LT = new Utf8String("<");
042    static private final Utf8String GT = new Utf8String(">");
043    static private final Utf8String ELT = new Utf8String("</");
044    static private final Utf8String EGT = new Utf8String("/>");

104     * Creates an XML wrapper of a writer
105     *
106     * @param writer the writer to which this writers (accessible from this by getWrapped)
107     * @param maxIndentLevel the max number of tag levels for which we'll continue to indent, or -1 to
108     *        never indent. The top level tag is level 0, etc.

119        this.markupIsAscii = markupIsAscii;

142    /** Writes the first line of an XML file */
143    public void xmlHeader(String encoding) {
144        w(ENCODING_START).w(encoding).w(ENCODING_END);
145    }
146
147    public XMLWriter openTag(String s) {
148        return openTag(new Utf8String(s));
149    }
150    public XMLWriter openTag(Utf8String tag) {
151        closeStartTag();
152        if (openTags.size()>0) {
153            w(LF);
154            if (isFirstInParent && openTags.size()<=maxLineSeparatorLevel) {
155                w(LF);
156            }
157            indent();
158        }
159        w(LT).w(tag);
160        openTags.add(tag);
161        inOpenStartTag=true;
162        currentIsMultiline=false;
163        isFirstInParent=true;
164        return this;
165    }
166
167    public XMLWriter closeTag() {
168        if (openTags.size()<=0) {
169            throw new RuntimeException("Called closeTag() when no tag was open");
170        }
171        Utf8String lastOpenTag=openTags.remove(openTags.size()-1);
172
173        if (inOpenStartTag) {// this tag has no content - use short form
174            w(EGT);
175        }
176        else {
177            if (currentIsMultiline) {
178                w(LF).indent();
179            }
180            w(ELT).w(lastOpenTag).w(GT);
181        }
182        if (openTags.size()==0 || openTags.size()<=maxLineSeparatorLevel) {
183            w(LF);
184        }
185        inOpenStartTag=false;
186        currentIsMultiline=true; // When we go up from a subtag we are at a multiline tag (because it contains subtags)
187        isFirstInParent=false;   // the next opened tag will not be first
188        return this;
189    }
190
191    private XMLWriter indent() {
192        for (int i=0; i<openTags.size() && i<maxIndentLevel; i++) {
193            w(INDENT);
194        }
195        return this;
196    }
197
198    /**
199     * Closes the start tag. Usually, it is not necessary to call this, as the other methods in this will do
200     * the right thing as needed. However, this can be called explicitly to allow content or subtags to be written
201     * by a regular write call which bypasses the logic in this.
202     * If a start tag is not currently open this has no effect.
203     */
204    public XMLWriter closeStartTag() {
205        if (!inOpenStartTag) return this;
206        w(GT);
207        inOpenStartTag=false;
208        return this;
209    }

211    /**
212     * Writes an attribute by XML.xmlEscape(value.toString(),false)
213     *
214     * @param name the name of the attribute. An exception is thrown if this is null
215     * @param value the value of the attribute. The empty string if the attribute is null or empty
216     */
217    public XMLWriter forceAttribute(Utf8String name, Object value) {
218        String stringValue = value!=null ? value.toString() : "";
219        allowAttribute();
220        return w(SPACE).w(name).w(ATTRIBUTE_START).wTranscode(XML.xmlEscape(stringValue,true)).w(ATTRIBUTE_END);
221    }
222
223    public XMLWriter forceAttribute(String name, Object value) {
224        return forceAttribute(new Utf8String(name), value);
225    }
226
227    private void allowAttribute() {
228        if (!inOpenStartTag) {
229            throw new RuntimeException("Called writeAttribute() while not in an open start tag");
230        }
231    }
232    /**
233     * Writes an attribute by its utf8 value
234     *
235     * @param name the name of the attribute. An exception is thrown if this is null
236     * @param value the value of the attribute. This method does nothing if the value is null or empty
237     */
238    public XMLWriter attribute(Utf8String name, AbstractUtf8Array value) {
239        if (value.isEmpty()) return this;
240        allowAttribute();
241        return w(SPACE).w(name).w(ATTRIBUTE_START).w(value).w(ATTRIBUTE_END);
242    }

249     */
250    public XMLWriter attribute(Utf8String name, Number value) {
251        if (value == null) return this;

289    /**
290     * Writes an attribute by XML.xmlEscape(value.toString(),false)
291     *
292     * @param name the name of the attribute. An exception is thrown if this is null
293     * @param value the value of the attribute. This method does nothing if the value is null or empty
294     */
295    public XMLWriter attribute(Utf8String name, String value) {
296        if ((value == null) || value.isEmpty()) return this;
297        allowAttribute();
298        return w(SPACE).w(name).w(ATTRIBUTE_START).wTranscode(XML.xmlEscape(value, true)).w(ATTRIBUTE_END);
299    }

301    public XMLWriter attribute(String name, Object value) {
302        if (value==null) return this;
303        return attribute(new Utf8String(name), value.toString());
304    }

306    /**
307     * XML escapes and writes the content.toString(). If the content is null this does nothing but closing the start tag.
308     *
309     * @param content the content - output by XML.xmlEscape(content.toString())
310     * @param multiline whether the content should be treated as multiline,
311     *        such that the following end tag should appear on a new line
312     */
313    public XMLWriter content(Object content,boolean multiline) {
314        closeStartTag();
315        return (content==null)
316                ? this
317                : escapedContent(XML.xmlEscape(content.toString(),false),multiline);
318    }
319
320    /**
321     * Writes the given string as-is. The content string <i>must</i> be XML escaped before calling this.
322     * If the content is null this does nothing but closing the start tag.
323     *
324     * @param content the content - output by XML.xmlEscape(content.toString())
325     * @param multiline whether the content should be treated as multiline,
326     *        such that the following end tag should appear on a new line
327     */
328    public XMLWriter escapedContent(String content,boolean multiline) {
329        closeStartTag();
330        if (content==null) return this;
331        if (multiline) currentIsMultiline=true;
332        return wTranscode(content);
333    }
334
335    /**
336     * Writes the given US-ASCII only string as-is.
337     * If the content is <b>not</b> US-ASCII <i>only</i> this <i>may</i> cause
338     * incorrectly encoded content to be written.
339     * This is faster than using escapedContent as transcoding is skipped.
340     * <p>
341     * The content string <i>must</i> be XML escaped before calling this.
342     * If the content is null this does nothing but closing the start tag.
343     *
344     * @param content the content - output by XML.xmlEscape(content.toString())
345     * @param multiline whether the content should be treated as multiline,
346     *        such that the following end tag should appear on a new line
347     */
348    public XMLWriter escapedAsciiContent(String content,boolean multiline) {
349        closeStartTag();
350        if (content==null) return this;
351        if (multiline) currentIsMultiline=true;
352        return w(content);
353    }
354
355    /**
356     * Writes the given string. If markup is us ascii (default), and the wrapped writer encodes in UTF, this will write
357     * the string <b>as is, with no transcoding</b> (for speed). Hence, this should never be used for just any content.
358     *
359     * @return this for consistency
360     */
361    private final XMLWriter w(String s) {
362        return markupIsAscii ? w(new Utf8String(s)) : w(s);
363    }
364
365    private final XMLWriter w(AbstractUtf8Array utf8) {
366        write(utf8);
367        return this;
368    }
369    private final XMLWriter w(long v) {
370        write(v);
371        return this;
372    }
373    private final XMLWriter w(boolean v) {
374        write(v);
375        return this;
376    }
377    private final XMLWriter w(double v) {
378        write(v);
379        return this;
380    }
381    private final XMLWriter w(Number v) {
382        write(v.toString());
383        return this;
384    }
385
386    /** Calls write(s) and returns this. Use this for general content which must be transcoded */
387    private final XMLWriter wTranscode(String s) {
388        write(s);
389        return this;
390    }

399    /**
400     * Returns true if the immediate parent (i.e the last element in openTags)
401     * is the tag with the given name
402     */
403    public boolean isIn(Utf8String containingTag) {
404        return (openTags.size()!=0) && openTags.get(openTags.size()-1).equals(containingTag);
405    }

407    public boolean isIn(String containingTag) {
408        return (openTags.size()!=0) && openTags.get(openTags.size()-1).equals(containingTag);
