002package com.yahoo.text;
003
004import java.util.Locale;
005
006/**
007 * The lower casing method to use in Vespa when doing string processing of data
008 * which is not to be handled as natural language data, e.g. field names or
009 * configuration paramaters.
010 *
011 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
012 */
013public final class Lowercase {
014
015    private static final char[] lowercase = new char[123];
016
017    static {
018        lowercase[0x41] = 'a';
019        lowercase[0x42] = 'b';
020        lowercase[0x43] = 'c';
021        lowercase[0x44] = 'd';
022        lowercase[0x45] = 'e';
023        lowercase[0x46] = 'f';
024        lowercase[0x47] = 'g';
025        lowercase[0x48] = 'h';
026        lowercase[0x49] = 'i';
027        lowercase[0x4A] = 'j';
028        lowercase[0x4B] = 'k';
029        lowercase[0x4C] = 'l';
030        lowercase[0x4D] = 'm';
031        lowercase[0x4E] = 'n';
032        lowercase[0x4F] = 'o';
033        lowercase[0x50] = 'p';
034        lowercase[0x51] = 'q';
035        lowercase[0x52] = 'r';
036        lowercase[0x53] = 's';
037        lowercase[0x54] = 't';
038        lowercase[0x55] = 'u';
039        lowercase[0x56] = 'v';
040        lowercase[0x57] = 'w';
041        lowercase[0x58] = 'x';
042        lowercase[0x59] = 'y';
043        lowercase[0x5A] = 'z';
044
045        lowercase[0x61] = 'a';
046        lowercase[0x62] = 'b';
047        lowercase[0x63] = 'c';
048        lowercase[0x64] = 'd';
049        lowercase[0x65] = 'e';
050        lowercase[0x66] = 'f';
051        lowercase[0x67] = 'g';
052        lowercase[0x68] = 'h';
053        lowercase[0x69] = 'i';
054        lowercase[0x6A] = 'j';
055        lowercase[0x6B] = 'k';
056        lowercase[0x6C] = 'l';
057        lowercase[0x6D] = 'm';
058        lowercase[0x6E] = 'n';
059        lowercase[0x6F] = 'o';
060        lowercase[0x70] = 'p';
061        lowercase[0x71] = 'q';
062        lowercase[0x72] = 'r';
063        lowercase[0x73] = 's';
064        lowercase[0x74] = 't';
065        lowercase[0x75] = 'u';
066        lowercase[0x76] = 'v';
067        lowercase[0x77] = 'w';
068        lowercase[0x78] = 'x';
069        lowercase[0x79] = 'y';
070        lowercase[0x7A] = 'z';
071    }
072
073    /**
074     * Return a lowercased version of the given string. Since this is language
075     * independent, this is more of a case normalization operation than
076     * lowercasing. Vespa code should <i>never</i> do lowercasing with implicit
077     * locale.
078     *
079     * @param in
080     *            a string to lowercase
081     * @return a string containing only lowercase character
082     */
083    public static String toLowerCase(String in) {
084        // def is picked from http://docs.oracle.com/javase/6/docs/api/java/lang/String.html#toLowerCase%28%29
085        String lower = toLowerCasePrintableAsciiOnly(in);
086        return (lower == null) ? in.toLowerCase(Locale.ENGLISH) : lower;
087    }
088    public static String toUpperCase(String in) {
089        // def is picked from http://docs.oracle.com/javase/6/docs/api/java/lang/String.html#toLowerCase%28%29
090        return in.toUpperCase(Locale.ENGLISH);
091    }
092
093    private static String toLowerCasePrintableAsciiOnly(String in) {
094        boolean anyUpper = false;
095        for (int i = 0; i < in.length(); i++) {
096            char c = in.charAt(i);
097            if (c < 0x41) {  //lower than A-Z
098                return null;
099            }
100            if (c > 0x5A && c < 0x61) {  //between A-Z and a-z
101                return null;
102            }
103            if (c > 0x7A) {  //higher than a-z
104                return null;
105            }
106            if (c != lowercase[c]) {
107                anyUpper = true;
108            }
109        }
110        if (!anyUpper) {
111            return in;
112        }
113        StringBuilder builder = new StringBuilder(in.length());
114        for (int i = 0; i < in.length(); i++) {
115            builder.append((char) (in.charAt(i) | ((char) 0x20)));
116        }
117        return builder.toString();
118    }
119}
