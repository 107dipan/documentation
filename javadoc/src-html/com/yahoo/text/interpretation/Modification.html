002package com.yahoo.text.interpretation;
003
004import java.util.HashMap;
005
006/**
007 * A modification of a text.
008 *
009 * This class represents a possible rewrite of an original text. Reasons for rewrite may be due to possible
010 * spelling errors in the text or to query expansion.
011 *
012 * @author <a href="mailto:arnebef@yahoo-inc.com">Arne Bergene Fossaa</a>
013 */
014public class Modification extends HashMap<String,Object>{
015
016    /**
017     *
018     */
019    private static final long serialVersionUID = -8522335044460396296L;
020
021
022    public final static AnnotationClass MODIFICATION_CLASS = new AnnotationClass("modification");
023
024
025    private String text;
026    private Annotations annotations;
027
028    public Modification(String text) {
029        this.text = text;
030        Span span = new Span(this);
031        this.annotations = span.annotate(MODIFICATION_CLASS);
032    }
033
034    public String getText() {
035        return text;
036    }
037
038    public Annotations getAnnotation() {
039        return annotations;
040    }
041
042}
