002package com.yahoo.text.interpretation;
003
004import java.util.Collections;
005import java.util.HashMap;
006import java.util.Map;
007
008/**
009 * An annotation is a description of a an area of text, with a given class. For example, an annotation for the
010 *
011 * @author <a href="mailto:arnebef@yahoo-inc.com">Arne Bergene Fossaa</a>
012 */
013public class Annotations {
014
015
016    private Span span;
017
018    protected Map<String,Object> annotations;
019
020
021    /**
022     * Adds an annotation to the the the set of annotations.
023     */
024    public void put(String key,Object o) {
025        if(annotations == null) {
026            annotations = new HashMap<>();
027        }
028        annotations.put(key,o);
029    }
030
031    public Map<String,Object> getMap() {
032        if(annotations == null) {
033            return Collections.emptyMap();
034        } else {
035            return annotations;
036        }
037    }
038
039    public Annotations(Span span) {
040        this.span = span;
041    }
042
043    public Object get(String key) {
044        return getMap().get(key);
045    }
046
047    /**
048     * The span that this annotation is for.
049     */
050    public Span getSpan() {
051        return span;
052    }
053
054    /**
055     * The text this annotation is for.
056     */
057    public String getSubString() {
058        return span.getText();
059    }
060
061
062    /**
063     * Helper function to get a Double annotation.
064     * <p>
065     * This function first checks if the Object in a map is a <code>Number</code>, and then calls doubleValue() on it
066     * If it is not, then  Double.parseDouble() is called on the string representation of the object. If the string
067     * is not parseable as a double, a NumberFormatException is thrown.
068     */
069    public Double getDouble(String key)  {
070        Object o = getMap().get(key);
071        if(o instanceof Number) {
072            return ((Number)o).doubleValue();
073        } else if(o == null) {
074            return null;
075        } else {
076            return Double.parseDouble(o.toString());
077        }
078    }
079
080    /**
081     * Helper function to get a String from the Annotation. This function will simply call <code>toString()</code> on the
082     * object saved in the Annotation or return null if the object is null;
083     */
084    public String getString(String key) {
085        Object o = getMap().get(key);
086        if(o == null) {
087            return null;
088        } else {
089            return o.toString();
090        }
091    }
092
093    /**
094    * Helper function to get a Double annotation.
095     * <p>
096     * This function first checks if the Object in a map is a <code>Number</code>, and intValue() is called on it.
097     * If it is not, then Double.parseDouble() is called on the string representation of the object. If the string
098     * is not parseable as a double, a NumberFormatException is thrown.
099     */
100    public Integer getInteger(String key) {
101        Object o = getMap().get(key);
102        if(o == null) {
103            return null;
104        } else if(o instanceof Number) {
105            return ((Number)o).intValue();
106        } else {
107            return Integer.parseInt(o.toString());
108        }
109    }
110
111    /**
112     * Helper function to get a Boolean annotation.
113     */
114    public Boolean getBoolean(String key) {
115        Object o = getMap().get(key);
116        if(o == null || !(o instanceof Boolean)) {
117            return null;
118        } else {
119            return (Boolean) o;
120        }
121    }
122
123
124}
