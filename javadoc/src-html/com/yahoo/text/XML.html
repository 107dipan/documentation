032     * The point of this weird class and the jumble of abstract methods is
033     * linking the scan for characters that must be quoted into the quoting
034     * table, and making it actual work to make them go out of sync again.
035     */
036    private static abstract class LegalCharacters {
037        // To quote http://www.w3.org/TR/REC-xml/ :
038        // Char ::= #x9 | #xA | #xD | [#x20-#xD7FF] | [#xE000-#xFFFD] |
039        // [#x10000-#x10FFFF]
040        final boolean isLegal(final int codepoint, final boolean escapeLow,
041                final int stripCodePoint, final boolean isAttribute) {
042            if (codepoint == stripCodePoint) {
043                return removeCodePoint();
044            } else if (codepoint < ' ') {
045                if (!escapeLow) {
046                    return true;
047                }
048                switch (codepoint) {
049                case 0x09:
050                case 0x0a:
051                case 0x0d:
052                    return true;
053                default:
054                    return ctrlEscapeCodePoint(codepoint);
055                }
056            } else if (codepoint >= 0x20 && codepoint <= 0xd7ff) {
057                switch (codepoint) {
058                case '&':
059                    return ampCodePoint();
060                case '<':
061                    return ltCodePoint();
062                case '>':
063                    return gtCodePoint();
064                case '"':
065                    return quotCodePoint(isAttribute);
066                default:
067                    return true;
068                }
069            } else if ((codepoint >= 0xe000 && codepoint <= 0xfffd)
070                    || (codepoint >= 0x10000 && codepoint <= 0x10ffff)) {
071                return true;
072            } else {
073                return filterCodePoint(codepoint);
074
075            }
076        }
077
078        private boolean quotCodePoint(final boolean isAttribute) {
079            if (isAttribute) {
080                quoteQuot();
081                return false;
082            } else {
083                return true;
084            }
085        }
086
087        private boolean filterCodePoint(final int codepoint) {
088            replace(codepoint);
089            return false;
090        }
091
092        private boolean gtCodePoint() {
093            quoteGt();
094            return false;
095        }
096
097        private boolean ltCodePoint() {
098            quoteLt();
099            return false;
100        }
101
102        private boolean ampCodePoint() {
103            quoteAmp();
104            return false;
105        }
106
107        private boolean ctrlEscapeCodePoint(final int codepoint) {
108            ctrlEscape(codepoint);
109            return false;
110        }
111
112        private boolean removeCodePoint() {
113            remove();
114            return false;
115        }
116
117        protected abstract void quoteQuot();
118
119        protected abstract void quoteGt();
120
121        protected abstract void quoteLt();
122
123        protected abstract void quoteAmp();
124
125        protected abstract void remove();
126
127        protected abstract void ctrlEscape(int codepoint);
128
129        protected abstract void replace(int codepoint);
130    }

340    /**
341     * Replaces the following:
342     * <ul>
343     * <li>all ascii codes less than 32 except 9 (tab), 10 (nl) and 13 (cr) if
344     * escapeLowAscii is <code>true</code>
345     * <li>ampersand (&amp;)
346     * <li>less than (&lt;)
347     * <li>larger than (&gt;)
348     * <li>double quotes (&quot;) if isAttribute is <code>true</code>
349     * </ul>
350     * with character entities.
351     *
352     * @param stripCodePoint
353     *            any occurrence of this character is removed from the string
354     */
355    public static String xmlEscape(String string, boolean isAttribute, boolean escapeLowAscii,
356                                   StringBuilder buffer, int stripCodePoint) {
357        // buffer and stripCodePoint changed order in the signature compared to
358        // the char based API to avoid wrong method being called
359
360        // This is inner loop stuff, so we sacrifice a little for speed -
361        // no copying will occur until a character needing escaping is found
362        boolean legalCharacter = true;
363        Quote escaper;
364        int i = 0;
365
366        for (i = 0; i < string.length() && legalCharacter; i = string.offsetByCodePoints(i, 1)) {
367            legalCharacter = scanner.isLegal(string.codePointAt(i), escapeLowAscii, stripCodePoint, isAttribute);
368        }
369        if (legalCharacter) {
370            return string;
371        }
372
373        i = string.offsetByCodePoints(i, -1); // Back to the char needing escaping
374        escaper = new Quote();
375
376        if (buffer == null) {
377            buffer = new StringBuilder((int) (string.length() * 1.2));
378        }
379
380        // ugly appending zero length strings
381        if (i > 0) {
382            buffer.append(string.substring(0, i));
383        }
384
385        // i is at the first codepoint which needs replacing
386        // Don't guard against double-escaping, as:
387        // don't try to be clever (LCJ).
388        for (; i < string.length(); i = string.offsetByCodePoints(i, 1)) {
389            int codepoint = string.codePointAt(i);
390            if (escaper.isLegal(codepoint, escapeLowAscii, stripCodePoint, isAttribute)) {
391                buffer.appendCodePoint(codepoint);
392            } else {
393                buffer.append(escaper.lastQuoted);
394            }
395        }
396        return buffer.toString();
397    }

401     *
402     * @throws RuntimeException
403     *             if the root Document cannot be returned
404     */
405    public static Document getDocument(Reader reader) {
406        try {
407            return getDocumentBuilder().parse(new InputSource(reader));
408        } catch (IOException e) {
409            throw new IllegalArgumentException("Could not read '" + reader + "'", e);
410        } catch (SAXParseException e) {
411            throw new IllegalArgumentException("Could not parse '" + reader + "', error at line " + e.getLineNumber() + ", column " + e.getColumnNumber(), e);
412        } catch (SAXException e) {
413            throw new IllegalArgumentException("Could not parse '" + reader + "'", e);
414        }
415    }

459     * Returns the child Element objects from a w3c dom spec
460     *
461     * @return List of elements. Empty list (never null) if none found or if the
462     *         given element is null
463     */
464    public static List<Element> getChildren(Element spec) {
465        List<Element> children = new ArrayList<>();
466        if (spec == null) {
467            return children;
468        }
469
470        NodeList childNodes = spec.getChildNodes();
471        for (int i = 0; i < childNodes.getLength(); i++) {
472            Node child = childNodes.item(i);
473            if (child instanceof Element) {
474                children.add((Element) child);
475            }
476        }
477        return children;
478    }

480    /**
481     * Returns the child Element objects with given name from a w3c dom spec
482     *
483     * @return List of elements. Empty list (never null) if none found or the
484     *         given element is null
485     */
486    public static List<Element> getChildren(Element spec, String name) {
487        List<Element> ret = new ArrayList<>();
488        if (spec == null) {
489            return ret;
490        }
491
492        NodeList children = spec.getChildNodes();
493        if (children == null) {
494            return ret;
495        }
496        for (int i = 0; i < children.getLength(); i++) {
497            Node child = children.item(i);
498            if (child != null && child instanceof Element) {
499                if (child.getNodeName().equals(name)) {
500                    ret.add((Element) child);
501                }
502            }
503        }
504        return ret;
505    }

507    /**
508     * Gets the string contents of the given Element. Returns "", never null if
509     * the element is null, or has no content
510     */
511    public static String getValue(Element e) {
512        if (e == null) {
513            return "";
514        }
515        Node child = e.getFirstChild();
516        if (child == null) {
517            return "";
518        }
519        return child.getNodeValue();
520    }

524        return (getChildren(e, name).size() >= 1) ? getChildren(e, name).get(0) : null;

527    /**
528     * Returns the path to the given xml node, where each node name is separated
529     * by the given separator string.
530     *
531     * @param n
532     *            The xml node to find path to
533     * @param sep
534     *            The separator string
535     * @return The path to the xml node as a String
536     */
537    public static String getNodePath(Node n, String sep) {
538        if (n == null) {
539            return "";
540        }
541        StringBuffer ret = new StringBuffer(n.getNodeName());
542        while ((n.getParentNode() != null) && !(n.getParentNode() instanceof Document)) {
543            n = n.getParentNode();
544            ret.insert(0, sep).insert(0, n.getNodeName());
545        }
546        return ret.toString();
547    }

550    private static boolean inclusiveWithin(int x, int low, int high) {
551        return low <= x && x <= high;
552    }

554    private static boolean nameStartSet(int codepoint) {
555        // NameStartChar ::= ":" | [A-Z] | "_" | [a-z] | [#xC0-#xD6] |
556        // [#xD8-#xF6] | [#xF8-#x2FF] | [#x370-#x37D] | [#x37F-#x1FFF] |
557        // [#x200C-#x200D] | [#x2070-#x218F] | [#x2C00-#x2FEF] | [#x3001-#xD7FF]
558        // | [#xF900-#xFDCF] | [#xFDF0-#xFFFD] | [#x10000-#xEFFFF]
559
560        boolean valid;
561        if (codepoint < 0xC0) {
562            valid = inclusiveWithin(codepoint, 'a', 'z')
563                    || inclusiveWithin(codepoint, 'A', 'Z') || codepoint == '_'
564                    || codepoint == ':';
565        } else {
566            valid = inclusiveWithin(codepoint, 0xC0, 0xD6)
567                    || inclusiveWithin(codepoint, 0xD8, 0xF6)
568                    || inclusiveWithin(codepoint, 0xF8, 0x2FF)
569                    || inclusiveWithin(codepoint, 0x370, 0x37D)
570                    || inclusiveWithin(codepoint, 0x37F, 0x1FFF)
571                    || inclusiveWithin(codepoint, 0x200C, 0x200D)
572                    || inclusiveWithin(codepoint, 0x2070, 0x218F)
573                    || inclusiveWithin(codepoint, 0x2C00, 0x2FEF)
574                    || inclusiveWithin(codepoint, 0x3001, 0xD7FF)
575                    || inclusiveWithin(codepoint, 0xF900, 0xFDCF)
576                    || inclusiveWithin(codepoint, 0xFDF0, 0xFFFD)
577                    || inclusiveWithin(codepoint, 0x10000, 0xEFFFF);
578        }
579        return valid;
580    }

582    private static boolean nameSetExceptStart(int codepoint) {
583        // "-" | "." | [0-9] | #xB7 | [#x0300-#x036F] | [#x203F-#x2040]
584        boolean valid;
585        if (codepoint < 0xB7) {
586            valid = inclusiveWithin(codepoint, '0', '9') || codepoint == '-'
587                    || codepoint == '.';
588        } else {
589
590            valid = codepoint == '\u00B7'
591                    || inclusiveWithin(codepoint, 0x300, 0x36F)
592                    || inclusiveWithin(codepoint, 0x023F, 0x2040);
593        }
594        return valid;
595    }
596
597    private static boolean nameChar(int codepoint, boolean first) {
598        // NameChar ::= NameStartChar | "-" | "." | [0-9] | #xB7 | [#x0300-#x036F] | [#x203F-#x2040]
599        boolean valid = nameStartSet(codepoint);
600        return first ? valid : valid || nameSetExceptStart(codepoint);
601    }

604    /**
605     * Check whether the name of a tag or attribute conforms to <a
606     * href="http://www.w3.org/TR/2006/REC-xml11-20060816/#sec-common-syn">XML
607     * 1.1 (Second Edition)</a>. This does not check against reserved names, it
608     * only checks the set of characters used.
609     *
610     * @param possibleName
611     *            a possibly valid XML name
612     * @return true if the name may be used as an XML tag or attribute name
613     */
614    public static boolean isName(CharSequence possibleName) {
615        final int barrier = possibleName.length();
616        int i = 0;
617        boolean valid = true;
618        boolean first = true;
619
620        if (barrier < 1) {
621            valid = false;
622        }
623
624        while (valid && i < barrier) {
625            char c = possibleName.charAt(i++);
626            if (Character.isHighSurrogate(c)) {
627                valid = nameChar(Character.toCodePoint(c, possibleName.charAt(i++)), first);
628            } else {
629                valid = nameChar((int) c, first);
630            }
631            first = false;
632        }
633        return valid;
634    }
