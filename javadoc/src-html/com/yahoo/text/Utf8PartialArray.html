004/**
005 * This wraps a window in a backing byte array. Without doing any copying.
006 * @author baldersheim
007 * @since 5.2
008 */
009public class Utf8PartialArray extends Utf8Array {
010    final int offset;
011    final int length;
012
013    /**
014     * Takes ownership of the given byte array. And keeps note of where
015     * the interesting utf8 sequence start and its length.
016     * @param utf8data The backing byte array.
017     * @param offset   The start of the utf8 sequence.
018     * @param bytes    The length of the utf8 sequence.
019     */
020    public Utf8PartialArray(byte[] utf8data, int offset, int bytes) {
021        super(utf8data);
022        this.offset = offset;
023        this.length = bytes;
024    }
025    @Override
026    public int getByteLength() {
027        return length;
028    }
029
030    @Override
031    protected int getByteOffset() {
032        return offset;
033    }
034
035}
