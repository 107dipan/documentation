002package com.yahoo.text;
003
004import java.io.IOException;
005
006/**
007 * Wraps another writer and also converting IOException to Exceptions.
008 *
009 * @author baldersheim
010 * @since 5.2
011 */
012public class ForwardWriter extends GenericWriter {
013
014    private final GenericWriter out;
015
016    public ForwardWriter(GenericWriter writer) {
017        super();
018        this.out = writer;
019    }
020
021    @Override
022    public void write(char[] c, int offset, int bytes) {
023        try {
024            out.write(c, offset, bytes);
025        } catch (IOException e) {
026            throw new RuntimeException(e);
027        }
028    }
029    @Override
030    public GenericWriter write(AbstractUtf8Array v) {
031        try {
032            out.write(v);
033        } catch (IOException e) {
034            throw new RuntimeException(e);
035        }
036        return this;
037    }
038    @Override
039    public void write(String v) {
040        try {
041            out.write(v);
042        } catch (IOException e) {
043            throw new RuntimeException(e);
044        }
045    }
046    @Override
047    public GenericWriter write(CharSequence c) {
048        try {
049            out.write(c);
050        } catch (IOException e) {
051            throw new RuntimeException(e);
052        }
053        return this;
054    }
055
056
057    @Override
058    public GenericWriter write(double d) {
059        try {
060            out.write(d);
061        } catch (IOException e) {
062            throw new RuntimeException(e);
063        }
064        return this;
065    }
066
067    @Override
068    public GenericWriter write(float f) {
069        try {
070            out.write(f);
071        } catch (IOException e) {
072            throw new RuntimeException(e);
073        }
074        return this;
075    }
076
077    @Override
078    public GenericWriter write(long v) {
079        try {
080            out.write(v);
081        } catch (IOException e) {
082            throw new RuntimeException(e);
083        }
084        return this;
085    }
086    @Override
087    public void write(int v) {
088        try {
089            out.write(v);
090        } catch (IOException e) {
091            throw new RuntimeException(e);
092        }
093    }
094    @Override
095    public GenericWriter write(short v) {
096        try {
097            out.write(v);
098        } catch (IOException e) {
099            throw new RuntimeException(e);
100        }
101        return this;
102    }
103    @Override
104    public GenericWriter write(char c) {
105        try {
106            out.write(c);
107        } catch (IOException e) {
108            throw new RuntimeException(e);
109        }
110        return this;
111    }
112    @Override
113    public GenericWriter write(byte b) {
114        try {
115            out.write(b);
116        } catch (IOException e) {
117            throw new RuntimeException(e);
118        }
119        return this;
120    }
121
122
123    @Override
124    public GenericWriter write(boolean v) {
125        try {
126            out.write(v);
127        } catch (IOException e) {
128            throw new RuntimeException(e);
129        }
130        return this;
131    }
132
133    @Override
134    public void flush() {
135        try {
136            out.flush();
137        } catch (IOException e) {
138            throw new RuntimeException(e);
139        }
140    }
141
142    @Override
143    public void close() {
144        try {
145            out.close();
146        } catch (IOException e) {
147            throw new RuntimeException(e);
148        }
149    }
150
151    /**
152     * Gives access to the wrapped writer.
153     * @return wrapped writer.
154     */
155    public GenericWriter getWriter() { return out; }
156
157}
