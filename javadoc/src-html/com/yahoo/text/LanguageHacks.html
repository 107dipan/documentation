002package com.yahoo.text;
003
004import static com.yahoo.text.Lowercase.toLowerCase;
005
006/**
007 * Language helper functions.
008 *
009 * @deprecated do not use
010 * @author  <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
011 */
012@Deprecated
013public class LanguageHacks {
014
015    /**
016     * Whether a language is in the CJK group.
017     */
018    public static boolean isCJK(String language) {
019        if (language == null) return false;
020
021        language = toLowerCase(language);
022        return "ja".equals(language)
023                || "ko".equals(language)
024                || language.startsWith("zh")
025                || language.startsWith("tw"); // TODO: tw is a bogus value?
026    }
027
028    /**
029     * Whether there is desegmenting in this language.
030     */
031    public static boolean yellDesegments(String language) {
032        if (language == null) return false;
033
034        language = toLowerCase(language);
035        return "de".equals(language) || isCJK(language);
036    }
037
038}
