076    /**
077     * Encode a long as its decimal representation, i.e. toAsciiBytes(15L) will
078     * return "15" encoded as UTF-8. In other words it is an optimized version
079     * of String.valueOf() followed by UTF-8 encoding. Avoid going through
080     * string in order to get a simple UTF-8 sequence.
081     *
082     * @param l
083     *            value to represent as a decimal number encded as utf8
084     * @return byte array
085     */
086    public static byte[] toAsciiBytes(long l) {
087        // Handle Long.MIN_VALUE specifically, since it breaks all the assumptions
088        if (Long.MIN_VALUE == l) {
089            return LONG_MIN_VALUE_BYTES;
090        }
091        int count=1;
092        for (long v= l<0 ? -l : l; v >= 10; v=v/10, count++);
093        byte [] buf = new byte [count + ((l<0) ? 1 : 0)];
094        int offset = 0;
095        if (l < 0) {
096            buf[offset++] = (byte) '-';
097            l = -l;
098        }
099        for (count--; count >= 0; l=l/10, count--) {
100            buf[count+offset] = (byte)(0x30 + l%10);
101        }
102        return buf;
103    }

106        return v ? TRUE : FALSE;

109    /**
110     * Will try an optimistic approach to utf8 encoding.
111     * That is 4.6x faster that the brute encode for ascii, not accounting for reduced memory footprint and GC.
112     * @param str The string to encode.
113     * @return Utf8 encoded array
114     */
115    public static byte[] toBytes(String str) {
116        byte [] utf8 = toBytesAscii(str);
117        return utf8 != null ? utf8 : str.getBytes(StandardCharsets.UTF_8);
118    }
119    /**
120     * Will try an optimistic approach to utf8 decoding.
121     *
122     * @param utf8 The string to encode.
123     * @return Utf8 encoded array
124     */
125    public static String toString(byte [] utf8) {
126        String s = toStringAscii(utf8, 0, utf8.length);
127        return s != null ? s : new String(utf8, StandardCharsets.UTF_8);
128    }

130    /**
131     * If String is purely ascii 7bit it will encode it as a byte array.
132     * @param str The string to encode
133     * @return Utf8 encoded array
134     */
135    private static byte[] toBytesAscii(final CharSequence str) {
136        byte [] utf8 = new byte[str.length()];
137        for (int i=0; i < utf8.length; i++) {
138            char c = str.charAt(i);
139            if ((c < 0) || (c >= 0x80)) {
140                return null;
141            }
142            utf8[i] = (byte)c;
143        }
144        return utf8;
145    }

147    private static String toStringAscii(byte [] b, int offset, int length) {
148        if (length > 0) {
149            char [] s = new char[length];
150            for (int i=0; i < length; i++) {
151                if (b[offset + i] >= 0) {
152                    s[i] = (char)b[offset+i];
153                } else {
154                    return null;
155                }
156            }
157            return new String(s);
158        } else {
159            return "";
160        }
161    }

163    /**
164     * Utility method as toBytes(String).
165     *
166     * @param str
167     *            String to encode
168     * @param offset
169     *            index of first character to encode
170     * @param length
171     *            number of characters to encode
172     * @return substring encoded as UTF-8
173     */
174    public static byte[] toBytes(String str, int offset, int length) {
175        CharBuffer c = CharBuffer.wrap(str, offset, offset + length);
176        ByteBuffer b = StandardCharsets.UTF_8.encode(c);
177        byte[] result = new byte[b.remaining()];
178        b.get(result);
179        return result;
180    }

183     * Direct encoding of a String into an array.
184     *
185     * @param str
186     *            string to encode
187     * @param srcOffset
188     *            index of first character in string to encode
189     * @param srcLen
190     *            number of characters in string to encode
191     * @param dst
192     *            destination for encoded data
193     * @param dstOffset
194     *            index of first position to write data
195     * @return the number of bytes written to the array.
196     */
197    public static int toBytes(String str, int srcOffset, int srcLen, byte[] dst, int dstOffset) {
198        CharBuffer c = CharBuffer.wrap(str, srcOffset, srcOffset + srcLen);
199        ByteBuffer b = StandardCharsets.UTF_8.encode(c);
200        int encoded = b.remaining();
201        b.get(dst, dstOffset, encoded);
202        return encoded;
203    }

205    /**
206     * Encode a string directly into a ByteBuffer instance.
207     *
208     * <p>
209     * This method is somewhat more cumbersome than the rest of the helper
210     * methods in this library, as it is intended for use cases in the following
211     * style, if extraneous copying is highly undesirable:
212     *
213     * <pre>
214     * String[] a = {"abc", "def", "ghi\u00e8"};
215     * int[] aLens = {3, 3, 5};
216     * CharsetEncoder ce = Utf8.getNewEncoder();
217     * ByteBuffer forWire = ByteBuffer.allocate(someNumber);
218     *
219     * for (int i = 0; i &lt; a.length; i++) {
220     *     forWire.putInt(aLens[i]);
221     *     Utf8.toBytes(a[i], 0, a[i].length(), forWire, ce);
222     * }
223     * </pre>
224     *
225     * @see Utf8#getNewEncoder()
226     *
227     * @param src the string to encode
228     * @param srcOffset index of first character to encode
229     * @param srcLen number of characters to encode
230     * @param dst the destination ByteBuffer
231     * @param encoder the character encoder to use
232     */
233    public static void toBytes(String src, int srcOffset, int srcLen, ByteBuffer dst, CharsetEncoder encoder) {
234        CharBuffer c = CharBuffer.wrap(src, srcOffset, srcOffset + srcLen);
235        encoder.encode(c, dst, true);
236    }

249     * Count the number of bytes needed to represent a given sequence of 16-bit
250     * char values as a UTF-8 encoded array. This method is written to be cheap
251     * to invoke.
252     *
253     * Note: It is strongly assumed to character sequence is valid.
254     */
255    public static int byteCount(CharSequence str) { return byteCount(str, 0, str.length()); }

257    /**
258     * Count the number of bytes needed to represent a given sequence of 16-bit
259     * char values as a UTF-8 encoded array. This method is written to be cheap
260     * to invoke.
261     *
262     * Note: It is strongly assumed to character sequence is valid.
263     */
264    public static int byteCount(CharSequence str, int offset, int length) {
265        int count = 0;
266        int barrier = offset + length;
267        int i = offset;
268        while (i < barrier) {
269            int codePoint = (int) str.charAt(i);
270            if (codePoint < 0x800) {
271                if (codePoint < 0x80) {
272                    ++count;
273                } else {
274                    count += 2;
275                }
276                ++i;
277            } else {
278                // bit masking to check (codePoint >= 0xd800 && codePoint <
279                // 0xe000)
280                if ((codePoint & 0xF800) == 0xD800) {
281                    count += 4;
282                    i += 2;
283                } else {
284                    count += 3;
285                    ++i;
286                }
287            }
288        }
289        return count;
290    }

292    /**
293     * Count the number of Unicode code units ("UTF-16 characters") needed to
294     * represent a given array of UTF-8 characters. This method is written to be
295     * cheap to invoke.
296     *
297     * Note: It is strongly assumed the sequence is valid.
298     */
299    public static int unitCount(byte[] utf8) { return unitCount(utf8, 0, utf8.length); }

301    /**
302     * Count the number of Unicode code units ("UTF-16 characters") needed to
303     * represent a given array of UTF-8 characters. This method is written to be
304     * cheap to invoke.
305     *
306     * Note: It is strongly assumed the sequence is valid.
307     *
308     * @param utf8
309     *            raw data
310     * @param offset
311     *            index of first byte of UTF-8 sequence to check
312     * @param length
313     *            number of bytes in the UTF-8 sequence to check
314     */
315    public static int unitCount(byte[] utf8, int offset, int length) {
316        int units = 0;
317        int barrier = offset + length;
318        int i = offset;
319        while (i < barrier) {
320            byte firstByte = utf8[i];
321            if (firstByte >= -16) {
322                if (firstByte >= 0) {
323                    ++units;
324                    ++i;
325                } else {
326                    units += 2;
327                    i += 4;
328                }
329            } else {
330                if (firstByte >= -32) {
331                    ++units;
332                    i += 3;
333                } else {
334                    ++units;
335                    i += 2;
336                }
337            }
338        }
339        return units;
340    }

342    /**
343     * Calculate the number of Unicode code units ("UTF-16 characters") needed
344     * to represent a given UTF-8 encoded code point.
345     *
346     * @param firstByte
347     *            the first byte of a character encoded as UTF-8
348     * @return the number of UTF-16 code units needed to represent the given
349     *         code point
350     */
351    public static int unitCount(byte firstByte) {
352        int units = 0;
353        if (firstByte >= -16) {
354            if (firstByte >= 0) {
355                units = 1;
356            } else {
357                units = 2;
358            }
359        } else {
360            units = 1;
361        }
362        return units;
363    }

365    /**
366     * Inspects a byte assumed to be the first byte in a UTF8 to check how many
367     * bytes in total the sequence of bytes will use.
368     *
369     * @param firstByte
370     *            the first byte of a UTF8 encoded character
371     * @return the number of bytes used to encode the character
372     */
373    // To avoid code duplication, this function should be used by unitCount(),
374    // but then unitCount(byte[], int, int) would not be as tight. This class is in general
375    // meant to be safe to use in performance sensitive code.
376    public static int totalBytes(byte firstByte) {
377        if (firstByte >= -16) {
378            if (firstByte >= 0) {
379                return 1;
380            } else {
381                return 4;
382            }
383        } else {
384            if (firstByte >= -32) {
385                return 3;
386            } else {
387                return 2;
388            }
389        }
390    }

392    /**
393     * Returns an integer array the length as the input string plus one. For
394     * every index in the array, the corresponding value gives the index into
395     * the UTF-8 byte sequence that can be created from the input.
396     *
397     * @param value
398     *            a String to generate UTF-8 byte indexes from
399     * @return an array containing corresponding UTF-8 byte indexes
400     */
401    public static int[] calculateBytePositions(CharSequence value) {
402        int[] positions = new int[value.length() + 1];
403
404        int bytePos = 0;
405        int barrier = value.length();
406        int i = 0;
407        int codepointNo = 0;
408        positions[codepointNo++] = bytePos;
409        while (i < barrier) {
410            int codePoint = (int) value.charAt(i);
411            if (codePoint < 0x800) {
412                if (codePoint < 0x80) {
413                    ++bytePos;
414                } else {
415                    bytePos += 2;
416                }
417                ++i;
418            } else {
419                // bit masking to check (codePoint >= 0xd800 && codePoint <
420                // 0xe000)
421                if ((codePoint & 0xF800) == 0xD800) {
422                    // double position write, as we have a surrogate pair
423                    positions[codepointNo++] = bytePos;
424                    bytePos += 4;
425                    i += 2;
426                } else {
427                    bytePos += 3;
428                    ++i;
429                }
430            }
431            positions[codepointNo++] = bytePos;
432        }
433        return positions;
434    }

436    /**
437     * Returns an array of the same length as the input array plus one. For
438     * every index in the array, the corresponding value gives the index into
439     * the Java string (UTF-16 sequence) that can be created from the input.
440     *
441     * @param utf8
442     *            a byte array containing a string encoded as UTF-8. Note: It is
443     *            strongly assumed that this sequence is correct.
444     * @return an array containing corresponding UTF-16 character indexes. If input
445     *            array is empty, returns an array containg a single zero.
446     */
447    public static int[] calculateStringPositions(byte[] utf8) {
448        if (utf8.length == 0) {
449            return new int[] { 0 };
450        }
451        int[] positions = new int[utf8.length + 1];
452        int utf8BytePos = 0;
453        int charPos = 0;
454        int lastUtf8SequencePos = 0;
455        int utf8SequenceLen = 0;
456        while (utf8BytePos < utf8.length) {
457            utf8SequenceLen = totalBytes(utf8[utf8BytePos]);
458            lastUtf8SequencePos = utf8BytePos;
459            for (int utf8SequenceCnt = 0; utf8SequenceCnt < utf8SequenceLen; utf8SequenceCnt++) {
460                positions[utf8BytePos + utf8SequenceCnt] = charPos;
461            }
462            utf8BytePos += utf8SequenceLen;
463            charPos++;
464        }
465        //we need to check if the last UTF-8 sequence resulted in a surrogate pair:
466        int lastCharLen = unitCount(utf8, lastUtf8SequencePos, utf8SequenceLen);
467        positions[utf8.length] = charPos + lastCharLen - 1;
468        return positions;
469    }

472    /**
473     * Encode a valid Unicode codepoint as a sequence of UTF-8 bytes into a new allocated array.
474     *
475     * @param codepoint Unicode codepoint to encode
476     * @return number of bytes written
477     * @throws IndexOutOfBoundsException if there is insufficient room for the encoded data in the given array
478     */
479    public static byte[] encode(int codepoint) {
480        byte[] destination = new byte[codePointAsUtf8Length(codepoint)];
481        encode(codepoint, destination, 0);
482        return destination;
483    }

485    /**
486     * Encode a valid Unicode codepoint as a sequence of UTF-8 bytes into an array.
487     *
488     * @param codepoint Unicode codepoint to encode
489     * @param destination array to write into
490     * @param offset index of first byte written
491     * @return index of the first byte after the last byte written (i.e. offset plus number of bytes written)
492     * @throws IndexOutOfBoundsException if there is insufficient room for the encoded data in the given array
493     */
494    public static int encode(int codepoint, byte[] destination, int offset) {
495        int writeOffset = offset;
496        byte firstByte = firstByte(codepoint);
497        int leftToWrite = codePointAsUtf8Length(codepoint) - 1;
498        destination[writeOffset++] = firstByte;
499        while (leftToWrite-- > 0) {
500            destination[writeOffset++] = trailingOctet(codepoint, leftToWrite);
501        }
502        return writeOffset;
503    }

505    /**
506     * Encode a valid Unicode codepoint as a sequence of UTF-8 bytes into a
507     * ByteBuffer.
508     *
509     * @param codepoint
510     *            Unicode codepoint to encode
511     * @param destination
512     *            buffer to write into
513     * @throws BufferOverflowException
514     *             if the buffer's limit is met while writing (propagated from
515     *             the ByteBuffer)
516     * @throws ReadOnlyBufferException
517     *             if the buffer is read only (propagated from the ByteBuffer)
518     */
519    public static void encode(int codepoint, ByteBuffer destination) {
520        byte firstByte = firstByte(codepoint);
521        int leftToWrite = codePointAsUtf8Length(codepoint) - 1;
522        destination.put(firstByte);
523        while (leftToWrite-- > 0) {
524            destination.put(trailingOctet(codepoint, leftToWrite));
525        }
526    }

528    /**
529     * Encode a valid Unicode codepoint as a sequence of UTF-8 bytes into an
530     * OutputStream.
531     *
532     * @param codepoint
533     *            Unicode codepoint to encode
534     * @param destination
535     *            buffer to write into
536     * @return number of bytes written
537     * @throws IOException
538     *             propagated from stream
539     */
540    public static int encode(int codepoint, OutputStream destination) throws IOException {
541        byte firstByte = firstByte(codepoint);
542        int toWrite = codePointAsUtf8Length(codepoint);
543        int leftToWrite = toWrite - 1;
544        destination.write(firstByte);
545        while (leftToWrite-- > 0) {
546            destination.write(trailingOctet(codepoint, leftToWrite));
547        }
548        return toWrite;
549    }

552    private static byte trailingOctet(int codepoint, int leftToWrite) {
553        return (byte) (0x80 | ((codepoint >> (6 * leftToWrite)) & 0x3F));
554    }
555
556    private static byte firstByte(int codepoint) {
557        if (codepoint < 0x800) {
558            if (codepoint < 0x80) {
559                return (byte) codepoint;
560            } else {
561                return (byte) (0xC0 | codepoint >> 6);
562            }
563        } else {
564            if (codepoint < 0x10000) {
565                return (byte) (0xE0 | codepoint >> 12);
566            } else {
567                return (byte) (0xF0 | codepoint >> 18);
568            }
569        }
570
571    }

573    /**
574     * Return the number of octets needed to encode a valid Unicode codepoint as UTF-8.
575     *
576     * @param codepoint the Unicode codepoint to inspect
577     * @return the number of bytes needed for UTF-8 representation
578     */
579    public static int codePointAsUtf8Length(int codepoint) {
580        if (codepoint < 0x800) {
581            if (codepoint < 0x80) {
582                return 1;
583            } else {
584                return 2;
585            }
586        } else {
587            if (codepoint < 0x10000) {
588                return 3;
589            } else {
590                return 4;
591            }
592        }
593    }
