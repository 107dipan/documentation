004/**
005 * This class is used to represent a legal identifier of [a-zA-Z_][a-zA-Z_0-9]*
006 *
007 * @author baldersheim
008 */
009public class Identifier extends Utf8Array {
010    public Identifier(String s) {
011        this(Utf8.toBytes(s));
012    }
013    public Identifier(AbstractUtf8Array utf8) {
014        this(utf8.getBytes());
015    }
016    public Identifier(byte [] utf8) {
017        super(verify(utf8));
018    }
019    private static byte [] verify(final byte [] utf8) {
020        if (utf8.length > 0) {
021            verifyFirst(utf8[0], utf8);
022            for (int i=1; i < utf8.length; i++) {
023                verifyAny(utf8[i], utf8);
024            }
025        }
026        return utf8;
027
028    }
029    private static boolean verifyFirst(byte c, byte [] identifier) {
030        if (!((c == '_') || ((c >= 'a') && (c <= 'z')) || ((c >= 'A') && (c <= 'Z')))) {
031            throw new IllegalArgumentException("Illegal starting character '" + (char)c + "' of identifier '" + new Utf8String(new Utf8Array(identifier)).toString() +"'.");
032        }
033        return true;
034    }
035    private static boolean verifyAny(byte c, byte [] identifier) {
036        if (!((c == '_') || ((c >= 'a') && (c <= 'z')) || ((c >= 'A') && (c <= 'Z')) || ((c >= '0') && (c <= '9')))) {
037            throw new IllegalArgumentException("Illegal character '" + (char)c + "' of identifier '" + new Utf8String(new Utf8Array(identifier)).toString() +"'.");
038        }
039        return true;
040    }
041}
