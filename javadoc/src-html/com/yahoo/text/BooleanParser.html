004/**
005 * Utility class parsing a string into a boolean.
006 * In contrast to Boolean.parseBoolean in the Java API this parser is strict.
007 *
008 * @author bratseth
009 */
010public class BooleanParser {
011
012    /**
013     * Returns true if the input string is case insensitive equal to "true" and
014     * false if it is case insensitive equal to "false".
015     * In any other case an exception is thrown.
016     *
017     * @param  s the string to parse
018     * @return true if s is "true", false if it is "false"
019     * @throws IllegalArgumentException if s is not null but neither "true" or "false"
020     * @throws NullPointerException if s is null
021     */
022    public static boolean parseBoolean(String s) {
023        if (s==null)
024            throw new NullPointerException("Expected 'true' or 'false', got NULL");
025        if (s.equalsIgnoreCase("false"))
026            return false;
027        if (s.equalsIgnoreCase("true"))
028            return true;
029        throw new IllegalArgumentException("Expected 'true' or 'false', got '" + s + "'");
030    }
031
032}
