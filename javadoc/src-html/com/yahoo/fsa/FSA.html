024    /**
025     * Thread local state object used to traverse a Finite-State Automaton.
026     */
027    public static class State {
028
029        FSA fsa;
030        int state = 0;
031        int hash = 0;
032
033        private State(FSA fsa) {
034            this.fsa = fsa;
035            start();
036        }
037
038        public void start(){
039            state = fsa.start();
040            hash = 0;
041        }
042
043        public void delta(byte symbol) {
044            hash += fsa.hashDelta(state,symbol);
045            state = fsa.delta(state,symbol);
046        }
047
048        /** Returns whether the given symbol would take us to a valid state, without changing the state */
049        public boolean peekDelta(byte symbol) {
050            return fsa.delta(state,symbol)!=0;
051        }
052
053        public boolean tryDelta(byte symbol) {
054            int lastHash=hash;
055            int lastState=state;
056            delta(symbol);
057            if (isValid()) return true;
058
059            hash=lastHash;
060            state=lastState;
061            return false;
062        }
063
064        public void delta(char chr){
065            CharBuffer chrbuf = CharBuffer.allocate(1);
066            chrbuf.put(0,chr);
067            ByteBuffer buf = fsa.encode(chrbuf);
068            while(state >0 && buf.position()<buf.limit()){
069                delta(buf.get());
070            }
071        }
072
073        /** Jumps ahead by string */
074        public void delta(String string){
075            ByteBuffer buf = fsa.encode(string);
076            while(state >0 && buf.position()<buf.limit()){
077                delta(buf.get());
078            }
079        }
080
081        /**
082         * Jumps ahead by string if that puts us into a valid state, does nothing otherwise
083         *
084         * @return whether we jumped to a valid state (true) or di nothing (false)
085         */
086        public boolean tryDelta(String string){
087            int lastHash=hash;
088            int lastState=state;
089            delta(string);
090            if (isValid()) return true;
091
092            hash=lastHash;
093            state=lastState;
094            return false;
095        }
096
097        /** Jumps ahead by a word -  if this is not the first word, it must be preceeded by space. */
098        public void deltaWord(String string){
099            if (state!=fsa.start()) {
100                delta((byte)' ');
101            }
102            delta(string);
103        }
104
105        /**
106         * Tries to jump ahead by one word. If the given string is not the next complete valid word, nothing is done.
107         */
108        public boolean tryDeltaWord(String string){
109            int lastHash=hash;
110            int lastState=state;
111            tryDelta((byte)' ');
112            delta(string);
113            if (isValid() && peekDelta((byte)' ')) return true;
114            if (isFinal()) return true;
115
116            hash=lastHash;
117            state=lastState;
118            return false;
119        }
120
121        public boolean isFinal(){
122            return fsa.isFinal(state);
123        }
124
125        public boolean isStartState() {
126            return fsa.start() == state;
127        }
128
129        public boolean isValid(){
130            return state !=0;
131        }
132
133        public ByteBuffer data(){
134            return fsa.data(state);
135        }
136
137        public String dataString(){
138            return fsa.dataString(state);
139        }
140
141        public int hash(){
142            return hash;
143        }
144
145        public ByteBuffer lookup(String str){
146            start();
147            delta(str);
148            return fsa.data(state);
149        }
150
151        public boolean hasPerfectHash(){
152            return fsa.hasPerfectHash();
153        }
154
155    }

157    /**
158     * Class used to iterate over all accepted strings in the fsa.
159     */
160    public static class Iterator implements java.util.Iterator<Iterator.Item> {
161        /**
162         * Internally, this class stores the state information for the iterator.
163         * Externally, it is used for accessing the data associated with the iterator position.
164         */
165        public static class Item {
166            private FSA fsa;
167            private java.util.Stack<Byte> string;
168            private int symbol;
169            private int state;
170            private java.util.Stack<Integer> stack;
171
172            /**
173             * Constructor
174             * @param fsa the FSA object the iterator is associated with.
175             * @param state the state used as start state.
176             */
177            public Item(FSA fsa, int state) {
178                this.fsa = fsa;
179                this.string = new java.util.Stack();
180                this.symbol = 0;
181                this.state = state;
182                this.stack = new java.util.Stack();
183            }
184
185            /**
186             * Copy constructor. (Does not copy the state stack)
187             */
188            public Item(Item item) {
189                this.fsa = item.fsa;
190                this.string = new java.util.Stack();
191                for (java.util.Iterator<Byte> itr = item.string.iterator(); itr.hasNext(); ) {
192                    byte b = itr.next();
193                    this.string.push(b);
194                }
195                this.symbol = item.symbol;
196                this.state = item.state;
197                // no need to fill the stack as this constructor is used by Iterator::next()
198                this.stack = null;
199            }
200
201            public String getString() {
202                ByteBuffer buffer = ByteBuffer.allocate(string.size());
203                for (java.util.Iterator<Byte> itr = string.iterator(); itr.hasNext(); ) {
204                    byte b = itr.next();
205                    buffer.put(b);
206                }
207                buffer.flip();
208                return fsa.decode(buffer);
209            }
210
211            public ByteBuffer getData() {
212                return fsa.data(state);
213            }
214
215            public String getDataString() {
216                return fsa.dataString(state);
217            }
218
219            public String toString() {
220                return "string: " + string + "(" + getString() + "), symbol: " + symbol + ", state: " + state;
221            }
222        }
223
224        private Item item;
225        boolean useInitState = false;
226
227        /**
228         * Constructor.
229         * @param state the state to create the iterator from.
230         */
231        public Iterator(State state) {
232            item = new Item(state.fsa, state.state);
233            if (state.isFinal()) {
234                useInitState = true;
235            } else {
236                findNext();
237            }
238        }
239
240        private void findNext() {
241            int nextState;
242            int depth;
243
244            if (item.symbol == 256 || item.fsa == null) {
245                throw new NoSuchElementException();
246            }
247
248            // flip the flag now that the first state has been returned
249            if (useInitState) {
250                useInitState = false;
251            }
252
253            // try to find the next final state
254            for(;;) {
255                item.symbol++;
256                if (item.symbol < 256) {
257                    byte symbol = (byte)item.symbol;
258                    nextState = item.fsa.delta(item.state, (byte)item.symbol);
259                    if (nextState != 0) {
260                        item.string.push((byte)item.symbol);
261                        item.stack.push(item.state);
262                        item.state = nextState;
263                        item.symbol = 0;
264                        if (item.fsa.isFinal(nextState)) {
265                            break;
266                        }
267                    }
268                } else { // backtrack
269                    if ((depth = item.string.size()) > 0) {
270                        byte b = item.string.pop(); // remove the last byte
271                        item.symbol = b < 0 ? b + 256 : b;
272                        item.state = item.stack.pop();
273                    } else {
274                        item.state = 0;
275                        break;
276                    }
277                }
278            }
279        }
280
281        public boolean hasNext() {
282            return item.state != 0 || useInitState;
283        }
284
285        public Item next() {
286            Item retval = new Item(item);
287            findNext();
288            return retval;
289        }
290
291        public void remove() {
292            throw new UnsupportedOperationException();
293        }
294    }

323    /**
324     * Loads an FSA from a resource file name, which is resolved from the class path of the
325     * class loader of the given class.
326     * <p>
327     * This is useful for loading fsa's deployed within OSGi bundles.
328     *
329     * @param  resourceFileName the name of the file, relative to any element on the classpath.
330     *         For example, if the classpath contains resources/ and the file is resources/myfsa.fsa
331     *         this argument should be myfsa.fsa
332     * @param  loadingClass a class which provides the class loader to use for loading. Any class which is loaded
333     *         from the same class path as the resource will do (e.g with OSGi - any class in the same bundle jar)
334     * @return the loaded FSA
335     * @throws RuntimeException if the class could not be loaded
336     */
337    public static FSA loadFromResource(String resourceFileName,Class loadingClass) {
338        URL fsaUrl=loadingClass.getResource(resourceFileName);
339        if ( ! "file".equals(fsaUrl.getProtocol()))
340            throw new RuntimeException("Could not open non-file url '" + fsaUrl + "' as a file input stream: " +
341                                       "The classloader of " + loadingClass + "' does not return file urls");
342        return new FSA(fsaUrl.getFile());
343    }

373    private void init(String filename, String charsetname){
374        try {
375            init(new FileInputStream(filename),charsetname);
376        }
377        catch (FileNotFoundException e) {
378            throw new IllegalArgumentException("Could not find FSA file '" + filename + "'",e);
379        }
380        catch (IOException e) {
381            throw new IllegalArgumentException("Could not read FSA file '" + filename + "'",e);
382        }
383    }
384
385    private void init(FileInputStream file, String charsetname) {
386        try {
387            _charset = Charset.forName(charsetname);
388
389            _header = file.getChannel().map(MapMode.READ_ONLY,0,256);
390            _header.order(ByteOrder.LITTLE_ENDIAN);
391            if (h_magic()!=2038637673) {
392                throw new IOException("Stream does not contain an FSA: Wrong file magic number " + h_magic());
393            }
394            _symbol_tab = file.getChannel().map(MapMode.READ_ONLY,
395                    256,h_size());
396            _symbol_tab.order(ByteOrder.LITTLE_ENDIAN);
397            _state_tab = file.getChannel().map(MapMode.READ_ONLY,
398                    256+h_size(),4*h_size());
399            _state_tab.order(ByteOrder.LITTLE_ENDIAN);
400            _data = file.getChannel().map(MapMode.READ_ONLY,
401                    256+5*h_size(),h_data_size());
402            _data.order(ByteOrder.LITTLE_ENDIAN);
403            if(h_has_phash()>0){
404                _phash = file.getChannel().map(MapMode.READ_ONLY,
405                        256+5*h_size()+h_data_size(),
406                        4*h_size());
407                _phash.order(ByteOrder.LITTLE_ENDIAN);
408            }
409            _ok=true;
410        }
411        catch (IOException e) {
412            throw new RuntimeException("IO error while reading FSA file",e);
413        }
414    }

496    protected int delta(int state, byte symbol){
497        int s=symbol;
498        if(s<0){
499            s+=256;
500        }
501        if(_ok && s>0 && s<255){
502            if(getSymbol(state+s)==s){
503                return _state_tab.getInt(4*(state+s));
504            }
505        }
506        return 0;
507    }
508
509    protected int hashDelta(int state, byte symbol){
510        int s=symbol;
511        if(s<0){
512            s+=256;
513        }
514        if(_ok && h_has_phash()==1 && s>0 && s<255){
515            if(getSymbol(state+s)==s){
516                return _phash.getInt(4*(state+s));
517            }
518        }
519        return 0;
520    }
521
522    protected boolean isFinal(int state){
523        if(_ok){
524            if(getSymbol(state+255)==255){
525                return true;
526            }
527        }
528        return false;
529    }

531    /**
532     * Retrieves data for the given state using the underlying fsa data buffer.
533     * @param state The fsa state to retrieve data from.
534     * @return A new buffer containing the data for the given state.
535     **/
536    protected ByteBuffer data(int state) {
537        if(_ok && isFinal(state)){
538            int offset = _state_tab.getInt(4*(state+255));
539            int length;
540            if(h_data_type()==1){
541                length = h_fixed_data_size();
542            }
543            else{
544                length = _data.getInt(offset);
545                offset += 4;
546            }
547            ByteBuffer meta = ByteBuffer.allocate(length);
548            meta.order(ByteOrder.LITTLE_ENDIAN);
549            byte[] dst = meta.array();
550            for (int i = 0; i < length; ++i) {
551                dst[i] = _data.get(i + offset);
552            }
553            return meta;
554        }
555        return null;
556    }
557
558    /**
559     * Retrieves data for the given state using the underlying fsa data buffer.
560     * @param state The fsa state to retrieve data from.
561     * @return A string representation of the data for the given state.
562     **/
563    protected String dataString(int state) {
564        ByteBuffer meta = data(state);
565        if(meta!=null){
566            // Remove trailing '\0' if it exists. This is usually the
567            // case for automata built with text format (makefsa -t)
568            String data = decode(meta);
569            if (data.endsWith("\0")) {
570                data = data.substring(0, data.length()-1);
571            }
572            return data;
573        }
574        return null;
575    }

577    /**
578     * Convenience method that returns the metadata string in the fsa
579     * for the input lookup String, or null if the input string does
580     * not exist in the fsa.
581     * @param str The string to look up.
582     * @return Metadata string from the fsa.  */
583    public String lookup(String str){
584        State s = getState();
585        s.lookup(str);
586        return s.dataString();
