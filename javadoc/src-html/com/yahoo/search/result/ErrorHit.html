002package com.yahoo.search.result;
003
004import java.util.Iterator;
005import java.util.Set;
006
007/**
008 * A hit which holds information on error conditions in a result.
009 * En error hit maintains a main error - the main error of the result.
010 *
011 * @author bratseth
012 */
013public interface ErrorHit extends Cloneable {
014
015    void setSource(String source);
016
017    /** Returns the main error of this result, never null */
018    @Deprecated // use: errors().iterator().next()
019    ErrorMessage getMainError();
020
021    /**
022     * Adds an error to this. This may change the main error
023     * and/or the list of detailed errors
024     */
025    void addError(ErrorMessage error);
026
027    /** Add all errors from another error hit to this */
028    void addErrors(ErrorHit errorHit);
029
030    /**
031     * Returns all the detail errors of this error hit, including the main error
032     */
033    Iterator<? extends ErrorMessage> errorIterator();
034
035    /** Returns a read-only set containing all the error of this, including the main error */
036    Set<ErrorMessage> errors();
037
038    /** Returns true - this is a meta hit containing information on other hits */
039    boolean isMeta();
040
041    /** Returns true if main error is the given error code or if main error
042        is general error 8 and all suberrors are the given error code */
043    boolean hasOnlyErrorCode(int code);
044
045    Object clone();
046
047}
