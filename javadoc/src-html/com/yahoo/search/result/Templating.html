002package com.yahoo.search.result;
003
004import java.util.Map;
005
006import com.yahoo.container.jdisc.HttpRequest;
007import com.yahoo.prelude.templates.SearchRendererAdaptor;
008import com.yahoo.prelude.templates.TemplateSet;
009import com.yahoo.prelude.templates.UserTemplate;
010import com.yahoo.processing.rendering.Renderer;
011import com.yahoo.search.Result;
012import com.yahoo.search.query.Presentation;
013
014/**
015 * Helper methods and data store for result attributes geared towards result
016 * rendering and presentation.
017 *
018 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
019 */
020public class Templating {
021
022    private final Result result;
023    private Renderer<Result> renderer;
024
025    public Templating(Result result) {
026        super();
027        this.result = result;
028    }
029
030    /**
031     * Returns The first hit presented in the result as an index into the global
032     * list of all hits generated by the user query.
033     */
034    public int getFirstHitNo() {
035        return result.getQuery().getOffset() + 1;
036    }
037
038    /**
039     * Returns the first hit of the next result page, 0 if there aren't any more
040     * hits available
041     */
042    public long getNextFirstHitNo() {
043        if (result.getQuery().getHits() > result.getConcreteHitCount()) {
044            return 0;
045        }
046
047        return Math.min(getLastHitNo() + 1, result.getTotalHitCount());
048    }
049
050    /**
051     * Returns the first hit of the next result page, 0 if there aren't any more
052     * hits available
053     */
054    public long getNextLastHitNo() {
055        if (result.getQuery().getHits() > result.getConcreteHitCount()) {
056            return 0;
057        }
058
059        return Math.min(getLastHitNo() + result.getConcreteHitCount(), result.getTotalHitCount());
060    }
061
062    /**
063     * Returns the number of the last result of the current hit page.
064     */
065    public int getLastHitNo() {
066        return getFirstHitNo() + result.getConcreteHitCount() - 1;
067    }
068
069    /**
070     * The first hit presented on the previous result page as an index into the
071     * global list of all hits generated by the user query
072     */
073    public int getPrevFirstHitNo() {
074        return Math.max(getFirstHitNo() - result.getQuery().getHits(), 1);
075    }
076
077    /**
078     * The last hit presented on the previous result page as an index into the
079     * global list of all hits generated by the user query
080     */
081    public int getPrevLastHitNo() {
082        return Math.max(getFirstHitNo() - 1, 0);
083    }
084
085    /**
086     * An URL that may be used to obtain the next result page.
087     */
088    public String getNextResultURL() {
089        HttpRequest request = result.getQuery().getHttpRequest();
090        StringBuilder nextURL = new StringBuilder();
091
092        nextURL.append(getPath(request)).append("?");
093        parametersExceptOffset(request, nextURL);
094
095        int offset = getLastHitNo();
096
097        nextURL.append("&").append("offset=").append(Integer.toString(offset));
098        return nextURL.toString();
099    }
100
101    /**
102     * An URL that may be used to obtain the previous result page.
103     */
104    public String getPreviousResultURL() {
105        HttpRequest request = result.getQuery().getHttpRequest();
106        StringBuilder prevURL = new StringBuilder();
107
108        prevURL.append(getPath(request)).append("?");
109        parametersExceptOffset(request, prevURL);
110        int offset = getPrevFirstHitNo() - 1;
111        prevURL.append("&").append("offset=").append(Integer.toString(offset));
112        return prevURL.toString();
113    }
114
115    public String getCurrentResultURL() {
116        HttpRequest request = result.getQuery().getHttpRequest();
117        StringBuilder thisURL = new StringBuilder();
118
119        thisURL.append(getPath(request)).append("?");
120        parameters(request, thisURL);
121        return thisURL.toString();
122    }
123
124    private String getPath(HttpRequest request) {
125        String path = request.getUri().getPath();
126        if (path == null) {
127            path = "";
128        }
129        return path;
130    }
131
132    private void parametersExceptOffset(HttpRequest request, StringBuilder nextURL) {
133        int startLength = nextURL.length();
134        for (Map.Entry<String, String> property : request.propertyMap().entrySet()) {
135            if (property.getKey().equals("offset")) continue;
136
137            if (nextURL.length() > startLength)
138                nextURL.append("&");
139            nextURL.append(property.getKey()).append("=").append(property.getValue());
140        }
141    }
142
143    private void parameters(HttpRequest request, StringBuilder nextURL) {
144        int startLength = nextURL.length();
145        for (Map.Entry<String, String> property : request.propertyMap().entrySet()) {
146            if (nextURL.length() > startLength)
147                nextURL.append("&");
148            nextURL.append(property.getKey()).append("=").append(property.getValue());
149        }
150    }
151
152    /**
153     * Returns the templates which will render the result. This is never null.
154     * If default rendering is used, it is a TemplateSet containing no
155     * templates.
156     */
157    @SuppressWarnings("rawtypes")
158    public UserTemplate getTemplates() {
159        if (renderer == null) {
160            return TemplateSet.getDefault();
161        } else if (renderer instanceof SearchRendererAdaptor) {
162            return ((SearchRendererAdaptor) renderer).getAdaptee();
163        } else {
164            throw new RuntimeException(
165                    "Please use getTemplate() instead of getTemplates() when using the new template api.");
166        }
167    }
168
169    /**
170     * Sets the template set which should render this result set
171     *
172     * @param templates
173     *                the templates which should render this result, or null to
174     *                use the default xml rendering
175     */
176    @SuppressWarnings("deprecation")
177    public void setTemplates(@SuppressWarnings("rawtypes") UserTemplate templates) {
178        if (templates == null) {
179            setTemplates(TemplateSet.getDefault());
180        } else {
181            setRenderer(new SearchRendererAdaptor(templates));
182        }
183    }
184
185    /**
186     * @deprecated since 5.1.21, use {@link Presentation#getRenderer()}
187     */
188    @Deprecated // OK Do not remove on Vespa 6. Remove when we move everything having to do with templates
189    public Renderer<Result> getRenderer() {
190        return renderer;
191    }
192
193    /**
194     * @deprecated since 5.1.21, use {@link Presentation#setRenderer(com.yahoo.component.ComponentSpecification)}
195     */
196    @Deprecated // OK Do not remove on Vespa 6. Remove when we move everything having to do with templates
197    public void setRenderer(Renderer<Result> renderer) {
198        this.renderer = renderer;
199    }
200
201    /**
202     * For internal use only.
203     */
204    public boolean usesDefaultTemplate() {
205        return renderer == null ||
206                (renderer instanceof SearchRendererAdaptor &&
207                ((SearchRendererAdaptor) renderer).getAdaptee().isDefaultTemplateSet());
208    }
