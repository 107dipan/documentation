002package com.yahoo.search.result;
003
004import com.yahoo.search.query.Sorting;
005
006import java.util.Collections;
007import java.util.Comparator;
008import java.util.List;
009
010/**
011 * A hit orderer which can be assigned to a HitGroup to keep that group's
012 * hit sorted in accordance with the sorting specification given when this is created.
013 *
014 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
015 */
016public class HitSortOrderer extends HitOrderer {
017
018    private final Comparator<Hit> fieldComparator;
019
020    /** Create a sort order from a sorting */
021    public HitSortOrderer(Sorting sorting) {
022        fieldComparator =
023                new MetaHitsFirstComparator(
024                        new HitGroupsLastComparator(
025                                new FieldComparator(sorting)));
026    }
027
028    /**
029     * Create a sort order from a comparator.
030     * This will be appended to the standard comparators used by this.
031     */
032    public HitSortOrderer(Comparator<Hit> comparator) {
033        fieldComparator = new MetaHitsFirstComparator(new HitGroupsLastComparator(comparator));
034    }
035
036    /**
037     * Orders the given list of hits according to the sorting given at construction
038     *
039     * Meta hits are sorted before concrete hits, but have no internal
040     * ordering. The sorting is stable.
041     */
042    public void order(List<Hit> hits) {
043        Collections.sort(hits, fieldComparator);
044    }
045
046    public Comparator<Hit> getComparator() {
047        return fieldComparator;
048    }
049
050}
