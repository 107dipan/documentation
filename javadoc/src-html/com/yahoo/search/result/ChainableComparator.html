002package com.yahoo.search.result;
003
004import java.util.Comparator;
005
006/**
007 * Superclass of hit comparators which delegates comparisons of hits which are
008 * equal according to this comparator, to a secondary comparator.
009 *
010 * @author bratseth
011 */
012public abstract class ChainableComparator implements Comparator<Hit> {
013
014    private final Comparator<Hit> secondaryComparator;
015
016    /** Creates this comparator, given a secondary comparator, or null if there is no secondary */
017    public ChainableComparator(Comparator<Hit> secondaryComparator) {
018        this.secondaryComparator=secondaryComparator;
019    }
020
021    /** Returns the comparator to use to compare hits which are equal according to this, or null if none */
022    public Comparator<Hit> getSecondaryComparator() { return secondaryComparator; }
023
024    /**
025     * Returns the comparison form the secondary comparison, or 0 if the secondary is null.
026     * When overriding this in the subclass, always <code>return super.compare(first,second)</code>
027     * at the end of the subclass' implementation.
028     */
029    public int compare(Hit first,Hit second) {
030        if (secondaryComparator==null) return 0;
031        return secondaryComparator.compare(first,second);
032    }
033
034}
