002package com.yahoo.search.result;
003
004import java.util.Comparator;
005
006/**
007 * Ensures that meta hits are sorted before normal hits. All meta hits are
008 * considered equal.
009 *
010 * @author tonytv
011 */
012public class MetaHitsFirstComparator extends ChainableComparator {
013
014    public MetaHitsFirstComparator(Comparator<Hit> secondaryComparator) {
015        super(secondaryComparator);
016    }
017
018    @Override
019    public int compare(Hit left, Hit right) {
020        if (left.isMeta() && right.isMeta()) {
021            return 0;
022        } else if (left.isMeta()) {
023            return -1;
024        } else if (right.isMeta()) {
025            return 1;
026        } else {
027            return super.compare(left, right);
028        }
029    }
030
031    @Override
032    public String toString() {
033        return getSecondaryComparator().toString();
034    }
035}
