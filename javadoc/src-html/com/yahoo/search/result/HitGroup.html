022/**
023 * <p>A group of ordered hits. Since hitGroup is itself a kind of Hit,
024 * this can compose hierarchies of grouped hits.</p>
025 *
026 * <p>Group hits has a relevancy just as other hits - they can be ordered
027 * between each other and in comparison to other hits.
028 *
029 * <p>Note that a group is by default a meta hit, but it can also contain its own content
030 * in addition to subgroup content, in which case it should be set to non-meta.</p>
031 *
032 * @author bratseth
033 */
034public class HitGroup extends Hit implements DataList<Hit>, Cloneable, Iterable<Hit> {
035
036    // This does its own book-keeping of its various state variables
037    // (see methods towards the end). For state variables which are recursive
038    // (depending on the state of hits in subgroups), the strategy is to do
039    // book-keeping on only this immediate level, but not do recursive calls to
040    // find the true recursive state when queried. This is sort of a middle ground
041    // between handling the complexity of recursive state book-keeping and the
042    // query cost of not doing any book-keeping.
043    // There is also a method, analyse which recursively updates the recursive
044    // state of the group and all subgroups. This should be called if the hits
045    // may have changed their own state in a way that may impact the recursive
046    // state of this.
047
048    private ListenableArrayList<Hit> hits = new ListenableArrayList<>(16);
049
050    transient private List<Hit> unmodifiableHits = Collections.unmodifiableList(hits);
051

053    private boolean hitsSorted = true;
054
055    /** Whether or not deletion of hits breaks the sorted ordering */
056    private boolean deletionBreaksOrdering = false;
057
058    /** Whether the hits should be sorted (again) */
059    private boolean orderedHits = false;

216     * Adds a hit to this group.
217     * If the given hit is an ErrorHit and this group already have an error hit,
218     * the errors in the given hit are merged into the errors of this.
219     *
220     * @return the resulting hit - this is usually the input hit, but if an error hit was added,
221     *         and there was already an error hit present, that hit, containing the merged information
222     *         is returned
223     */
224    @Override
225    public Hit add(Hit hit) {
226        if (hit.isMeta() && hit instanceof ErrorHit) {
227            boolean add = mergeErrors((ErrorHit) hit);
228            if (!add) return (Hit)errorHit;
229        }
230        handleNewHit(hit);
231        hits.add(hit);
232        return hit;
233    }

265    /**
266     * Returns the hit with the given id, or null if there is no hit with this id
267     * in this group or any subgroup.
268     * This method is o(min(number of nested hits in this result,depth)).
269     *
270     * @param id the id of the hit to return from this or any nested group
271     * @param depth the max depth to recurse into nested groups: -1: Recurse infinitely deep, 0: Only look at hits in
272     *        the list of this group, 1: Look at hits in this group, and the hits of any immediate nested HitGroups,
273     *        etc.
274     * @return The hit, or null if not found.
275     */
276    public Hit get(URI id, int depth) {
277        updateHits();
278        for (Iterator<Hit> i = unorderedIterator(); i.hasNext();) {
279            Hit hit = i.next();
280            URI hitUri = hit.getId();
281
282            if (hitUri != null && hitUri.equals(id)) {
283                return hit;
284            }
285
286            if (hit instanceof HitGroup && depth!=0) {
287                Hit found=((HitGroup)hit).get(id,depth-1);
288                if (found!=null) return found;
289            }
290        }
291        return null;
292    }

294    /**
295     * Inserts the given hit at the specified index in this group.
296     */
297    public void set(int index, Hit hit) {
298        updateHits();
299        if (hit instanceof ErrorHit) { // Merge instead
300            add(hit);
301            return;
302        }
303
304        handleNewHit(hit);
305        Hit oldHit = hits.set(index, hit);
306
307        if (oldHit!=null)
308            handleRemovedHit(oldHit);
309    }

311    /**
312     * Adds a hit to this group in the specified index,
313     * all existing hits on this index and higher will have their index
314     * increased by one.
315     * <b>Note:</b> If the group was sorted, it will still be considered sorted
316     * after this call.
317     */
318    public void add(int index, Hit hit) {
319        if (hit instanceof ErrorHit) { // Merge instead
320            add(hit);
321            return;
322        }
323
324        boolean wasSorted = hitsSorted;
325        handleNewHit(hit);
326        hits.add(index, hit);
327        hitsSorted = wasSorted;
328    }

340    /**
341     * Removes a hit from this group or any subgroup.
342     *
343     * @param uri The uri of the hit to remove.
344     * @return The hit removed, or null if not found.
345     */
346    public Hit remove(URI uri) {
347        for (Iterator<Hit> it = hits.iterator(); it.hasNext(); ) {
348            Hit hit = it.next();
349            if (uri.equals(hit.getId())) {
350                it.remove();
351                handleRemovedHit(hit);
352                return hit;
353            }
354            if (hit instanceof HitGroup) {
355                Hit removed = ((HitGroup)hit).remove(uri);
356                if (removed != null) {
357                    return removed;
358                }
359            }
360        }
361        return null;
362    }

404    /**
405     * Returns the first error in this result,
406     * or null if no searcher has produced an error AND the query doesn't contain an error
407     */
408    public ErrorMessage getError() {
409        // See updateHits if this method is changed
410        if (errorHit != null) {
411            return errorHit.errors().iterator().next();
412        }
413
414        if (getQuery() != null && getQuery().errors().size() != 0) {
415            updateHits();
416        } // Pull them over
417
418        if (errorHit == null) {
419            return null;
420        }
421
422        return errorHit.errors().iterator().next();
423    }

425    /**
426     * Handles the addition of a new error hit, whether or not we already have one
427     *
428     * @return true if this shouls also be added to the list of hits of this reslt
429     */
430    private boolean mergeErrors(ErrorHit newHit) {
431        if (errorHit == null) {
432            errorHit = newHit;
433            return true;
434        } else {
435            errorHit.addErrors(newHit);
436            return false;
437        }
438    }

440    /**
441     * Must be called before the list of hits, or anything dependent on the list of hits, is removed.
442     * Merges errors from the query if there is one set for this group
443     */
444    private void updateHits() {
445        if (getQuery()==null) return;
446
447        if (getQuery().errors().size() == 0) return;
448
449        if (errorHit == null) // Creates an error hit where the first error is "main"
450            add((Hit)createErrorHit(toSearchError(getQuery().errors().get(0))));
451
452        // Add the rest of the errors
453        for (int i=1; i<getQuery().errors().size(); i++)
454            errorHit.addError(toSearchError(getQuery().errors().get(i)));
455        getQuery().errors().clear(); // TODO: Really clear them from here?
456    }

468    /**
469     * Remove the first <code>offset</code> <i>concrete</i> hits in this group,
470     * and hits beyond <code>offset+numHits</code>
471     */
472    public void trim(int offset, int numHits) {
473        updateHits();
474        ensureSorted();
475
476        int highBound = numHits + offset; // Largest offset +1
477
478        int currentIndex = -1;
479
480        for (Iterator<Hit> i = hits.iterator(); i.hasNext();) {
481            Hit hit = i.next();
482
483            if (hit.isAuxiliary()) continue;
484
485            currentIndex++;
486            if (currentIndex < offset || currentIndex >= highBound) {
487                i.remove();
488                handleRemovedHit(hit);
489            }
490        }
491    }

552    /**
553     * Force hit sorting now.
554     * This is not normally useful because a group will stay sorted automatically,
555     * but it is in the case where
556     * the hits have changed their internal state in a way that should change ordering
557     */
558    public void sort() {
559        if (hitOrderer == null) {
560            Collections.sort(hits);
561            hitsSorted = true;
562        } else {
563            // This may or may not lead to a sorted result set, but
564            // it's a best effort
565            hitOrderer.order(hits);
566            if (likelyHitsHaveCorrectValueForSortFields()) {
567                hitsSorted = true;
568            }
569        }
570    }
571
572    private boolean likelyHitsHaveCorrectValueForSortFields() {
573        if (hitOrderer == null) {
574            return true;
575        } else {
576            Set<String> filledFields = getFilled();
577            return filledFields == null || !filledFields.isEmpty();
578        }
579    }

619    /**
620     * Returns true if all the hits recursively contained in this
621     * is cached
622     */
623    public @Override boolean isCached() {
624        if (notCachedCount<1) return true;
625        if (subgroupCount<1) return false; // No need to check below
626
627        // Else check recursively
628        for (Hit hit : hits) {
629            if (hit instanceof HitGroup) {
630                if (hit.isCached()) return true;
631            }
632        }
633        return false;
634    }

661    /** Ensures result invariants. Must be called when a hit is added to this result. */
662    private void handleNewHit(Hit hit) {
663        if (!hit.isAuxiliary())
664            concreteHitCount++;
665
666        if (hit.getAddNumber() < 0) {
667            hit.setAddNumber(size());
668        }
669
670        hitsSorted = false;
671        Set<String> hitFilled = hit.getFilled();
672
673        if (hitFilled != null) {
674            Set<String> filled = getFilledInternal();
675            if (filled == null) {
676                if (hitFilled.isEmpty()) {
677                    filled = null;
678                } else if (hitFilled.size() == 1) {
679                    filled = Collections.singleton(hitFilled.iterator().next());
680                } else {
681                    filled = new HashSet<>(hitFilled);
682                }
683                setFilledInternal(filled);
684            } else {
685                if (filled.size() == 1) {
686                    if ( ! hitFilled.contains(filled.iterator().next())) {
687                        filled = null; // No intersection
688                        setFilledInternal(filled);
689                    }
690                } else {
691                    filled.retainAll(hitFilled);
692                }
693            }
694        }
695
696        if (hit instanceof HitGroup) {
697            subgroupCount++;
698        }
699        if (!hit.isCached()) {
700            notCachedCount++;
701        }
702    }
703
704    // Filled is not kept in sync at removal
705    private void handleRemovedHit(Hit hit) {
706        if (!hit.isAuxiliary()) {
707            concreteHitCount--;
708            if (!hit.isCached())
709                notCachedCount--;
710        }
711        else if (hit instanceof HitGroup) {
712            subgroupCount--;
713        }
714
715        if (deletionBreaksOrdering) {
716            hitsSorted = false;
717        }
718    }

720    private void analyzeHit(Hit hit) {
721        if (hit instanceof HitGroup) {
722            ((HitGroup)hit).analyze();
723        }
724        if (!hit.isAuxiliary())
725            concreteHitCount++;
726
727        if (!hit.isCached())
728            notCachedCount++;
729    }

731    /**
732     * Update concreteHitCount, cached and filled by iterating trough the hits of this result.
733     * Recursively also update all subgroups.
734     */
735    public void analyze() {
736        concreteHitCount=0;
737        setFilledInternal(null);
738        notCachedCount=0;
739        Set<String> filled = getFilledInternal();
740
741        Iterator<Hit> i = unorderedIterator();
742        while (filled == null && i.hasNext()) {
743            Hit hit = i.next();
744            analyzeHit(hit);
745            Set<String> hitFilled = hit.getFilled();
746            if (hitFilled != null) {
747                filled = (hitFilled.size() == 1)
748                        ? Collections.singleton(hitFilled.iterator().next())
749                        : hitFilled.isEmpty() ? null : new HashSet<>(hitFilled);
750                setFilledInternal(filled);
751            }
752        }
753        String singleKey = null;
754        if (filled != null && filled.size() == 1) {
755            singleKey = filled.iterator().next();
756        }
757
758
759        for (; i.hasNext();) {
760            Hit hit = i.next();
761            analyzeHit(hit);
762
763            if (filled != null) {
764                Set<String> hitFilled = hit.getFilled();
765                if (hitFilled == null) {
766                    // Intentionally empty. Strange semantic, null -> matches everything
767                } else if (hitFilled.isEmpty()) {
768                    filled = null; // No intersection
769                    setFilledInternal(filled);
770                } else {
771                    if (filled.size() == 1) {
772                        if ( ! hitFilled.contains(singleKey)) {
773                            filled = null; // No intersection
774                            setFilledInternal(filled);
775                            singleKey = null;
776                        }
777                    } else {
778                        filled.retainAll(hitFilled);
779                        if (filled.size() == 1) {
780                            singleKey = filled.iterator().next();
781                        }
782                    }
783                }
784            }
785        }
786    }

788    public HitGroup clone() {
789        HitGroup hitGroupClone = (HitGroup) super.clone();
790        hitGroupClone.hits = new ListenableArrayList<>(this.hits.size());
791        hitGroupClone.unmodifiableHits = Collections.unmodifiableList(hitGroupClone.hits);
792        for (Iterator<Hit> i = this.hits.iterator(); i.hasNext();) {
793            Hit hitClone = i.next().clone();
794            hitGroupClone.hits.add(hitClone);
795        }
796        if (this.errorHit!=null) { // Find the cloned error and assign it
797            for (Hit hit : hitGroupClone.asList()) {
798                if (hit instanceof ErrorHit)
799                    hitGroupClone.errorHit=(ErrorHit)hit;
800            }
801        }
802
803        if (this.getFilledInternal()!=null) {
804            hitGroupClone.setFilledInternal(new HashSet<>(this.getFilledInternal()));
805        }
806
807        return hitGroupClone;
808    }

822    @Override
823    public Set<String> getFilled() {
824        Iterator<Hit> hitIterator = hits.iterator();
825        Set<String> firstSummaryNames = getSummaryNamesNextFilledHit(hitIterator);
826        if (firstSummaryNames == null || firstSummaryNames.isEmpty())
827            return firstSummaryNames;
828
829        Set<String> intersection = firstSummaryNames;
830        while (true) {
831            Set<String> summaryNames = getSummaryNamesNextFilledHit(hitIterator);
832            if (summaryNames == null)
833                break;
834
835            if (intersection.size() == 1)
836                return getFilledSingle(first(intersection), hitIterator);
837
838
839            boolean notInSet = false;
840            if (intersection == firstSummaryNames) {
841                if (intersection.size() == summaryNames.size()) {
842                    for(String s : summaryNames) {
843                        if ( ! intersection.contains(s)) {
844                            intersection = new HashSet<>(firstSummaryNames);
845                            notInSet = true;
846                            break;
847                        }
848                    }
849                }
850            }
851            if (notInSet) {
852                intersection.retainAll(summaryNames);
853            }
854
855        }
856
857        return intersection;
858    }

860    private Set<String> getSummaryNamesNextFilledHit(Iterator<Hit> hitIterator) {
861        while (hitIterator.hasNext()) {
862            Set<String> filled = hitIterator.next().getFilled();
863            if (filled != null)
864                return filled;
865        }
866        return null;
867    }
868
869    private Set<String> getFilledSingle(String summaryName, Iterator<Hit> hitIterator) {
870        while (true) {
871            Set<String> summaryNames = getSummaryNamesNextFilledHit(hitIterator);
872            if (summaryNames == null) {
873                return Collections.singleton(summaryName);
874            } else if (!summaryNames.contains(summaryName)) {
875                return Collections.emptySet();
876            }
877        }
878    }
