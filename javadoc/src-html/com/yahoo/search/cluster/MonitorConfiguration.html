002package com.yahoo.search.cluster;
003
004/**
005 * The configuration of a cluster monitor instance
006 *
007 * @author bratseth
008 */
009public class MonitorConfiguration  {
010
011    /**
012     * The interval in ms between consecutive checks of the monitored
013     * nodes
014     */
015    private long checkInterval=1000;
016
017    /**
018     * The number of times a failed node must respond before getting
019     * traffic again
020     */
021    private int responseAfterFailLimit=3;
022
023    /**
024     * The number of ms a node is allowed to stay idle before it is
025     * pinged
026     */
027    private long idleLimit=3000;
028
029    /**
030     * The number of milliseconds to attempt to complete a request
031     * before giving up
032     */
033    private long requestTimeout = 5000;
034
035    /**
036     * The number of milliseconds a node is allowed to fail before we
037     * mark it as not working
038     */
039    private long failLimit=5000;
040
041    /**
042     * The number of times a node is allowed to fail in one hour
043     * before it is quarantined for an hour
044     */
045    private int failQuarantineLimit=3;
046
047    /**
048     * The number of ms to quarantine an unstable node
049     */
050    private long quarantineTime=1000*60*60;
051
052    /**
053     * Sets the interval between each ping of idle or failing nodes
054     * Default is 1000ms
055     */
056    public void setCheckInterval(long intervalMs) {
057        this.checkInterval=intervalMs;
058    }
059
060    /**
061     * Returns the interval between each ping of idle or failing nodes
062     * Default is 1000ms
063     */
064    public long getCheckInterval() {
065        return checkInterval;
066    }
067
068    /**
069     * Sets the number of times a failed node must respond before it is put
070     * back in service. Default is 3.
071     */
072    public void setResponseAfterFailLimit(int responseAfterFailLimit) {
073        this.responseAfterFailLimit=responseAfterFailLimit;
074    }
075
076    /**
077     * Sets the number of ms a node (failing or working) is allowed to
078     * stay idle before it is pinged. Default is 3000
079     */
080    public void setIdleLimit(int idleLimit) {
081        this.idleLimit=idleLimit;
082    }
083
084    /**
085     * Gets the number of ms a node (failing or working)
086     * is allowed to stay idle before it is pinged. Default is 3000
087     */
088    public long getIdleLimit() {
089        return idleLimit;
090    }
091
092    /**
093     * Returns the number of milliseconds to attempt to service a request
094     * (at different nodes) before giving up. Default is 5000 ms.
095     */
096    public long getRequestTimeout() { return requestTimeout; }
097
098    /**
099     * Sets the number of milliseconds a node is allowed to fail before we
100     * mark it as not working
101     */
102    public void setFailLimit(long failLimit) { this.failLimit=failLimit; }
103
104    /**
105     * Returns the number of milliseconds a node is allowed to fail before we
106     * mark it as not working
107     */
108    public long getFailLimit() { return failLimit; }
109
110    /**
111     * The number of times a node must fail in one hour to be placed
112     * in quarantine.  Once in quarantine it won't be put back in
113     * productuion before quarantineTime has expired even if it is
114     * working. Default is 3
115     */
116    public void setFailQuarantineLimit(int failQuarantineLimit) {
117        this.failQuarantineLimit=failQuarantineLimit;
118    }
119
120    /**
121     * The number of ms an unstable node is quarantined. Default is
122     * 100*60*60
123     */
124    public void setQuarantineTime(long quarantineTime) {
125        this.quarantineTime=quarantineTime;
126    }
127
128    public String toString() {
129        return "monitor configuration [" +
130            "checkInterval: " + checkInterval +
131            " responseAfterFailLimit: " + responseAfterFailLimit +
132            " idleLimit: " + idleLimit +
133            " requestTimeout " + requestTimeout +
134            " feilLimit " + failLimit +
135            " failQuerantineLimit " + failQuarantineLimit +
136            " quarantineTime " + quarantineTime +
137            "]";
138    }
139
140}
