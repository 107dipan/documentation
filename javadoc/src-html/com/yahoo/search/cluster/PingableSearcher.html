002package com.yahoo.search.cluster;
003
004import com.yahoo.component.ComponentId;
005import com.yahoo.prelude.Ping;
006import com.yahoo.prelude.Pong;
007import com.yahoo.search.Searcher;
008import com.yahoo.search.searchchain.Execution;
009
010/**
011 * A searcher to which we can send a ping to probe if it is alive
012 *
013 * @author bratseth
014 */
015public abstract class PingableSearcher extends Searcher {
016
017    public PingableSearcher() {
018    }
019
020    public PingableSearcher(ComponentId id) {
021        super(id);
022    }
023
024    /** Send a ping request downwards to probe if this searcher chain is in functioning order */
025    public Pong ping(Ping ping, Execution execution) {
026        return execution.ping(ping);
027    }
028
029}
