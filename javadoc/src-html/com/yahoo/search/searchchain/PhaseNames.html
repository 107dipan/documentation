002package com.yahoo.search.searchchain;
003
004/**
005 * Helper class for ordering searchers. Searchers may use these names in their
006 * {@literal @}Before and {@literal @}After annotations, though in general
007 * a searcher should depend on some explicit functionality, not these
008 * checkpoints.
009 *
010 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
011 */
012public final class PhaseNames {
013    private PhaseNames() {
014    }
015
016    /**
017     * A checkpoint where the query is not yet transformed in any way. RAW_QUERY
018     * is the first checkpoint not provided by some searcher.
019     */
020    public static final String RAW_QUERY = "rawQuery";
021
022    /**
023     * A checkpoint where as many query transformers as practically possible has
024     * been run. TRANSFORMED_QUERY is the first checkpoint after RAW_QUERY.
025     */
026    public static final String TRANSFORMED_QUERY = "transformedQuery";
027
028    /**
029     * A checkpoint where results from different backends have been flattened
030     * into a single result. BLENDED_RESULT is the first checkpoint after
031     * TRANSFORMED_QUERY.
032     */
033    public static final String BLENDED_RESULT = "blendedResult";
034
035    /**
036     * A checkpoint where data from different backends are not yet merged.
037     * UNBLENDED_RESULT is the first checkpoint after BLENDED_RESULT.
038     */
039    public static final String UNBLENDED_RESULT = "unblendedResult";
040
041    /**
042     * The last checkpoint in a search chain not provided by any searcher.
043     */
044    public static final String BACKEND = "backend";
045
046}
