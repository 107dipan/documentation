002package com.yahoo.search.searchchain;
003
004import com.yahoo.component.ComponentId;
005import com.yahoo.component.chain.Chain;
006import com.yahoo.search.Searcher;
007
008import java.util.Collection;
009
010/**
011 * Searchers which invokes other search chains should override this.
012 *
013 * @author bratseth
014 */
015public abstract class ForkingSearcher extends Searcher {
016
017    public ForkingSearcher() {}
018
019    /** A search chain with a comment about when it is used. */
020    public static class CommentedSearchChain {
021        public final String comment;
022        public final Chain<Searcher> searchChain;
023
024        public CommentedSearchChain(String comment, Chain<Searcher> searchChain) {
025            this.comment = comment;
026            this.searchChain = searchChain;
027        }
028    }
029
030    /** Returns which searchers this searcher may forward to, for debugging and tracing */
031    public abstract Collection<CommentedSearchChain> getSearchChainsForwarded(SearchChainRegistry registry);
032
033}
