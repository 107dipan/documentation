024/**
025 * <p>An execution of a search chain. This keeps track of the call state for an execution (in the calling thread)
026 * of the searchers of a search chain.</p>
027 *
028 * <p>To execute a search chain, simply do
029 * <pre>
030 *     Result result = new Execution(mySearchChain, execution.context()).search(query)
031 * </pre>
032 *
033 *
034 * <p>See also {@link AsyncExecution}, which performs an execution in a separate thread than the caller.</p>
035 *
036 * <p>Execution instances should not be reused for multiple separate executions.</p>
037 *
038 * @author bratseth
039 */
040public class Execution extends com.yahoo.processing.execution.Execution {
041
042    public static final String ATTRIBUTEPREFETCH = "attributeprefetch";

044    /**
045     * The execution context is the search chain's current view of the indexes,
046     * search chain registrys, etc. Searcher instances may set values here to
047     * change the behavior of the rest of the search chain.
048     * <p>
049     * The Context class simply carries a set of objects which define the
050     * environment for the search. <b>Important:</b> All objects available through context need to
051     * be either truly immutable or support the freeze pattern.
052     * <p>
053     * If you are implementing a searcher where you need to create a new Context
054     * instance to create an Execution, you should use the context from the
055     * execution the searcher was invoked from. You can also copy
056     * (Context.shallowCopy()) the incoming context if it is necessary to do
057     * more. In other words, a minimal example would be:<br>
058     * new Execution(searchChain, execution.context())
059     */
060    public static final class Context {

169        /**
170         * Populate missing values in this from the given context.
171         * Values which are non-null in this will not be overwritten.
172         *
173         * @param sourceContext the context from which to get the parameters
174         */
175        public void populateFrom(Context sourceContext) {
176            // breakdown and detailedDiagnostics has no unset state, so they are always copied
177            detailedDiagnostics = sourceContext.detailedDiagnostics;
178            breakdown = sourceContext.breakdown;
179            if (indexFacts == null) {
180                indexFacts = sourceContext.indexFacts;
181            }
182            if (tokenRegistry == null) {
183                tokenRegistry = sourceContext.tokenRegistry;
184            }
185            if (searchChainRegistry == null) {
186                searchChainRegistry = sourceContext.searchChainRegistry;
187            }
188            if (rendererRegistry == null) {
189                rendererRegistry = sourceContext.rendererRegistry;
190            }
191            if (linguistics == null) {
192                linguistics = sourceContext.linguistics;
193            }
194        }
195
196        /**
197         * The brutal version of populateFrom().
198         *
199         * @param other a Context instance this will copy all state from
200         */
201        void fill(Context other) {
202            searchChainRegistry = other.searchChainRegistry;
203            indexFacts = other.indexFacts;
204            tokenRegistry = other.tokenRegistry;
205            rendererRegistry = other.rendererRegistry;
206            detailedDiagnostics = other.detailedDiagnostics;
207            breakdown = other.breakdown;
208            linguistics = other.linguistics;
209        }

211        public boolean equals(Context other) {
212            // equals() needs to be cheap, that's yet another reason we can only
213            // allow immutables and frozen objects in the context
214            return other.indexFacts == indexFacts
215                    && other.rendererRegistry == rendererRegistry
216                    && other.tokenRegistry == tokenRegistry
217                    && other.searchChainRegistry == searchChainRegistry
218                    && other.detailedDiagnostics == detailedDiagnostics
219                    && other.breakdown == breakdown
220                    && other.linguistics == linguistics;
221        }

223        @Override
224        public boolean equals(Object other) {
225            if (other == null) {
226                return false;
227            }
228            if (other.getClass() != Context.class) {
229                return false;
230            } else {
231                return equals((Context) other);
232            }
233        }

236         * Standard shallow copy, the new instance will carry the same
237         * references as this.
238         *
239         * @return a new instance which is a shallow copy
240         */
241        public Context shallowCopy() {
242            Context c = new Context();
243            c.fill(this);
244            return c;
245        }

247        /**
248         * This is used when building the Context stack. If Context has been
249         * changed since last time, build a new object. Otherwise simply return
250         * the previous snapshot.
251         *
252         * @param previous another Context instance to compare with
253         * @return a copy of this, or previous
254         */
255        Context copyIfChanged(Context previous) {
256            if (equals(previous)) {
257                return previous;
258            } else {
259                return shallowCopy();
260            }
261        }

427     * The usually best way of creating a new execution for a search chain. This
428     * is the one suitable for a production environment. It is safe to use the
429     * incoming context from the search directly:
430     *
431     * <pre>
432     * public Result search(Query query, Execution execution) {
433     *     SearchChain searchChain = fancyChainSelectionRoutine(query);
434     *     if (searchChain != null) {
435     *         return new Execution(searchChain, execution.context());

446     *            context is not changed nor retained by this), or null to not
447     *            populate from a context

460    /**
461     * Creates a new execution for a search chain or a single searcher. private
462     * to ensure only searchChain or searcher is null (and because it's long and
463     * cumbersome).
464     *
465     * @param searchChain
466     *            the search chain to execute, must be null if searcher is set
467     * @param context
468     *            execution context for the search
469     * @param searcherIndex
470     *            index of the first searcher to invoke, see
471     *            Execution(Execution)
472     * @throws IllegalArgumentException
473     *             if searchChain is null
474     */
475    @SuppressWarnings("unchecked")
476    private Execution(Chain<? extends Processor> searchChain,Context context, int searcherIndex) {
477        // Create a new Execution which is placed in the context of the execution of the given Context if any
478        // "if any" because a context may, or may not, belong to an execution.
479        // This is decided at the creation time of the Context - Context instances which do not belong
480        // to an execution plays the role of data carriers between executions.
481        super(searchChain,searcherIndex,context.createChildTrace(),context.createChildEnvironment());
482        this.context.fill(context);
483        contextCache = new Context[searchChain.components().size()];
484        entryIndex=searcherIndex;
485        timer = new TimeTracker(searchChain, searcherIndex);
486    }

494    /** Calls search on the next searcher in this chain. If there is no next, an empty result is returned. */
495    public Result search(Query query) {
496        timer.sampleSearch(nextIndex(), context.getDetailedDiagnostics());
497
498        // Transfer state between query and execution as the execution constructors does not do that completely
499        query.getModel().setExecution(this);
500        trace().setTraceLevel(query.getTraceLevel());
501
502        return (Result)super.process(query);
503    }
504
505    @Override
506    protected void onInvoking(Request request, Processor processor) {
507        super.onInvoking(request,processor);
508        final int traceDependencies = 6;
509        Query query = (Query) request;
510        if (query.getTraceLevel() >= traceDependencies) {
511            query.trace(new StringBuilder().append(processor.getId())
512                    .append(" ").append(processor.getDependencies().toString())
513                    .toString(), traceDependencies);
514        }
515    }

558    /** Calls fill on the next searcher in this chain. If there is no next, nothing is done. */
559    public void fill(Result result,String summaryClass) {
560        timer.sampleFill(nextIndex(), context.getDetailedDiagnostics());
561        Searcher next = (Searcher)next(); // TODO: Allow but skip processors which are not searchers
562        if (next==null) return;
563
564        try {
565            nextProcessor();
566            next.ensureFilled(result, summaryClass, this);
567        }
568        finally {
569            previousProcessor();
570            timer.sampleFillReturn(nextIndex(), context.getDetailedDiagnostics(), result);
571        }
572    }

574    /** Calls ping on the next search in this chain. If there is no next, a Pong is created and returned. */
575    public Pong ping(Ping ping) {
576        // return this reference, not directly. It's needed for adding time data
577        Pong annotationReference = null;
578
579        timer.samplePing(nextIndex(), context.getDetailedDiagnostics());
580        Searcher next = (Searcher)next(); // TODO: Allow but skip processors which are not searchers
581        if (next==null) {
582            annotationReference = new Pong();
583            return annotationReference;
584        }
585
586        try {
587            nextProcessor();
588            annotationReference = invokePing(ping, next);
589            return annotationReference;
590        }
591        finally {
592            previousProcessor();
593            timer.samplePingReturn(nextIndex(), context.getDetailedDiagnostics(), annotationReference);
594        }
595    }

597    @Override
598    protected void onReturning(Request request, Processor processor,Response response) {
599        super.onReturning(request, processor, response);
600        timer.sampleSearchReturn(nextIndex(), context.getDetailedDiagnostics(), (Result)response);
601    }

603    @Override
604    protected void previousProcessor() {
605        super.previousProcessor();
606        popContext();
607    }

609    @Override
610    protected void nextProcessor() {
611        pushContext();
612        super.nextProcessor();
613    }
614
615    private void popContext() {
616        context.fill(contextCache[nextIndex()]);
617        contextCache[nextIndex()] = null;
618    }
619
620    private void pushContext() {
621        final Context contextToPush;
622        // Do note: Never put this.context in the cache. It would be totally
623        // meaningless, since it's a final.
624        if (nextIndex() == entryIndex) {
625            contextToPush = context.shallowCopy();
626        } else {
627            contextToPush = context.copyIfChanged(contextCache[nextIndex() - 1]);
628        }
629        contextCache[nextIndex()] = contextToPush;
630    }
631
632    private Pong invokePing(Ping ping, Searcher next) {
633        Pong annotationReference;
634        if (next instanceof PingableSearcher) {
635            annotationReference = ((PingableSearcher) next).ping(ping, this);
636        } else {
637            annotationReference = ping(ping);
638        }
639        return annotationReference;
640    }
