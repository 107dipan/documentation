002package com.yahoo.search.query.properties;
003
004import com.yahoo.processing.request.CompoundName;
005import com.yahoo.search.query.Properties;
006
007import java.util.Map;
008
009/**
010 * Turns get(name) into get(name,request) using the request given at construction time.
011 * This is used to allow the query's request to be supplied to all property requests
012 * without forcing users of the query.properties() to supply this explicitly.
013 *
014 * @author bratseth
015 */
016public class RequestContextProperties extends Properties {
017
018    private final Map<String,String> requestMap;
019
020    public RequestContextProperties(Map<String, String> properties) {
021        this.requestMap=properties;
022    }
023
024    @Override
025    public Object get(CompoundName name,Map<String,String> context,
026                      com.yahoo.processing.request.Properties substitution) {
027        return super.get(name,context==null ? requestMap : context,substitution);
028    }
029
030    @Override
031    public void set(CompoundName name,Object value,Map<String,String> context) {
032        super.set(name,value,context==null ? requestMap : context);
033    }
034
035    @Override
036    public Map<String, Object> listProperties(CompoundName path,Map<String,String> context,
037                                              com.yahoo.processing.request.Properties substitution) {
038        return super.listProperties(path,context==null ? requestMap : context,substitution);
039    }
040
041}
