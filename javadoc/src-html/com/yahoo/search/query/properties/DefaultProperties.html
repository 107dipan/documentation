002package com.yahoo.search.query.properties;
003
004import com.yahoo.processing.request.CompoundName;
005import com.yahoo.search.query.Properties;
006import com.yahoo.search.query.profile.types.FieldDescription;
007import com.yahoo.search.query.profile.types.QueryProfileType;
008
009import java.util.Map;
010
011/**
012 * Default values for properties that are meant to be customized in query profiles.
013 * @author tonytv
014 */
015public final class DefaultProperties extends Properties  {
016    public static final CompoundName MAX_OFFSET = new CompoundName("maxOffset");
017    public static final CompoundName MAX_HITS = new CompoundName("maxHits");
018
019
020    public static final QueryProfileType argumentType = new QueryProfileType("DefaultProperties");
021    static {
022        argumentType.setBuiltin(true);
023
024        argumentType.addField(new FieldDescription(MAX_OFFSET.toString(), "integer"));
025        argumentType.addField(new FieldDescription(MAX_HITS.toString(), "integer"));
026
027        argumentType.freeze();
028    }
029
030    @Override
031    public Object get(CompoundName name, Map<String, String> context, com.yahoo.processing.request.Properties substitution) {
032        if (MAX_OFFSET.equals(name)) {
033            return 1000;
034        } else if (MAX_HITS.equals(name)) {
035            return 400;
036        } else {
037            return super.get(name, context, substitution);
038        }
039    }
040}
