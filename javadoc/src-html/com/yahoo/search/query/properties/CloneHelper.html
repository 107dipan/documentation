001package com.yahoo.search.query.properties;
002
003import com.yahoo.search.result.Hit;
004
005/**
006 * Extends com.yahoo.processing.request.CloneHelper with fastpath for
007 *  - com.yahoo.search.result.Hit
008 *
009 * Created by balder on 13/02/2017.
010 */
011public class CloneHelper extends com.yahoo.processing.request.CloneHelper {
012    @Override
013    protected Object objectClone(Object object) {
014        if (object instanceof Hit) {
015            return ((Hit)object).clone();
016        }
017        return super.objectClone(object);
018    }
019}
