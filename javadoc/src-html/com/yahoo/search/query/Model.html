122    /**
123     * Gets the language to use for parsing. If this is explicitly set in the model, that language is returned.
124     * Otherwise, if a query tree is already produced and any node in it specifies a language the first such 
125     * node encountered in a depth first 
126     * left to right search is returned. Otherwise the language is guessed from the query string. 
127     * If this does not yield an actual language, English is returned as the default.
128     *
129     * @return the language determined, never null
130     */
131    // TODO: We can support multiple languages per query by changing searchers which call this
132    //       to look up the query to use at each point from item.getLanguage
133    //       with this as fallback for query branches where no parent item specifies language
134    public Language getParsingLanguage(String languageDetectionText) {
135        Language language = getLanguage();
136        if (language != null) return language;
137
138        language = Language.fromEncoding(encoding);
139        if (language != Language.UNKNOWN) return language;
140
141        if (queryTree != null)
142            language = languageBelow(queryTree);
143        if (language != Language.UNKNOWN) return language;
144        
145        Linguistics linguistics = execution.context().getLinguistics();
146        if (linguistics != null)
147            language = linguistics.getDetector().detect(languageDetectionText, null).getLanguage(); // TODO: Set language if detected
148        if (language != Language.UNKNOWN) return language;
149
150        return Language.ENGLISH;
151    }
152    
153    private Language languageBelow(Item item) {
154        if (item.getLanguage() != Language.UNKNOWN) return item.getLanguage();
155        if (item instanceof CompositeItem) {
156            for (Iterator<Item> i = ((CompositeItem) item).getItemIterator(); i.hasNext(); ) {
157                Language childLanguage = languageBelow(i.next());
158                if (childLanguage != Language.UNKNOWN) return childLanguage;
159            }
160        }
161        return Language.UNKNOWN;
162    }

256    public QueryTree getQueryTree() {
257        if (queryTree == null) {
258            Parser parser = ParserFactory.newInstance(type, ParserEnvironment.fromExecutionContext(execution.context()));
259            queryTree = parser.parse(Parsable.fromQueryModel(this));
260            if (parent.getTraceLevel() >= 2) {
261                parent.trace("Query parsed to: " + parent.yqlRepresentation(), 2);
262            }
263        }
264        return queryTree;
265    }

313    public boolean equals(Object o) {
314        if ( ! (o instanceof Model)) return false;
315
316        Model other = (Model) o;
317        if ( ! (
318                QueryHelper.equals(other.encoding, this.encoding) &&
319                QueryHelper.equals(other.language, this.language) &&
320                QueryHelper.equals(other.searchPath, this.searchPath) &&
321                QueryHelper.equals(other.sources, this.sources) &&
322                QueryHelper.equals(other.restrict, this.restrict) &&
323                QueryHelper.equals(other.defaultIndex, this.defaultIndex) &&
324                QueryHelper.equals(other.type, this.type) ))
325            return false;
326
327        if (other.queryTree == null && this.queryTree == null) // don't cause query parsing
328            return QueryHelper.equals(other.queryString, this.queryString) &&
329                   QueryHelper.equals(other.filter, this.filter);
330        else // make sure we compare a parsed variant of both
331            return QueryHelper.equals(other.getQueryTree(), this.getQueryTree());
332    }

341    public Object clone() {
342        try {
343            Model clone = (Model) super.clone();
344            if (queryTree != null)
345                clone.queryTree = this.queryTree.clone();
346            if (sources !=null)
347                clone.sources = new LinkedHashSet<>(this.sources);
348            if (restrict !=null)
349                clone.restrict = new LinkedHashSet<>(this.restrict);
350            return clone;
351        }
352        catch (CloneNotSupportedException e) {
353            throw new RuntimeException("Someone inserted a noncloneable superclass",e);
354        }
355    }

404    /** Sets the execution working on this. For internal use. */
405    public void setExecution(Execution execution) {
406        if (execution==this.execution) return;
407
408        // If not already coupled, bind the trace of the new execution into the existing execution trace
409        if (execution.trace().traceNode().isRoot()
410            && execution.trace().traceNode() != this.execution.trace().traceNode().root()) {
411            this.execution.trace().traceNode().add(execution.trace().traceNode());
412        }
413
414        this.execution = execution;
415    }

448    private void prepareRankFeaturesFromModel(Ranking ranking) {
449        Item root = getQueryTree().getRoot();
450        if (root != null) {
451            List<Item> tagged = setUniqueIDs(root);
452            addLabels(tagged, ranking);
453            addConnectivityRankProperties(tagged, ranking);
454            addSignificances(tagged, ranking);
455        }
456    }
457
458    private List<Item> setUniqueIDs(Item root) {
459        List<Item> items = new ArrayList<>();
460        collectTaggableItems(root, items);
461        int id = 1;
462        for (Item i : items) {
463            TaggableItem t = (TaggableItem) i;
464            t.setUniqueID(id++);
465        }
466        return items;
467    }
468
469    private void addLabels(List<Item> candidates, Ranking ranking) {
470        for (Item candidate : candidates) {
471            String label = candidate.getLabel();
472            if (label != null) {
473                String name = "vespa.label." + label + ".id";
474                TaggableItem t = (TaggableItem) candidate;
475                ranking.getProperties().put(name, String.valueOf(t.getUniqueID()));
476            }
477        }
478    }
479
480    private void addConnectivityRankProperties(List<Item> connectedItems, Ranking ranking) {
481        for (Item link : connectedItems) {
482            TaggableItem t = (TaggableItem) link;
483            Item connectedTo = t.getConnectedItem();
484            if (connectedTo != null && strictContains(connectedTo, connectedItems)) {
485                TaggableItem t2 = (TaggableItem) connectedTo;
486                String name = "vespa.term." + t.getUniqueID() + ".connexity";
487                ranking.getProperties().put(name, String.valueOf(t2.getUniqueID()));
488                ranking.getProperties().put(name, String.valueOf(t.getConnectivity()));
489            }
490        }
491    }
492
493    private void addSignificances(List<Item> candidates, Ranking ranking) {
494        for (Item  candidate : candidates) {
495            TaggableItem t = (TaggableItem) candidate;
496            if ( ! t.hasExplicitSignificance()) continue;
497            String name = "vespa.term." + t.getUniqueID() + ".significance";
498            ranking.getProperties().put(name, String.valueOf(t.getSignificance()));
499        }
500    }
501
502    private void collectTaggableItems(Item root, List<Item> terms) {
503        if (root == null) return;
504
505        if (root instanceof TaggableItem) {
506            // This is tested before descending, as phrases are viewed
507            // as leaf nodes in the ranking code in the backend
508            terms.add(root);
509        } else if (root instanceof CompositeItem) {
510            CompositeItem c = (CompositeItem) root;
511            for (Iterator<Item> i = c.getItemIterator(); i.hasNext();) {
512                collectTaggableItems(i.next(), terms);
513            }
514        } else {} // nop
515    }
516
517    private boolean strictContains(Object needle, Collection<?> haystack) {
518        for (Object pin : haystack)
519            if (pin == needle) return true;
520        return false;
521    }
