002package com.yahoo.search.query.profile;
003
004import com.yahoo.processing.request.CompoundName;
005import com.yahoo.search.query.profile.types.FieldType;
006import com.yahoo.search.query.properties.PropertyMap;
007
008/**
009 * A map which stores all types which cannot be stored in a query profile
010 * that is rich model objects.
011 * <p>
012 * This map will deep copy not only the model object map, but also each
013 * clonable member in the map.
014 *
015 * @author bratseth
016 */
017public class ModelObjectMap extends PropertyMap {
018
019    /** Returns true if the class of the value is not acceptable as a query profile value */
020    @Override
021    protected boolean shouldSet(CompoundName name,Object value) {
022        if (value==null) return true;
023        return FieldType.fromClass(value.getClass())==null;
024    }
025
026}
