019/**
020 * A query profile is a data container with an id and a class (type). More precisely, it contains
021 * <ul>
022 *   <li>An id, on the form name:version, where the version is optional, and follows the same rules as for other search container components.
023 *   <li>A class id referring to the class defining this profile (see Query Profile Classes below)
024 *   <li>A (possibly empty) list of ids of inherited query profiles
025 *   <li>A (possibly empty) list of declarative predicates over search request parameters which defines when this query profile is applicable (see Query Profile Selection below)
026 *   <li>The data content, which consists of
027 *       <ul>
028 *         <li>named values
029 *         <li>named references to other profiles
030 *       </ul>
031 * </ul>
032 *
033 * This serves the purpose of an intermediate format between configuration and runtime structures - the runtime
034 * structure used is QueryProfileProperties.
035 *
036 * @author bratseth
037 */
038public class QueryProfile extends FreezableSimpleComponent implements Cloneable {

113    /** Adds a profile to the end of the inherited list of this for the given variant. Throws an exception if this is frozen. */
114    public void addInherited(QueryProfile profile, DimensionValues dimensionValues) {
115        ensureNotFrozen();
116
117        DimensionBinding dimensionBinding=DimensionBinding.createFrom(getDimensions(),dimensionValues);
118        if (dimensionBinding.isNull()) {
119            if (inherited==null)
120                inherited=new ArrayList<>();
121            inherited.add(profile);
122        }
123        else {
124            if (variants==null)
125                variants=new QueryProfileVariants(dimensionBinding.getDimensions(), this);
126            variants.inherit(profile,dimensionBinding.getValues());
127        }
128    }

225    /**
226     * Adds all objects that start with the given path prefix to the given value map. Use "" to list all.
227     * <p>
228     * For example, if {a.d =&gt; "a.d-value" ,a.e =&gt; "a.e-value", b.d =&gt; "b.d-value", then calling listValues("a")
229     * will return {"d" =&gt; "a.d-value","e" =&gt; "a.e-value"}
230     */
231    public Map<String, Object> listValues(CompoundName prefix, Map<String, String> context, Properties substitution) {
232        DimensionBinding dimensionBinding=DimensionBinding.createFrom(getDimensions(),context);
233
234        AllValuesQueryProfileVisitor visitor=new AllValuesQueryProfileVisitor(prefix);
235        accept(visitor,dimensionBinding, null);
236        Map<String,Object>  values=visitor.getResult();
237
238        if (substitution==null) return values;
239        for (Map.Entry<String,Object> entry : values.entrySet()) {
240            if (entry.getValue().getClass()==String.class) continue; // Shortcut
241            if (entry.getValue() instanceof SubstituteString)
242                entry.setValue(((SubstituteString)entry.getValue()).substitute(context,substitution));
243        }
244        return values;
245    }

389    /**
390     * Switches this from write-only to read-only mode.
391     * This profile can never be modified again after this method returns.
392     * Calling this on an already frozen profile has no effect.
393     * <p>
394     * Calling this will also freeze any profiles inherited and referenced by this.
395     */
396    // TODO: Remove/simplify as query profiles are not used at query time
397    public synchronized void freeze() {
398        if (isFrozen()) return;
399
400        resolvedDimensions=getDimensions();
401
402        if (variants !=null)
403            variants.freeze();
404
405        if (inherited!=null) {
406            for (QueryProfile inheritedProfile : inherited)
407                inheritedProfile.freeze();
408        }
409
410        content.freeze();
411
412        inherited= inherited==null ? ImmutableList.of() : ImmutableList.copyOf(inherited);
413
414        super.freeze();
415    }

422    /**
423     * Returns a clone of this. The clone will not be frozen and will contain copied inherited and content collections
424     * pointing to the same values as this.
425     */
426    @Override
427    public QueryProfile clone() {
428        if (isFrozen()) return this;
429        QueryProfile clone=(QueryProfile)super.clone();
430        if (variants !=null)
431            clone.variants = variants.clone();
432        if (inherited!=null)
433            clone.inherited=new ArrayList<>(inherited);
434
435        if (this.content!=null)
436            clone.content=content.clone();
437
438        return clone;
439    }

495        if (type!=null) return type.isOverridable(localName);
496        return true;

514    void acceptAndEnter(String key, QueryProfileVisitor visitor,DimensionBinding dimensionBinding, QueryProfile owner) {
515        boolean allowContent=visitor.enter(key);
516        accept(allowContent, visitor, dimensionBinding, owner);
517        if (allowContent)
518            visitor.leave(key);

521    /**
522     * Visit the profiles and values referenced from this in order of decreasing precedence
523     *
524     * @param allowContent whether content in this should be visited
525     * @param visitor the visitor
526     * @param dimensionBinding the dimension binding to use
527     */
528    final void accept(boolean allowContent,QueryProfileVisitor visitor, DimensionBinding dimensionBinding, QueryProfile owner) {
529        visitor.onQueryProfile(this, dimensionBinding, owner);
530        if (visitor.isDone()) return;
531
532        visitVariants(allowContent,visitor,dimensionBinding);
533        if (visitor.isDone()) return;
534
535        if (allowContent) {
536            visitContent(visitor,dimensionBinding);
537            if (visitor.isDone()) return;
538        }
539
540        if (visitor.visitInherited())
541            visitInherited(allowContent, visitor, dimensionBinding, owner);
542    }
543
544    protected void visitVariants(boolean allowContent,QueryProfileVisitor visitor,DimensionBinding dimensionBinding) {
545        if (getVariants()!=null)
546            getVariants().accept(allowContent, getType(), visitor, dimensionBinding);
547    }
548
549    protected void visitInherited(boolean allowContent,QueryProfileVisitor visitor,DimensionBinding dimensionBinding, QueryProfile owner) {
550        if (inherited==null) return;
551        for (QueryProfile inheritedProfile : inherited) {
552            inheritedProfile.accept(allowContent,visitor,dimensionBinding.createFor(inheritedProfile.getDimensions()), owner);
553            if (visitor.isDone()) return;
554        }
555    }
556
557    private void visitContent(QueryProfileVisitor visitor,DimensionBinding dimensionBinding) {
558        String contentKey=visitor.getLocalKey();
559
560        // Visit this' content
561        if (contentKey!=null) { // Get only the content of the current key
562            if (type!=null)
563                contentKey=type.unalias(contentKey);
564            visitor.acceptValue(contentKey, getContent(contentKey), dimensionBinding, this);
565        }
566        else { // get all content in this
567            for (Map.Entry<String,Object> entry : getContent().entrySet()) {
568                visitor.acceptValue(entry.getKey(), entry.getValue(), dimensionBinding, this);
569                if (visitor.isDone()) return;
570            }
571        }
572    }

598     * be added (usually because the new value was added to the existing).
599     */
600    static Object combineValues(Object newValue, Object existingValue) {
601        if (newValue instanceof QueryProfile) {
602            QueryProfile newProfile=(QueryProfile)newValue;
603            if ( existingValue==null || ! (existingValue instanceof QueryProfile)) {
604                if (!isModifiable(newProfile))
605                    newProfile=new BackedOverridableQueryProfile(newProfile); // Make the query profile reference overridable
606                newProfile.value=existingValue;
607                return newProfile;
608            }
609
610            // if both are profiles:
611            return combineProfiles(newProfile,(QueryProfile)existingValue);
612        }
613        else {
614            if (existingValue instanceof QueryProfile) { // we need to set a non-leaf value on a query profile
615                QueryProfile existingProfile=(QueryProfile)existingValue;
616                if (isModifiable(existingProfile)) {
617                    existingProfile.setValue(newValue);
618                    return null;
619                }
620                else {
621                    QueryProfile existingOverridable = new BackedOverridableQueryProfile((QueryProfile)existingValue);
622                    existingOverridable.setValue(newValue);
623                    return existingOverridable;
624                }
625            }
626            else {
627                return newValue;
628            }
629        }
630    }
631
632    private static QueryProfile combineProfiles(QueryProfile newProfile,QueryProfile existingProfile) {
633        QueryProfile returnValue=null;
634        QueryProfile existingModifiable;
635
636        // Ensure the existing profile is modifiable
637        if (existingProfile.getClass()==QueryProfile.class) {
638            existingModifiable = new BackedOverridableQueryProfile(existingProfile);
639            returnValue=existingModifiable;
640        }
641        else { // is an overridable wrapper
642            existingModifiable=existingProfile; // May be used as-is
643        }
644
645        // Make the existing profile inherit the new one
646        if (existingModifiable instanceof BackedOverridableQueryProfile)
647            ((BackedOverridableQueryProfile)existingModifiable).addInheritedHere(newProfile);
648        else
649            existingModifiable.addInherited(newProfile);
650
651        // Remove content from the existing which the new one does not allow overrides of
652        if (existingModifiable.content!=null) {
653            for (String key : existingModifiable.content.unmodifiableMap().keySet()) {
654                if ( ! newProfile.isLocalOverridable(key, null)) {
655                    existingModifiable.content.remove(key);
656                }
657            }
658        }
659
660        return returnValue;
661    }

663    /** Returns whether the given profile may be modified from this profile */
664    private static boolean isModifiable(QueryProfile profile) {
665        if (profile.isFrozen()) return false;
666        if ( ! profile.isExplicit()) return true; // Implicitly defined from this - ok to modify then
667        if (! (profile instanceof BackedOverridableQueryProfile)) return false;
668        return true;
669    }

671    /**
672     * Converts to the type of the receiving field, if possible and necessary.
673     *
674     * @return the value to be assigned: the original or a converted value
675     * @throws IllegalArgumentException if the assignment is illegal
676     */
677    protected Object checkAndConvertAssignment(String localName, Object value, QueryProfileRegistry registry) {
678        if (type==null) return value; // no type checking
679
680        FieldDescription fieldDescription=type.getField(localName);
681        if (fieldDescription==null) {
682            if (type.isStrict())
683                throw new IllegalArgumentException("'" + localName + "' is not declared in " + type + ", and the type is strict");
684            return value;
685        }
686
687        if (registry == null && (fieldDescription.getType() instanceof QueryProfileFieldType))
688            throw new IllegalArgumentException("A registry was not passed: Query profile references is not supported");
689        Object convertedValue = fieldDescription.getType().convertFrom(value, registry);
690        if (convertedValue == null)
691            throw new IllegalArgumentException("'" + value + "' is not a " + fieldDescription.getType().toInstanceDescription());
692        return convertedValue;
693    }

704    /** Do a variant-aware content lookup in this */
705    protected Object localLookup(String name, DimensionBinding dimensionBinding) {
706        Object node=null;
707        if ( variants!=null && !dimensionBinding.isNull())
708            node=variants.get(name,type,true,dimensionBinding);
709        if (node==null)
710            node=content==null ? null : content.get(name);
711        return node;

716    private Boolean isDeclaredOverridable(CompoundName name,DimensionBinding dimensionBinding) {
717        QueryProfile parent= lookupParentExact(name, true, dimensionBinding);
718        if (parent.overridable==null) return null;
719        return parent.overridable.get(name.last());
720    }

722    /**
723     * Sets the overridability of a field in this profile,
724     * this overrides the corresponding setting in the type (if any)
725     */
726    private void setOverridable(CompoundName fieldName,boolean overridable,DimensionBinding dimensionBinding) {
727        QueryProfile parent= lookupParentExact(fieldName, true, dimensionBinding);
728        if (parent.overridable==null)
729            parent.overridable=new HashMap<>();
730        parent.overridable.put(fieldName.last(),overridable);
731    }

733    /** Sets a value to a (possibly non-local) node. The parent query profile holding the value is returned */
734    private void setNode(CompoundName name, Object value, QueryProfileType parentType,
735                         DimensionBinding dimensionBinding, QueryProfileRegistry registry) {
736        ensureNotFrozen();
737        if (name.isCompound()) {
738            QueryProfile parent= getQueryProfileExact(name.first(), true, dimensionBinding);
739            parent.setNode(name.rest(), value,parentType, dimensionBinding.createFor(parent.getDimensions()), registry);
740        }
741        else {
742            setLocalNode(name.toString(), value,parentType, dimensionBinding, registry);
743        }
744    }

746    /**
747     * Looks up and, if necessary, creates, the query profile which should hold the given local name portion of the
748     * given name. If the name contains no dots, this is returned.
749     *
750     * @param name the name of the variable to lookup the parent of
751     * @param create whether or not to create the parent if it is not present
752     * @return the parent, or null if not present and created is false
753     */
754    private QueryProfile lookupParentExact(CompoundName name, boolean create, DimensionBinding dimensionBinding) {
755        CompoundName rest = name.rest();
756        if (rest.isEmpty()) return this;
757
758        QueryProfile topmostParent = getQueryProfileExact(name.first(), create, dimensionBinding);
759        if (topmostParent == null) return null;
760        return topmostParent.lookupParentExact(rest, create, dimensionBinding.createFor(topmostParent.getDimensions()));
761    }

763    /**
764     * Returns a query profile from this by name
765     *
766     * @param localName the local name of the profile in this, this is never a compound
767     * @param create whether the profile should be created if missing
768     * @return the created profile, or null if not present, and create is false
769     */
770    private QueryProfile getQueryProfileExact(String localName, boolean create, DimensionBinding dimensionBinding) {
771        Object node=localExactLookup(localName, dimensionBinding);
772        if (node!=null && node instanceof QueryProfile) {
773            return (QueryProfile)node;
774        }
775        if (!create) return null;
776
777        QueryProfile queryProfile=createSubProfile(localName,dimensionBinding);
778        if (type != null) {
779            Class<?> legalClass=type.getValueClass(localName);
780            if (legalClass == null || ! legalClass.isInstance(queryProfile))
781                throw new RuntimeException("'" + localName + "' is not a legal query profile reference name in " + this);
782            queryProfile.setType(type.getType(localName));
783        }
784        localPut(localName,queryProfile,dimensionBinding);
785        return queryProfile;
786    }
787
788    /** Do a variant-aware content lookup in this - without looking in any wrapped content. But by matching variant bindings exactly only */
789    private Object localExactLookup(String name, DimensionBinding dimensionBinding) {
790        if (dimensionBinding.isNull()) return content == null ? null : content.get(name);
791        if (variants == null) return null;
792        QueryProfileVariant variant = variants.getVariant(dimensionBinding.getValues(),false);
793        if (variant == null) return null;
794        return variant.values().get(name);
795    }
796
797    /** Sets a value directly in this query profile (unless frozen) */
798    private void localPut(String localName,Object value, DimensionBinding dimensionBinding) {
799        ensureNotFrozen();
800
801        if (type != null)
802            localName = type.unalias(localName);
803
804        validateName(localName);
805        value = convertToSubstitutionString(value);
806
807        if (dimensionBinding.isNull()) {
808            Object combinedValue;
809            if (value instanceof QueryProfile)
810                combinedValue = combineValues(value,content==null ? null : content.get(localName));
811            else
812                combinedValue = combineValues(value, localLookup(localName, dimensionBinding));
813
814            if (combinedValue!=null)
815                content.put(localName,combinedValue);
816        }
817        else {
818            if (variants == null)
819                variants = new QueryProfileVariants(dimensionBinding.getDimensions(), this);
820            variants.set(localName,dimensionBinding.getValues(),value);
821        }
822    }
