011/**
012 * This class represent a set of query profiles virtually - rather
013 * than storing and instantiating each profile this structure represents explicitly only
014 * the values set in the various virtual profiles. The set of virtual profiles are defined by a set of
015 * <i>dimensions</i>. Values may be set for any point in this multi-dimensional space, and may also be set for
016 * any regular hyper-region by setting values for any point in certain of these dimensions.
017 * The set of virtual profiles defined by this consists of all the combinations of dimension points for
018 * which one or more values is set in this, as well as any possible less specified regions.
019 * <p>
020 * A set of virtual profiles are always owned by a single profile, which is also their parent
021 * in the inheritance hierarchy.
022 *
023 * @author bratseth
024 */
025public class QueryProfileVariants implements Freezable, Cloneable {
026
027    private boolean frozen=false;
028
029    /** Properties indexed by name, to support fast lookup of single values */
030    private Map<String,FieldValues> fieldValuesByName=new HashMap<>();
031
032    /** The inherited profiles for various dimensions settings - a set of fieldvalues of List&lt;QueryProfile&gt; */
033    private FieldValues inheritedProfiles=new FieldValues();
034
035    /**
036     * Field and inherited profiles sorted by specificity used for all-value visiting.
037     * This is the same as how the source data looks (apart from the sorting).
038     */
039    private List<QueryProfileVariant> variants=new ArrayList<>();
040
041    /**
042     * The names of the dimensions (which are possible properties in the context given on lookup) of this.
043     * Order matters - more specific values to the left in this list are more significant than more specific values
044     * to the right
045     */
046    private List<String> dimensions;
047
048    /** The query profile this variants of */
049    private QueryProfile owner;
050
051    /**
052     * Creates a set of virtual query profiles which may return varying values over the set of dimensions given.
053     * Each dimension is a name for which a key-value may be supplied in the context properties
054     * on lookup time to influence the value returned.
055     */
056    public QueryProfileVariants(String[] dimensions, QueryProfile owner) {
057        this(Arrays.asList(dimensions), owner);
058    }
059
060    /**
061     * Creates a set of virtual query profiles which may return varying values over the set of dimensions given.
062     * Each dimension is a name for which a key-value may be supplied in the context properties
063     * on lookup time to influence the value returned.
064     *
065     * @param dimensions the dimension names this may vary over. The list gets owned by this, so it must not be further
066     *        modified from outside). This will not modify the list.
067     */
068    public QueryProfileVariants(List<String> dimensions, QueryProfile owner) {
069        // Note: This is not made unmodifiable (here or in freeze) because we depend on map identity comparisons of this
070        // list (in dimensionBinding) for performance reasons.
071        this.dimensions = dimensions;
072        this.owner = owner;
073    }
074
075    /** Irreversibly prevents any further modifications to this */
076    public void freeze() {
077        if (frozen) return;
078        for (FieldValues fieldValues : fieldValuesByName.values())
079            fieldValues.freeze();
080        fieldValuesByName = ImmutableMap.copyOf(fieldValuesByName);
081        inheritedProfiles.freeze();
082
083        Collections.sort(variants);
084        for (QueryProfileVariant variant : variants)
085            variant.freeze();
086        variants = ImmutableList.copyOf(variants);
087
088        frozen=true;
089    }
090
091    @Override
092    public boolean isFrozen() {
093        return frozen;
094    }
095
096    /** Visits the most specific match to the dimension binding of each variable (or the one named by the visitor) */
097    void accept(boolean allowContent,QueryProfileType type,QueryProfileVisitor visitor,DimensionBinding dimensionBinding) {
098        String contentName=null;
099        if (allowContent)
100            contentName=visitor.getLocalKey();
101
102        if (contentName!=null) {
103            if (type!=null)
104                contentName=type.unalias(contentName);
105            acceptSingleValue(contentName,allowContent,visitor,dimensionBinding); // Special cased for performance
106        }
107        else {
108            acceptAllValues(allowContent,visitor,type,dimensionBinding);
109        }
110    }
111
112    // PERF: 90%
113    void acceptSingleValue(String name,boolean allowContent,QueryProfileVisitor visitor,DimensionBinding dimensionBinding) {
114        FieldValues fieldValues=fieldValuesByName.get(name);
115        if (fieldValues==null || !allowContent)
116            fieldValues=new FieldValues();
117
118        fieldValues.sort();
119        inheritedProfiles.sort();
120
121        int inheritedIndex=0;
122        int fieldIndex=0;
123        // Go through both the fields and the inherited profiles at the same time and try the single must specific pick
124        // from either of the lists at each step
125        while(fieldIndex<fieldValues.size() || inheritedIndex<inheritedProfiles.size()) { // PERF: 8% - fieldValues.size()
126            // Get the next most specific from field and inherited
127            FieldValue fieldValue=fieldValues.getIfExists(fieldIndex); // PERF: 11% - getIfExists
128            FieldValue inheritedProfileValue=inheritedProfiles.getIfExists(inheritedIndex); // PERF: 11% - getIfExists
129
130            // Try the most specific first, then the other
131            if (inheritedProfileValue==null || (fieldValue!=null && fieldValue.compareTo(inheritedProfileValue)<=0)) { // Field is most specific, or both are equally specific
132                if (fieldValue.matches(dimensionBinding.getValues())) { // PERF: 42% - matches, together with the other matches
133                    visitor.acceptValue(name, fieldValue.getValue(), dimensionBinding, owner);
134                }
135                if (visitor.isDone()) return;
136                fieldIndex++;
137            }
138            else if (inheritedProfileValue!=null) { // Inherited is most specific at this point
139                if (inheritedProfileValue.matches(dimensionBinding.getValues())) { // PERF: 42% - matches, together with the other matches
140                    @SuppressWarnings("unchecked")
141                    List<QueryProfile> inheritedProfileList=(List<QueryProfile>)inheritedProfileValue.getValue();
142                    for (QueryProfile inheritedProfile : inheritedProfileList) {
143                        if (visitor.visitInherited()) {
144                            inheritedProfile.accept(allowContent,visitor,dimensionBinding.createFor(inheritedProfile.getDimensions()), owner);
145                        }
146                        if (visitor.isDone()) return;
147                    }
148                }
149                inheritedIndex++;
150            }
151            if (visitor.isDone()) return;
152        }
153    }
154
155    void acceptAllValues(boolean allowContent,QueryProfileVisitor visitor, QueryProfileType type,DimensionBinding dimensionBinding) {
156        if (!frozen)
157            Collections.sort(variants);
158        for (QueryProfileVariant variant : variants) {
159            if (variant.matches(dimensionBinding.getValues()))
160                variant.accept(allowContent,type,visitor,dimensionBinding);
161            if (visitor.isDone()) return;
162        }
163    }

165    /**
166     * Returns the most specific matching value of a name for a given set of <b>canonical</b> dimension values.
167     *
168     * @param name the name to return the best matching value of
169     * @param dimensionBinding the dimension bindings to use in this
170     */
171    public Object get(String name, QueryProfileType type, boolean allowQueryProfileResult, DimensionBinding dimensionBinding) {
172        SingleValueQueryProfileVisitor visitor=new SingleValueQueryProfileVisitor(Collections.singletonList(name),allowQueryProfileResult);
173        visitor.enter("");
174        accept(true,type,visitor,dimensionBinding);
175        visitor.leave("");
176        return visitor.getResult();
177    }

179    /** Inherits a particular profile in a variant of this */
180    public void inherit(QueryProfile profile,DimensionValues dimensionValues) {
181        ensureNotFrozen();
182
183        // Update variant
184        getVariant(dimensionValues,true).inherit(profile);
185
186        // Update per-variable optimized structure
187        @SuppressWarnings("unchecked")
188        List<QueryProfile> inheritedAtDimensionValues=(List<QueryProfile>)inheritedProfiles.getExact(dimensionValues);
189        if (inheritedAtDimensionValues==null) {
190            inheritedAtDimensionValues=new ArrayList<>();
191            inheritedProfiles.put(dimensionValues,inheritedAtDimensionValues);
192        }
193        inheritedAtDimensionValues.add(profile);
194    }

196    /**
197     * Sets a value to this
198     *
199     * @param fieldName the name of the field to set. This cannot be a compound (dotted) name
200     * @param binding the dimension values for which this value applies.
201     *         The dimensions must be canonicalized, and ownership is transferred to this.
202     * @param value the value to set
203     */
204    /**
205     * Sets a value to this
206     *
207     * @param fieldName the name of the field to set. This cannot be a compound (dotted) name
208     * @param dimensionValues the dimension values for which this value applies
209     * @param value the value to set
210     */
211    public void set(String fieldName,DimensionValues dimensionValues,Object value) {
212        ensureNotFrozen();
213
214        // Update variant
215        getVariant(dimensionValues,true).set(fieldName,value);
216
217        // Update per-variable optimized structure
218        FieldValues fieldValues=fieldValuesByName.get(fieldName);
219        if (fieldValues==null) {
220            fieldValues=new FieldValues();
221            fieldValuesByName.put(fieldName,fieldValues);
222        }
223
224        Object combinedValue=QueryProfile.combineValues(value,fieldValues.getExact(dimensionValues));
225        if (combinedValue!=null)
226            fieldValues.put(dimensionValues,combinedValue);
227    }

229    /**
230     * Returns the dimensions over which the virtual profiles in this may return different values.
231     * Each dimension is a name for which a key-value may be supplied in the context properties
232     * on lookup time to influence the value returned.
233     * The dimensions may not be modified - the returned list is always read only.
234     */
235    // Note: A performance optimization in DimensionBinding depends on the identity of the list returned from this
236    public List<String> getDimensions() { return dimensions; }
237
238    /** Returns the map of field values of this indexed by field name. */
239    public Map<String,FieldValues> getFieldValues() { return fieldValuesByName; }
240
241    /** Returns the profiles inherited from various variants of this */
242    public FieldValues getInherited() { return inheritedProfiles; }
243
244    /**
245     * Returns all the variants of this, sorted by specificity. This is content as declared.
246     * The returned list is always unmodifiable.
247     */
248    public List<QueryProfileVariant> getVariants() {
249        if (frozen) return variants; // Already unmodifiable
250        return Collections.unmodifiableList(variants);
251    }
252
253    public QueryProfileVariants clone() {
254        try {
255            if (frozen) return this;
256            QueryProfileVariants clone=(QueryProfileVariants)super.clone();
257            clone.inheritedProfiles=inheritedProfiles.clone();
258
259            clone.variants=new ArrayList<>();
260            for (QueryProfileVariant variant : variants)
261                clone.variants.add(variant.clone());
262
263            clone.fieldValuesByName=new HashMap<>();
264            for (Map.Entry<String,FieldValues> entry : fieldValuesByName.entrySet())
265                clone.fieldValuesByName.put(entry.getKey(),entry.getValue().clone(entry.getKey(),clone.variants));
266
267            return clone;
268        }
269        catch (CloneNotSupportedException e) {
270            throw new RuntimeException(e);
271        }
272    }
273
274    /** Throws an IllegalStateException if this is frozen */
275    protected void ensureNotFrozen() {
276        if (frozen)
277            throw new IllegalStateException(this + " is frozen and cannot be modified");
278    }
279
280    /**
281     * Returns the query profile variant having exactly the given dimensions, and creates it if create is set and
282     * it is missing
283     *
284     * @param  dimensionValues the dimension values
285     * @param  create whether or not to create the variant if missing
286     * @return the profile variant, or null if not found and create is false
287     */
288    public QueryProfileVariant getVariant(DimensionValues dimensionValues,boolean create) {
289        for (QueryProfileVariant profileVariant : variants)
290            if (profileVariant.getDimensionValues().equals(dimensionValues))
291                return profileVariant;
292
293        // Not found
294        if (!create) return null;
295        QueryProfileVariant variant=new QueryProfileVariant(dimensionValues, owner);
296        variants.add(variant);
297        return variant;
298    }

300    public static class FieldValues implements Freezable, Cloneable {
301
302        private List<FieldValue> resolutionList=null;
303
304        private boolean frozen=false;
305
306        @Override
307        public void freeze() {
308            if (frozen) return;
309            sort();
310            if (resolutionList != null)
311                resolutionList = ImmutableList.copyOf(resolutionList);
312            frozen = true;
313        }
314
315        @Override
316        public boolean isFrozen() {
317            return frozen;
318        }
319
320        public void put(DimensionValues dimensionValues,Object value) {
321            ensureNotFrozen();
322            if (resolutionList==null) resolutionList=new ArrayList<>();
323            FieldValue fieldValue=getExactFieldValue(dimensionValues);
324            if (fieldValue!=null) // Replace
325                fieldValue.setValue(value);
326            else
327                resolutionList.add(new FieldValue(dimensionValues,value));
328        }
329
330        /** Returns the value having exactly the given dimensions, or null if none */
331        public Object getExact(DimensionValues dimensionValues) {
332            FieldValue value=getExactFieldValue(dimensionValues);
333            if (value==null) return null;
334            return value.getValue();
335        }
336
337        /** Returns the field value having exactly the given dimensions, or null if none */
338        private FieldValue getExactFieldValue(DimensionValues dimensionValues) {
339            for (FieldValue fieldValue : asList())
340                if (fieldValue.getDimensionValues().equals(dimensionValues))
341                    return fieldValue;
342            return null;
343        }
344
345        /** Returns the field values (values for various dimensions) for this field as a read-only list (never null) */
346        public List<FieldValue> asList() {
347            if (resolutionList==null) return Collections.emptyList();
348            return resolutionList;
349        }
350
351        public FieldValue getIfExists(int index) {
352            if (index>=size()) return null;
353            return resolutionList.get(index);
354        }
355
356        public void sort() {
357            if (frozen) return ; // sorted already
358            if (resolutionList!=null)
359                Collections.sort(resolutionList);
360        }
361
362        /** Same as asList().size() */
363        public int size() {
364            if (resolutionList==null) return 0;
365            return resolutionList.size();
366        }
367
368        /** Throws an IllegalStateException if this is frozen */
369        protected void ensureNotFrozen() {
370            if (frozen)
371                throw new IllegalStateException(this + " is frozen and cannot be modified");
372        }
373
374        /** Clone by filling in values from the given variants */
375        public FieldValues clone(String fieldName,List<QueryProfileVariant> clonedVariants) {
376            try {
377                if (frozen) return this;
378                FieldValues clone=(FieldValues)super.clone();
379
380                if (resolutionList!=null) {
381                    clone.resolutionList=new ArrayList<>(resolutionList.size());
382                    for (FieldValue value : resolutionList)
383                        clone.resolutionList.add(value.clone(fieldName,clonedVariants));
384                }
385
386                return clone;
387            }
388            catch (CloneNotSupportedException e) {
389                throw new RuntimeException(e);
390            }
391        }
392
393        public @Override FieldValues clone() {
394            try {
395                if (frozen) return this;
396                FieldValues clone=(FieldValues)super.clone();
397
398                if (resolutionList!=null) {
399                    clone.resolutionList=new ArrayList<>(resolutionList.size());
400                    for (FieldValue value : resolutionList)
401                        clone.resolutionList.add(value.clone());
402                }
403
404                return clone;
405            }
406            catch (CloneNotSupportedException e) {
407                throw new RuntimeException(e);
408            }
409        }
410
411    }

413    public static class FieldValue implements Comparable<FieldValue>, Cloneable {
414
415        private DimensionValues dimensionValues;
416        private Object value;
417
418        public FieldValue(DimensionValues dimensionValues,Object value) {
419            this.dimensionValues=dimensionValues;
420            this.value=value;
421        }
422
423        /**
424         * Returns the dimension values for which this value should be used.
425         * The dimension array is always of the exact size of the dimensions specified by the owning QueryProfileVariants,
426         * and the values appear in the order defined. "Wildcard" dimensions are represented by a null.
427         */
428        public DimensionValues getDimensionValues() { return dimensionValues; }
429
430        /** Returns the value to use for this set of dimension values */
431        public Object getValue() { return value; }
432
433        /** Sets the value to use for this set of dimension values */
434        public void setValue(Object value) { this.value=value; }
435
436        public boolean matches(DimensionValues givenDimensionValues) {
437            return dimensionValues.matches(givenDimensionValues);
438        }
439
440        /**
441         * Implements the sort order of this which is based on specificity
442         * where dimensions to the left are more significant.
443         * <p>
444         * <b>Note:</b> This ordering is not consistent with equals - it returns 0 when the same dimensions
445         * are <i>set</i>, regardless of what they are set <i>to</i>.
446         */
447        public @Override int compareTo(FieldValue other) {
448            return this.dimensionValues.compareTo(other.dimensionValues);
449        }
450
451        /** Clone by filling in the value from the given variants */
452        public FieldValue clone(String fieldName,List<QueryProfileVariant> clonedVariants) {
453            try {
454                FieldValue clone=(FieldValue)super.clone();
455                if (this.value instanceof QueryProfile)
456                    clone.value=lookupInVariants(fieldName,dimensionValues,clonedVariants);
457                // Otherwise the value is immutable, so keep it as-is
458                return clone;
459            }
460            catch (CloneNotSupportedException e) {
461                throw new RuntimeException(e);
462            }
463        }
464
465        public FieldValue clone() {
466            try {
467                FieldValue clone=(FieldValue)super.clone();
468                clone.value=QueryProfile.cloneIfNecessary(this.value);
469                return clone;
470            }
471            catch (CloneNotSupportedException e) {
472                throw new RuntimeException(e);
473            }
474        }
475
476        private Object lookupInVariants(String fieldName,DimensionValues dimensionValues,List<QueryProfileVariant> variants) {
477            for (QueryProfileVariant variant : variants) {
478                if ( ! variant.getDimensionValues().equals(dimensionValues)) continue;
479                return variant.values().get(fieldName);
480            }
481            return null;
482        }
483
484    }
