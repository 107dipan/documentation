002package com.yahoo.search.query;
003
004import com.yahoo.search.Query;
005
006/**
007 * Object properties keyed by name which can be looked up using default values and
008 * with conversion to various primitive wrapper types.
009 * <p>
010 * Multiple property implementations can be chained to provide unified access to properties
011 * backed by multiple sources as a Chain of Responsibility.
012 * <p>
013 * For better performance, prefer CompoundName argument constants over Strings.
014 * <p>
015 * Properties can be cloned. Cloning a properties instance returns a new instance
016 * which chains new instances of all chained instances. The content within each instance
017 * is cloned to the extent determined appropriate by that implementation.
018 * <p>
019 * This base class simply passes all access on to the next in chain.
020 *
021 * @author bratseth
022 */
023public abstract class Properties extends com.yahoo.processing.request.Properties {
024
025    @Override
026    public Properties chained() { return (Properties)super.chained(); }
027
028    @Override
029    public Properties clone() {
030        return (Properties)super.clone();
031    }
032
033    /** The query owning this property object.
034     *  Only guaranteed to work if this instance is accessible as query.properties()
035     */
036    public Query getParentQuery() {
037        if (chained() == null) {
038            throw new RuntimeException("getParentQuery should only be called on a properties instance accessible as query.properties()");
039        } else {
040            return chained().getParentQuery();
041        }
042    }
043
044    /**
045     * Invoked during deep cloning of the parent query.
046     */
047    public void setParentQuery(Query query) {
048        if (chained() != null)
049            chained().setParentQuery(query);
050    }
051}
