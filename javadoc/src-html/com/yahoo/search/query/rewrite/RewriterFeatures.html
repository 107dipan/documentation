026    /**
027     * <p>Add proximity boosting to original query by modifying
028     * the query tree directly</p>
029     * e.g. original Query Tree: (AND aa bb)<br>
030     *      if keepOriginalQuery: true<br>
031     *      new Query tree: (OR (AND aa bb) "aa bb")<br>
032     *      if keepOriginalQuery: false<br>
033     *      new Query Tree: "aa bb"<br><br>
034     *
035     *      original Query Tree: (OR (AND aa bb) (AND cc dd))<br>
036     *      boostingQuery: cc dd<br>
037     *      if keepOriginalQuery: true<br>
038     *      new Query Tree: (OR (AND aa bb) (AND cc dd) "cc dd")<br>
039     *      if keepOriginalQuery: false<br>
040     *      new Query Tree: (OR (AND aa bb) "cc dd") <br>
041     *
042     * @param query Query object from searcher
043     * @param boostingQuery query to be boosted
044     * @param keepOriginalQuery whether to keep original unboosted query as equiv
045     * @return Modified Query object, return original query object
046     *          on error
047     */
048    public static Query addUnitToOriginalQuery(Query query, String boostingQuery,
049                                               boolean keepOriginalQuery)
050                                               throws RuntimeException {
051        RewriterUtils.log(logger, query, "Adding proximity boosting to [" + boostingQuery + "]");
052
053        Model queryModel = query.getModel();
054        QueryTree qTree = queryModel.getQueryTree();
055        Item oldRoot = qTree.getRoot();
056
057        if (oldRoot == null) {
058            RewriterUtils.error(logger, query, "Error retrieving query tree root");
059            throw new RuntimeException("Error retrieving query tree root");
060        }
061
062        // Convert original query to query tree item
063        Item origQueryItem = convertStringToQTree(query, boostingQuery);
064
065        // Boost proximity by phrasing the original query
066        // query tree structure: (AND aa bb)
067        if (oldRoot instanceof AndItem &&
068            oldRoot.equals(origQueryItem)) {
069            PhraseItem phrase = convertAndToPhrase((AndItem)oldRoot);
070
071            if(!keepOriginalQuery) {
072                qTree.setRoot(phrase);
073            } else {
074                OrItem newRoot = new OrItem();
075                newRoot.addItem(oldRoot);
076                newRoot.addItem(phrase);
077                qTree.setRoot(newRoot);
078                queryModel.setType(Query.Type.ADVANCED); //set type=adv
079            }
080            RewriterUtils.log(logger, query, "Added proximity boosting successfully");
081            return query;
082
083        // query tree structure: (OR (AND aa bb) (AND cc dd))
084        } else if (oldRoot instanceof OrItem &&
085                   ((OrItem)oldRoot).getItemIndex(origQueryItem)!=-1 &&
086                   origQueryItem instanceof AndItem) {
087
088            // Remove original unboosted query
089            if(!keepOriginalQuery)
090                ((OrItem)oldRoot).removeItem(origQueryItem);
091
092            // Check if the tree already contained the phrase item
093            PhraseItem pI = convertAndToPhrase((AndItem)origQueryItem);
094            if(((OrItem)oldRoot).getItemIndex(pI)==-1) {
095                ((OrItem)oldRoot).addItem(convertAndToPhrase((AndItem)origQueryItem));
096                RewriterUtils.log(logger, query, "Added proximity boosting successfully");
097                return query;
098            }
099        }
100        RewriterUtils.log(logger, query, "No proximity boosting added");
101        return query;
102    }

104    /**
105     * <p>Add query expansion to the query tree</p>
106     * e.g. origQuery: aa bb<br>
107     *      matchingStr: aa bb<br>
108     *      rewrite: cc dd, ee ff<br>
109     *      if addUnitToRewrites: false<br>
110     *      new query tree: (OR (AND aa bb) (AND cc dd) (AND ee ff))<br>
111     *      if addUnitToRewrites: true<br>
112     *      new query tree: (OR (AND aa bb) "cc dd" "ee ff") <br>
113     *
114     * @param query Query object from searcher
115     * @param matchingStr string used to retrieve the rewrite
116     * @param rewrites The rewrite string retrieved from
117     *                 dictionary
118     * @param addUnitToRewrites Whether to add unit to rewrites
119     * @param maxNumRewrites Max number of rewrites to be added,
120     *                       0 if no limit
121     * @return Modified Query object, return original query object
122     *         on error
123     */
124    public static Query addRewritesAsEquiv(Query query, String matchingStr,
125                                           String rewrites,
126                                           boolean addUnitToRewrites,
127                                           int maxNumRewrites) throws RuntimeException {
128        String normalizedQuery = RewriterUtils.getNormalizedOriginalQuery(query);
129
130        RewriterUtils.log(logger, query,
131                          "Adding rewrites [" + rewrites +
132                          "] to the query [" + normalizedQuery + "]");
133        if (rewrites.equalsIgnoreCase(normalizedQuery) || rewrites.equalsIgnoreCase("n/a")) {
134            RewriterUtils.log(logger, query, "No rewrite added");
135            return query;
136        }
137
138        Model queryModel = query.getModel();
139        QueryTree qTree = queryModel.getQueryTree();
140        Item oldRoot = qTree.getRoot();
141
142        if (oldRoot == null) {
143            RewriterUtils.error(logger, query, "Error retrieving query tree root");
144            throw new RuntimeException("Error retrieving query tree root");
145        }
146
147        StringTokenizer rewrite_list = new StringTokenizer(rewrites, "\t");
148        Item rI = null;
149
150        // Convert matching string to query tree item
151        Item matchingStrItem = convertStringToQTree(query, matchingStr);
152        PhraseItem matchingStrPhraseItem = null;
153        if(matchingStrItem instanceof AndItem) {
154            matchingStrPhraseItem = convertAndToPhrase(((AndItem)matchingStrItem));
155        }
156
157        // Add rewrites as OR item to the query tree
158        // Only should rewrite in this case:
159        // - origQuery: (OR (AND aa bb) (AND cc dd))
160        // - matchingStr: (AND aa bb)
161        // Or in this case:
162        // - origQuery: (AND aa bb)
163        // - matching Str: (AND aa bb)
164        // Should not rewrite in this case:
165        // - origQuery: (OR (AND cc (OR dd (AND aa bb)) ee)
166        // - matchingStr: (AND aa bb)
167        // - for this case, should use getNonOverlappingMatches instead
168        OrItem newRoot;
169        if(oldRoot instanceof OrItem) {
170            if(((OrItem)oldRoot).getItemIndex(matchingStrItem)==-1) {
171                RewriterUtils.log(logger, query, "Whole query matching is used, skipping rewrite");
172                return query;
173            }
174            newRoot = (OrItem)oldRoot;
175        } else if(oldRoot.equals(matchingStrItem) || oldRoot.equals(matchingStrPhraseItem)) {
176            newRoot = new OrItem();
177            newRoot.addItem(oldRoot);
178        } else {
179            RewriterUtils.log(logger, query, "Whole query matching is used, skipping rewrite");
180            return query;
181        }
182        int numRewrites = 0;
183        while(rewrite_list.hasMoreTokens() &&
184              (maxNumRewrites==0 || numRewrites < maxNumRewrites)) {
185            rI = convertStringToQTree(query, rewrite_list.nextToken());
186            if(addUnitToRewrites && rI instanceof AndItem) {
187                rI = convertAndToPhrase((AndItem)rI);
188            }
189            if(newRoot.getItemIndex(rI)==-1) {
190                newRoot.addItem(rI);
191                numRewrites++;
192            } else {
193                RewriterUtils.log(logger, query, "Rewrite already exist, skipping");
194            }
195        }
196        qTree.setRoot(newRoot);
197        queryModel.setType(Query.Type.ADVANCED); //set type=adv
198        RewriterUtils.log(logger, query, "Added rewrite successfully");
199
200        return query;
201    }

203    /**
204     * <p>Retrieve the longest, from left to right non overlapping full
205     * phrase substrings in query based on FSA dictionary</p>
206     *
207     * e.g. query: ((modern AND new AND york AND city AND travel) OR travel) AND
208     *             ((sunny AND travel AND agency) OR nyc)<br>
209     *      dictionary: <br>
210     *                  mny\tmodern new york<br>
211     *                  mo\tmodern<br>
212     *                  modern\tn/a<br>
213     *                  modern\tnew york\tn/a<br>
214     *                  new york\tn/a<br>
215     *                  new york city\tn/a<br>
216     *                  new york city travel\tn/a<br>
217     *                  new york company\tn/a<br>
218     *                  ny\tnew york<br>
219     *                  nyc\tnew york city\tnew york company<br>
220     *                  nyct\tnew york city travel<br>
221     *                  ta\ttravel agency<br>
222     *                  travel agency\tn/a<br>
223     *      return: nyc
224     * @param phraseMatcher PhraseMatcher object loaded with FSA dict
225     * @param query Query object from the searcher
226     * @return Matching phrases
227     */
228    public static Set<PhraseMatcher.Phrase> getNonOverlappingFullPhraseMatches(PhraseMatcher phraseMatcher,
229                                                                               Query query)
230                                                                               throws RuntimeException {
231        RewriterUtils.log(logger, query, "Retrieving longest non-overlapping full phrase matches");
232        if(phraseMatcher==null)
233            return null;
234
235        Item root = query.getModel().getQueryTree().getRoot();
236        List<PhraseMatcher.Phrase> matches = phraseMatcher.matchPhrases(root);
237        if (matches==null || matches.isEmpty())
238            return null;
239
240        Set<PhraseMatcher.Phrase> resultMatches = new HashSet<>();
241        ListIterator<Phrase> matchesIter = matches.listIterator();
242
243        // Iterate through all matches
244        while(matchesIter.hasNext()) {
245            PhraseMatcher.Phrase phrase = matchesIter.next();
246            RewriterUtils.log(logger, query, "Working on phrase: " + phrase);
247            CompositeItem currOwner = phrase.getOwner();
248
249            // Check if this is full phrase
250            // If phrase is not an AND item, only keep those that are single word
251            // in order to eliminate cases such as (new RANK york) from being treated
252            // as match if only new york but not new or york is in the dictionary
253            if((currOwner!=null &&
254                ((phrase.isComplete() && currOwner instanceof AndItem) ||
255                 (phrase.getLength()==1 && currOwner instanceof OrItem) ||
256                 (phrase.getLength()==1 && currOwner instanceof RankItem && phrase.getStartIndex()==0))) ||
257               (currOwner==null && phrase.getLength()==1)) {
258               resultMatches.add(phrase);
259               RewriterUtils.log(logger, query, "Keeping phrase: " + phrase);
260            }
261        }
262
263        RewriterUtils.log(logger, query, "Successfully Retrieved longest non-overlapping full phrase matches");
264        return resultMatches;
265    }

268    /**
269     * <p>Retrieve the longest, from left to right non overlapping partial
270     * phrase substrings in query based on FSA dictionary</p>
271     *
272     * e.g. query: ((modern AND new AND york AND city AND travel) OR travel) AND
273     *             ((sunny AND travel AND agency) OR nyc)<br>
274     *      dictionary: <br>
275     *                  mny\tmodern new york<br>
276     *                  mo\tmodern<br>
277     *                  modern\tn/a<br>
278     *                  modern new york\tn/a<br>
279     *                  new york\tn/a<br>
280     *                  new york city\tn/a<br>
281     *                  new york city travel\tn/a<br>
282     *                  new york company\tn/a<br>
283     *                  ny\tnew york<br>
284     *                  nyc\tnew york city\tnew york company<br>
285     *                  nyct\tnew york city travel<br>
286     *                  ta\ttravel agency<br>
287     *                  travel agency\tn/a<br>
288     *      return: <br>
289     *              modern<br>
290     *              new york city travel<br>
291     *              travel agency<br>
292     *              nyc<br>
293     * @param phraseMatcher PhraseMatcher object loaded with FSA dict
294     * @param query Query object from the searcher
295     * @return Matching phrases
296     */
297    public static Set<PhraseMatcher.Phrase> getNonOverlappingPartialPhraseMatches(PhraseMatcher phraseMatcher,
298                                                                                  Query query)
299                                                                                  throws RuntimeException {
300        RewriterUtils.log(logger, query, "Retrieving longest non-overlapping partial phrase matches");
301        if(phraseMatcher==null)
302            return null;
303
304        Item root = query.getModel().getQueryTree().getRoot();
305        List<PhraseMatcher.Phrase> matches = phraseMatcher.matchPhrases(root);
306        if (matches==null || matches.isEmpty())
307            return null;
308
309        Set<PhraseMatcher.Phrase> resultMatches = new HashSet<>();
310        ArrayList<PhraseMatcher.Phrase> phrasesInSubTree = new ArrayList<>();
311        CompositeItem prevOwner = null;
312        ListIterator<PhraseMatcher.Phrase> matchesIter = matches.listIterator();
313
314        // Iterate through all matches
315        while(matchesIter.hasNext()) {
316            PhraseMatcher.Phrase phrase = matchesIter.next();
317            RewriterUtils.log(logger, query, "Working on phrase: " + phrase);
318            CompositeItem currOwner = phrase.getOwner();
319
320            // Check if previous is AND item and this phrase is in a different item
321            // If so, work on the previous set to eliminate overlapping matches
322            if(!phrasesInSubTree.isEmpty() && currOwner!=null &&
323               prevOwner!=null && !currOwner.equals(prevOwner)) {
324                RewriterUtils.log(logger, query, "Previous phrase is in different AND item");
325                List<PhraseMatcher.Phrase> subTreeMatches
326                    = getNonOverlappingMatchesInAndItem(phrasesInSubTree, query);
327                if(subTreeMatches==null) {
328                    RewriterUtils.error(logger, query, "Error retrieving matches from subtree");
329                    throw new RuntimeException("Error retrieving matches from subtree");
330                }
331                resultMatches.addAll(subTreeMatches);
332                phrasesInSubTree.clear();
333            }
334
335            // Check if this is an AND item
336            if(currOwner!=null && currOwner instanceof AndItem) {
337                phrasesInSubTree.add(phrase);
338            // If phrase is not an AND item, only keep those that are single word
339            // in order to eliminate cases such as (new RANK york) from being treated
340            // as match if only new york but not new or york is in the dictionary
341            } else if (phrase.getLength()==1 &&
342                       !(currOwner!=null && currOwner instanceof RankItem && phrase.getStartIndex()!=0)) {
343                resultMatches.add(phrase);
344            }
345
346            prevOwner = currOwner;
347        }
348
349        // Check if last item is AND item
350        // If so, work on the previous set to elimate overlapping matches
351        if(!phrasesInSubTree.isEmpty()) {
352            RewriterUtils.log(logger, query, "Last phrase is in AND item");
353            List<PhraseMatcher.Phrase> subTreeMatches
354                = getNonOverlappingMatchesInAndItem(phrasesInSubTree, query);
355            if(subTreeMatches==null) {
356                RewriterUtils.error(logger, query, "Error retrieving matches from subtree");
357                throw new RuntimeException("Error retrieving matches from subtree");
358            }
359            resultMatches.addAll(subTreeMatches);
360        }
361        RewriterUtils.log(logger, query, "Successfully Retrieved longest non-overlapping partial phrase matches");
362        return resultMatches;
363    }

365    /**
366     * <p>Retrieve the longest, from left to right non overlapping substrings in
367     * AndItem based on FSA dictionary</p>
368     *
369     * e.g. subtree: (modern AND new AND york AND city AND travel)<br>
370     *      dictionary:<br>
371     *                  mny\tmodern new york<br>
372     *                  mo\tmodern<br>
373     *                  modern\tn/a<br>
374     *                  modern new york\tn/a<br>
375     *                  new york\tn/a<br>
376     *                  new york city\tn/a<br>
377     *                  new york city travel\tn/a<br>
378     *                  new york company\tn/a<br>
379     *                  ny\tnew york<br>
380     *                  nyc\tnew york city\tnew york company<br>
381     *                  nyct\tnew york city travel<br>
382     *      allMatches:<br>
383     *                  modern<br>
384     *                  modern new york<br>
385     *                  new york<br>
386     *                  new york city<br>
387     *                  new york city travel<br>
388     *      return: <br>
389     *              modern<br>
390     *              new york city travel<br>
391     * @param allMatches All matches within the subtree
392     * @param query Query object from the searcher
393     * @return Matching phrases
394     */
395    public static List<PhraseMatcher.Phrase> getNonOverlappingMatchesInAndItem(
396                                             List<PhraseMatcher.Phrase> allMatches,
397                                             Query query)
398                                             throws RuntimeException {
399        RewriterUtils.log(logger, query, "Retrieving longest non-overlapping matches in subtree");
400
401        if (allMatches==null || allMatches.isEmpty())
402            return null;
403
404        if(allMatches.size()==1) {
405            RewriterUtils.log(logger, query, "Only one match in subtree");
406            return allMatches;
407        }
408
409        // Phrase are sorted based on length, if both have the
410        // same length, the lefter one ranks higher
411        RewriterUtils.log(logger, query, "Sorting the phrases");
412        PhraseLength phraseLength = new PhraseLength();
413        Collections.sort(allMatches, phraseLength);
414
415        // Create a bitset with length equal to the number of
416        // items in the subtree
417        int numWords = allMatches.get(0).getOwner().getItemCount();
418        BitSet matchPos = new BitSet(numWords);
419
420        // Removing matches that are overlapping with previously selected ones
421        RewriterUtils.log(logger, query, "Removing matches that are overlapping " +
422                                         "with previously selected ones");
423        ListIterator<Phrase> allMatchesIter = allMatches.listIterator();
424        while(allMatchesIter.hasNext()) {
425            PhraseMatcher.Phrase currMatch = allMatchesIter.next();
426            PhraseMatcher.Phrase.MatchIterator matchIter = currMatch.itemIterator();
427            if(matchIter.hasNext() && matchIter.next().isFilter()) {
428                RewriterUtils.log(logger, query, "Removing filter item" + currMatch);
429                allMatchesIter.remove();
430                continue;
431            }
432
433            BitSet currMatchPos = new BitSet(numWords);
434            currMatchPos.set(currMatch.getStartIndex(),
435                             currMatch.getLength()+currMatch.getStartIndex());
436            if(currMatchPos.intersects(matchPos)) {
437                RewriterUtils.log(logger, query, "Removing " + currMatch);
438                allMatchesIter.remove();
439            } else {
440                RewriterUtils.log(logger, query, "Keeping " + currMatch);
441                matchPos.or(currMatchPos);
442            }
443        }
444        return allMatches;
445    }

447    /**
448     * <p>Add Expansions to the matching phrases</p>
449     *
450     * e.g. Query: nyc travel agency<br>
451     *      matching phrase: nyc\tnew york city\tnew york company
452     *                       travel agency\tn/a<br>
453     *      if expandIndex is not null and removeOriginal is true<br>
454     *      New Query: ((new york city) OR ([expandIndex]:new york city)
455     *                  OR (new york company) OR
456     *                  ([expandIndex]:new york company)) AND
457     *                 ((travel agency) OR ([expandIndex]:travel agency))<br>
458     *      if expandIndex is null and removeOriginal is true<br>
459     *      New Query: ((new york city) OR (new york company)) AND
460     *                 travel agency<br>
461     *      if expandIndex is null and removeOriginal is false<br>
462     *      New Query: (nyc OR (new york city) OR (new york company)) AND
463     *                 travel agency<br>
464     *
465     * @param query Query object from searcher
466     * @param matches Set of longest non-overlapping matches
467     * @param expandIndex Name of expansion index or null if
468     *                    default index
469     * @param maxNumRewrites Max number of rewrites to be added,
470     *                       0 if no limit
471     * @param removeOriginal Whether to remove the original matching phrase
472     * @param addUnitToRewrites Whether to add rewrite as phrase
473     */
474    public static Query addExpansions(Query query, Set<PhraseMatcher.Phrase> matches,
475                                      String expandIndex, int maxNumRewrites,
476                                      boolean removeOriginal, boolean addUnitToRewrites)
477                                      throws RuntimeException {
478
479        if(matches==null) {
480            RewriterUtils.log(logger, query, "No expansions to be added");
481            return query;
482        }
483
484        RewriterUtils.log(logger, query, "Adding expansions to matching phrases");
485        Model queryModel = query.getModel();
486        QueryTree qTree = queryModel.getQueryTree();
487        Iterator<Phrase> matchesIter = matches.iterator();
488        CompositeItem parent = null;
489
490        // Iterate through all matches
491        while(matchesIter.hasNext()) {
492            PhraseMatcher.Phrase match = matchesIter.next();
493            RewriterUtils.log(logger, query, "Working on phrase: " + match);
494
495            // Retrieve expansion phrases
496            String expansionStr = match.getData();
497            if(expansionStr.equalsIgnoreCase("n/a") && expandIndex==null) {
498                continue;
499            }
500            StringTokenizer expansions = new StringTokenizer(expansionStr,"\t");
501
502            // Create this structure for all expansions of this match
503            // (OR (AND expandsion1) indexName:expansion1
504            //  (AND expansion2) indexName:expansion2..)
505            OrItem expansionGrp = new OrItem();
506            int numRewrites = 0;
507            String matchStr = convertMatchToString(match);
508            while(expansions.hasMoreTokens() &&
509                  (maxNumRewrites==0 || numRewrites < maxNumRewrites)) {
510                String expansion = expansions.nextToken();
511                RewriterUtils.log(logger, query, "Working on expansion: " + expansion);
512                if(expansion.equalsIgnoreCase("n/a")) {
513                    expansion = matchStr;
514                }
515                // (AND expansion) or "expansion"
516                Item expansionItem = convertStringToQTree(query, expansion);
517                if(addUnitToRewrites && expansionItem instanceof AndItem) {
518                   expansionItem = convertAndToPhrase((AndItem)expansionItem);
519                }
520                expansionGrp.addItem(expansionItem);
521
522                if(expandIndex!=null) {
523                    // indexName:expansion
524                    WordItem expansionIndexItem = new WordItem(expansion, expandIndex);
525                    expansionGrp.addItem(expansionIndexItem);
526                }
527                numRewrites++;
528                RewriterUtils.log(logger, query, "Adding expansion: " + expansion);
529            }
530
531            if(!removeOriginal) {
532                //(AND original)
533                Item matchItem = convertStringToQTree(query, matchStr);
534                if(expansionGrp.getItemIndex(matchItem)==-1) {
535                    expansionGrp.addItem(matchItem);
536                }
537            }
538
539            parent = match.getOwner();
540            int matchIndex = match.getStartIndex();
541            if(parent!=null) {
542                // Remove matching phrase from original query
543                for(int i=0; i<match.getLength(); i++) {
544                    parent.removeItem(matchIndex);
545                }
546                // Adding back expansions
547                parent.addItem(matchIndex, expansionGrp);
548            } else {
549               RewriterUtils.log(logger, query, "Single root item");
550               // If there's no parent, i.e. single root item
551               qTree.setRoot(expansionGrp);
552               break;
553            }
554        }
555
556        // Not root single item
557        if(parent!=null) {
558            // Cleaning up the query after rewrite to remove redundant tags
559            // e.g. (AND (OR (AND a b) c)) => (OR (AND a b) c)
560            String cleanupError = QueryCanonicalizer.canonicalize(qTree);
561            if(cleanupError!=null) {
562                RewriterUtils.error(logger, query, "Error canonicalizing query tree");
563                throw new RuntimeException("Error canonicalizing query tree");
564            }
565        }
566        queryModel.setType(Query.Type.ADVANCED); //set type=adv
567        RewriterUtils.log(logger, query, "Successfully added expansions to matching phrases");
568        return query;
569    }

571    /**
572     * Convert Match to String
573     *
574     * @param phrase Match from PhraseMatcher
575     * @return String format of the phrase
576     */
577    public static String convertMatchToString(PhraseMatcher.Phrase phrase) {
578        StringBuilder buffer = new StringBuilder();
579        for (Iterator<Item> i = phrase.itemIterator(); i.hasNext();) {
580            buffer.append(i.next().toString());
581            if (i.hasNext()) {
582                buffer.append(" ");
583            }
584        }
585        return buffer.toString();
586    }

588    /**
589     * Convert String to query tree
590     *
591     * @param stringToParse The string to be converted to a
592     *                      query tree
593     * @param query Query object from searcher
594     * @return Item The resulting query tree
595     */
596    static Item convertStringToQTree(Query query, String stringToParse) {
597        RewriterUtils.log(logger, query, "Converting string [" + stringToParse + "] to query tree");
598        if(stringToParse==null) {
599            return new NullItem();
600        }
601        Model model = query.getModel();
602        CustomParser parser = (CustomParser) ParserFactory.newInstance(model.getType(),
603                ParserEnvironment.fromExecutionContext(query.getModel().getExecution().context()));
604        IndexFacts indexFacts = new IndexFacts();
605        Item item = parser.parse(stringToParse, null, model.getParsingLanguage(),
606                                 indexFacts.newSession(model.getSources(), model.getRestrict()),
607                                 model.getDefaultIndex());
608        RewriterUtils.log(logger, query, "Converted string: [" + item.toString() + "]");
609        return item;
610    }

612    /**
613     * Convert AndItem to PhraseItem<br>
614     *
615     * e.g. (AND a b) to "a b"
616     * @param andItem query tree to be converted
617     * @return converted PhraseItem
618     */
619    private static PhraseItem convertAndToPhrase(AndItem andItem) {
620        PhraseItem result = new PhraseItem();
621        Iterator<Item> subItems = andItem.getItemIterator();
622        while(subItems.hasNext()) {
623            Item curr = (subItems.next());
624            if(curr instanceof IntItem) {
625                WordItem numItem = new WordItem(((IntItem)curr).stringValue());
626                result.addItem(numItem);
627            } else {
628                result.addItem(curr);
629            }
630        }
631        return result;
632    }

634    /**
635     * Class for comparing phrase.
636     * A phrase is larger if its length is longer.
637     * If both phrases are of the same length, the lefter one
638     * is considered larger
639     */
640    private static class PhraseLength implements Comparator<PhraseMatcher.Phrase> {
641        public int compare(PhraseMatcher.Phrase phrase1, PhraseMatcher.Phrase phrase2) {
642            if((phrase2.getLength()>phrase1.getLength()) ||
643               (phrase2.getLength()==phrase1.getLength() &&
644                phrase2.getStartIndex()<=phrase1.getStartIndex())) {
645                return 1;
646            } else {
647                return -1;
648            }
649        }
650    }
