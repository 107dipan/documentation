095    private XMLWriter wrapWriter(Writer writer) {
096        return XMLWriter.from(writer, 10, -1);
097    }

099    /**
100     * Renders this result
101     */
102    public void render(Writer writer, Result result) throws IOException {
103        XMLWriter xmlWriter = wrapWriter(writer);
104
105        context.put("context", context);
106        context.put("result", result);
107        context.setBoldOpenTag(defaultTemplate.getBoldOpenTag());
108        context.setBoldCloseTag(defaultTemplate.getBoldCloseTag());
109        context.setSeparatorTag(defaultTemplate.getSeparatorTag());
110
111        try {
112            header(xmlWriter, result);
113        } catch (Exception e) {
114            handleException(e);
115        }
116
117        if (result.hits().getError() != null || result.hits().getQuery().errors().size() > 0) {
118            error(xmlWriter, result);
119        }
120
121        if (result.getConcreteHitCount() == 0) {
122            emptyResult(xmlWriter, result);
123        }
124
125        if (result.getContext(false) != null) {
126            queryContext(xmlWriter, result.getContext(false), result.getQuery());
127        }
128
129        renderHitGroup(xmlWriter, result.hits(), result.hits().getQuery().getOffset() + 1);
130
131        endResult(xmlWriter, result);
132    }
133
134    private void header(XMLWriter writer, Result result) throws IOException {
135        // TODO: move setting this to Result
136        context.setUtf8Output("utf-8".equalsIgnoreCase(getRequestedEncoding(result.getQuery())));
137        writer.xmlHeader(getRequestedEncoding(result.getQuery()));
138        writer.openTag(RESULT).attribute(TOTAL_HIT_COUNT,String.valueOf(result.getTotalHitCount()));
139        renderCoverageAttributes(result.getCoverage(false), writer);
140        renderTime(writer, result);
141        writer.closeStartTag();
142    }
143
144    private void renderTime(XMLWriter writer, Result result) {
145        if (!result.getQuery().getPresentation().getTiming()) {
146            return;
147        }
148
149        final String threeDecimals = "%.3f";
150        final double milli = .001d;
151        final long now = System.currentTimeMillis();
152        final long searchTime = now - result.getQuery().getStartTime();
153        final double searchSeconds = ((double) searchTime) * milli;
154
155        if (result.getElapsedTime().firstFill() != 0L) {
156            final long queryTime = result.getElapsedTime().firstFill() - result.getQuery().getStartTime();
157            final long summaryFetchTime = now - result.getElapsedTime().firstFill();
158            final double querySeconds = ((double) queryTime) * milli;
159            final double summarySeconds = ((double) summaryFetchTime) * milli;
160            writer.attribute(QUERY_TIME, String.format(threeDecimals, querySeconds));
161            writer.attribute(SUMMARY_FETCH_TIME, String.format(threeDecimals, summarySeconds));
162        }
163        writer.attribute(SEARCH_TIME, String.format(threeDecimals, searchSeconds));
164    }

166    protected static void renderCoverageAttributes(Coverage coverage, XMLWriter writer) throws IOException {
167        if (coverage == null) return;
168        writer.attribute(COVERAGE_DOCS,coverage.getDocs());
169        writer.attribute(COVERAGE_NODES,coverage.getNodes());
170        writer.attribute(COVERAGE_FULL,coverage.getFull());
171        writer.attribute(COVERAGE,coverage.getResultPercentage());
172        writer.attribute(RESULTS_FULL,coverage.getFullResultSets());
173        writer.attribute(RESULTS,coverage.getResultSets());
174    }

176    public void endResult(XMLWriter writer, Result result) throws IOException {
177        try {
178            writer.closeTag();
179        } catch (Exception e) {
180            handleException(e);
181        }
182    }

184    public void error(XMLWriter writer, Result result) throws IOException {
185        try {
186            ErrorMessage error = result.hits().getError();
187            writer.openTag(ERROR).attribute(CODE,error.getCode()).content(error.getMessage(),false).closeTag();
188        } catch (Exception e) {
189            handleException(e);
190        }
191    }

196    public void queryContext(XMLWriter writer, QueryContext queryContext, Query owner) throws IOException {
197        try {
198            if (owner.getTraceLevel()!=0) {
199                XMLWriter xmlWriter=XMLWriter.from(writer);
200                xmlWriter.openTag("meta").attribute("type", QueryContext.ID);
201                TraceNode traceRoot = owner.getModel().getExecution().trace().traceNode().root();
202                traceRoot.accept(new RenderingVisitor(xmlWriter, owner.getStartTime()));
203                xmlWriter.closeTag();
204            }
205        } catch (Exception e) {
206            handleException(e);
207        }
208    }

210    private  void renderHitGroup(XMLWriter writer, HitGroup hitGroup, int hitnumber)
211            throws IOException {
212        for (Hit hit : hitGroup.asList()) {
213            renderHit(writer, hit, hitnumber);
214            if (!hit.isAuxiliary())
215                hitnumber++;
216        }
217    }

220    /**
221     * Renders this hit as xml. The default implementation will call the simpleRender()
222     * hook. If it returns true, nothing more is done, otherwise the
223     * given template set will be used for rendering.
224     *
225     *
226     * @param writer      the XMLWriter to append this hit to
227     * @throws java.io.IOException if rendering fails
228     */
229    public void renderHit(XMLWriter writer, Hit hit, int hitno) throws IOException {
230        renderRegularHit(writer, hit, hitno);
231    }
232
233    private void renderRegularHit(XMLWriter writer, Hit hit, int hitno) throws IOException {
234        boolean renderedSimple = simpleRenderHit(writer, hit);
235
236        if (renderedSimple) {
237            return;
238        }
239
240        try {
241            if (hit instanceof HitGroup) {
242                renderHitGroup(writer, (HitGroup) hit);
243            } else {
244                renderSingularHit(writer, hit);
245            }
246        } catch (Exception e) {
247            handleException(e);
248        }
249
250        if (hit instanceof HitGroup)
251            renderHitGroup(writer, (HitGroup) hit, hitno);
252
253        try {
254            writer.closeTag();
255        } catch (Exception e) {
256            handleException(e);
257        }
258    }
259
260    private void renderSingularHit(XMLWriter writer, Hit hit) throws IOException {
261        writer.openTag(HIT);
262        renderHitAttributes(writer, hit);
263        writer.closeStartTag();
264        renderHitFields(writer, hit);
265    }
266
267    private void renderHitFields(XMLWriter writer, Hit hit) throws IOException {
268        renderSyntheticRelevanceField(writer, hit);
269        for (Iterator<Map.Entry<String, Object>> it = hit.fieldIterator(); it.hasNext(); ) {
270            renderField(writer, hit, it);
271        }
272    }
273
274    private void renderField(XMLWriter writer, Hit hit, Iterator<Map.Entry<String, Object>> it) throws IOException {
275        Map.Entry<String, Object> entry = it.next();
276        boolean isProbablyNotDecoded = false;
277        if (hit instanceof FastHit) {
278            FastHit f = (FastHit) hit;
279            isProbablyNotDecoded = f.fieldIsNotDecoded(entry.getKey());
280        }
281        renderGenericFieldPossiblyNotDecoded(writer, hit, entry, isProbablyNotDecoded);
282    }
283
284    private void renderGenericFieldPossiblyNotDecoded(XMLWriter writer, Hit hit, Map.Entry<String, Object> entry, boolean probablyNotDecoded) throws IOException {
285        String fieldName = entry.getKey();
286
287        if (!shouldRenderField(hit, fieldName)) return;
288        if (fieldName.startsWith("$")) return; // Don't render fields that start with $ // TODO: Move to should render
289
290        writeOpenFieldElement(writer, fieldName);
291        renderFieldContentPossiblyNotDecoded(writer, hit, probablyNotDecoded, fieldName);
292        writeCloseFieldElement(writer);
293    }
294
295    private void renderFieldContentPossiblyNotDecoded(XMLWriter writer, Hit hit, boolean probablyNotDecoded, String fieldName) throws IOException {
296        boolean dumpedRaw = false;
297        if (probablyNotDecoded && (hit instanceof FastHit)) {
298            writer.closeStartTag();
299            if ((writer.getWriter() instanceof ByteWriter) && context.isUtf8Output()) {
300                dumpedRaw = UserTemplate.dumpBytes((ByteWriter) writer.getWriter(), (FastHit) hit, fieldName);
301            }
302            if (dumpedRaw) {
303                writer.content("", false); // let the xml writer note that this tag had content
304            }
305        }
306        if (!dumpedRaw) {
307            String xmlval = hit.getFieldXML(fieldName);
308            if (xmlval == null) {
309                xmlval = "(null)";
310            }
311            writer.escapedContent(xmlval, false);
312        }
313    }
314
315    private void renderSyntheticRelevanceField(XMLWriter writer, Hit hit) throws IOException {
316        final String relevancyFieldName = "relevancy";
317        final Relevance relevance = hit.getRelevance();
318
319        if (shouldRenderField(hit, relevancyFieldName) && relevance != null) {
320            renderSimpleField(writer, relevancyFieldName, relevance);
321        }
322    }
323
324    private void renderSimpleField(XMLWriter writer, String relevancyFieldName, Relevance relevance) throws IOException {
325        writeOpenFieldElement(writer, relevancyFieldName);
326        writer.content(relevance.toString(), false);
327        writeCloseFieldElement(writer);
328    }
329
330    private void writeCloseFieldElement(XMLWriter writer) throws IOException {
331        writer.closeTag();
332    }
333
334    private void writeOpenFieldElement(XMLWriter writer, String relevancyFieldName) throws IOException {
335        Utf8String utf8 = fieldNameMap.get(relevancyFieldName);
336        if (utf8 == null) {
337            utf8 = new Utf8String(relevancyFieldName);
338            fieldNameMap.put(relevancyFieldName, utf8);
339        }
340        writer.openTag(FIELD).attribute(NAME, utf8);
341        writer.closeStartTag();
342    }
343
344    private boolean shouldRenderField(Hit hit, String relevancyFieldName) {
345        // skip depending on hit type
346        return true;
347    }

349    private void renderHitAttributes(XMLWriter writer, Hit hit) throws IOException {
350        writer.attribute(TYPE, hit.getTypeString());
351        if (hit.getRelevance() != null) {
352            writer.attribute(RELEVANCY, hit.getRelevance().toString());
353}
354        writer.attribute(SOURCE, hit.getSource());
355    }

357    private void renderHitGroup(XMLWriter writer, HitGroup hit) throws IOException {
358        if (HitRenderer.renderHeader((HitGroup) hit, writer)) {
359            // empty
360        } else if (((HitGroup) hit).types().contains("grouphit")) {
361            // TODO Keep this?
362            renderHitGroupOfTypeGroupHit(writer, hit);
363        } else {
364            renderGroup(writer, hit);
365        }
366    }
367
368    private void renderGroup(XMLWriter writer, HitGroup hit) throws IOException {
369        writer.openTag(GROUP);
370        renderHitAttributes(writer, (HitGroup) hit);
371        writer.closeStartTag();
372    }
373
374    private void renderHitGroupOfTypeGroupHit(XMLWriter writer, HitGroup hit) throws IOException {
375        writer.openTag(HIT);
376        renderHitAttributes(writer, (HitGroup) hit);
377        renderId(writer, hit);
378        writer.closeStartTag();
379    }
380
381    private void renderId(XMLWriter writer, HitGroup hit) throws IOException {
382        URI uri = hit.getId();
383        if (uri != null) {
384            writer.openTag(ID).content(uri.stringValue(),false).closeTag();
385        }
386    }

388    private boolean simpleRenderHit(XMLWriter writer, Hit hit) throws IOException {
389        if (hit instanceof DefaultErrorHit) {
390            return simpleRenderDefaultErrorHit(writer, (DefaultErrorHit) hit);
391        } else if (hit instanceof GroupingListHit) {
392            return true;
393        } else {
394            return false;
395        }
396    }
397
398    public static boolean simpleRenderDefaultErrorHit(XMLWriter writer, ErrorHit defaultErrorHit) throws IOException {
399        writer.openTag("errordetails");
400        for (Iterator i = defaultErrorHit.errorIterator(); i.hasNext();) {
401            ErrorMessage error = (ErrorMessage) i.next();
402            renderMessageDefaultErrorHit(writer, error);
403        }
404        writer.closeTag();
405        return true;
406    }
407
408    public static void renderMessageDefaultErrorHit(XMLWriter writer, ErrorMessage error) throws IOException {
409        writer.openTag("error");
410        writer.attribute("source", error.getSource());
411        writer.attribute("error", error.getMessage());
412        writer.attribute("code", Integer.toString(error.getCode()));
413        writer.content(error.getDetailedMessage(), false);
414        if (error.getCause()!=null) {
415            writer.openTag("cause");
416            writer.content("\n", true);
417            StringWriter stackTrace=new StringWriter();
418            error.getCause().printStackTrace(new PrintWriter(stackTrace));
419            writer.content(stackTrace.toString(), true);
420            writer.closeTag();
421        }
422        writer.closeTag();
423    }
424
425    private void handleException(Exception e) throws IOException {
426        if (e instanceof IOException) {
427            throw (IOException) e;
428        } else {
429            log.log(LogLevel.WARNING, "Exception thrown when rendering the result:", e);
430        }
431    }

433    public static final class RenderingVisitor extends TraceVisitor {
434
435        private static final String tag = "p";
436        private final XMLWriter writer;
437        private long baseTime;
438
439        public RenderingVisitor(XMLWriter writer,long baseTime) {
440            this.writer=writer;
441            this.baseTime=baseTime;
442        }
443
444        @Override
445        public void entering(TraceNode node) {
446            if (node.isRoot()) return;
447            writer.openTag(tag);
448        }
449
450        @Override
451        public void leaving(TraceNode node) {
452            if (node.isRoot()) return;
453            writer.closeTag();
454        }
455
456        @Override
457        public void visit(TraceNode node) {
458            if (node.isRoot()) return;
459            if (node.payload()==null) return;
460
461            writer.openTag(tag);
462            if (node.timestamp()!=0)
463                writer.content(node.timestamp()-baseTime,false).content(" ms: ", false);
464            writer.content(node.payload().toString(),false);
465            writer.closeTag();
466        }
467
468    }
