140    private class TraceRenderer extends TraceVisitor {
141        private final long basetime;
142        private boolean hasFieldName = false;
143        int emittedChildNesting = 0;
144        int currentChildNesting = 0;
145        private boolean insideOpenObject = false;
146
147        TraceRenderer(long basetime) {
148            this.basetime = basetime;
149        }
150
151        @Override
152        public void entering(TraceNode node) {
153            ++currentChildNesting;
154        }
155
156        @Override
157        public void leaving(TraceNode node) {
158            conditionalEndObject();
159            if (currentChildNesting == emittedChildNesting) {
160                try {
161                    generator.writeEndArray();
162                    generator.writeEndObject();
163                } catch (IOException e) {
164                    throw new TraceRenderWrapper(e);
165                }
166                --emittedChildNesting;
167            }
168            --currentChildNesting;
169        }
170
171        @Override
172        public void visit(TraceNode node) {
173            try {
174                doVisit(node.timestamp(), node.payload(), node.children().iterator().hasNext());
175            } catch (IOException e) {
176                throw new TraceRenderWrapper(e);
177            }
178        }
179
180        private void doVisit(final long timestamp, final Object payload, final boolean hasChildren) throws IOException {
181            boolean dirty = false;
182            if (timestamp != 0L) {
183                header();
184                generator.writeStartObject();
185                generator.writeNumberField(TRACE_TIMESTAMP, timestamp - basetime);
186                dirty = true;
187            }
188            if (payload != null) {
189                if (!dirty) {
190                    header();
191                    generator.writeStartObject();
192                }
193                generator.writeStringField(TRACE_MESSAGE, payload.toString());
194                dirty = true;
195            }
196            if (dirty) {
197                if (!hasChildren) {
198                    generator.writeEndObject();
199                } else {
200                    setInsideOpenObject(true);
201                }
202            }
203        }
204
205        private void header() {
206            fieldName();
207            for (int i = 0; i < (currentChildNesting - emittedChildNesting); ++i) {
208                startChildArray();
209            }
210            emittedChildNesting = currentChildNesting;
211        }
212
213        private void startChildArray() {
214            try {
215                conditionalStartObject();
216                generator.writeArrayFieldStart(TRACE_CHILDREN);
217            } catch (IOException e) {
218                throw new TraceRenderWrapper(e);
219            }
220        }
221
222        private void conditionalStartObject() throws IOException {
223            if (!isInsideOpenObject()) {
224                generator.writeStartObject();
225            } else {
226                setInsideOpenObject(false);
227            }
228        }
229
230        private void conditionalEndObject() {
231            if (isInsideOpenObject()) {
232                // This triggers if we were inside a data node with payload and
233                // subnodes, but none of the subnodes contained data
234                try {
235                    generator.writeEndObject();
236                    setInsideOpenObject(false);
237                } catch (IOException e) {
238                    throw new TraceRenderWrapper(e);
239                }
240            }
241        }
242
243        private void fieldName() {
244            if (hasFieldName) {
245                return;
246            }
247
248            try {
249                generator.writeFieldName(TRACE);
250            } catch (IOException e) {
251                throw new TraceRenderWrapper(e);
252            }
253            hasFieldName = true;
254        }
255
256        boolean isInsideOpenObject() {
257            return insideOpenObject;
258        }
259
260        void setInsideOpenObject(boolean insideOpenObject) {
261            this.insideOpenObject = insideOpenObject;
262        }
263    }

313    @Override
314    public void beginResponse(OutputStream stream) throws IOException {
315        beginJsonCallback(stream);
316        generator = generatorFactory.createGenerator(stream, JsonEncoding.UTF8);
317        renderedChildren = new ArrayDeque<>();
318        debugRendering = getDebugRendering(getResult().getQuery());
319        generator.writeStartObject();
320        renderTrace(getExecution().trace());
321        renderTiming();
322        generator.writeFieldName(ROOT);
323    }
324
325    private void renderTiming() throws IOException {
326        if (!getResult().getQuery().getPresentation().getTiming()) return;
327
328        double milli = .001d;
329        long now = timeSource.getAsLong();
330        long searchTime = now - getResult().getElapsedTime().first();
331        double searchSeconds = searchTime * milli;
332
333        generator.writeObjectFieldStart(TIMING);
334        if (getResult().getElapsedTime().firstFill() != 0L) {
335            long queryTime = getResult().getElapsedTime().weightedSearchTime();
336            long summaryFetchTime = getResult().getElapsedTime().weightedFillTime();
337            double querySeconds = queryTime * milli;
338            double summarySeconds = summaryFetchTime * milli;
339            generator.writeNumberField(QUERY_TIME, querySeconds);
340            generator.writeNumberField(SUMMARY_FETCH_TIME, summarySeconds);
341        }
342
343        generator.writeNumberField(SEARCH_TIME, searchSeconds);
344        generator.writeEndObject();
345    }

347    private boolean getDebugRendering(Query q) {
348        return q != null && q.properties().getBoolean(DEBUG_RENDERING_KEY, false);
349    }

351    private void renderTrace(Trace trace) throws IOException {
352        if (!trace.traceNode().children().iterator().hasNext()) return;
353        if (getResult().getQuery().getTraceLevel() == 0) return;
354
355        try {
356            long basetime = trace.traceNode().timestamp();
357            if (basetime == 0L)
358                basetime = getResult().getElapsedTime().first();
359            trace.accept(new TraceRenderer(basetime));
360        } catch (TraceRenderWrapper e) {
361            throw new IOException(e);
362        }
363    }

366    public void beginList(DataList<?> list) throws IOException {
367        Preconditions.checkArgument(list instanceof HitGroup,
368                                    "Expected subclass of com.yahoo.search.result.HitGroup, got %s.",

374    protected void moreChildren() throws IOException {
375        if (!renderedChildren.isEmpty())
376            childrenArray();
377
378        renderedChildren.push(0);
379    }
380
381    private void childrenArray() throws IOException {
382        if (renderedChildren.peek() == 0)
383            generator.writeArrayFieldStart(CHILDREN);
384        renderedChildren.push(renderedChildren.pop() + 1);
385    }
386
387    private void lessChildren() throws IOException {
388        int lastRenderedChildren = renderedChildren.pop();
389        if (lastRenderedChildren > 0) {
390            generator.writeEndArray();
391        }
392    }

394    private void renderHitGroupHead(HitGroup hitGroup) throws IOException {
395        generator.writeStartObject();
396
397        renderHitContents(hitGroup);
398        if (getRecursionLevel() == 1)
399            renderCoverage();
400
401        ErrorHit errorHit = hitGroup.getErrorHit();
402        if (errorHit != null)
403            renderErrors(errorHit.errors());
404
405        // the framework will invoke begin methods as needed from here
406    }
407
408    private void renderErrors(Set<ErrorMessage> errors) throws IOException {
409        if (errors.isEmpty()) return;
410
411        generator.writeArrayFieldStart(ERRORS);
412        for (ErrorMessage e : errors) {
413            String summary = e.getMessage();
414            String source = e.getSource();
415            Throwable cause = e.getCause();
416            String message = e.getDetailedMessage();
417            generator.writeStartObject();
418            generator.writeNumberField(ERROR_CODE, e.getCode());
419            generator.writeStringField(ERROR_SUMMARY, summary);
420            if (source != null) {
421                generator.writeStringField(ERROR_SOURCE, source);
422            }
423            if (message != null) {
424                generator.writeStringField(ERROR_MESSAGE, message);
425            }
426            if (cause != null && cause.getStackTrace().length > 0) {
427                StringWriter s = new StringWriter();
428                PrintWriter p = new PrintWriter(s);
429                cause.printStackTrace(p);
430                p.close();
431                generator.writeStringField(ERROR_STACK_TRACE, s.toString());
432            }
433            generator.writeEndObject();
434        }
435        generator.writeEndArray();
436
437
438    }

440    private void renderCoverage() throws IOException {
441        Coverage c = getResult().getCoverage(false);
442        if (c == null) return;
443
444        generator.writeObjectFieldStart(COVERAGE);
445        generator.writeNumberField(COVERAGE_COVERAGE, c.getResultPercentage());
446        generator.writeNumberField(COVERAGE_DOCUMENTS, c.getDocs());
447        if (c.isDegraded()) {
448            generator.writeObjectFieldStart(COVERAGE_DEGRADE);
449            generator.writeBooleanField(COVERAGE_DEGRADE_MATCHPHASE, c.isDegradedByMatchPhase());
450            generator.writeBooleanField(COVERAGE_DEGRADE_TIMEOUT, c.isDegradedByTimeout());
451            generator.writeBooleanField(COVERAGE_DEGRADE_ADAPTIVE_TIMEOUT, c.isDegradedByAdapativeTimeout());
452            generator.writeBooleanField(COVERAGE_DEGRADED_NON_IDEAL_STATE, c.isDegradedByNonIdealState());
453            generator.writeEndObject();
454        }
455        generator.writeBooleanField(COVERAGE_FULL, c.getFull());
456        generator.writeNumberField(COVERAGE_NODES, c.getNodes());
457        generator.writeNumberField(COVERAGE_RESULTS, c.getResultSets());
458        generator.writeNumberField(COVERAGE_RESULTS_FULL, c.getFullResultSets());
459        generator.writeEndObject();
460    }

462    private void renderHit(Hit hit) throws IOException {
463        if (!shouldRender(hit)) return;
464
465        childrenArray();
466        generator.writeStartObject();
467        renderHitContents(hit);
468        generator.writeEndObject();

475    private boolean fieldsStart(boolean hasFieldsField) throws IOException {
476        if (hasFieldsField) return true;
477        generator.writeObjectFieldStart(FIELDS);
478        return true;
479    }

481    private void fieldsEnd(boolean hasFieldsField) throws IOException {
482        if (!hasFieldsField) return;
483        generator.writeEndObject();

486    private void renderHitContents(Hit hit) throws IOException {
487        String id = hit.getDisplayId();
488        if (id != null)
489            generator.writeStringField(ID, id);
490
491        generator.writeNumberField(RELEVANCE, hit.getRelevance().getScore());
492
493        if (hit.types().size() > 0) { // TODO: Remove types rendering on Vespa 7
494            generator.writeArrayFieldStart(TYPES);
495            for (String t : hit.types()) {
496                generator.writeString(t);
497            }
498            generator.writeEndArray();
499        }
500
501        String source = hit.getSource();
502        if (source != null)
503            generator.writeStringField(SOURCE, hit.getSource());
504
505        renderSpecialCasesForGrouping(hit);
506
507        renderAllFields(hit);
508    }
509
510    private void renderAllFields(Hit hit) throws IOException {
511        boolean hasFieldsField = false;
512
513        hasFieldsField |= renderTotalHitCount(hit, hasFieldsField);
514        hasFieldsField |= renderStandardFields(hit, hasFieldsField);
515        fieldsEnd(hasFieldsField);
516    }
517
518    private boolean renderStandardFields(Hit hit, boolean initialHasFieldsField) throws IOException {
519        boolean hasFieldsField = initialHasFieldsField;
520        for (String fieldName : hit.fieldKeys()) {
521            if (!shouldRender(fieldName, hit)) continue;
522
523            // We can't look at the size of fieldKeys() and know whether we need
524            // the fields object, as all fields may be hidden.
525            hasFieldsField |= fieldsStart(hasFieldsField);
526            renderField(fieldName, hit);
527        }
528        return hasFieldsField;
529    }
530
531    private boolean shouldRender(String fieldName, Hit hit) {
532        if (debugRendering) return true;
533
534        if (fieldName.startsWith(VESPA_HIDDEN_FIELD_PREFIX)) return false;
535
536        RenderDecision r = lazyRenderAwareCheck(fieldName, hit);
537        if (r != RenderDecision.DO_NOT_KNOW) return r.booleanValue();
538
539        // this will trigger field decoding, so it is important the lazy decoding magic is done first
540        Object field = hit.getField(fieldName);
541
542        if (field instanceof CharSequence && ((CharSequence) field).length() == 0) return false;
543
544        // StringFieldValue cannot hold a null, so checking length directly is OK:
545        if (field instanceof StringFieldValue && ((StringFieldValue) field).getString().isEmpty()) return false;
546
547        if (field instanceof NanNumber) return false;
548
549        return true;
550    }
551
552    private RenderDecision lazyRenderAwareCheck(String fieldName, Hit hit) {
553        if ( ! (hit instanceof FastHit)) return RenderDecision.DO_NOT_KNOW;
554
555        FastHit asFastHit = (FastHit) hit;
556        if (asFastHit.fieldIsNotDecoded(fieldName)) {
557            FastHit.RawField rawField = asFastHit.fetchFieldAsUtf8(fieldName);
558            if (rawField != null)
559                return rawField.getUtf8().length == 0 ? RenderDecision.NO : RenderDecision.YES;
560        }
561        return RenderDecision.DO_NOT_KNOW;
562    }

564    private void renderSpecialCasesForGrouping(Hit hit) throws IOException {
565        if (hit instanceof AbstractList) {
566            renderGroupingListSyntheticFields((AbstractList) hit);
567        } else if (hit instanceof Group) {
568            renderGroupingGroupSyntheticFields(hit);
569        }

573        renderGroupMetadata(((Group) hit).getGroupId());
574        if (hit instanceof RootGroup) {
575            renderContinuations(Collections.singletonMap(
576                    Continuation.THIS_PAGE, ((RootGroup) hit).continuation()));
577        }

589    private void renderContinuations(Map<String, Continuation> continuations) throws IOException {
590        if (continuations.isEmpty()) return;
591
592        generator.writeObjectFieldStart(CONTINUATION);
593        for (Map.Entry<String, Continuation> e : continuations.entrySet()) {
594            generator.writeStringField(e.getKey(), e.getValue().toString());
595        }
596        generator.writeEndObject();
597    }

599    private void renderGroupMetadata(GroupId id) throws IOException {
600        if (!(id instanceof ValueGroupId || id instanceof BucketGroupId)) return;
601
602        if (id instanceof ValueGroupId) {
603            ValueGroupId<?> valueId = (ValueGroupId<?>) id;
604            generator.writeStringField(GROUPING_VALUE, getIdValue(valueId));
605        } else {
606            BucketGroupId<?> bucketId = (BucketGroupId<?>) id;
607            generator.writeObjectFieldStart(BUCKET_LIMITS);
608            generator.writeStringField(BUCKET_FROM, getBucketFrom(bucketId));
609            generator.writeStringField(BUCKET_TO, getBucketTo(bucketId));
610            generator.writeEndObject();
611        }
612    }

626    private boolean renderTotalHitCount(Hit hit, boolean hasFieldsField) throws IOException {
627        if ( ! (getRecursionLevel() == 1 && hit instanceof HitGroup)) return false;
628
629        fieldsStart(hasFieldsField);
630        generator.writeNumberField(TOTAL_COUNT, getResult().getTotalHitCount());

634    private void renderField(String fieldName, Hit hit) throws IOException {
635        generator.writeFieldName(fieldName);
636        if ( ! tryDirectRendering(fieldName, hit)) {
637            renderFieldContents(hit.getField(fieldName));
638        }
639    }

641    private void renderFieldContents(Object field) throws IOException {
642        if (field == null) {
643            generator.writeNull();
644        } else if (field instanceof Number) {
645            renderNumberField((Number) field);
646        } else if (field instanceof TreeNode) {
647            generator.writeTree((TreeNode) field);
648        } else if (field instanceof Tensor) {
649            renderTensor(Optional.of((Tensor)field));
650        } else if (field instanceof JsonProducer) {
651            generator.writeRawValue(((JsonProducer) field).toJson());
652        } else if (field instanceof Inspectable) {
653            StringBuilder intermediate = new StringBuilder();
654            JsonRender.render((Inspectable) field, intermediate, true);
655            generator.writeRawValue(intermediate.toString());
656        } else if (field instanceof StringFieldValue) {
657            // This needs special casing as JsonWriter hides empty strings now
658            generator.writeString(((StringFieldValue)field).getString());
659        } else if (field instanceof TensorFieldValue) {
660            renderTensor(((TensorFieldValue)field).getTensor());
661        } else if (field instanceof FieldValue) {
662            // the null below is the field which has already been written
663            ((FieldValue) field).serialize(null, new JsonWriter(generator));
664        } else if (field instanceof JSONArray || field instanceof JSONObject) {
665            // org.json returns null if the object would not result in
666            // syntactically correct JSON
667            String s = field.toString();
668            if (s == null) {
669                generator.writeNull();
670            } else {
671                generator.writeRawValue(s);
672            }
673        } else {
674            generator.writeString(field.toString());
675        }
676    }
677
678    private void renderNumberField(Number field) throws IOException {
679        if (field instanceof Integer) {
680            generator.writeNumber(field.intValue());
681        }  else if (field instanceof Float) {
682            generator.writeNumber(field.floatValue());
683        }  else if (field instanceof Double) {
684            generator.writeNumber(field.doubleValue());
685        } else if (field instanceof Long) {
686            generator.writeNumber(field.longValue());
687        } else if (field instanceof Byte || field instanceof Short) {
688            generator.writeNumber(field.intValue());
689        } else if (field instanceof BigInteger) {
690            generator.writeNumber((BigInteger) field);
691        } else if (field instanceof BigDecimal) {
692            generator.writeNumber((BigDecimal) field);
693        } else {
694            generator.writeNumber(field.doubleValue());
695        }
696    }

698    private void renderTensor(Optional<Tensor> tensor) throws IOException {
699        generator.writeStartObject();
700        generator.writeArrayFieldStart("cells");
701        if (tensor.isPresent()) {
702            for (Iterator<Tensor.Cell> i = tensor.get().cellIterator(); i.hasNext(); ) {
703                Tensor.Cell cell = i.next();
704
705                generator.writeStartObject();
706
707                generator.writeObjectFieldStart("address");
708                for (int d = 0; d < cell.getKey().size(); d++)
709                    generator.writeObjectField(tensor.get().type().dimensions().get(d).name(), cell.getKey().label(d));
710                generator.writeEndObject();
711
712                generator.writeObjectField("value", cell.getValue());
713
714                generator.writeEndObject();
715            }
716        }
717        generator.writeEndArray();
718        generator.writeEndObject();
719    }

722     * Really a private method, but package access for testability.
723     */
724    boolean tryDirectRendering(String fieldName, Hit hit) throws IOException {
725        boolean renderedAsUtf8 = false;
726        if (hit instanceof FastHit) {
727            FastHit f = (FastHit) hit;
728            if (f.fieldIsNotDecoded(fieldName)) {
729                FastHit.RawField r = f.fetchFieldAsUtf8(fieldName);
730                if (r != null) {
731                    byte[] utf8 = r.getUtf8();
732
733                    generator.writeUTF8String(utf8, 0, utf8.length);
734                    renderedAsUtf8 = true;
735                }
736            }
737        }
738        return renderedAsUtf8;
739    }

779    /**
780     * Adds JSONP (Json with padding) support.
781     *
782     * Basically, if the JSON renderer receives a query parameter "jsoncallback=...",
783     * the JSON response will be wrapped in a function call with the name specified
784     * by the client. This side-steps the same-origin policy, thus supports calling
785     * Vespa from javascript loaded from a different domain then the Vespa instance.
786     */
787    private void beginJsonCallback(OutputStream stream) throws IOException {
788        if (shouldRenderJsonCallback()) {
789            String jsonCallback = getJsonCallback() + "(";
790            stream.write(jsonCallback.getBytes(StandardCharsets.UTF_8));
791            this.stream = stream;
792        }
793    }
794
795    private void endJsonCallback() throws IOException {
796        if (shouldRenderJsonCallback() && stream != null) {
797            stream.write(");".getBytes(StandardCharsets.UTF_8));
798        }
799    }
800
801    private boolean shouldRenderJsonCallback() {
802        String jsonCallback = getJsonCallback();
803        return jsonCallback != null && !"".equals(jsonCallback);
804    }
805
806    private String getJsonCallback() {
807        Result result = getResult();
808        if (result != null) {
809            Query query = result.getQuery();
810            if (query != null) {
811                return query.properties().getString(JSON_CALLBACK, null);
812            }
813        }
814        return null;
815    }
