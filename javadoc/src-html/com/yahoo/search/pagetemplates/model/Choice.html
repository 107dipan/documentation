002package com.yahoo.search.pagetemplates.model;
003
004import java.util.*;
005
006/**
007 * A choice between some alternative lists of page elements.
008 *
009 * @author bratseth
010 */
011public final class Choice extends AbstractChoice {
012
013    private List<List<PageElement>> alternatives=new ArrayList<>(3);
014
015    /** Creates an empty choice */
016    public Choice() { }
017
018    /** Creates a choice having a single alternative having a single page element */
019    public static Choice createSingleton(PageElement singletonAlternative) {
020        Choice choice=new Choice();
021        choice.alternatives().add(createSingletonList(singletonAlternative));
022        return choice;
023    }
024
025    /** Creates a choice in which each alternative consists of a single element */
026    public static Choice createSingletons(List<PageElement> alternatives) {
027        Choice choice=new Choice();
028        for (PageElement alternative : alternatives)
029            choice.alternatives().add(createSingletonList(alternative));
030        return choice;
031    }
032
033    private static List<PageElement> createSingletonList(PageElement member) {
034        List<PageElement> list=new ArrayList<>();
035        list.add(member);
036        return list;
037    }
038
039    /**
040     * Creates a choice between some alternatives. This method takes a copy of the given lists.
041     */
042    public Choice(List<List<PageElement>> alternatives) {
043        for (List<PageElement> alternative : alternatives)
044            this.alternatives.add(new ArrayList<>(alternative));
045    }
046
047    /**
048     * Returns the alternatives of this as a live reference to the alternatives of this.
049     * The list and elements may be modified unless this is frozen. This is never null.
050     */
051    public List<List<PageElement>> alternatives() { return alternatives; }
052
053    /** Convenience shorthand of <code>return alternatives().get(index)</code> */
054    public List<PageElement> get(int index) {
055        return alternatives.get(index);
056    }
057
058    /** Convenience shorthand for <code>if (alternative!=null) alternatives().add(alternative)</code> */
059    public void add(List<PageElement> alternative) {
060        if (alternative!=null)
061            alternatives.add(new ArrayList<>(alternative));
062    }
063
064    /** Returns true only if there are no alternatives in this */
065    public boolean isEmpty() { return alternatives.size()==0; }
066
067    /** Answers true if this is either a choice between the given class, or between Lists of the given class */
068    @SuppressWarnings({ "rawtypes", "unchecked" })
069    @Override
070    public boolean isChoiceBetween(Class pageTemplateModelElementClass) {
071        List firstNonEmpty=null;
072        for (List<PageElement> value : alternatives) {
073            if (pageTemplateModelElementClass.isAssignableFrom(value.getClass())) return true;
074            if (value instanceof List) {
075                List listValue=(List)value;
076                if (listValue.size()>0)
077                    firstNonEmpty=listValue;
078            }
079        }
080        if (firstNonEmpty==null) return false;
081        return (pageTemplateModelElementClass.isAssignableFrom(firstNonEmpty.get(0).getClass()));
082    }
083
084    @Override
085    public void freeze() {
086        if (isFrozen()) return;
087        super.freeze();
088        for (ListIterator<List<PageElement>> i=alternatives.listIterator(); i.hasNext(); ) {
089            List<PageElement> alternative=i.next();
090            for (PageElement alternativeElement : alternative)
091                alternativeElement.freeze();
092            i.set(Collections.unmodifiableList(alternative));
093        }
094        alternatives= Collections.unmodifiableList(alternatives);
095    }
096
097    /** Accepts a visitor to this structure */
098    @Override
099    public void accept(PageTemplateVisitor visitor) {
100        visitor.visit(this);
101        for (List<PageElement> alternative : alternatives) {
102            for (PageElement alternativeElement : alternative)
103                alternativeElement.accept(visitor);
104        }
105    }
106
107    @Override
108    public String toString() {
109        if (alternatives.isEmpty()) return "(empty choice)";
110        if (alternatives.size()==1) return alternatives.get(0).toString();
111        return "a choice between " + alternatives;
112    }
113
114}
