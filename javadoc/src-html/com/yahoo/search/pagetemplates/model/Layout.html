004/**
005 * The layout of a section
006 *
007 * @author bratseth
008 */
009// This is not made an enum, to allow the value set to be extendible.
010// It is not explicitly made immutable
011// to enable adding of internal state later (esp. parameters).
012// If this becomes mutable, the creation scheme must be changed
013// such that each fromString returns a unique instance, and
014// the name must become a (immutable) type.
015public class Layout {
016
017    /** The built in "column" layout */
018    public static final Layout column=new Layout("column");
019    /** The built in "row" layout */
020    public static final Layout row=new Layout("row");
021
022    private String name;
023
024    public Layout(String name) {
025        this.name=name;
026    }
027
028    public String getName() { return name; }
029
030    public @Override int hashCode() { return name.hashCode(); }
031
032    public @Override boolean equals(Object o) {
033        if (o==this) return true;
034        if (! (o instanceof Layout)) return false;
035        Layout other=(Layout)o;
036        return this.name.equals(other.name);
037    }
038
039    /** Returns a layout having this string as name, or null if the given string is null or empty */
040    public static Layout fromString(String layout) {
041        //if (layout==null) return null;
042        //if (layout)
043        if (layout.equals("column")) return column;
044        if (layout.equals("row")) return row;
045        return new Layout(layout);
046    }
047
048    public @Override String toString() { return "layout '" + name + "'"; }
049
050}
