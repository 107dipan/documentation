002package com.yahoo.search.pagetemplates.model;
003
004import com.yahoo.component.provider.FreezableClass;
005
006/**
007 * Abstract superclass of various kinds of choices.
008 *
009 * @author bratseth
010 */
011public abstract class AbstractChoice extends FreezableClass implements PageElement {
012
013    private String method;
014
015    /**
016     * Returns the choice method to use - a string interpreted by the resolver in use,
017     * or null to use any available method
018     */
019    public String getMethod() { return method; }
020
021    public void setMethod(String method) {
022        ensureNotFrozen();
023        this.method=method;
024    }
025
026    // TODO: is this really choices between classes in general, or e.g. subclasses of Section?
027    /** Returns true if this choice is (partially or completely) a choice between the given type */
028    @SuppressWarnings("rawtypes")
029    public abstract boolean isChoiceBetween(Class pageTemplateModelClass);
030
031}
