002package com.yahoo.search.pagetemplates.engine.resolvers;
003
004import com.google.inject.Inject;
005import com.yahoo.component.AbstractComponent;
006import com.yahoo.component.ComponentId;
007import com.yahoo.component.provider.ComponentRegistry;
008import com.yahoo.search.pagetemplates.engine.Resolver;
009
010import java.util.List;
011import java.util.logging.Logger;
012
013/**
014 * A registry of available resolver components
015 *
016 * @author bratseth
017 */
018public class ResolverRegistry extends ComponentRegistry<Resolver> {
019
020    private final Resolver defaultResolver;
021
022    public ResolverRegistry(List<Resolver> resolvers) {
023        addBuiltInResolvers();
024        for (Resolver component : resolvers)
025            registerResolver(component);
026        defaultResolver = decideDefaultResolver();
027        freeze();
028    }
029
030    private void addBuiltInResolvers() {
031        registerResolver(createNativeDeterministicResolver());
032        registerResolver(createNativeRandomResolver());
033    }
034
035    private Resolver decideDefaultResolver() {
036        Resolver defaultResolver = getComponent("default");
037        if (defaultResolver != null) return defaultResolver;
038        return getComponent("native.random");
039    }
040
041    private Resolver createNativeRandomResolver() {
042        RandomResolver resolver = new RandomResolver();
043        resolver.initId(ComponentId.fromString(RandomResolver.nativeId));
044        return resolver;
045    }
046
047    private DeterministicResolver createNativeDeterministicResolver() {
048        DeterministicResolver resolver = new DeterministicResolver();
049        resolver.initId(ComponentId.fromString(DeterministicResolver.nativeId));
050        return resolver;
051    }
052
053    private void registerResolver(Resolver resolver) {
054        super.register(resolver.getId(), resolver);
055    }
056
057    public Resolver defaultResolver() { return defaultResolver; }
058
059}
