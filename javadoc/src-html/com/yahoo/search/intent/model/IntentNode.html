002package com.yahoo.search.intent.model;
003
004/**
005 * An intent in an intent model tree. The intent node score is the <i>probability</i> of this intent
006 * given the parent interpretation.
007 *
008 * @author bratseth
009 */
010public class IntentNode extends ParentNode<SourceNode> {
011
012    private Intent intent;
013
014    public IntentNode(Intent intent,double probabilityScore) {
015        super(probabilityScore);
016        this.intent=intent;
017    }
018
019    /** Returns the intent of this node, this is never null */
020    public Intent getIntent() { return intent; }
021
022    public void setIntent(Intent intent) { this.intent=intent; }
023
024    /** Returns intent:probability */
025    public @Override String toString() {
026        return intent + ":" + getScore();
027    }
028
029}
