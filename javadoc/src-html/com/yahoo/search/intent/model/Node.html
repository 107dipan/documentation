002package com.yahoo.search.intent.model;
003
004import java.util.Map;
005
006/**
007 * A node in the <a href="TODO">intent model tree</a>
008 *
009 * @author bratseth
010 */
011public abstract class Node implements Comparable<Node> {
012
013    /**
014     * The score, unless getScore/setScore is overridden which is the case with interpretations,
015     * so DO NOT ACCESS SCORE DIRECTLY, ALWAYS USE GET/SET
016     */
017    private double score;
018
019    public Node(double score) {
020        this.score=score;
021    }
022
023    /** Returns the normalized (0-1) score of this node */
024    public double getScore() { return score; }
025
026    /** Sets the normalized (0-1) score of this node */
027    public void setScore(double score) { this.score=score; }
028
029    /** Increases this score by an increment and returns the new score */
030    public double increaseScore(double increment) {
031        setScore(getScore()+increment);
032        return getScore();
033    }
034
035    public int compareTo(Node other) {
036        if (this.getScore()<other.getScore()) return 1;
037        if (this.getScore()>other.getScore()) return -1;
038        return 0;
039    }
040
041    /**
042     * Adds the sources at (and beneath) this node to the given
043     * sparsely represented source vector, weighted by the score of this node
044     * times the given weight from the parent path
045     */
046    abstract void addSources(double weight,Map<Source,SourceNode> sources);
047
048}
