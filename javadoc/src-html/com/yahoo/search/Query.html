060/**
061 * A search query containing all the information required to produce a Result.
062 * <p>
063 * The Query contains:
064 * <ul>
065 * <li>the selection criterion received in the request - which may be a structured boolean tree of operators,
066 * an annotated piece of natural language text received from a user, or a combination of both
067 * <li>a set of field containing the additional general parameters of a query - number of hits,
068 * ranking, presentation etc.
069 * <li>a Map of properties, which can be of any object type
070 * </ul>
071 *
072 * <p>
073 * The properties has three sources
074 * <ol>
075 * <li>They may be set in some Searcher component already executed for this Query - the properties acts as
076 * a blackboard for communicating arbitrary objects between Searcher components.
077 * <li>Properties set in the search Request received - the properties acts as a way to parametrize Searcher
078 * components from the Request.
079 * <li>Properties defined in the selected {@link com.yahoo.search.query.profile.QueryProfile} - this provides
080 * defaults for the parameters to Searcher components. Note that by using query profile types, the components may
081 * define the set of parameters they support.
082 * </ol>
083 * When looked up, the properties are accessed in the priority order listed above.
084 * <p>
085 * The identity of a query is determined by its content.
086 *
087 * @author Arne Bergene Fossaa
088 * @author bratseth
089 */
090public class Query extends com.yahoo.processing.Request implements Cloneable {

096    public enum Type {
097
098        ALL(0,"all"),
099        ANY(1,"any"),
100        PHRASE(2,"phrase"),
101        ADVANCED(3,"adv"),
102        WEB(4,"web"),
103        PROGRAMMATIC(5, "prog"),
104        YQL(6, "yql");
105
106        private final int intValue;
107        private final String stringValue;
108
109        Type(int intValue,String stringValue) {
110            this.intValue = intValue;
111            this.stringValue = stringValue;
112        }
113
114        /** Converts a type argument value into a query type */
115        public static Type getType(String typeString) {
116            for (Type type:Type.values())
117                if(type.stringValue.equals(typeString))
118                    return type;
119            return ALL;
120        }
121
122        public int asInt() { return intValue; }
123
124        public String toString() { return stringValue; }

299    private void init(Map<String, String> requestMap, CompiledQueryProfile queryProfile) {
300        startTime = System.currentTimeMillis();
301        if (queryProfile != null) {
302            // Move all request parameters to the query profile just to validate that the parameter settings are legal
303            Properties queryProfileProperties=new QueryProfileProperties(queryProfile);
304            properties().chain(queryProfileProperties);
305            // TODO: Just checking legality rather than actually setting would be faster
306            setPropertiesFromRequestMap(requestMap, properties()); // Adds errors to the query for illegal set attempts
307
308            // Create the full chain
309            properties().chain(new QueryProperties(this, queryProfile.getRegistry())).
310                         chain(new ModelObjectMap()).
311                         chain(new RequestContextProperties(requestMap)).
312                         chain(queryProfileProperties).
313                         chain(new DefaultProperties());
314
315            // Pass the values from the query profile which maps through a field in the Query object model
316            // through the property chain to cause those values to be set in the Query object model
317            setFieldsFrom(queryProfileProperties, requestMap);
318        }
319        else { // bypass these complications if there is no query profile to get values from and validate against
320            properties().
321                    chain(new QueryProperties(this, CompiledQueryProfileRegistry.empty)).
322                    chain(new PropertyMap()).
323                    chain(new DefaultProperties());
324            setPropertiesFromRequestMap(requestMap, properties());
325        }
326
327        properties().setParentQuery(this);
328        traceProperties();
329    }

353     * Calls properties().set on each value in the given properties which is declared in this query or
354     * one of its dependent objects. This will ensure the appropriate setters are called on this and all
355     * dependent objects for the appropriate subset of the given property values

357    private void setFieldsFrom(Properties properties, Map<String,String> context) {
358        setFrom(properties,Query.getArgumentType(), context);
359        setFrom(properties,Model.getArgumentType(), context);
360        setFrom(properties,Presentation.getArgumentType(), context);
361        setFrom(properties,Ranking.getArgumentType(), context);

364    /**
365     * For each field in the given query profile type, take the corresponding value from originalProperties
366     * (if any) set it to properties().
367     */
368    private void setFrom(Properties originalProperties,QueryProfileType arguments,Map<String,String> context) {
369        String prefix=getPrefix(arguments);
370        for (FieldDescription field : arguments.fields().values()) {
371            String fullName=prefix + field.getName();
372            if (field.getType() == FieldType.genericQueryProfileType) {
373                for (Map.Entry<String, Object> entry : originalProperties.listProperties(fullName,context).entrySet()) {
374                    try {
375                        properties().set(fullName + "." + entry.getKey(), entry.getValue(), context);
376                    } catch (IllegalArgumentException e) {
377                        throw new QueryException("Invalid request parameter", e);
378                    }
379                }
380            } else {
381                Object value=originalProperties.get(fullName,context);
382                if (value!=null) {
383                    try {
384                        properties().set(fullName,value,context);
385                    } catch (IllegalArgumentException e) {
386                        throw new QueryException("Invalid request parameter", e);
387                    }
388                }
389            }
390        }
391    }

393    /** Calls properties.set on all entries in requestMap */
394    private void setPropertiesFromRequestMap(Map<String, String> requestMap, Properties properties) {
395        for (Map.Entry<String, String> entry : requestMap.entrySet()) {
396            try {
397                if (entry.getKey().equals("queryProfile")) continue;
398                properties.set(entry.getKey(), entry.getValue(), requestMap);
399            }
400            catch (IllegalArgumentException e) {
401                throw new QueryException("Invalid request parameter", e);
402            }
403        }
404    }

410    /**
411     * Traces how properties was resolved and from where. Done after the fact to avoid special handling
412     * of tracelevel, which is the property deciding whether this needs to be done
413     */
414    private void traceProperties() {
415        if (traceLevel==0) return;
416        CompiledQueryProfile profile=null;
417        QueryProfileProperties profileProperties=properties().getInstance(QueryProfileProperties.class);
418        if (profileProperties!=null)
419            profile=profileProperties.getQueryProfile();
420
421        if (profile==null)
422            trace("No query profile is used", false, 1);
423        else
424            trace("Using " + profile.toString(), false, 1);
425        if (traceLevel<4) return;
426
427        StringBuilder b=new StringBuilder("Resolved properties:\n");
428        Set<String> mentioned=new HashSet<>();
429        for (Map.Entry<String,String> requestProperty : requestProperties().entrySet() ) {
430            Object resolvedValue = properties().get(requestProperty.getKey(), requestProperties());
431            if (resolvedValue == null && requestProperty.getKey().equals("queryProfile"))
432                resolvedValue = requestProperty.getValue();
433
434            b.append(requestProperty.getKey());
435            b.append("=");
436            b.append(String.valueOf(resolvedValue)); // (may be null)
437            b.append(" (");
438
439            if (profile != null && ! profile.isOverridable(new CompoundName(requestProperty.getKey()), requestProperties()))
440                b.append("value from query profile - unoverridable, ignoring request value");
441            else
442                b.append("value from request");
443            b.append(")\n");
444            mentioned.add(requestProperty.getKey());
445        }
446        if (profile!=null) {
447            appendQueryProfileProperties(profile,mentioned,b);
448        }
449        trace(b.toString(),false,4);
450    }

452    private Map<String, String> requestProperties() {
453        return httpRequest.propertyMap();
454    }

456    private void appendQueryProfileProperties(CompiledQueryProfile profile,Set<String> mentioned,StringBuilder b) {
457        for (Map.Entry<String,Object> property : profile.listValues("",requestProperties()).entrySet()) {
458            if ( ! mentioned.contains(property.getKey()))
459                b.append(property.getKey() + "=" + property.getValue() + " (value from query profile)<br/>\n");
460        }
461    }

463    /**
464     * Validates this query
465     *
466     * @return the reason if it is invalid, null if it is valid
467     */
468    public String validate() {
469        // Validate the query profile
470        QueryProfileProperties queryProfileProperties = properties().getInstance(QueryProfileProperties.class);
471        if (queryProfileProperties == null) return null; // Valid
472        StringBuilder missingName = new StringBuilder();
473        if (! queryProfileProperties.isComplete(missingName, httpRequest.propertyMap()))
474            return "Incomplete query: Parameter '" + missingName + "' is mandatory in " +
475                   queryProfileProperties.getQueryProfile() + " but is not set";
476        else
477            return null; // is valid
478    }

677     * @param traceLevel   the context level of the message, this method will do nothing
678     *                     if the traceLevel of the query is lower than this value
679     */
680    public void trace(String message, boolean includeQuery, int traceLevel) {
681        if ( ! isTraceable(traceLevel)) return;
682
683        if (includeQuery)
684            message += ": [" + queryTreeText() + "]";
685
686        log.log(LogLevel.DEBUG,message);
687
688        // Pass 0 as traceLevel as the trace level check is already done above,
689        // and it is not propagated to trace until execution has started
690        // (it is done in the execution.search method)
691        getContext(true).trace(message, 0);
692    }

694    /**
695     * Adds a trace message to this query
696     * if the trace level of the query is sufficiently high.
697     *
698     * @param includeQuery true to append the query root stringValue at the end of the message
699     * @param traceLevel   the context level of the message, this method will do nothing
700     *                     if the traceLevel of the query is lower than this value
701     * @param messages     the messages whose toStrings will be concatenated into the trace message.
702     *                     Concatenation will only happen if the trace level is sufficiently high.
703     */
704    public void trace(boolean includeQuery, int traceLevel, Object... messages) {
705        if ( ! isTraceable(traceLevel)) return;
706
707        StringBuilder concatenated = new StringBuilder();
708        for (Object message : messages)
709            concatenated.append(String.valueOf(message));
710        trace(concatenated.toString(), includeQuery, traceLevel);
711    }

713    /**
714     * Set the context information for another query to be part of this query's
715     * context information. This is to be used if creating fresh query objects as
716     * part of a plug-in's execution. The query should be attached before it is
717     * used, in case an exception causes premature termination. This is enforced
718     * by an IllegalStateException. In other words, intended use is create the
719     * new query, and attach the context to the invoking query as soon as the new
720     * query is properly initialized.
721     *
722     * <p>
723     * This method will always set the argument query's context level to the context
724     * level of this query.
725     *
726     * @param query
727     *                The query which should be traced as a part of this query.
728     * @throws IllegalStateException
729     *                 If the query given as argument already has context
730     *                 information.
731     */
732    public void attachContext(Query query) throws IllegalStateException {
733        query.setTraceLevel(getTraceLevel());
734        if (context == null) {
735            // Nothing to attach to. This is about the same as
736            // getTraceLevel() == 0,
737            // but is a direct test of what will make the function superfluous.
738            return;
739        }
740        if (query.getContext(false) != null) {
741            // If we added the other query's context info as a subnode in this
742            // query's context tree, we would have to check for loops in the
743            // context graph. If we simply created a new node without checking,
744            // we might silently overwrite useful information.
745            throw new IllegalStateException("Query to attach already has context information stored.");
746        }
747        query.context = context;
748    }

750    private String queryTreeText() {
751        QueryTree root = getModel().getQueryTree();
752
753        if (getTraceLevel() < 2)
754            return root.toString();
755        if (getTraceLevel() < 6)
756            return yqlRepresentation();
757        else
758            return "\n" + yqlRepresentation() + "\n" + new TextualQueryRepresentation(root.getRoot()) + "\n";
759    }

781    private void commaSeparated(StringBuilder yql, Set<String> fields) {
782        int initLen = yql.length();
783        for (String field : fields) {
784            if (yql.length() > initLen) {
785                yql.append(", ");
786            }
787            yql.append(field);
788        }
789    }

791    /**
792     * Serialize this query as YQL+. This will create a string representation
793     * which should always be legal YQL+. If a problem occurs, a
794     * RuntimeException is thrown.
795     *
796     * @param segmenterVersion
797     *            linguistics metadata used in federation, set to null if the
798     *            annotation is not necessary
799     * @param includeHitsAndOffset
800     *            whether to include hits and offset parameters converted to a
801     *            offset/limit slice
802     * @return a valid YQL+ query string
803     * @throws RuntimeException if there is a problem serializing the query tree
804     */
805    public String yqlRepresentation(@Nullable Tuple2<String, Version> segmenterVersion, boolean includeHitsAndOffset) {
806        String q = VespaSerializer.serialize(this);
807
808        Set<String> sources = getModel().getSources();
809        Set<String> fields = getPresentation().getSummaryFields();
810        StringBuilder yql = new StringBuilder("select ");
811        if (fields.isEmpty()) {
812            yql.append('*');
813        } else {
814            commaSeparated(yql, fields);
815        }
816        yql.append(" from ");
817        if (sources.isEmpty()) {
818            yql.append("sources *");
819        } else {
820            if (sources.size() > 1) {
821                yql.append("sources ");
822            }
823            commaSeparated(yql, sources);
824        }
825        yql.append(" where ");
826        if (segmenterVersion != null) {
827            yql.append("[{\"segmenter\": {\"version\": \"")
828                    .append(segmenterVersion.second.toString())
829                    .append("\", \"backend\": \"")
830                    .append(segmenterVersion.first).append("\"}}](");
831        }
832        yql.append(q);
833        if (segmenterVersion != null) {
834            yql.append(')');
835        }
836        if (getRanking().getSorting() != null && getRanking().getSorting().fieldOrders().size() > 0) {
837            serializeSorting(yql);
838        }
839        if (includeHitsAndOffset) {
840            if (getOffset() != 0) {
841                yql.append(" limit ")
842                        .append(Integer.toString(getHits() + getOffset()))
843                        .append(" offset ")
844                        .append(Integer.toString(getOffset()));
845            } else if (getHits() != 10) {
846                yql.append(" limit ").append(Integer.toString(getHits()));
847            }
848        }
849        if (getTimeout() != 5000L) {
850            yql.append(" timeout ").append(Long.toString(getTimeout()));
851        }
852        yql.append(';');
853        return yql.toString();
854    }
855
856    private void serializeSorting(StringBuilder yql) {
857        yql.append(" order by ");
858        int initLen = yql.length();
859        for (FieldOrder f : getRanking().getSorting().fieldOrders()) {
860            if (yql.length() > initLen) {
861                yql.append(", ");
862            }
863            final Class<? extends AttributeSorter> sorterType = f.getSorter()
864                    .getClass();
865            if (sorterType == Sorting.RawSorter.class) {
866                yql.append("[{\"").append(YqlParser.SORTING_FUNCTION)
867                        .append("\": \"").append(Sorting.RAW).append("\"}]");
868            } else if (sorterType == Sorting.LowerCaseSorter.class) {
869                yql.append("[{\"").append(YqlParser.SORTING_FUNCTION)
870                        .append("\": \"").append(Sorting.LOWERCASE)
871                        .append("\"}]");
872            } else if (sorterType == Sorting.UcaSorter.class) {
873                Sorting.UcaSorter uca = (Sorting.UcaSorter) f.getSorter();
874                String ucaLocale = uca.getLocale();
875                Sorting.UcaSorter.Strength ucaStrength = uca.getStrength();
876                yql.append("[{\"").append(YqlParser.SORTING_FUNCTION)
877                        .append("\": \"").append(Sorting.UCA).append("\"");
878                if (ucaLocale != null) {
879                    yql.append(", \"").append(YqlParser.SORTING_LOCALE)
880                            .append("\": \"").append(ucaLocale).append('"');
881                }
882                if (ucaStrength != Sorting.UcaSorter.Strength.UNDEFINED) {
883                    yql.append(", \"").append(YqlParser.SORTING_STRENGTH)
884                            .append("\": \"").append(ucaStrength.name())
885                            .append('"');
886                }
887                yql.append("}]");
888            }
889            yql.append(f.getFieldName());
890            if (f.getSortOrder() == Order.DESCENDING) {
891                yql.append(" desc");
892            }
893        }
894    }

995    /**
996     * Encodes properties of this query.
997     *
998     * @param buffer the buffer to encode to
999     * @param encodeQueryData true to encode all properties, false to only include session information, not actual query data
1000     * @return the encoded length
1001     */
1002    public int encodeAsProperties(ByteBuffer buffer, boolean encodeQueryData) {
1003        // Make sure we don't encode anything here if we have turned the property feature off
1004        // Due to sendQuery we sometimes end up turning this feature on and then encoding a 0 int as the number of
1005        // property maps - that's ok (probably we should simplify by just always turning the feature on)
1006        if (! hasEncodableProperties()) return 0;
1007
1008        int start = buffer.position();
1009
1010        int mapCountPosition = buffer.position();
1011        buffer.putInt(0); // map count will go here
1012
1013        int mapCount = 0;
1014
1015        // TODO: Push down
1016        mapCount += ranking.getProperties().encode(buffer, encodeQueryData);
1017        if (encodeQueryData) mapCount += ranking.getFeatures().encode(buffer);
1018
1019        // TODO: Push down
1020        if (encodeQueryData && presentation.getHighlight() != null) mapCount += MapEncoder.encodeStringMultiMap(Highlight.HIGHLIGHTTERMS, presentation.getHighlight().getHighlightTerms(), buffer);
1021
1022        // TODO: Push down
1023        if (encodeQueryData) mapCount += MapEncoder.encodeSingleValue("model", "searchpath", model.getSearchPath(), buffer);
1024        mapCount += MapEncoder.encodeSingleValue(DocumentDatabase.MATCH_PROPERTY, DocumentDatabase.SEARCH_DOC_TYPE_KEY, model.getDocumentDb(), buffer);
1025
1026        mapCount += MapEncoder.encodeMap("caches", createCacheSettingMap(), buffer);
1027
1028        buffer.putInt(mapCountPosition, mapCount);
1029
1030        return buffer.position() - start;
1031    }

1034        if (getGroupingSessionCache() && ranking.getQueryCache()) {
