123    private static class Tag {
124        public final String name;
125
126        /**
127         * Offset is a number which is generated for all data and tags inside
128         * fields, used to determine whether a tag was closed without enclosing
129         * any characters or other tags.
130         */
131        public final int offset;
132
133        public Tag(final String name, final int offset) {
134            this.name = name;
135            this.offset = offset;
136        }
137
138        @Override
139        public String toString() {
140            return name + '(' + Integer.valueOf(offset) + ')';
141        }
142    }

155    public static XMLReader createParser() {
156        ClassLoader savedContextClassLoader = Thread.currentThread().getContextClassLoader();
157        Thread.currentThread().setContextClassLoader(org.apache.xerces.parsers.SAXParser.class.getClassLoader());
158
159        try {
160            XMLReader reader = XMLReaderFactory.createXMLReader(DEFAULT_PARSER_NAME);
161            setParserFeatures(reader);
162            return reader;
163        } catch (Exception e) {
164            throw new RuntimeException("error: Unable to instantiate parser ("
165                    + DEFAULT_PARSER_NAME + ")", e);
166        } finally {
167            Thread.currentThread().setContextClassLoader(savedContextClassLoader);
168        }
169    }

171    private static void setParserFeatures(XMLReader reader) {
172        try {
173            reader.setFeature(NAMESPACES_FEATURE_ID, DEFAULT_NAMESPACES);
174        } catch (SAXException e) {
175            log.log(LogLevel.WARNING, "warning: Parser does not support feature ("
176                    + NAMESPACES_FEATURE_ID + ")");
177        }
178        try {
179            reader.setFeature(NAMESPACE_PREFIXES_FEATURE_ID,
180                    DEFAULT_NAMESPACE_PREFIXES);
181        } catch (SAXException e) {
182            log.log(LogLevel.WARNING, "warning: Parser does not support feature ("
183                    + NAMESPACE_PREFIXES_FEATURE_ID + ")");
184        }
185        try {
186            reader.setFeature(VALIDATION_FEATURE_ID, DEFAULT_VALIDATION);
187        } catch (SAXException e) {
188            log.log(LogLevel.WARNING, "warning: Parser does not support feature ("
189                    + VALIDATION_FEATURE_ID + ")");
190        }
191        try {
192            reader.setFeature(SCHEMA_VALIDATION_FEATURE_ID,
193                    DEFAULT_SCHEMA_VALIDATION);
194        } catch (SAXNotRecognizedException e) {
195            log.log(LogLevel.WARNING, "warning: Parser does not recognize feature ("
196                    + SCHEMA_VALIDATION_FEATURE_ID + ")");
197
198        } catch (SAXNotSupportedException e) {
199            log.log(LogLevel.WARNING, "warning: Parser does not support feature ("
200                    + SCHEMA_VALIDATION_FEATURE_ID + ")");
201        }
202
203        try {
204            reader.setFeature(DYNAMIC_VALIDATION_FEATURE_ID,
205                    DEFAULT_DYNAMIC_VALIDATION);
206        } catch (SAXNotRecognizedException e) {
207            log.log(LogLevel.WARNING, "warning: Parser does not recognize feature ("
208                    + DYNAMIC_VALIDATION_FEATURE_ID + ")");
209
210        } catch (SAXNotSupportedException e) {
211            log.log(LogLevel.WARNING, "warning: Parser does not support feature ("
212                    + DYNAMIC_VALIDATION_FEATURE_ID + ")");
213        }
214    }

216    @Override
217    public void startDocument() throws SAXException {
218        reset();
219        result = new Result(query);
220        hitGroups.addFirst(result.hits());
221        location.addFirst(ResultPart.ROOT);
222        return;
223    }

225    private void reset() {
226        result = null;
227        fieldLevel = 0;
228        hasLiteralTags = false;
229        tagStack = null;
230        fieldContent = null;
231        offset = 0;
232        currentError = null;
233        currentErrorCode = null;
234        hitGroups.clear();
235        location.clear();
236    }

238    @Override
239    public void startElement(String uri, String local, String raw,
240            Attributes attrs) throws SAXException {
241        // "Everybody" wants this switch to be moved into the
242        // enum class instead, but in this case, I find the classic
243        // approach more readable.
244        switch (location.peekFirst()) {
245        case HIT:
246            if (fieldLevel > 0) {
247                tagInField(raw, attrs, FIELD);
248                ++offset;
249                return;
250            }
251            if (FIELD.equals(raw)) {
252                ++fieldLevel;
253                fieldName = attrs.getValue("name");
254                fieldContent = new StringBuilder();
255                hasLiteralTags = false;
256            }
257            break;
258        case ERRORDETAILS:
259            if (fieldLevel > 0) {
260                tagInField(raw, attrs, ERROR);
261                ++offset;
262                return;
263            }
264            if (ERROR.equals(raw)) {
265                if (attrs != null) {
266                    currentErrorCode = attrs.getValue("code");
267                    currentError = attrs.getValue("error");
268                }
269                ++fieldLevel;
270                fieldContent = new StringBuilder();
271                hasLiteralTags = false;
272            }
273            break;
274        case HITGROUP:
275            if ("hit".equals(raw)) {
276                startHit(attrs);
277            } else if ("group".equals(raw)) {
278                startHitGroup(attrs);
279            }
280            break;
281        case ROOT:
282            if ("hit".equals(raw)) {
283                startHit(attrs);
284            } else if ("errordetails".equals(raw)) {
285                location.addFirst(ResultPart.ERRORDETAILS);
286            } else if ("result".equals(raw)) {
287                if (attrs != null) {
288                    String total = attrs.getValue("total-hit-count");
289                    if (total != null) {
290                        result.setTotalHitCount(Long.valueOf(total));
291                    }
292                }
293            } else if ("group".equals(raw)) {
294                startHitGroup(attrs);
295            } else if (ERROR.equals(raw)) {
296                if (attrs != null) {
297                    currentErrorCode = attrs.getValue("code");
298                    fieldContent = new StringBuilder();
299                }
300            }
301            break;
302        }
303        ++offset;
304    }

306    private void startHitGroup(Attributes attrs) {
307        HitGroup g = new HitGroup();
308        Set<String> types = g.types();
309
310        final String source;
311        if (attrs != null) {
312            String groupType = attrs.getValue("type");
313            if (groupType != null) {
314                for (String s : groupType.split(" ")) {
315                    if (s.length() > 0) {
316                        types.add(s);
317                    }
318                }
319            }
320
321            source = attrs.getValue("source");
322        } else {
323            source = null;
324        }
325
326        g.setId((source != null) ? source : "dummy");
327
328        hitGroups.peekFirst().add(g);
329        hitGroups.addFirst(g);
330        location.addFirst(ResultPart.HITGROUP);
331    }
332
333    private void startHit(Attributes attrs) {
334        hitFields.clear();
335        location.addFirst(ResultPart.HIT);
336        if (attrs != null) {
337            hitRelevance = attrs.getValue("relevancy");
338            hitSource = attrs.getValue("source");
339            hitType = attrs.getValue("type");
340        } else {
341            hitRelevance = null;
342            hitSource = null;
343            hitType = null;
344        }
345    }

347    private void tagInField(String tag, Attributes attrs, String enclosingTag) {
348        if (!hasLiteralTags) {
349            hasLiteralTags = true;
350            String fieldTillNow = XML.xmlEscape(fieldContent.toString(), false);
351            fieldContent = new StringBuilder(fieldTillNow);
352            tagStack = new ArrayList<>();
353        }
354        if (enclosingTag.equals(tag)) {
355            ++fieldLevel;
356        }
357        if (tagStack.size() > 0) {
358            Tag prevTag = tagStack.get(tagStack.size() - 1);
359            if (prevTag != null && (prevTag.offset + 1) == offset) {
360                fieldContent.append(">");
361            }
362        }
363        fieldContent.append("<").append(tag);
364        if (attrs != null) {
365            int attrCount = attrs.getLength();
366            for (int i = 0; i < attrCount; i++) {
367                fieldContent.append(" ").append(attrs.getQName(i))
368                        .append("=\"").append(
369                                XML.xmlEscape(attrs.getValue(i), true)).append(
370                                "\"");
371            }
372        }
373        tagStack.add(new Tag(tag, offset));
374    }
375
376    private void endElementInField(String qName, String enclosingTag) {
377        Tag prevTag = tagStack.get(tagStack.size() - 1);
378        if (qName.equals(prevTag.name) && offset == (prevTag.offset + 1)) {
379            fieldContent.append(" />");
380        } else {
381            fieldContent.append("</").append(qName).append('>');
382        }
383        if (prevTag.name.equals(qName)) {
384            tagStack.remove(tagStack.size() - 1);
385        }
386    }
387
388    private void endElementInHitField(String qName) {
389        if (FIELD.equals(qName) && --fieldLevel == 0) {
390            Object content;
391            if (hasLiteralTags) {
392                content = new XMLString(fieldContent.toString());
393            } else {
394                content = fieldContent.toString();
395            }
396            hitFields.put(fieldName, content);
397            if ("collapseId".equals(fieldName)) {
398                hitFields.put(fieldName, Integer.valueOf(content.toString()));
399            }
400            fieldName = null;
401            fieldContent = null;
402            tagStack = null;
403        } else {
404            Tag prevTag = tagStack.get(tagStack.size() - 1);
405            if (qName.equals(prevTag.name) && offset == (prevTag.offset + 1)) {
406                fieldContent.append(" />");
407            } else {
408                fieldContent.append("</").append(qName).append('>');
409            }
410            if (prevTag.name.equals(qName)) {
411                tagStack.remove(tagStack.size() - 1);
412            }
413        }
414    }
415    @Override
416    public void characters(char ch[], int start, int length)
417            throws SAXException {
418
419        switch (location.peekFirst()) {
420        case ERRORDETAILS:
421        case HIT:
422            if (fieldLevel > 0) {
423                if (hasLiteralTags) {
424                    if (tagStack.size() > 0) {
425                        Tag tag = tagStack.get(tagStack.size() - 1);
426                        if (tag != null && (tag.offset + 1) == offset) {
427                            fieldContent.append(">");
428                        }
429                    }
430                    fieldContent.append(
431                            XML.xmlEscape(new String(ch, start, length), false));
432                } else {
433                    fieldContent.append(ch, start, length);
434                }
435            }
436            break;
437        default:
438            if (fieldContent != null) {
439                fieldContent.append(ch, start, length);
440            }
441            break;
442        }
443        ++offset;
444    }

458    @Override
459    public void endElement(String namespaceURI, String localName, String qName)
460            throws SAXException {
461        switch (location.peekFirst()) {
462        case HITGROUP:
463            if ("group".equals(qName)) {
464                hitGroups.removeFirst();
465                location.removeFirst();
466            }
467            break;
468        case HIT:
469            if (fieldLevel > 0) {
470                endElementInHitField(qName);
471            }  else if ("hit".equals(qName)) {
472                 //assert(hitKeys.size() == hitValues.size());
473                 //We try to get either uri or documentID and use that as id
474                 Object docId = extractDocumentID();
475                 Hit newHit = new Hit(docId.toString());
476                 if (hitRelevance != null) newHit.setRelevance(new Relevance(DoubleParser.parse(hitRelevance)));
477                 if(hitSource != null) newHit.setSource(hitSource);
478                 if(hitType != null) {
479                     for(String type: hitType.split(" ")) {
480                         newHit.types().add(type);
481                     }
482                 }
483                 for(Map.Entry<String, Object> field : hitFields.entrySet()) {
484                     newHit.setField(field.getKey(), field.getValue());
485                 }
486
487                 hitGroups.peekFirst().add(newHit);
488                 location.removeFirst();
489            }
490            break;
491        case ERRORDETAILS:
492            if (fieldLevel == 1 && ERROR.equals(qName)) {
493                ErrorMessage error = new ErrorMessage(Integer.valueOf(currentErrorCode),
494                        currentError,
495                        fieldContent.toString());
496                hitGroups.peekFirst().addError(error);
497                currentError = null;
498                currentErrorCode = null;
499                fieldContent = null;
500                tagStack = null;
501                fieldLevel = 0;
502            } else if (fieldLevel > 0) {
503                endElementInField(qName, ERROR);
504            } else if ("errordetails".equals(qName)) {
505                location.removeFirst();
506            }
507            break;
508        case ROOT:
509            if (ERROR.equals(qName)) {
510                ErrorMessage error = new ErrorMessage(Integer.valueOf(currentErrorCode),
511                        fieldContent.toString());
512                hitGroups.peekFirst().setError(error);
513                currentErrorCode = null;
514                fieldContent = null;
515            }
516            break;
517        default:
518            break;
519        }
520        ++offset;
521    }

523    private Object extractDocumentID() {
524        Object docId = null;
525        if (hitFields.containsKey("uri")) {
526            docId = hitFields.get("uri");
527        } else {
528            final String documentId = "documentId";
529            if (hitFields.containsKey(documentId)) {
530                docId = hitFields.get(documentId);
531            } else {
532                final String lcDocumentId = toLowerCase(documentId);
533                for (Map.Entry<String, Object> e : hitFields.entrySet()) {
534                    String key = e.getKey();
535                    // case insensitive matching, checking length first hoping to avoid some lowercasing
536                    if (documentId.length() == key.length() && lcDocumentId.equals(toLowerCase(key))) {
537                        docId = e.getValue();
538                        break;
539                    }
540                }
541            }
542        }
543        if (docId == null) {
544            docId = "dummy";
545            log.info("Results from vespa backend did not contain either uri or documentId");
546        }
547        return docId;
548    }

566    /** Prints the error message. */
567    protected void printError(String type, SAXParseException ex) {
568        StringBuilder errorMessage = new StringBuilder();
569
570        errorMessage.append(type);
571        if (ex != null) {
572            String systemId = ex.getSystemId();
573            if (systemId != null) {
574                int index = systemId.lastIndexOf('/');
575                if (index != -1)
576                    systemId = systemId.substring(index + 1);
577                errorMessage.append(' ').append(systemId);
578            }
579        }
580        errorMessage.append(':')
581            .append(ex.getLineNumber())
582            .append(':')
583            .append(ex.getColumnNumber())
584            .append(": ")
585            .append(ex.getMessage());
586        log.log(LogLevel.WARNING, errorMessage.toString());
587
588    }

590    public Result parse(String identifier, Query query) {
591        Result toReturn;
592
593        setQuery(query);
594        try {
595            parser.parse(identifier);
596        } catch (SAXParseException e) {
597            // ignore
598        } catch (Exception e) {
599            log.log(LogLevel.WARNING, "Error parsing result from Vespa",e);
600            Exception se = e;
601            if (e instanceof SAXException) {
602                se = ((SAXException) e).getException();
603            }
604            if (se != null)
605                se.printStackTrace(System.err);
606            else
607                e.printStackTrace(System.err);
608        }
609        toReturn = result;
610        reset();
611        return toReturn;
612    }

622        } catch (Exception e) {

624            Exception se = e;
625            if (e instanceof SAXException) {

627            }
628            if (se != null)
629                se.printStackTrace(System.err);
630            else
631                e.printStackTrace(System.err);
632        }
633        toReturn = result;
634        reset();
635        return toReturn;
636    }
