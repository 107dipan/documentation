002package com.yahoo.search.federation.selection;
003
004import com.yahoo.processing.execution.chain.ChainRegistry;
005import com.yahoo.search.Query;
006import com.yahoo.search.Result;
007import com.yahoo.search.Searcher;
008import com.yahoo.search.federation.selection.FederationTarget;
009
010import java.util.Collection;
011
012/**
013 * Allows adding extra targets that the federation searcher should federate to.
014 *
015 * For each federation search call, the federation searcher will call targetSelector.getTargets.
016 *
017 * Then, for each target, it will:
018 * 1) call modifyTargetQuery(target, query)
019 * 2) call modifyTargetResult(target, result)
020 *
021 * @author tonytv
022 */
023public interface TargetSelector<T> {
024
025    Collection<FederationTarget<T>> getTargets(Query query, ChainRegistry<Searcher> searcherChainRegistry);
026
027    /**
028     * For modifying the query before sending it to a the target
029     */
030    void modifyTargetQuery(FederationTarget<T> target, Query query);
031
032    /**
033     * For modifying the result produced by the target.
034     */
035    void modifyTargetResult(FederationTarget<T> target, Result result);
036
037}
