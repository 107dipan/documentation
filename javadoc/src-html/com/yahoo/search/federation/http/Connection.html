002package com.yahoo.search.federation.http;
003
004/**
005 * Represents a connection to a particular node (host/port).
006 * Right now this is just a container of connection parameters, but might be extended to
007 * contain an open connection later.
008 * The host and port state is immutable.
009 *
010 * @author bratseth
011 */
012public class Connection {
013
014    private String host;
015    private int port;
016
017    public Connection(String host,int port) {
018        this.host=host;
019        this.port=port;
020    }
021
022    public String getHost() { return host; }
023
024    public int getPort() { return port; }
025
026    public String toString() {
027        return "http connection '" + host + ":" + port + "'";
028    }
029
030}
