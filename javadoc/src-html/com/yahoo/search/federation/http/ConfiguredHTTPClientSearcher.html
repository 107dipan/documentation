002package com.yahoo.search.federation.http;
003
004import java.util.Collections;
005
006import com.yahoo.component.ComponentId;
007import com.yahoo.search.federation.ProviderConfig;
008import com.yahoo.search.Result;
009import com.yahoo.search.searchchain.Execution;
010import com.yahoo.statistics.Statistics;
011
012
013/**
014 * Superclass for http client searchers which depends on config. All this is doing is translating
015 * the provider and cache configurations to parameters which are passed upwards.
016 *
017 * @author bratseth
018 */
019public abstract class ConfiguredHTTPClientSearcher extends HTTPClientSearcher {
020
021    /** Create this from a configuraton */
022    public ConfiguredHTTPClientSearcher(final ComponentId id, final ProviderConfig providerConfig, Statistics manager) {
023        super(id, ConfiguredSearcherHelper.toConnectionList(providerConfig), new HTTPParameters(providerConfig), manager);
024    }
025
026    /** Create an instance from direct parameters having a single connection. Useful for testing */
027    public ConfiguredHTTPClientSearcher(String idString,String host,int port,String path, Statistics manager) {
028        super(new ComponentId(idString), Collections.singletonList(new Connection(host,port)),path, manager);
029    }
030
031    /** Forwards to the next in chain fill(result,summaryName) */
032    public @Override void fill(Result result,String summaryName, Execution execution,Connection connection) {
033        execution.fill(result,summaryName);
034    }
035
036}
