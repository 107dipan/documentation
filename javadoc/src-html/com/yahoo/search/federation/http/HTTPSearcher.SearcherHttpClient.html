062/**
063 * Generic superclass of searchers making connections to some HTTP service. This
064 * supports clustered connections - a list of alternative servers may be given,
065 * requests will be hashed across these and failed over in case some are down.
066 * <p>
067 * This simply provides some utility methods for working with http connections
068 * and implements ping against the service.
069 *
070 * <p>This searcher contains code from the Apache httpcomponents client library,
071 * licensed to the Apache Software Foundation under the Apache License, Version
072 * 2.0. Please refer to http://www.apache.org/licenses/LICENSE-2.0 for details.
073 *
074 * <p>This class automatically adds a meta hit containing latency and other
075 * meta information about the obtained HTTP data using createRequestMeta().
076 * The fields available in the hit are:</p>
077 *
078 * <dl><dt>
079 * HTTPSearcher.LOG_LATENCY_START
080 * <dd>
081 *     The latency of the external provider answering a request.
082 * <dt>
083 * HTTPSearcher.LOG_LATENCY_FINISH
084 * <dd>
085 *     Total time of the HTTP traffic, but also decoding of the data, as this
086 *     happens at the same time.
087 * <dt>
088 * HTTPSearcher.LOG_HITCOUNT
089 * <dd>
090 *     Number of concrete hits in the result returned by this provider.
091 * <dt>
092 * HTTPSearcher.LOG_URI
093 * <dd>
094 *     The complete URI used for external service.
095 * <dt>
096 * HTTPSearcher.LOG_SCHEME
097 * <dd>
098 *     The scheme of the request URI sent.
099 * <dt>
100 * HTTPSearcher.LOG_HOST
101 * <dd>
102 *     The host used for the request URI sent.
103 * <dt>
104 * HTTPSearcher.LOG_PORT
105 * <dd>
106 *     The port used for the request URI sent.
107 * <dt>
108 * HTTPSearcher.LOG_PATH
109 * <dd>
110 *     Path element of the request URI sent.
111 * <dt>
112 * HTTPSearcher.LOG_STATUS
113 * <dd>
114 *     Status code of the HTTP response.
115 * <dt>
116 * HTTPSearcher.LOG_PROXY_TYPE
117 * <dd>
118 *     The proxy type used, if any. Default is "http".
119 * <dt>
120 * HTTPSearcher.LOG_PROXY_HOST
121 * <dd>
122 *     The proxy host, if any.
123 * <dt>
124 * HTTPSearcher.LOG_PROXY_PORT
125 * <dd>
126 *     The proxy port, if any.
127 * <dt>
128 * HTTPSearcher.LOG_HEADER_PREFIX prepended to request header field name
129 * <dd>
130 *     The content of any additional request header fields.
131 * <dt>
132 * HTTPSearcher.LOG_RESPONSE_HEADER_PREFIX prepended to response header field name
133 * <dd>
134 *     The content of any additional response header fields.
135 * </dl>
136 *
137 * @author Arne Bergene Fossaa
138 */

221    /**
222     * Creates a http searcher
223     *
224     * @param componentId the id of this instance
225     * @param connections the connections to establish to the backend nodes
226     * @param parameters the http parameters to use. This object will be frozen if it isn't already
227     */
228    @Inject
229    public HTTPSearcher(ComponentId componentId, List<Connection> connections, HTTPParameters parameters,
230                        Statistics statistics, CertificateStore certificateStore) {
231        super(componentId,connections,false);
232        String suffix = "_" + getId().getName().replace('.', '_');
233
234        connectTimeouts = new Counter(LOG_CONNECT_TIMEOUT_PREFIX + suffix, statistics, false);
235
236        parameters.freeze();
237        this.httpParameters = parameters;
238        this.certificateStore = certificateStore;
239
240        if (parameters.getPersistentConnections()) {
241            HttpParams params=parameters.toHttpParams();
242            HttpProtocolParams.setVersion(params, HttpVersion.HTTP_1_1);
243            ConnManagerParams.setTimeout(params, 10);
244            sharedConnectionManager = new ThreadSafeClientConnManager(params, schemeRegistry);
245            Thread connectionPurgerThread = new Thread(() -> {
246                //this is the default value in yahoo jvm installations
247                long DNSTTLSec = 120;
248                while (true) {
249                    try {
250                        Thread.sleep(DNSTTLSec * 1000);
251                        if (sharedConnectionManager == null)
252                            continue;
253
254                        sharedConnectionManager.closeExpiredConnections();
255                        DNSTTLSec = Long.valueOf(java.security.Security
256                                .getProperty("networkaddress.cache.ttl"));
257                        //No DNS TTL, no need to close idle connections
258                        if (DNSTTLSec <= 0) {
259                            DNSTTLSec = 120;
260                            continue;
261                        }
262                        sharedConnectionManager.closeIdleConnections(2 * DNSTTLSec, TimeUnit.SECONDS);
263                    } catch (InterruptedException e) {
264                        return;
265                    } catch (NumberFormatException e) {
266                        continue;
267                    }
268                }
269            });
270            connectionPurgerThread.setDaemon(true);
271            connectionPurgerThread.start();
272
273        }
274        else {
275            singleClientConnManagerThreadLocal =new ThreadLocal<>();
276        }
277
278        initializeYCA(httpParameters, certificateStore);
279    }

281    /**
282     * Initialize YCA certificate and proxy if they have been set to non-null,
283     * non-empty values. It will wrap thrown exceptions from the YCA layer into
284     * RuntimeException and propagate them.
285     */
286    private void initializeYCA(HTTPParameters parameters, CertificateStore certificateStore) {
287        String applicationId = parameters.getYcaApplicationId();
288        String proxy = parameters.getYcaProxy();
289        int port = parameters.getYcaPort();
290        long ttl = parameters.getYcaTtl();
291        long retry = parameters.getYcaRetry();
292
293        if (applicationId != null && !applicationId.trim().isEmpty()) {
294            initializeCertificate(applicationId, ttl, retry, certificateStore);
295        }
296
297        if (parameters.getYcaUseProxy()) {
298            initializeProxy(proxy, port);
299        }
300    }

316    /**
317     * Initialize the YCA certificate.
318     * This will warn but not throw if certificates could not be loaded, as the certificates
319     * are external state which can fail independently.
320     */
321    private void initializeCertificate(String applicationId, long ttl, long retry, CertificateStore certificateStore) {
322        try {
323            // get the certificate, i.e. init the cache and check integrity
324            String certificate = certificateStore.getCertificate(applicationId, ttl, retry);
325            if (certificate == null) {
326                getLogger().log(LogLevel.WARNING, "No certificate found for application '" + applicationId + "'");
327                return;
328            }
329
330            this.useCertificate = true;
331            this.ycaApplicationId = applicationId;
332            this.ycaTtl = ttl;
333            this.ycaRetry = retry;
334            getLogger().log(LogLevel.CONFIG, "Got certificate: " + certificate);
335        }
336        catch (Exception e) {
337            getLogger().log(LogLevel.WARNING,"Exception while initializing certificate for application '" +
338                                             applicationId + "' in " + this, e);
339        }
340    }

364    /**
365     * Creates the URI for a query.
366     * Populates the {@code requestMeta} meta hit with the created URI HTTP properties.
367     *
368     * @param requestMeta a meta hit that holds logging information about this request (may be {@code null}).
369     */
370    protected URI getURI(Query query, Hit requestMeta, Connection connection)
371            throws MalformedURLException, URISyntaxException {
372        StringBuilder parameters = new StringBuilder();
373
374        Map<String, String> queries = getQueryMap(query);
375        if (queries.size() > 0) {
376            Iterator<Map.Entry<String, String>> mapIterator = queries.entrySet().iterator();
377            parameters.append("?");
378            try {
379                Map.Entry<String, String> entry;
380                while (mapIterator.hasNext()) {
381                    entry = mapIterator.next();
382
383                    if (requestMeta != null)
384                        requestMeta.setField(LOG_QUERY_PARAM_PREFIX
385                                + entry.getKey(), entry.getValue());
386
387                    parameters.append(entry.getKey() + "=" + URLEncoder.encode(entry.getValue(),
388                                                                               httpParameters.getInputEncoding()));
389                    if (mapIterator.hasNext()) {
390                        parameters.append("&");
391                    }
392                }
393            } catch (UnsupportedEncodingException e) {
394                throw new RuntimeException("Unknown input encoding set in " + this, e);
395            }
396        }
397
398        URI uri = new URL(httpParameters.getSchema(), connection.getHost(),
399                          connection.getPort(), getPath() + parameters.toString()).toURI();
400        if (requestMeta != null) {
401            requestMeta.setField(LOG_URI, uri.toString());
402            requestMeta.setField(LOG_SCHEME, uri.getScheme());
403            requestMeta.setField(LOG_HOST, uri.getHost());
404            requestMeta.setField(LOG_PORT, uri.getPort());
405            requestMeta.setField(LOG_PATH, uri.getPath());
406        }
407        return uri;
408    }

444    /**
445     * Checks if the response is valid.
446     * @param response The response from the ping request
447     * @param pong The pong result to return back to the calling method. This method
448     * will add an error to the pong result (using addError) if the status of the HTTP response is 300 or above.
449     */
450    protected void checkPing(HttpResponse response, Pong pong) {
451        if (response.getStatusLine().getStatusCode() >= 300) {
452            pong.addError(com.yahoo.search.result.ErrorMessage.createBackendCommunicationError(
453                    "Got error " + response.getStatusLine().getStatusCode()
454                    + " when contacting backend")
455            );
456        }
457    }

459    /**
460     * Pinging in HTTPBackend is done by creating a PING uri from http://host:port/path.
461     * If this returns a status that is below 300, the ping is considered good.
462     *
463     * If another uri is needed for pinging, reimplement getPingURI.
464     *
465     * Override either this method to change how ping
466     */
467    @Override
468    public Pong ping(Ping ping, Connection connection) {
469        URI uri = null;
470        Pong pong = new Pong();
471        HttpResponse response = null;
472
473        if (httpParameters.getPingOption() == PingOption.DISABLE)
474            return pong;
475
476        try {
477            uri = getPingURI(connection);
478            if (uri == null)
479                pong.addError(ErrorMessage.createIllegalQuery("Ping uri is null"));
480            if (uri.getHost()==null) {
481                pong.addError(ErrorMessage.createIllegalQuery("Ping uri has no host"));
482                uri=null;
483            }
484        } catch (MalformedURLException | URISyntaxException e) {
485            pong.addError(ErrorMessage.createIllegalQuery("Malformed ping uri '" + uri + "': " +
486                                                          Exceptions.toMessageString(e)));
487        } catch (RuntimeException e) {
488            log.log(Level.WARNING,"Unexpected exception while attempting to ping " + connection + 
489                                  " using uri '" + uri + "'",e);
490            pong.addError(ErrorMessage.createIllegalQuery("Unexpected problem with ping uri '" + uri + "': " +
491                                                          Exceptions.toMessageString(e)));
492        }
493
494        if (uri == null) return pong;
495        pong.setPingInfo("using uri '" + uri + "'");
496
497        try {
498            response = getPingResponse(uri, ping);
499            checkPing(response, pong);
500        } catch (IOException e) {
501            // We do not have a valid ping
502            pong.addError(ErrorMessage.createBackendCommunicationError(
503                          "Exception thrown when pinging with url '" + uri + "': " + Exceptions.toMessageString(e)));
504        } catch (TimeoutException e) {
505            pong.addError(ErrorMessage.createTimeout("Timeout for ping " + uri + " in " + this + ": " + e.getMessage()));
506        } catch (RuntimeException e) {
507            log.log(Level.WARNING,"Unexpected exception while attempting to ping " + connection + " using uri '" + uri + "'",e);
508            pong.addError(ErrorMessage.createIllegalQuery("Unexpected problem with ping uri '" + uri + "': " +
509                                                          Exceptions.toMessageString(e)));
510        } finally {
511            if (response != null) {
512                cleanupHttpEntity(response.getEntity());
513            }
514        }
515
516        return pong;
517    }
518
519    private HttpResponse getPingResponse(URI uri, Ping ping) throws IOException {
520        long timeLeft = ping.getTimeout();
521        int connectionTimeout = (int) (timeLeft / 4L);
522        int readTimeout = (int) (timeLeft * 3L / 4L);
523
524        Map<String, String> requestHeaders = null;
525        if (httpParameters.getPingOption() == PingOption.YCA)
526            requestHeaders = generateYCAHeaders();
527
528        return getResponse(uri, null, requestHeaders, null, connectionTimeout, readTimeout);
529    }

542    /**
543     * Gets the HTTP entity that holds the response contents.
544     * @param uri the request URI.
545     * @param requestMeta a meta hit that holds logging information about this request (may be {@code null}).
546     * @param query the originating query
547     * @return the http entity, or null if none
548     * @throws java.io.IOException Whenever HTTP status code is in the 300 or higher range.
549     * @throws TimeoutException If query.timeLeft() equal to or lower than 0
550     */
551    protected HttpEntity getEntity(URI uri, Hit requestMeta, Query query) throws IOException {
552        if (query.getTimeLeft() <= 0) {
553            throw new TimeoutException("No time left for querying external backend.");
554        }
555        HttpResponse response = getResponse(uri, requestMeta, query);
556        StatusLine statusLine = response.getStatusLine();
557
558        // Logging
559        if (requestMeta != null) {
560                requestMeta.setField(LOG_STATUS, statusLine.getStatusCode());
561                for (HeaderIterator headers = response.headerIterator(); headers.hasNext(); ) {
562                        Header h = headers.nextHeader();
563                        requestMeta.setField(LOG_RESPONSE_HEADER_PREFIX + h.getName(), h.getValue());
564                }
565        }
566
567        if (statusLine.getStatusCode() >= 300) {
568            HttpEntity entity = response.getEntity();
569            String message = createServerReporterErrorMessage(statusLine, entity);
570            cleanupHttpEntity(response.getEntity());
571            throw new IOException(message);
572        }
573
574        return response.getEntity();
575    }

577    private String createServerReporterErrorMessage(StatusLine statusLine, HttpEntity entity) {
578        String message = "Error when trying to connect to HTTP backend: "
579                         + statusLine.getStatusCode() + " : " + statusLine.getReasonPhrase();
580
581        try {
582            if (entity != null) {
583                message += "(Message = " + EntityUtils.toString(entity) + ")";
584            }
585        } catch (Exception e) {
586            log.log(LogLevel.WARNING, "Could not get message.", e);
587        }
588
589        return message;
590    }

603    protected void cleanupHttpEntity(HttpEntity entity) {
604        if (entity == null) return;
605
606        try {
607            entity.consumeContent();
608        } catch (IOException e) {
609            // It is ok if do not consume it, the resource will be freed after
610            // timeout.
611            // But log it just in case.
612            log.log(LogLevel.getVespaLogLevel(LogLevel.DEBUG),
613                    "Not able to consume after processing: " + Exceptions.toMessageString(e));
614        }
615    }

624    /**
625     * Executes an HTTP request and gets the response.
626     * @param uri the request URI.
627     * @param requestMeta a meta hit that holds logging information about this request (may be {@code null}).
628     * @param query the originating query, used to calculate timeouts
629     */
630    protected HttpResponse getResponse(URI uri, Hit requestMeta, Query query) throws IOException {
631        long timeLeft = query.getTimeLeft();
632        int connectionTimeout = (int) (timeLeft / 4L);
633        int readTimeout = (int) (timeLeft * 3L / 4L);
634        connectionTimeout = connectionTimeout <= 0 ? 1 : connectionTimeout;
635        readTimeout = readTimeout <= 0 ? 1 : readTimeout;
636        HttpEntity reqEntity = getRequestEntity(query, requestMeta);
637        Map<String, String> reqHeaders = getRequestHeaders(query, requestMeta);
638        if ((reqEntity == null) && (reqHeaders == null)) {
639            return getResponse(uri, requestMeta, connectionTimeout, readTimeout);
640        } else {
641            return getResponse(uri, reqEntity, reqHeaders, requestMeta, connectionTimeout, readTimeout);
642        }
643    }

682    /**
683     * Executes an HTTP request and gets the response.
684     * @param uri the request URI.
685     * @param requestMeta a meta hit that holds logging information about this request (may be {@code null}).
686     * @param connectionTimeout how long to wait for getting a connection
687     * @param readTimeout timeout for reading HTTP data
688     */
689    protected HttpResponse getResponse(URI uri, HttpEntity reqEntity,
690                                       Map<String, String> reqHeaders, Hit requestMeta,
691                                       int connectionTimeout, int readTimeout) throws IOException {
692
693        HttpParams httpParams = httpParameters.toHttpParams(connectionTimeout, readTimeout);
694        HttpClient httpClient = createClient(httpParams);
695        long start = 0L;
696        HttpUriRequest request;
697        if (httpParameters.getEnableProxy() && "http".equals(httpParameters.getProxyType())) {
698            HttpHost proxy = new HttpHost(httpParameters.getProxyHost(),
699                                          httpParameters.getProxyPort(), httpParameters.getProxyType());
700            httpClient.getParams().setParameter(ConnRoutePNames.DEFAULT_PROXY, proxy);
701            // Logging
702            if (requestMeta != null) {
703                requestMeta.setField(LOG_PROXY_TYPE, httpParameters.getProxyType());
704                requestMeta.setField(LOG_PROXY_HOST, httpParameters.getProxyHost());
705                requestMeta.setField(LOG_PROXY_PORT, httpParameters.getProxyPort());
706            }
707        }
708        if (reqEntity == null) {
709            request = createRequest(httpParameters.getMethod(), uri);
710        } else {
711            request = createRequest(httpParameters.getMethod(), uri, reqEntity);
712        }
713
714        if (reqHeaders != null) {
715            for (Entry<String, String> entry : reqHeaders.entrySet()) {
716                if (entry.getValue() == null || isAscii(entry.getValue())) {
717                    request.addHeader(entry.getKey(), entry.getValue());
718                } else {
719                    byte[] asBytes = Utf8.toBytes(entry.getValue());
720                    String asLyingString = new String(asBytes, 0, asBytes.length, iso8859Charset);
721                    request.addHeader(entry.getKey(), asLyingString);
722                }
723            }
724        }
725
726        // Logging
727        if (requestMeta != null) {
728            for (HeaderIterator headers = request.headerIterator(); headers.hasNext();) {
729                Header h = headers.nextHeader();
730                requestMeta.setField(LOG_HEADER_PREFIX + h.getName(), h.getValue());
731            }
732            start = System.currentTimeMillis();
733        }
734
735        HttpResponse response;
736
737        try {
738            HttpContext context = new BasicHttpContext();
739            response = httpClient.execute(request, context);
740
741            if (requestMeta != null) {
742                requestMeta.setField(LOG_IP_ADDRESS, getIpAddress(context));
743            }
744         } catch (ConnectTimeoutException e) {
745            connectTimeouts.increment();
746            throw e;
747        }
748
749        // Logging
750        long latencyStart = System.currentTimeMillis() - start;
751        if (requestMeta != null) {
752            requestMeta.setField(LOG_LATENCY_START, latencyStart);
753        }
754        logResponseLatency(latencyStart);
755        return response;
756    }

758    private String getIpAddress(HttpContext context) {
759        HttpConnection connection = (HttpConnection) context.getAttribute(ExecutionContext.HTTP_CONNECTION);
760        if (connection instanceof HttpInetConnection) {
761            InetAddress address = ((HttpInetConnection) connection).getRemoteAddress();
762            String hostAddress = address.getHostAddress();
763            return hostAddress == null ?
764                    IP_ADDRESS_UNKNOWN:
765                    hostAddress;
766        } else {
767            getLogger().log(LogLevel.DEBUG, "Unexpected connection type: " + connection.getClass().getName());
768            return IP_ADDRESS_UNKNOWN;
769        }
770    }

772    private boolean isAscii(String value) {
773        char[] scanBuffer = new char[value.length()];
774        value.getChars(0, value.length(), scanBuffer, 0);
775        for (char c: scanBuffer)
776            if (c > 127) return false;
777        return true;
778    }

838    protected static class SearcherHttpClient extends DefaultHttpClient {
839
840        private final int retries;
841
842        public SearcherHttpClient(final ClientConnectionManager conman, final HttpParams params) {
843            super(conman, params);
844            retries = params.getIntParameter(HTTPParameters.RETRIES, 1);
845            addRequestInterceptor((request, context) -> {
846                if (!request.containsHeader("Accept-Encoding")) {
847                    request.addHeader("Accept-Encoding", "gzip");
848                }
849            });
850            addResponseInterceptor((response, context) -> {
851                HttpEntity entity = response.getEntity();
852                if (entity == null) return;
853                Header ceheader = entity.getContentEncoding();
854                if (ceheader == null) return;
855                for (HeaderElement codec : ceheader.getElements()) {
856                    if (codec.getName().equalsIgnoreCase("gzip")) {
857                        response.setEntity(new GzipDecompressingEntity(response.getEntity()));
858                        return;
859                    }
860                }
861            });
862        }
863
864        @Override
865        protected HttpRequestExecutor createRequestExecutor() {
866            return new HttpRequestExecutor();
867        }
868
869        @Override
870        protected HttpRoutePlanner createHttpRoutePlanner() {
871            return new DefaultHttpRoutePlanner(getConnectionManager().getSchemeRegistry());
872        }
873
874        @Override
875        protected HttpRequestRetryHandler createHttpRequestRetryHandler() {
876            return new SearcherHttpRequestRetryHandler(retries);
877        }
878    }

880    /** A retry handler which avoids retrying forever on errors misclassified as transient */
881    private static class SearcherHttpRequestRetryHandler implements HttpRequestRetryHandler {
882        private final int retries;
883
884        public SearcherHttpRequestRetryHandler(int retries) {
885            this.retries = retries;
886        }
887
888        @Override
889        public boolean retryRequest(IOException e, int executionCount, HttpContext httpContext) {
890            if (e == null) {
891                throw new IllegalArgumentException("Exception parameter may not be null");
892            }
893            if (executionCount > retries) {
894                return false;
895            }
896            if (e instanceof NoHttpResponseException) {
897                // Retry if the server dropped connection on us
898                return true;
899            }
900            if (e instanceof InterruptedIOException) {
901                // Timeout from federation layer
902                return false;
903            }
904            if (e instanceof UnknownHostException) {
905                // Unknown host
906                return false;
907            }
908            if (e instanceof SSLHandshakeException) {
909                // SSL handshake exception
910                return false;
911            }
912            return true;
913        }
