002package com.yahoo.search.grouping.result;
003
004/**
005 * This abstract class represents the id of a single group in the grouping result model. A subclass corresponding to the
006 * evaluation result of generating {@link com.yahoo.search.grouping.request.GroupingExpression} is contained in all
007 * {@link Group} objects. It is used by {@link com.yahoo.search.grouping.GroupingRequest} to identify its root result
008 * group, and by all client code for identifying groups.
009 * <p>
010 * The {@link #toString()} method of this class generates a URI-compatible string on the form
011 * "group:&lt;typeName&gt;:&lt;subclassSpecific&gt;".
012 *
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
014 */
015public abstract class GroupId {
016
017    private final String type;
018    private final String image;
019
020    protected GroupId(String type, Object... args) {
021        this.type = type;
022
023        StringBuilder image = new StringBuilder("group:");
024        image.append(type);
025        for (Object arg : args) {
026            image.append(":").append(arg);
027        }
028        this.image = image.toString();
029    }
030
031    /**
032     * Returns the type name of this group id. This is the second part of the {@link #toString()} value of this.
033     *
034     * @return The type name.
035     */
036    public String getTypeName() {
037        return type;
038    }
039
040    @Override
041    public String toString() {
042        return image;
043    }
044}
