002package com.yahoo.search.grouping.result;
003
004import static com.yahoo.text.Lowercase.toLowerCase;
005
006/**
007 * This abstract class is used in {@link Group} instances where the identifying expression evaluated to a singe value.
008 *
009 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
010 */
011public abstract class ValueGroupId<T> extends GroupId {
012
013    private final T value;
014
015    /**
016     * Constructs a new instance of this class.
017     *
018     * @param type  The type of this id's value.
019     * @param value The identifying value.
020     */
021    public ValueGroupId(String type, T value) {
022        this(type, value, String.valueOf(value.toString()));
023    }
024
025    /**
026     * Constructs a new instance of this class.
027     *
028     * @param type       The type of this id's value.
029     * @param value      The identifying value.
030     * @param valueImage The String representation of the <tt>value</tt> argument.
031     */
032    public ValueGroupId(String type, T value, String valueImage) {
033        super(type, valueImage);
034        this.value = value;
035    }
036
037    /**
038     * Returns the identifying value.
039     *
040     * @return The value.
041     */
042    public T getValue() {
043        return value;
044    }
045}
