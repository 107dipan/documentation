002package com.yahoo.search.grouping.result;
003
004import com.yahoo.search.result.Hit;
005import com.yahoo.search.result.HitGroup;
006import com.yahoo.search.result.Relevance;
007
008/**
009 * This class represents a single group in the grouping result model. A group may contain any number of results (stored
010 * as fields, use {@link #getField(String)} to access), {@link GroupList} and {@link HitList}. Use the {@link
011 * com.yahoo.search.grouping.GroupingRequest#getResultGroup(com.yahoo.search.Result)} to retrieve an instance of this.
012 *
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
014 */
015public class Group extends HitGroup {
016
017    private static final long serialVersionUID = 2122928012157537800L;
018    private final GroupId groupId;
019
020    /**
021     * Creates a new instance of this class.
022     *
023     * @param groupId The id to assign to this group.
024     * @param rel     The relevance of this group.
025     */
026    public Group(GroupId groupId, Relevance rel) {
027        super(groupId.toString(), rel);
028        this.groupId = groupId;
029    }
030
031    /**
032     * Returns the id of this group. This is a model of the otherwise flattened {@link #getId() hit id}.
033     *
034     * @return The group id.
035     */
036    public GroupId getGroupId() {
037        return groupId;
038    }
039
040    /**
041     * Returns the {@link HitList} with the given label. The label is the one given to the {@link
042     * com.yahoo.search.grouping.request.EachOperation} that generated the list. This method returns null if no such
043     * list was found.
044     *
045     * @param label The label of the list to return.
046     * @return The requested list, or null.
047     */
048    public HitList getHitList(String label) {
049        for (Hit hit : this) {
050            if (!(hit instanceof HitList)) {
051                continue;
052            }
053            HitList lst = (HitList)hit;
054            if (!label.equals(lst.getLabel())) {
055                continue;
056            }
057            return lst;
058        }
059        return null;
060    }
061
062    /**
063     * Returns the {@link GroupList} with the given label. The label is the one given to the {@link
064     * com.yahoo.search.grouping.request.EachOperation} that generated the list. This method returns null if no such
065     * list was found.
066     *
067     * @param label The label of the list to return.
068     * @return The requested list, or null.
069     */
070    public GroupList getGroupList(String label) {
071        for (Hit hit : this) {
072            if (!(hit instanceof GroupList)) {
073                continue;
074            }
075            GroupList lst = (GroupList)hit;
076            if (!label.equals(lst.getLabel())) {
077                continue;
078            }
079            return lst;
080        }
081        return null;
082    }
083}
