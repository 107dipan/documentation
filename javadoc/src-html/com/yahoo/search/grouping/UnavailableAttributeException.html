002package com.yahoo.search.grouping;
003
004/**
005 * This exception is thrown by the {@link GroupingValidator} if it a {@link GroupingRequest} contains a reference to an
006 * unavailable attribute.
007 *
008 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
009 */
010@SuppressWarnings("serial")
011public class UnavailableAttributeException extends RuntimeException {
012
013    private final String clusterName;
014    private final String attributeName;
015
016    /**
017     * Constructs a new instance of this class.
018     *
019     * @param clusterName   The name of the cluster for which the request is illegal.
020     * @param attributeName The name of the attribute which is referenced but not available.
021     */
022    public UnavailableAttributeException(String clusterName, String attributeName) {
023        super("Grouping request references attribute '" + attributeName + "' which is not available " +
024              "in cluster '" + clusterName + "'.");
025        this.clusterName = clusterName;
026        this.attributeName = attributeName;
027    }
028
029    /**
030     * Returns the name of the cluster for which the request is illegal.
031     *
032     * @return The cluster name.
033     */
034    public String getClusterName() {
035        return clusterName;
036    }
037
038    /**
039     * Returns the name of the attribute which is referenced but not available.
040     *
041     * @return The attribute name.
042     */
043    public String getAttributeName() {
044        return attributeName;
045    }
046}
