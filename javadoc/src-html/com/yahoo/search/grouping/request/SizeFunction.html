002package com.yahoo.search.grouping.request;
003
004import java.util.Arrays;
005
006/**
007 * This class represents a size-function in a {@link GroupingExpression}. It evaluates to a number that equals the
008 * number of elements in the result of the argument (e.g. the number of elements in an array).
009 *
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
011 */
012public class SizeFunction extends FunctionNode {
013
014    /**
015     * Constructs a new instance of this class.
016     *
017     * @param exp The expression to evaluate.
018     */
019    public SizeFunction(GroupingExpression exp) {
020        super("size", Arrays.asList(exp));
021    }
022}
