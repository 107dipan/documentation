002package com.yahoo.search.grouping.request;
003
004import java.util.Arrays;
005
006/**
007 * This class represents debug_wait function in a {@link GroupingExpression}. For each hit evaluated,
008 * it waits for the time specified as the second argument. The third argument specifies if the wait
009 * should be a busy-wait or not. The first argument is then evaluated.
010 *
011 * @author <a href="mailto:lulf@yahoo-inc.com">Ulf Lilleengen</a>
012 */
013public class DebugWaitFunction extends FunctionNode {
014
015    /**
016     * Constructs a new instance of this class.
017     *
018     * @param arg1 The first compulsory argument, the expression to proxy.
019     * @param arg2 The second compulsory argument, must evaluate to a positive number.
020     * @param arg3 The third compulsory argument, specifying busy wait or not.
021     */
022    public DebugWaitFunction(GroupingExpression arg1, DoubleValue arg2, BooleanValue arg3) {
023        super("debugwait", Arrays.asList(arg1, arg2, arg3));
024    }
025
026    /**
027     * Returns the time to wait when evaluating this function.
028     *
029     * @return the number of seconds to wait.
030     */
031    public double getWaitTime() {
032        return ((DoubleValue)getArg(1)).getValue();
033    }
034
035    /**
036     * Returns whether or not the debug node should busy-wait.
037     *
038     * @return true if busy-wait, false if not.
039     */
040    public boolean getBusyWait() {
041        return ((BooleanValue)getArg(2)).getValue();
042    }
043}
