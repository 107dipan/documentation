002package com.yahoo.search.grouping.request;
003
004import java.util.Arrays;
005
006/**
007 * This class represents an md5-function in a {@link GroupingExpression}. It evaluates to a long that equals the md5 of
008 * the result of the argument.
009 *
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
011 */
012public class Md5Function extends FunctionNode {
013
014    /**
015     * Constructs a new instance of this class.
016     *
017     * @param exp     The expression to evaluate.
018     * @param numBits The number of bits of the md5 to include.
019     */
020    public Md5Function(GroupingExpression exp, int numBits) {
021        super("md5", Arrays.asList(exp, new LongValue(numBits)));
022    }
023
024    /**
025     * Returns the number of bits of the md5 to include in the evaluated result.
026     *
027     * @return The bit count.
028     */
029    public int getNumBits() {
030        return ((LongValue)getArg(1)).getValue().intValue();
031    }
032}
