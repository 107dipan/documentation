002package com.yahoo.search.grouping.request;
003
004import java.util.List;
005
006/**
007 * This class represents a mod-function in a {@link GroupingExpression}. It evaluates to a number that equals the result
008 * of mod'ing the results of all arguments in the order they were given to the constructor (modulo first argument by
009 * second, result by third, ...).
010 *
011 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
012 */
013public class ModFunction extends FunctionNode {
014
015    /**
016     * Constructs a new instance of this class.
017     *
018     * @param arg1 The first compulsory argument, must evaluate to a number.
019     * @param arg2 The second compulsory argument, must evaluate to a number.
020     * @param argN The optional arguments, must evaluate to a number.
021     */
022    public ModFunction(GroupingExpression arg1, GroupingExpression arg2, GroupingExpression... argN) {
023        this(asList(arg1, arg2, argN));
024    }
025
026    private ModFunction(List<GroupingExpression> args) {
027        super("mod", args);
028    }
029
030    /**
031     * Constructs a new instance of this class from a list of arguments.
032     *
033     * @param args The arguments to pass to the constructor.
034     * @return The created instance.
035     * @throws IllegalArgumentException Thrown if the number of arguments is less than 2.
036     */
037    public static ModFunction newInstance(List<GroupingExpression> args) {
038        if (args.size() < 2) {
039            throw new IllegalArgumentException("Expected 2 or more arguments, got " + args.size() + ".");
040        }
041        return new ModFunction(args);
042    }
043}
