002package com.yahoo.search.grouping.request;
003
004import java.util.Arrays;
005
006/**
007 * This class represents an uca-function in a {@link GroupingExpression}.
008 *
009 * @author <a href="mailto:lulf@yahoo-inc.com">Ulf Lilleengen</a>
010 */
011public class UcaFunction extends FunctionNode {
012
013    private final String locale;
014    private final String strength;
015
016
017    /**
018     * Constructs a new instance of this class.
019     *
020     * @param exp     The expression to evaluate.
021     * @param locale  The locale to used for sorting.
022     */
023    public UcaFunction(GroupingExpression exp, String locale) {
024        super("uca", Arrays.asList(exp, new StringValue(locale)));
025        this.locale = locale;
026        this.strength = "TERTIARY";
027    }
028
029    /**
030     * Constructs a new instance of this class.
031     *
032     * @param exp     The expression to evaluate.
033     * @param locale  The locale to used for sorting.
034     * @param strength The strength level to use.
035     */
036    public UcaFunction(GroupingExpression exp, String locale, String strength) {
037        super("uca", Arrays.asList(exp, new StringValue(locale), new StringValue(strength)));
038        if (!validStrength(strength)) {
039            throw new IllegalArgumentException("Not a valid UCA strength: " + strength);
040        }
041        this.locale = locale;
042        this.strength = strength;
043    }
044
045    private boolean validStrength(String strength) {
046        return (strength.equals("PRIMARY") ||
047                strength.equals("SECONDARY") ||
048                strength.equals("TERTIARY") ||
049                strength.equals("QUATERNARY") ||
050                strength.equals("IDENTICAL"));
051    }
052
053    public String getLocale() {
054        return locale;
055    }
056
057    public String getStrength() {
058        return strength;
059    }
060}
