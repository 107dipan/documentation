002package com.yahoo.search.grouping.request;
003
004/**
005 * This class represents a document value in a {@link GroupingExpression}. As such, the subclasses of this can only be
006 * used as document-level expressions (i.e. level 0, see {@link GroupingExpression#resolveLevel(int)}).
007 *
008 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
009 */
010public abstract class DocumentValue extends GroupingExpression {
011
012    protected DocumentValue(String image) {
013        super(image);
014    }
015
016    @Override
017    public void resolveLevel(int level) {
018        if (level != 0) {
019            throw new IllegalArgumentException("Expression '" + this + "' not applicable for " +
020                                               GroupingOperation.getLevelDesc(level) + ".");
021        }
022        super.resolveLevel(level);
023    }
024}
