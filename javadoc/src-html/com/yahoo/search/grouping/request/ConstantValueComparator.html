002package com.yahoo.search.grouping.request;
003
004import java.util.Comparator;
005
006/**
007 * This class compares two constant values, and takes into account that one of
008 * the arguments may be the very special infinity value.
009 *
010 * @author <a href="mailto:lulf@yahoo-inc.com">Ulf Lilleengen</a>
011 */
012@SuppressWarnings("rawtypes")
013public class ConstantValueComparator implements Comparator<ConstantValue> {
014    @SuppressWarnings("unchecked")
015    @Override
016    public int compare(ConstantValue lhs, ConstantValue rhs) {
017        // Run infinite comparison method if one of the arguments are infinite.
018        if (rhs instanceof InfiniteValue) {
019            return (-1 * rhs.getValue().compareTo(lhs));
020        }
021        return (lhs.getValue().compareTo(rhs.getValue()));
022    }
023
024}
