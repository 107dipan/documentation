002package com.yahoo.search.grouping.request;
003
004import com.google.common.annotations.Beta;
005
006/**
007 * Represents access of array element in a document attribute in a {@link GroupingExpression}.
008 *
009 * The first argument should be the name of an array attribute in the
010 * input {@link com.yahoo.search.result.Hit}, while the second
011 * argument is evaluated as an integer and used as the index in that array.
012 * If the index argument is less than 0 returns the first array element;
013 * if the index is greater than or equal to size(array) returns the last array element;
014 * if the array is empty returns 0 (or NaN?).
015 * @author arnej27959
016 */
017@Beta
018public class ArrayAtLookup extends DocumentValue {
019
020    private final String attributeName;
021    private final GroupingExpression arg2;
022
023    /**
024     * Constructs a new instance of this class.
025     *
026     * @param attributeName The attribute name to assign to this.
027     */
028    public ArrayAtLookup(String attributeName, GroupingExpression indexArg) {
029        super("array.at(" + attributeName + ", " + indexArg + ")");
030        this.attributeName = attributeName;
031        this.arg2 = indexArg;
032    }
033
034    /**
035     * Returns the name of the attribute to retrieve from the input hit.
036     *
037     * @return The attribute name.
038     */
039    public String getAttributeName() {
040        return attributeName;
041    }
042
043    /**
044     * get the expression to evaluate before indexing
045     * @return grouping expression argument
046     */
047    public GroupingExpression getIndexArgument() {
048        return arg2;
049    }
050
051}
