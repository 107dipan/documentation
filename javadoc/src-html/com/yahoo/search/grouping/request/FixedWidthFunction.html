002package com.yahoo.search.grouping.request;
003
004import java.util.Arrays;
005
006/**
007 * This class represents a fixed-width bucket-function in a {@link GroupingExpression}. It maps the input into the given
008 * number of buckets by the result of the argument expression.
009 *
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
011 */
012public class FixedWidthFunction extends FunctionNode {
013
014    /**
015     * Constructs a new instance of this class.
016     *
017     * @param exp        The expression to evaluate.
018     * @param width The width of each bucket.
019     */
020    public FixedWidthFunction(GroupingExpression exp, Number width) {
021        super("fixedwidth", Arrays.asList(exp, width instanceof Double ? new DoubleValue(width.doubleValue()) : new LongValue(width.longValue())));
022    }
023
024    /**
025     * Returns the number of buckets to divide the result into.
026     *
027     * @return The bucket count.
028     */
029    public Number getWidth() {
030        GroupingExpression w = getArg(1);
031        return (w instanceof LongValue) ? ((LongValue)w).getValue() : ((DoubleValue)w).getValue();
032    }
033}
