002package com.yahoo.search.grouping.request;
003
004import com.google.common.annotations.Beta;
005
006/**
007 * This class represents a lookup in a multivalue document
008 * attribute in a {@link GroupingExpression}.  It takes the
009 * attribute (assumed to contain a sorted array) from the input
010 * {@link com.yahoo.search.result.Hit} and finds the index that
011 * the second (lookup) argument expression would have, with linear
012 * interpolation when the lookup argument is between two array
013 * element values.
014 *
015 * @author arnej27959
016 */
017@Beta
018public class InterpolatedLookup extends DocumentValue {
019
020    private final String attributeName;
021    private final GroupingExpression arg2;
022
023    /**
024     * Constructs a new instance of this class.
025     *
026     * @param attributeName The attribute name the lookup should happen in
027     * @param lookupArg Expression giving a floating-point value for the lookup argument
028     */
029    public InterpolatedLookup(String attributeName, GroupingExpression lookupArg) {
030        super("interpolatedlookup(" + attributeName + ", " + lookupArg + ")");
031        this.attributeName = attributeName;
032        this.arg2 = lookupArg;
033    }
034
035    /**
036     * Get the name of the attribute to be retrieved from the input hit.
037     * @return The attribute name.
038     */
039    public String getAttributeName() {
040        return attributeName;
041    }
042
043    /**
044     * Get the expression that will be evaluated before lookup.
045     * @return grouping expression argument
046     */
047    public GroupingExpression getLookupArgument() {
048        return arg2;
049    }
050
051}
