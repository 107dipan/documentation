002package com.yahoo.search.grouping.request;
003import java.text.ChoiceFormat;
004
005/**
006 * This class represents a {@link Double} bucket in a {@link PredefinedFunction}.
007 *
008 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
009 */
010public class DoubleBucket extends BucketValue {
011
012    /**
013     * Returns the next distinct value.
014     *
015     * @param value The base value.
016     * @return the next value.
017     */
018    public static DoubleValue nextValue(DoubleValue value) {
019        return (new DoubleValue(ChoiceFormat.nextDouble(value.getValue())));
020    }
021
022    /**
023     * Constructs a new instance of this class.
024     *
025     * @param from          The from-value to assign to this.
026     * @param to            The to-value to assign to this.
027     */
028    public DoubleBucket(double from, double to) {
029        super(new DoubleValue(from), new DoubleValue(to));
030    }
031
032    /**
033     * Constructs a new instance of this class.
034     *
035     * @param from          The from-value to assign to this.
036     * @param to            The to-value to assign to this.
037     */
038    public DoubleBucket(ConstantValue<?> from, ConstantValue<?> to) {
039        super(from, to);
040    }
041}
