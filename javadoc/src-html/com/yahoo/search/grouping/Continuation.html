002package com.yahoo.search.grouping;
003
004import com.yahoo.search.grouping.vespa.ContinuationDecoder;
005
006/**
007 * <p>This class represents a piece of data stored by the grouping framework within a grouping result, which can
008 * subsequently be sent back along with the original request to navigate across a large result set. It is an opaque
009 * data object that is not intended to be human readable.</p>
010 *
011 * <p>To render a Cookie within a result set, you simply need to call {@link #toString()}.</p>
012 *
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
014 */
015public abstract class Continuation {
016
017    public static final String NEXT_PAGE = "next";
018    public static final String PREV_PAGE = "prev";
019    public static final String THIS_PAGE = "this";
020
021    public static Continuation fromString(String str) {
022        return ContinuationDecoder.decode(str);
023    }
024}
