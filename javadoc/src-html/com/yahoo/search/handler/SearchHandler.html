117    private final class MeanConnections implements Callback {
118
119        @Override
120        public void run(Handle h, boolean firstTime) {
121            if (firstTime) {
122                metric.set(SEARCH_CONNECTIONS, 0.0d, null);
123                return;
124            }
125            Value v = (Value) h;
126            metric.set(SEARCH_CONNECTIONS, v.getMean(), null);
127        }
128    }

130    @Inject
131    public SearchHandler(
132            final ChainsConfig chainsConfig,
133            final IndexInfoConfig indexInfo,
134            final QrSearchersConfig clusters,
135            final SpecialtokensConfig specialtokens,
136            final Statistics statistics,
137            final Linguistics linguistics,
138            final Metric metric,
139            final ComponentRegistry<Renderer> renderers,
140            final Executor executor,
141            final AccessLog accessLog,
142            final QueryProfilesConfig queryProfileConfig,
143            final ComponentRegistry<Searcher> searchers,
144            final ContainerHttpConfig containerHttpConfig) {
145        super(executor, accessLog, metric, true);
146        log.log(LogLevel.DEBUG, "SearchHandler.init " + System.identityHashCode(this));
147        searchChainRegistry = new SearchChainRegistry(searchers);
148        setupSearchChainRegistry(searchers, chainsConfig);
149        indexFacts = new IndexFacts(new IndexModel(indexInfo, clusters));
150        indexFacts.freeze();
151        specialTokens = new SpecialTokenRegistry(specialtokens);
152        rendererRegistry = new RendererRegistry(renderers.allComponents());
153        QueryProfileRegistry queryProfileRegistry = QueryProfileConfigurer.createFromConfig(queryProfileConfig);
154        this.queryProfileRegistry = queryProfileRegistry.compile();
155
156        this.linguistics = linguistics;
157        this.maxThreads = examineExecutor(executor);
158
159        searchConnections = new Value(SEARCH_CONNECTIONS, statistics,
160                                      new Value.Parameters().setLogRaw(true).setLogMax(true)
161                                              .setLogMean(true).setLogMin(true)
162                                              .setNameExtension(true)
163                                              .setCallback(new MeanConnections()));
164        
165        this.hostResponseHeaderKey = containerHttpConfig.hostResponseHeaderKey().equals("") ?
166                                     Optional.empty() : Optional.of( containerHttpConfig.hostResponseHeaderKey());
167    }

223    private static int examineExecutor(Executor executor) {
224        if (executor instanceof ThreadPoolExecutor) {
225            return ((ThreadPoolExecutor) executor).getMaximumPoolSize();
226        }
227        return Integer.MAX_VALUE; // assume unbound
228    }

230    @Override
231    public final HttpResponse handle(com.yahoo.container.jdisc.HttpRequest request) {
232        requestsInFlight.incrementAndGet();
233        try {
234            try {
235                return handleBody(request);
236            } catch (QueryException e) {
237                return (e.getCause() instanceof IllegalArgumentException)
238                        ? invalidParameterResponse(request, e)
239                        : illegalQueryResponse(request, e);
240            } catch (RuntimeException e) { // Make sure we generate a valid XML response even on unexpected errors
241                log.log(Level.WARNING, "Failed handling " + request, e);
242                return internalServerErrorResponse(request, e);
243            }
244        } finally {
245            requestsInFlight.decrementAndGet();
246        }
247    }

249    private int getHttpResponseStatus(com.yahoo.container.jdisc.HttpRequest httpRequest, Result result) {
250        boolean benchmarkOutput = VespaHeaders.benchmarkOutput(httpRequest);
251        if (benchmarkOutput) {
252            return VespaHeaders.getEagerErrorStatus(result.hits().getError(),
253                                                    SearchResponse.getErrorIterator(result.hits().getErrorHit()));
254        } else {
255            return VespaHeaders.getStatus(SearchResponse.isSuccess(result),
256                                          result.hits().getError(),
257                                          SearchResponse.getErrorIterator(result.hits().getErrorHit()));
258        }
259
260    }

264        Query query = new Query();
265        Result result = new Result(query, errorMessage);
266        Renderer renderer = getRendererCopy(ComponentSpecification.fromString(request.getProperty("format")));

268        result.getTemplating().setRenderer(renderer); // Pre-Vespa 6 Result.getEncoding() expects this TODO: Remove

270        return new HttpSearchResponse(getHttpResponseStatus(request, result), result, query, renderer);
271    }

285    private HttpSearchResponse handleBody(HttpRequest request) {
286        // Find query profile
287        String queryProfileName = request.getProperty("queryProfile");
288        CompiledQueryProfile queryProfile = queryProfileRegistry.findQueryProfile(queryProfileName);
289        boolean benchmarkOutput = VespaHeaders.benchmarkOutput(request);
290
291        // Create query
292        Query query = new Query(request, queryProfile);
293
294        boolean benchmarkCoverage = VespaHeaders.benchmarkCoverage(benchmarkOutput, request.getJDiscRequest().headers());
295
296        // Find and execute search chain if we have a valid query
297        String invalidReason = query.validate();
298        Chain<Searcher> searchChain = null;
299        String searchChainName = null;
300        if (invalidReason == null) {
301            Tuple2<String, Chain<Searcher>> nameAndChain = resolveChain(query.properties().getString(Query.SEARCH_CHAIN));
302            searchChainName = nameAndChain.first;
303            searchChain = nameAndChain.second;
304        }
305
306        // Create the result
307        Result result;
308        if (invalidReason != null) {
309            result = new Result(query, ErrorMessage.createIllegalQuery(invalidReason));
310        } else if (queryProfile == null && queryProfileName != null) {
311            result = new Result(
312                    query,
313                    ErrorMessage.createIllegalQuery("Could not resolve query profile '" + queryProfileName + "'"));
314        } else if (searchChain == null) {
315            result = new Result(
316                    query,
317                    ErrorMessage.createInvalidQueryParameter("No search chain named '" + searchChainName + "' was found"));
318        } else {
319            String pathAndQuery = UriTools.rawRequest(request.getUri());
320            result = search(pathAndQuery, query, searchChain, searchChainRegistry);
321        }
322
323        Renderer renderer;
324        if (result.getTemplating().usesDefaultTemplate()) {
325            renderer = toRendererCopy(query.getPresentation().getRenderer());
326            result.getTemplating().setRenderer(renderer); // pre-Vespa 6 Result.getEncoding() expects this to be set. TODO: Remove
327        }
328        else { // somebody explicitly assigned a old style template
329            renderer = perRenderingCopy(result.getTemplating().getRenderer());
330        }
331
332        // Transform result to response
333        HttpSearchResponse response = new HttpSearchResponse(getHttpResponseStatus(request, result), 
334                                                             result, query, renderer);
335        if (hostResponseHeaderKey.isPresent())
336            response.headers().add(hostResponseHeaderKey.get(), selfHostname);
337
338        if (benchmarkOutput)
339            VespaHeaders.benchmarkOutput(response.headers(), benchmarkCoverage, response.getTiming(),
340                                         response.getHitCounts(), getErrors(result), response.getCoverage());
341
342        return response;
343    }

347    }

356    private Tuple2<String, Chain<Searcher>> resolveChain(String explicitChainName) {
357        String chainName = explicitChainName;
358        if (chainName == null) {
359            chainName = defaultSearchChainName;
360        }
361
362        Chain<Searcher> searchChain = searchChainRegistry.getChain(chainName);
363        if (searchChain == null && explicitChainName == null) { // explicit chain not found should cause error
364            chainName = fallbackSearchChain;
365            searchChain = searchChainRegistry.getChain(chainName);
366        }
367        return new Tuple2<>(chainName, searchChain);
368    }

370    /** Used from container SDK, for internal use only */
371    public Result searchAndFill(Query query, Chain<? extends Searcher> searchChain, SearchChainRegistry registry) {
372        Result errorResult = validateQuery(query);
373        if (errorResult != null) return errorResult;
374
375        Renderer<Result> renderer = rendererRegistry.getRenderer(query.getPresentation().getRenderer());
376
377        // docsumClass null means "unset", so we set it (it might be null
378        // here too in which case it will still be "unset" after we
379        // set it :-)
380        if (query.getPresentation().getSummary() == null && renderer instanceof com.yahoo.search.rendering.Renderer)
381            query.getPresentation().setSummary(((com.yahoo.search.rendering.Renderer) renderer).getDefaultSummaryClass());
382
383        Execution execution = new Execution(searchChain,
384                                            new Execution.Context(registry, indexFacts, specialTokens, rendererRegistry, linguistics));
385        query.getModel().setExecution(execution);
386        execution.trace().setForceTimestamps(query.properties().getBoolean(FORCE_TIMESTAMPS, false));
387        if (query.properties().getBoolean(DETAILED_TIMING_LOGGING, false)) {
388            // check and set (instead of set directly) to avoid overwriting stuff from prepareForBreakdownAnalysis()
389            execution.context().setDetailedDiagnostics(true);
390        }
391        Result result = execution.search(query);
392
393        if (result.getTemplating() == null)
394            result.getTemplating().setRenderer(renderer);
395
396        ensureQuerySet(result, query);
397        execution.fill(result, result.getQuery().getPresentation().getSummary());
398
399        traceExecutionTimes(query, result);
400        traceVespaVersion(query);
401        traceRequestAttributes(query);
402        return result;
403    }

437    private Result search(String request, Query query, Chain<Searcher> searchChain, SearchChainRegistry registry) {
438        if (query.getTraceLevel() >= 2) {
439            query.trace("Invoking " + searchChain, false, 2);
440        }
441
442        if (searchConnections != null) {
443            connectionStatistics();
444        } else {
445            log.log(LogLevel.WARNING,
446                    "searchConnections is a null reference, probably a known race condition during startup.",
447                    new IllegalStateException("searchConnections reference is null."));
448        }
449        try {
450            return searchAndFill(query, searchChain, registry);
451        } catch (ParseException e) {
452            ErrorMessage error = ErrorMessage.createIllegalQuery("Could not parse query [" + request + "]: "
453                                                                 + Exceptions.toMessageString(e));
454            log.log(LogLevel.DEBUG, () -> error.getDetailedMessage());
455            return new Result(query, error);
456        } catch (IllegalArgumentException e) {
457            ErrorMessage error = ErrorMessage.createBadRequest("Invalid search request [" + request + "]: "
458                                                               + Exceptions.toMessageString(e));
459            log.log(LogLevel.DEBUG, () -> error.getDetailedMessage());
460            return new Result(query, error);
461        } catch (LinkageError e) {
462            // Should have been an Exception in an OSGi world - typical bundle dependency issue problem
463            ErrorMessage error = ErrorMessage.createErrorInPluginSearcher(
464                            "Error executing " + searchChain + "]: " + Exceptions.toMessageString(e), e);
465            log(request, query, e);
466            return new Result(query, error);
467        } catch (StackOverflowError e) { // Also recoverable
468            ErrorMessage error = ErrorMessage.createErrorInPluginSearcher(
469                            "Error executing " + searchChain + "]: " + Exceptions.toMessageString(e), e);
470            log(request, query, e);
471            return new Result(query, error);
472        } catch (Exception e) {
473            Result result = new Result(query);
474            log(request, query, e);
475            result.hits().setError(
476                    ErrorMessage.createUnspecifiedError("Failed searching: " + Exceptions.toMessageString(e), e));
477            return result;
478        }
479    }

481    private void connectionStatistics() {
482        int connections = requestsInFlight.intValue();
483        searchConnections.put(connections);
484        if (maxThreads > 3) {
485            // cast to long to avoid overflows if maxThreads is at no
486            // log value (maxint)
487            final long maxThreadsAsLong = maxThreads;
488            final long connectionsAsLong = connections;
489            // only log when exactly crossing the limit to avoid
490            // spamming the log
491            if (connectionsAsLong < maxThreadsAsLong * 9L / 10L) {
492                // NOP
493            } else if (connectionsAsLong == maxThreadsAsLong * 9L / 10L) {
494                log.log(Level.WARNING, threadConsumptionMessage(connections, maxThreads, "90"));
495            } else if (connectionsAsLong == maxThreadsAsLong * 95L / 100L) {
496                log.log(Level.WARNING, threadConsumptionMessage(connections, maxThreads, "95"));
497            } else if (connectionsAsLong == maxThreadsAsLong) {
498                log.log(Level.WARNING, threadConsumptionMessage(connections, maxThreads, "100"));
499            }
500        }
501    }
502
503    private String threadConsumptionMessage(int connections, int maxThreads, String percentage) {
504        return percentage + "% of possible search connections (" + connections +
505               " of maximum " + maxThreads + ") currently active.";
506    }

508    private void log(String request, Query query, Throwable e) {
509        // Attempted workaround for missing stack traces
510        if (e.getStackTrace().length == 0) {
511            log.log(LogLevel.ERROR,
512                    "Failed executing " + query.toDetailString() + " [" + request
513                            + "], received exception with no context", e);
514        } else {
515            log.log(LogLevel.ERROR,
516                    "Failed executing " + query.toDetailString() + " [" + request + "]", e);
517        }
518    }

520    private Result validateQuery(Query query) {
521        if (query.getHttpRequest().getProperty(DefaultProperties.MAX_HITS.toString()) != null)
522            throw new RuntimeException(DefaultProperties.MAX_HITS + " must be specified in a query profile.");
523
524        if (query.getHttpRequest().getProperty(DefaultProperties.MAX_OFFSET.toString()) != null)
525            throw new RuntimeException(DefaultProperties.MAX_OFFSET + " must be specified in a query profile.");
526
527        int maxHits = query.properties().getInteger(DefaultProperties.MAX_HITS);
528        int maxOffset = query.properties().getInteger(DefaultProperties.MAX_OFFSET);
529
530        if (query.getHits() > maxHits) {
531            return new Result(query, ErrorMessage.createIllegalQuery(query.getHits() +
532                              " hits requested, configured limit: " + maxHits + "."));
533
534        } else if (query.getOffset() > maxOffset) {
535            return new Result(query,
536                    ErrorMessage.createIllegalQuery("Offset of " + query.getOffset() +
537                                                    " requested, configured limit: " + maxOffset + "."));
538        }
539        return null;
540    }

542    private void traceExecutionTimes(Query query, Result result) {
543        if (query.getTraceLevel() < 3) return;
544
545        ElapsedTime elapsedTime = result.getElapsedTime();
546        long now = System.currentTimeMillis();
547        if (elapsedTime.firstFill() != 0) {
548            query.trace("Query time " + query + ": " + (elapsedTime.firstFill() - elapsedTime.first()) + " ms", false, 3);
549            query.trace("Summary fetch time " + query + ": " + (now - elapsedTime.firstFill()) + " ms", false, 3);
550        } else {
551            query.trace("Total search time " + query + ": " + (now - elapsedTime.first()) + " ms", false, 3);
552        }
553    }
