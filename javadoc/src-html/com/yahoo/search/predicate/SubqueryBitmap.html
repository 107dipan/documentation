004/**
005 * Constants related to the subquery bitmap in predicate/boolean search.

008public interface SubqueryBitmap {

012     */
013    long ALL_SUBQUERIES = 0xFFFF_FFFF_FFFF_FFFFl;
014
015    /**
016     * Default subquery bitmap.
017     */
018    long DEFAULT_VALUE = ALL_SUBQUERIES;
019
020
021}
