002package com.yahoo.messagebus;
003
004import com.yahoo.log.LogLevel;
005
006import java.util.concurrent.atomic.AtomicBoolean;
007import java.util.logging.Logger;
008
009/**
010 * A session which supports receiving, forwarding and acknowledgement of messages. An intermediate session is expacted
011 * to either forward or acknowledge every message received.
012 *
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
014 */
015public final class IntermediateSession implements MessageHandler, ReplyHandler {
016
017    private static final Logger log = Logger.getLogger(IntermediateSession.class.getName());
018    private final AtomicBoolean destroyed = new AtomicBoolean(false);
019    private final String name;
020    private final boolean broadcastName;
021    private final MessageHandler msgHandler;
022    private final ReplyHandler replyHandler;
023    private final MessageBus mbus;
024
025    /**
026     * This constructor is declared package private since only MessageBus is supposed to instantiate it.
027     *
028     * @param mbus   The message bus that created this instance.
029     * @param params The parameter object for this session.
030     */
031    IntermediateSession(MessageBus mbus, IntermediateSessionParams params) {
032        this.mbus = mbus;
033        this.name = params.getName();
034        this.broadcastName = params.getBroadcastName();
035        this.msgHandler = params.getMessageHandler();
036        this.replyHandler= params.getReplyHandler();
037    }
038
039    @Override
040    protected void finalize() throws Throwable {
041        try {
042            if (destroy()) {
043                log.log(LogLevel.WARNING, "IntermediateSession destroyed by finalizer, please review application shutdown logic.");
044            }
045        } finally {
046            super.finalize();
047        }
048    }
049
050    /**
051     * Sets the destroyed flag to true. The very first time this method is called, it cleans up all its dependencies.
052     * Even if you retain a reference to this object, all of its content is allowed to be garbage collected.
053     *
054     * @return True if content existed and was destroyed.
055     */
056    public boolean destroy() {
057        if (!destroyed.getAndSet(true)) {
058            close();
059            return true;
060        }
061        return false;
062    }
063
064    /**
065     * This method unregisters this session from message bus, effectively disabling any more messages from being
066     * delivered to the message handler. After unregistering, this method calls {@link com.yahoo.messagebus.MessageBus#sync()}
067     * as to ensure that there are no threads currently entangled in the handler.
068     *
069     * This method will deadlock if you call it from the message or reply handler.
070     */
071    public void close() {
072        mbus.unregisterSession(name, broadcastName);
073        mbus.sync();
074    }
075
076    /**
077     * Forwards a routable to the next hop in its route. This method will never block.
078     * @param routable the routable to forward.
079     */
080    public void forward(Routable routable) {
081        if (routable instanceof Reply) {
082            Reply reply = (Reply)routable;
083            ReplyHandler handler = reply.popHandler();
084            handler.handleReply(reply);
085        } else {
086            routable.pushHandler(this);
087            mbus.handleMessage((Message)routable);
088        }
089    }
090
091    /**
092     * Returns the message handler of this session.
093     *
094     * @return The message handler.
095     */
096    public MessageHandler getMessageHandler() {
097        return msgHandler;
098    }
099
100    /**
101     * Returns the reply handler of this session.
102     *
103     * @return The reply handler.
104     */
105    public ReplyHandler getReplyHandler() {
106        return replyHandler;
107    }
108
109    /**
110     * Returns the connection spec string for this session. This returns a combination of the owning message bus' own
111     * spec string and the name of this session.
112     *
113     * @return The connection string.
114     */
115    public String getConnectionSpec() {
116        return mbus.getConnectionSpec() + "/" + name;
117    }
118
119    /**
120     * Returns the name of this session.
121     *
122     * @return The session name.
123     */
124    public String getName() {
125        return name;
126    }
127
128    @Override
129    public void handleMessage(Message msg) {
130        msgHandler.handleMessage(msg);
131    }
132
133    @Override
134    public void handleReply(Reply reply) {
135        if (destroyed.get()) {
136            reply.discard();
137        } else {
138            replyHandler.handleReply(reply);
139        }
140    }
141}
