002package com.yahoo.messagebus;
003
004import com.yahoo.component.Version;
005import com.yahoo.messagebus.metrics.MetricSet;
006import com.yahoo.messagebus.routing.RoutingPolicy;
007
008/**
009 * Interface implemented by the concrete application message protocol.
010 *
011 * @author bratseth
012 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
013 */
014public interface Protocol {
015
016    /**
017     * Returns a global unique name for this protocol.
018     *
019     * @return The name.
020     */
021    public String getName();
022
023    /**
024     * Encodes the protocol specific data of a routable into a byte array.
025     *
026     * @param version  The version to encode for.
027     * @param routable The routable to encode.
028     * @return The encoded data.
029     */
030    public byte[] encode(Version version, Routable routable);
031
032    /**
033     * Decodes the protocol specific data into a routable of the correct type.
034     *
035     * @param version The version of the serialized routable.
036     * @param payload The payload to decode from.
037     * @return The decoded routable.
038     */
039    public Routable decode(Version version, byte[] payload);
040
041    /**
042     * Create a policy of the named type with the named param passed to the constructor of that policy.
043     *
044     * @param name  The name of the policy to create.
045     * @param param The parameter to that policy's constructor.
046     * @return The created policy.
047     */
048    public RoutingPolicy createPolicy(String name, String param);
049
050    /**
051     * Returns the metrics associated with this protocol.
052     */
053    MetricSet getMetrics();
054}
