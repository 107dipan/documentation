002package com.yahoo.messagebus;
003
004import com.yahoo.log.LogLevel;
005
006import java.util.ArrayList;
007import java.util.Collections;
008import java.util.List;
009import java.util.logging.Logger;
010
011/**
012 * This class contains the actual trace information of a {@link Trace} object. A trace node can be encoded to, and
013 * decoded from a string representation to allow transport across the network. Each node contains a list of children, a
014 * strictness flag and an optional note. The child list is what forms the trace tree, the strictness flag dictates
015 * whether or not the ordering of the children is important, and the note is the actual traced data.
016 *
017 * The most important feature to notice is the {@link #normalize()} method that will compact, sort and 'rootify' the
018 * trace tree so that trees become well-formed (and can be compared for equality).
019 *
020 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
021 */
022public class TraceNode implements Comparable<TraceNode> {
023
024    private static final Logger log = Logger.getLogger(TraceNode.class.getName());
025    private TraceNode parent = null;
026    private boolean strict = true;
027    private String note = null;
028    private List<TraceNode> children = new ArrayList<TraceNode>();
029
030    /**
031     * Create an empty trace tree.
032     */
033    public TraceNode() {
034        // empty
035    }
036
037    /**
038     * Create a leaf node with the given note.
039     *
040     * @param note The note to assign to this.
041     */
042    private TraceNode(String note) {
043        this.note = note;
044    }
045
046    /**
047     * Create a trace tree which is a copy of another.
048     *
049     * @param rhs The tree to copy.
050     */
051    TraceNode(TraceNode rhs) {
052        strict = rhs.strict;
053        note = rhs.note;
054        addChildren(rhs.children);
055    }
056
057    /**
058     * Swap the internals of this tree with another.
059     *
060     * @param other The tree to swap internals with.
061     * @return This, to allow chaining.
062     */
063    public TraceNode swap(TraceNode other) {
064        TraceNode parent = this.parent;
065        this.parent = other.parent;
066        other.parent = parent;
067
068        boolean strict = this.strict;
069        this.strict = other.strict;
070        other.strict = strict;
071
072        String note = this.note;
073        this.note = other.note;
074        other.note = note;
075
076        List<TraceNode> children = this.children;
077        this.children = other.children;
078        for (TraceNode child : this.children) {
079            child.parent = this;
080        }
081        other.children = children;
082        for (TraceNode child : other.children) {
083            child.parent = other;
084        }
085
086        return this;
087    }
088
089    /**
090     * Remove all trace information from this tree.
091     *
092     * @return This, to allow chaining.
093     */
094    public TraceNode clear() {
095        parent = null;
096        strict = true;
097        note = null;
098        children.clear();
099        return this;
100    }
101
102    /**
103     * Sort non-strict children recursively down the tree.
104     *
105     * @return This, to allow chaining.
106     */
107    public TraceNode sort() {
108        if (!isLeaf()) {
109            for (TraceNode child : children) {
110                child.sort();
111            }
112            if (!isStrict()) {
113                Collections.sort(children);
114            }
115        }
116        return this;
117    }
118
119    @Override
120    public int compareTo(TraceNode rhs) {
121        if (isLeaf() || rhs.isLeaf()) {
122            if (isLeaf() && rhs.isLeaf()) {
123                return note.compareTo(rhs.getNote());
124            } else {
125                return isLeaf() ? -1 : 1;
126            }
127        }
128        if (children.size() != rhs.children.size()) {
129            return children.size() < rhs.children.size() ? -1 : 1;
130        }
131        for (int i = 0; i < children.size(); ++i) {
132            int cmp = children.get(i).compareTo(rhs.children.get(i));
133            if (cmp != 0) {
134                return cmp;
135            }
136        }
137        return -1;
138    }
139
140    /**
141     * Compact this tree. This will reduce the height of this tree as much as possible without removing information
142     * stored in it.
143     *
144     * @return This, to allow chaining.
145     */
146    public TraceNode compact() {
147        if (isLeaf()) {
148            return this;
149        }
150        List<TraceNode> tmp = this.children;
151        this.children = new ArrayList<TraceNode>();
152        for (TraceNode child : tmp) {
153            child.compact();
154            if (child.isEmpty()) {
155                // ignore
156            } else if (child.isLeaf()) {
157                addChild(child);
158            } else if (strict == child.strict) {
159                addChildren(child.children);
160            } else if (child.getNumChildren() == 1) {
161                TraceNode grandChild = child.getChild(0);
162                if (grandChild.isEmpty()) {
163                    // ignore
164                } else if (grandChild.isLeaf() || strict != grandChild.strict) {
165                    addChild(grandChild);
166                } else {
167                    addChildren(grandChild.children);
168                }
169            } else {
170                addChild(child);
171            }
172        }
173        return this;
174    }
175
176    /**
177     * Normalize this tree. This will transform all equivalent trees into the same form. Note that this will also
178     * perform an implicit compaction of the tree.
179     *
180     * @return This, to allow chaining.
181     */
182    public TraceNode normalize() {
183        compact();
184        sort();
185        if (note != null || !strict) {
186            TraceNode child = new TraceNode();
187            child.swap(this);
188            addChild(child);
189            strict = true;
190        }
191        return this;
192    }
193
194    /**
195     * Check whether or not this is a root node.
196     *
197     * @return True if this has no parent.
198     */
199    public boolean isRoot() {
200        return parent == null;
201    }
202
203    /**
204     * Check whether or not this is a leaf node.
205     *
206     * @return True if this has no children.
207     */
208    public boolean isLeaf() {
209        return children.isEmpty();
210    }
211
212    /**
213     * Check whether or not this node is empty, i.e. it has no note and no children.
214     *
215     * @return True if this node is empty.
216     */
217    public boolean isEmpty() {
218        return note == null && children.isEmpty();
219    }
220
221    /**
222     * Check whether or not the children of this node are strictly ordered.
223     *
224     * @return True if this node is strict.
225     */
226    public boolean isStrict() {
227        return strict;
228    }
229
230    /**
231     * Sets whether or not the children of this node are strictly ordered.
232     *
233     * @param strict True to order children strictly.
234     * @return This, to allow chaining.
235     */
236    public TraceNode setStrict(boolean strict) {
237        this.strict = strict;
238        return this;
239    }
240
241    /**
242     * Returns whether or not a note is assigned to this node.
243     *
244     * @return True if a note is assigned.
245     */
246    public boolean hasNote() {
247        return note != null;
248    }
249
250    /**
251     * Returns the note assigned to this node.
252     *
253     * @return The note.
254     */
255    public String getNote() {
256        return note;
257    }
258
259    /**
260     * Returns the number of child nodes of this.
261     *
262     * @return The number of children.
263     */
264    public int getNumChildren() {
265        return children.size();
266    }
267
268    /**
269     * Returns the child trace node at the given index.
270     *
271     * @param i The index of the child to return.
272     * @return The child at the given index.
273     */
274    public TraceNode getChild(int i) {
275        return children.get(i);
276    }
277
278    /**
279     * Convenience method to add a child node containing a note to this.
280     *
281     * @param note The note to assign to the child.
282     * @return This, to allow chaining.
283     */
284    public TraceNode addChild(String note) {
285        return addChild(new TraceNode(note));
286    }
287
288    /**
289     * Adds a child node to this.
290     *
291     * @param child The child to add.
292     * @return This, to allow chaining.
293     */
294    public TraceNode addChild(TraceNode child) {
295        if (note != null) {
296            throw new IllegalStateException("Nodes with notes are leaf nodes, you can not add children to it.");
297        }
298        TraceNode node = new TraceNode(child);
299        node.parent = this;
300        children.add(node);
301        return this;
302    }
303
304    /**
305     * Adds a list of child nodes to this.
306     *
307     * @param children The children to add.
308     * @return This, to allow chaining.
309     */
310    public TraceNode addChildren(List<TraceNode> children) {
311        for (TraceNode child : children) {
312            addChild(child);
313        }
314        return this;
315    }
316
317    // Overrides Object.
318    @Override
319    public String toString() {
320        return toString(Integer.MAX_VALUE);
321    }
322
323    /**
324     * Generates a non-parseable, human-readable string representation
325     * of this trace node.
326     *
327     * @return generated string
328     * @param limit soft limit for maximum string size
329     **/
330    public String toString(int limit) {
331        StringBuilder out = new StringBuilder();
332        if (!writeString(out, "", limit)) {
333            out.append("...\n");
334        }
335        return out.toString();
336    }
337
338    /**
339     * Writes a non-parseable, human-readable string representation of
340     * this trace node to the given string builder using the given
341     * indent string for every written line.
342     *
343     * @return false if written string was capped
344     * @param ret    The string builder to write to.
345     * @param indent The indent to use.
346     * @param limit soft limit for maximum string size
347     */
348    private boolean writeString(StringBuilder ret, String indent, int limit) {
349        if (ret.length() >= limit) {
350            return false;
351        }
352        if (note != null) {
353            ret.append(indent).append(note).append("\n");
354        } else {
355            String name = isStrict() ? "trace" : "fork";
356            ret.append(indent).append("<").append(name).append(">\n");
357            for (TraceNode child : children) {
358                if (!child.writeString(ret, indent + "    ", limit)) {
359                    return false;
360                }
361            }
362            if (ret.length() >= limit) {
363                return false;
364            }
365            ret.append(indent).append("</").append(name).append(">\n");
366        }
367        return true;
368    }

370    /**
371     * Returns a parseable (using {@link #decode(String)}) string representation of this trace node.
372     *
373     * @return A string representation of this tree.
374     */
375    public String encode() {
376        StringBuilder ret = new StringBuilder();
377        encode(ret);
378        return ret.toString();
379    }
380
381    /**
382     * Writes a parseable string representation of this trace node to the given string builder.
383     *
384     * @param ret The string builder to write to.
385     */
386    private void encode(StringBuilder ret) {
387        if (note != null) {
388            ret.append("[");
389            for (int i = 0, len = note.length(); i < len; ++i) {
390                char c = note.charAt(i);
391                if (c == '\\' || c == ']') {
392                    ret.append('\\');
393                }
394                ret.append(note.charAt(i));
395            }
396            ret.append("]");
397        } else {
398            ret.append(strict ? "(" : "{");
399            for (TraceNode child : children) {
400                child.encode(ret);
401            }
402            ret.append(strict ? ")" : "}");
403        }
404    }
405
406    /**
407     * Build a trace tree from the given string representation (possibly encoded using {@link #encode()}).
408     *
409     * @param str The string to parse.
410     * @return The corresponding trace tree, or an empty node if parsing failed.
411     */
412    public static TraceNode decode(String str) {
413        if (str == null || str.isEmpty()) {
414            return new TraceNode();
415        }
416        TraceNode proxy = new TraceNode();
417        TraceNode node = proxy;
418        StringBuilder note = null;
419        boolean inEscape = false;
420        for (int i = 0, len = str.length(); i < len; ++i) {
421            char c = str.charAt(i);
422            if (note != null) {
423                if (inEscape) {
424                    note.append(c);
425                    inEscape = false;
426                } else if (c == '\\') {
427                    inEscape = true;
428                } else if (c == ']') {
429                    node.addChild(note.toString());
430                    note = null;
431                } else {
432                    note.append(c);
433                }
434            } else {
435                if (c == '[') {
436                    note = new StringBuilder();
437                } else if (c == '(' || c == '{') {
438                    node.addChild(new TraceNode());
439                    node = node.getChild(node.getNumChildren() - 1);
440                    node.setStrict(c == '(');
441                } else if (c == ')' || c == '}') {
442                    if (node == null) {
443                        log.log(LogLevel.WARNING, "Unexpected closing brace in trace '" + str + "' at position " + i + ".");
444                        return new TraceNode();
445                    }
446                    if (node.isStrict() != (c == ')')) {
447                        log.log(LogLevel.WARNING, "Mismatched closing brace in trace '" + str + "' at position " + i + ".");
448                        return new TraceNode();
449                    }
450                    node = node.parent;
451                }
452            }
453        }
454        if (note != null) {
455            log.log(LogLevel.WARNING, "Unterminated note in trace '" + str + "'.");
456            return new TraceNode();
457        }
458        if (node != proxy) {
459            log.log(LogLevel.WARNING, "Missing closing brace in trace '" + str + "'.");
460            return new TraceNode();
461        }
462        if (proxy.getNumChildren() == 0) {
463            log.log(LogLevel.WARNING, "No nodes found in trace '" + str + "'.");
464            return new TraceNode();
465        }
466        if (proxy.getNumChildren() != 1) {
467            return proxy; // best-effort recovery from malformed input
468        }
469        TraceNode ret = proxy.children.remove(0);
470        ret.parent = null;
471        return ret;
472    }
