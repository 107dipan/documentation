002package com.yahoo.messagebus;
003
004import com.yahoo.messagebus.network.Network;
005
006import java.util.concurrent.atomic.AtomicBoolean;
007
008/**
009 * The combination of a messagebus and a network over which it may send data.
010 *
011 * @author bratseth
012 */
013public class NetworkMessageBus {
014
015    private final Network network;
016    private final MessageBus messageBus;
017
018    private final AtomicBoolean destroyed = new AtomicBoolean(false);
019
020    public NetworkMessageBus(Network network, MessageBus messageBus) {
021        this.network = network;
022        this.messageBus = messageBus;
023    }
024
025    /** Returns the contained message bus object */
026    public MessageBus getMessageBus() { return messageBus; }
027
028    /** Returns the network of this as a Network */
029    public Network getNetwork() { return network; }
030
031    /**
032     * Irreversibly destroys the content of this.
033     *
034     * @return whether this destroyed anything, or if it was already destroyed
035     */
036    public boolean destroy() {
037        if ( destroyed.getAndSet(true)) return false;
038
039        getMessageBus().destroy();
040        return true;
041    }
042
043}
