002package com.yahoo.messagebus;
003
004import com.yahoo.text.Utf8String;
005
006/**
007 * The empty reply is the only concrete implementation of a message that is offered by the MessageBus. It is used to
008 * generate replies to events that occur within the messagebus, and since the messagebus by design knows nothing about
009 * the messages that have been implemented by the users it requires a class such as this.
010 *
011 * @author Simon Thoresen
012 */
013public final class EmptyReply extends Reply {
014
015    private final Utf8String PROTOCOL = new Utf8String("");
016
017    /**
018     * Implements the getType() function of the root class Routable to identify this reply as the reserved type '0'.
019     *
020     * @return The number '0'.
021     */
022    public int getType() {
023        return 0;
024    }
025
026    /**
027     * Implements the getProtocol() function of Routable to identify this reply as the reserved type. This is done by an
028     * empty string.
029     *
030     * @return The string "".
031     */
032    public Utf8String getProtocol() {
033        return PROTOCOL;
034    }
035
036}
