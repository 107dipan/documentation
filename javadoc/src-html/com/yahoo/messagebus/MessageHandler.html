002package com.yahoo.messagebus;
003
004/**
005 * All classes that wants to handle messages that move through the messagebus need to implement this interface.
006 * As opposed to the {@link ReplyHandler} which handles replies as they return from the receiver to the sender, this
007 * interface is intended for handling messages as they travel from the sender to the receiver.
008 *
009 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
010 */
011public interface MessageHandler {
012
013    /**
014     * This function is called when a message arrives.
015     *
016     * @param message The message that arrived.
017     */
018    public void handleMessage(Message message);
019}
