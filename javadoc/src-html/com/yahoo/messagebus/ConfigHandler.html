002package com.yahoo.messagebus;
003
004import com.yahoo.messagebus.routing.RoutingSpec;
005
006/**
007 * This class declares those methods required to be a handler for an instance of the {@link ConfigAgent} class.
008 * Instead of declaring separate subscribers and handlers for all types of configurations, this pair is intended to hold
009 * everything. Extend this handler whenever new configs are added to {@link ConfigAgent}.
010 *
011 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
012 */
013public interface ConfigHandler {
014
015    /**
016     * Sets the routing specification for this client. This will be done synchronously during initialization, and then
017     * subsequently whenever an updated configuration is available.
018     *
019     * @param spec The routing specification.
020     */
021    public void setupRouting(RoutingSpec spec);
022}
