002package com.yahoo.messagebus;
003
004/**
005 * This interface contains the reserved error codes that are used for errors that occur within the messagebus.
006 *
007 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
008 */
009public final class ErrorCode {
010
011    /** The code is here for completeness. */
012    public static final int NONE                   = 0;
013
014    /** A general transient error, resending is possible. */
015    public static final int TRANSIENT_ERROR        = 100000;
016
017    /** Sending was rejected because throttler capacity is full. */
018    public static final int SEND_QUEUE_FULL        = TRANSIENT_ERROR + 1;
019
020    /** No addresses found for the services of the message route. */
021    public static final int NO_ADDRESS_FOR_SERVICE = TRANSIENT_ERROR + 2;
022
023    /** A connection problem occured while sending. */
024    public static final int CONNECTION_ERROR       = TRANSIENT_ERROR + 3;
025
026    /** The session specified for the message is unknown. */
027    public static final int UNKNOWN_SESSION        = TRANSIENT_ERROR + 4;
028
029    /** The recipient session is busy. */
030    public static final int SESSION_BUSY           = TRANSIENT_ERROR + 5;
031
032    /** Sending aborted by route verification. */
033    public static final int SEND_ABORTED           = TRANSIENT_ERROR + 6;
034
035    /** Version handshake failed for any reason. */
036    public static final int HANDSHAKE_FAILED       = TRANSIENT_ERROR + 7;
037
038    /** An application specific transient error. */
039    public static final int APP_TRANSIENT_ERROR    = TRANSIENT_ERROR + 50000;
040
041    /** A general non-recoverable error, resending is not possible. */
042    public static final int FATAL_ERROR            = 200000;
043
044    /** Sending was rejected because throttler is closed. */
045    public static final int SEND_QUEUE_CLOSED      = FATAL_ERROR + 1;
046
047    /** The route of the message is illegal. */
048    public static final int ILLEGAL_ROUTE          = FATAL_ERROR + 2;
049
050    /** No services found for the message route. */
051    public static final int NO_SERVICES_FOR_ROUTE  = FATAL_ERROR + 3;
052
053    /** The selected service was out of service. */
054    public static final int SERVICE_OOS            = FATAL_ERROR + 4;
055
056    /** An error occured while encoding the message. */
057    public static final int ENCODE_ERROR           = FATAL_ERROR + 5;
058
059    /** A fatal network error occured while sending. */
060    public static final int NETWORK_ERROR          = FATAL_ERROR + 6;
061
062    /** The protocol specified for the message is unknown. */
063    public static final int UNKNOWN_PROTOCOL       = FATAL_ERROR + 7;
064
065    /** An error occured while decoding the message. */
066    public static final int DECODE_ERROR           = FATAL_ERROR + 8;
067
068    /** A timeout occured while sending. */
069    public static final int TIMEOUT                = FATAL_ERROR + 9;
070
071    /** The target is running an incompatible version. */
072    public static final int INCOMPATIBLE_VERSION   = FATAL_ERROR + 10;
073
074    /** The policy specified in a route is unknown. */
075    public static final int UNKNOWN_POLICY         = FATAL_ERROR + 11;
076
077    /** The network was shut down when attempting to send. */
078    public static final int NETWORK_SHUTDOWN       = FATAL_ERROR + 12;
079
080    /** Exception thrown by routing policy. */
081    public static final int POLICY_ERROR           = FATAL_ERROR + 13;
082
083    /** An error occured while sequencing a message. */
084    public static final int SEQUENCE_ERROR         = FATAL_ERROR + 14;
085
086    /** An application specific non-recoverable error. */
087    public static final int APP_FATAL_ERROR        = FATAL_ERROR + 50000;
088
089    /** No error codes are allowed to be this big. */
090    public static final int ERROR_LIMIT            = APP_FATAL_ERROR + 50000;
091
092    /**
093     * Translates the given error code into its symbolic name.
094     *
095     * @param error The error code to translate.
096     * @return The symbolic name.
097     */
098    public static String getName(int error) {
099        switch (error) {
100        case APP_FATAL_ERROR        : return "APP_FATAL_ERROR";
101        case APP_TRANSIENT_ERROR    : return "APP_TRANSIENT_ERROR";
102        case CONNECTION_ERROR       : return "CONNECTION_ERROR";
103        case DECODE_ERROR           : return "DECODE_ERROR";
104        case ENCODE_ERROR           : return "ENCODE_ERROR";
105        case FATAL_ERROR            : return "FATAL_ERROR";
106        case HANDSHAKE_FAILED       : return "HANDSHAKE_FAILED";
107        case ILLEGAL_ROUTE          : return "ILLEGAL_ROUTE";
108        case INCOMPATIBLE_VERSION   : return "INCOMPATIBLE_VERSION";
109        case NETWORK_ERROR          : return "NETWORK_ERROR";
110        case NETWORK_SHUTDOWN       : return "NETWORK_SHUTDOWN";
111        case NO_ADDRESS_FOR_SERVICE : return "NO_ADDRESS_FOR_SERVICE";
112        case NO_SERVICES_FOR_ROUTE  : return "NO_SERVICES_FOR_ROUTE";
113        case NONE                   : return "NONE";
114        case POLICY_ERROR           : return "POLICY_ERROR";
115        case SEND_ABORTED           : return "SEND_ABORTED";
116        case SEND_QUEUE_CLOSED      : return "SEND_QUEUE_CLOSED";
117        case SEND_QUEUE_FULL        : return "SEND_QUEUE_FULL";
118        case SEQUENCE_ERROR         : return "SEQUENCE_ERROR";
119        case SERVICE_OOS            : return "SERVICE_OOS";
120        case SESSION_BUSY           : return "SESSION_BUSY";
121        case TIMEOUT                : return "TIMEOUT";
122        case TRANSIENT_ERROR        : return "TRANSIENT_ERROR";
123        case UNKNOWN_POLICY         : return "UNKNOWN_POLICY";
124        case UNKNOWN_PROTOCOL       : return "UNKNOWN_PROTOCOL";
125        case UNKNOWN_SESSION        : return "UNKNOWN_SESSION";
126        default                     : return "UNKNOWN(" + error + ")";
127        }
128    }
129}
