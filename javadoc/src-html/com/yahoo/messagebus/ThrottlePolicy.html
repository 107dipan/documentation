002package com.yahoo.messagebus;
003
004/**
005 * An implementation of this interface is used by {@link SourceSession} to throttle output. Every message entering
006 * {@link SourceSession#send(Message)} needs to be accepted by this interface's {@link #canSend(Message, int)} method.
007 * All messages accepted are passed through the {@link #processMessage(Message)} method, and the corresponding replies
008 * are passed through the {@link #processReply(Reply)} method.
009 *
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
011 */
012public interface ThrottlePolicy {
013
014    /**
015     * Returns whether or not the given message can be sent according to the current state of this policy.
016     *
017     * @param msg          The message to evaluate.
018     * @param pendingCount The current number of pending messages.
019     * @return True to send the message.
020     */
021    public boolean canSend(Message msg, int pendingCount);
022
023    /**
024     * This method is called once for every message that was accepted by {@link #canSend(Message, int)} and sent.
025     *
026     * @param msg The message beint sent.
027     */
028    public void processMessage(Message msg);
029
030    /**
031     * This method is called once for every reply that is received.
032     *
033     * @param reply The reply received.
034     */
035    public void processReply(Reply reply);
036}
