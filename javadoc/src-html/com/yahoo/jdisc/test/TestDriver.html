025/**
026 * <p>This class provides a unified way to set up and run unit tests on jDISC components. In short, it is a programmable
027 * {@link BootstrapLoader} that provides convenient access to the {@link ContainerActivator} and {@link
028 * CurrentContainer} interfaces. A typical test case using this class looks as follows:</p>
029 * <pre>
030 * {@literal @}Test
031 * public void requireThatMyComponentIsWellBehaved() {
032 *     TestDriver driver = TestDriver.newSimpleApplicationInstanceWithoutOsgi();
033 *     ContainerBuilder builder = driver.newContainerBuilder();
034 *     (... configure builder ...)
035 *     driver.activateContainer(builder);
036 *     (... run tests ...)
037 *     assertTrue(driver.close());
038 * }
039 * </pre>
040 * <p>One of the most important things to remember when using this class is to always call {@link #close()} at the end
041 * of your test case. This ensures that the tested configuration does not prevent graceful shutdown. If close() returns
042 * FALSE, it means that either your components or the test case itself does not conform to the reference counting
043 * requirements of {@link Request}, {@link RequestHandler}, {@link ContentChannel}, or {@link CompletionHandler}.</p>
044 *
045 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
046 */
047public class TestDriver implements ContainerActivator, CurrentContainer {

102    /**
103     * <p>Convenience method to create and {@link Request#connect(ResponseHandler)} a {@link Request} on the {@link
104     * CurrentContainer}. This method will either return the corresponding {@link ContentChannel} or throw the
105     * appropriate exception (see {@link Request#connect(ResponseHandler)}).</p>
106     *
107     * @param requestUri      The URI string to parse and pass to the Request constructor.
108     * @param responseHandler The ResponseHandler to pass to {@link Request#connect(ResponseHandler)}.
109     * @return The ContentChannel returned by {@link Request#connect(ResponseHandler)}.
110     * @throws NullPointerException     If the URI string or the {@link ResponseHandler} is null.
111     * @throws IllegalArgumentException If the URI string violates RFC&nbsp;2396.
112     * @throws BindingNotFoundException If the corresponding call to {@link Container#resolveHandler(Request)}
113     *                                  returns null.
114     * @throws RequestDeniedException   If the corresponding call to {@link RequestHandler#handleRequest(Request,
115     *                                  ResponseHandler)} returns null.
116     */
117    public ContentChannel connectRequest(String requestUri, ResponseHandler responseHandler) {
118        return newRequestDispatch(requestUri, responseHandler).connect();
119    }
120
121    /**
122     * <p>Convenience method to create a {@link Request}, connect it to a {@link RequestHandler}, and close the returned
123     * {@link ContentChannel}. This is the same as calling:</p>
124     * <pre>
125     * connectRequest(uri, responseHandler).close(null);
126     * </pre>
127     *
128     * @param requestUri      The URI string to parse and pass to the Request constructor.
129     * @param responseHandler The ResponseHandler to pass to {@link Request#connect(ResponseHandler)}.
130     * @return A waitable Future that provides access to the corresponding {@link Response}.
131     * @throws NullPointerException     If the URI string or the {@link ResponseHandler} is null.
132     * @throws IllegalArgumentException If the URI string violates RFC&nbsp;2396.
133     * @throws BindingNotFoundException If the corresponding call to {@link Container#resolveHandler(Request)}
134     *                                  returns null.
135     * @throws RequestDeniedException   If the corresponding call to {@link RequestHandler#handleRequest(Request,
136     *                                  ResponseHandler)} returns null.
137     */
138    public Future<Response> dispatchRequest(String requestUri, ResponseHandler responseHandler) {
139        return newRequestDispatch(requestUri, responseHandler).dispatch();
140    }
141
142    /**
143     * <p>Initiates the shut down of this TestDriver in another thread. By doing this in a separate thread, it allows
144     * other code to monitor its progress. Unless you need the added monitoring capability, you should use {@link
145     * #close()} instead.</p>
146     *
147     * @see #awaitClose(long, TimeUnit)
148     */
149    public void scheduleClose() {
150        new Thread(closeTask, "TestDriver.Closer").start();
151    }
152
153    /**
154     * <p>Waits for shut down of this TestDriver to complete. This call must be preceded by a call to {@link
155     * #scheduleClose()}.</p>
156     *
157     * @param timeout The maximum time to wait.
158     * @param unit    The time unit of the timeout argument.
159     * @return True if shut down completed within the allocated time.
160     */
161    public boolean awaitClose(long timeout, TimeUnit unit) {
162        try {
163            closeTask.get(timeout, unit);
164            return true;
165        } catch (TimeoutException e) {
166            return false;
167        } catch (Exception e) {
168            throw e instanceof RuntimeException ? (RuntimeException)e : new RuntimeException(e);
169        }
170    }
171
172    /**
173     * <p>Initiatiates shut down of this TestDriver and waits for it to complete. If shut down fails to complete within
174     * 60 seconds, this method throws an exception.</p>
175     *
176     * @return True if shut down completed within the allocated time.
177     * @throws IllegalStateException If shut down failed to complete within the allocated time.
178     */
179    public boolean close() {
180        scheduleClose();
181        if ( ! awaitClose(600, TimeUnit.SECONDS)) {
182            throw new IllegalStateException("Application failed to terminate within allocated time.");
183        }
184        return true;
185    }

188     * <p>Creates a new {@link RequestDispatch} that dispatches a {@link Request} with the given URI and {@link
189     * ResponseHandler}.</p>
190     *
191     * @param requestUri      The uri of the Request to create.
192     * @param responseHandler The ResponseHandler to use for the dispather.
193     * @return The created RequestDispatch.
194     */
195    public RequestDispatch newRequestDispatch(final String requestUri, final ResponseHandler responseHandler) {
196        return new RequestDispatch() {
197
198            @Override
199            protected Request newRequest() {
200                return new Request(loader, URI.create(requestUri));
201            }
202
203            @Override
204            public ContentChannel handleResponse(Response response) {
205                return responseHandler.handleResponse(response);
206            }
207        };
208    }

339    private class CloseTask implements Callable<Boolean> {
340
341        @Override
342        public Boolean call() throws Exception {
343            loader.stop();
344            loader.destroy();
345            return true;
346        }
347    }
348
349    private static TestDriver newInstance(OsgiFramework osgiFramework, String bundleLocation, boolean privileged,
350                                          Iterable<? extends Module> guiceModules) {
351        ApplicationLoader loader = new ApplicationLoader(osgiFramework, guiceModules);
352        try {
353            loader.init(bundleLocation, privileged);
354        } catch (Exception e) {
355            throw e instanceof RuntimeException ? (RuntimeException)e : new RuntimeException(e);
356        }
357        try {
358            loader.start();
359        } catch (Exception e) {
360            loader.destroy();
361            throw e instanceof RuntimeException ? (RuntimeException)e : new RuntimeException(e);
362        }
363        return new TestDriver(loader);
364    }
365
366    private static List<Module> newModuleList(final Application app, final Class<? extends Application> appClass,
367                                              Module... guiceModules) {
368        List<Module> lst = new LinkedList<>();
369        lst.addAll(Arrays.asList(guiceModules));
370        lst.add(new AbstractModule() {
371
372            @Override
373            public void configure() {
374                AnnotatedBindingBuilder<Application> builder = bind(Application.class);
375                if (app != null) {
376                    builder.toInstance(app);
377                } else {
378                    builder.to(appClass);
379                }
380            }
381        });
382        return lst;
383    }
