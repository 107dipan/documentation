002package com.yahoo.jdisc;
003
004/**
005 * Utility class for working with {@link SharedResource}s and {@link ResourceReference}s.
006 *
007 * @author bakksjo
008 */
009public class References {
010    // Prevents instantiation.
011    private References() {
012    }
013
014    /**
015     * A {@link ResourceReference} that does nothing.
016     * Useful for e.g. testing of resource types when reference counting is not the focus.
017     */
018    public static final ResourceReference NOOP_REFERENCE = new ResourceReference() {
019        @Override
020        public void close() {
021        }
022    };
023
024    /**
025     * <p>Returns a {@link ResourceReference} that invokes {@link SharedResource#release()} on
026     * {@link ResourceReference#close() close}. Useful for treating the "main" reference of a {@link SharedResource}
027     * just as any other reference obtained by calling {@link SharedResource#refer()}. Example:</p>
028     * <pre>
029     *     final Request request = new Request(...);
030     *     try (final ResourceReference ref = References.fromResource(request)) {
031     *         ....
032     *     }
033     *     // The request will be released on exit from the try block.
034     * </pre>
035     *
036     * @param resource The resource to create a ResourceReference for.
037     * @return a ResourceReference whose close() method will call release() on the given resource.
038     */
039    public static ResourceReference fromResource(final SharedResource resource) {
040        return new ResourceReference() {
041            @Override
042            public void close() {
043                resource.release();
044            }
045        };
046    }
047}
