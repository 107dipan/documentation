002package com.yahoo.jdisc.application;
003
004import com.yahoo.jdisc.Container;
005
006/**
007 * <p>This interface defines the API for changing the active {@link Container} of a jDISC application. An instance of
008 * this class is typically injected into the {@link Application} constructor. If injection is unavailable due to an
009 * Application design, an instance of this class is also available as an OSGi service under the full ContainerActivator
010 * class name.</p>
011 *
012 * <p>This interface allows one to create and active a new Container. To do so, one has to 1) call {@link
013 * #newContainerBuilder()}, 2) configure the returned {@link ContainerBuilder}, and 3) pass the builder to the {@link
014 * #activateContainer(ContainerBuilder)} method.</p>
015 *
016 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
017 */
018public interface ContainerActivator {
019
020    /**
021     * <p>This method creates and returns a new {@link ContainerBuilder} object that has the necessary references to the
022     * application and its internal components.</p>
023     *
024     * @return The created builder.
025     */
026    public ContainerBuilder newContainerBuilder();
027
028    /**
029     * <p>Creates and activates a {@link Container} based on the provided {@link ContainerBuilder}. By providing a
030     * <em>null</em> argument, this method can be used to deactivate the current Container. The returned object can be
031     * used to schedule a cleanup task that is executed once the the deactivated Container has terminated.</p>
032     *
033     * @param builder The builder to activate.
034     * @return The previous container, if any.
035     * @throws ApplicationNotReadyException If this method is called before {@link Application#start()} or after {@link
036     *                                      Application#stop()}.
037     */
038    public DeactivatedContainer activateContainer(ContainerBuilder builder);
039}
