002package com.yahoo.jdisc.application;
003
004import com.google.common.collect.ImmutableList;
005import org.osgi.framework.Bundle;
006import org.osgi.framework.BundleException;
007
008import java.util.ArrayList;
009import java.util.Collection;
010import java.util.Collections;
011import java.util.List;
012
013/**
014 * <p>This exception is thrown by {@link OsgiFramework#installBundle(String)} if installation failed. Because </p>
015 *
016 * <p>Please see commentary on {@link OsgiFramework#installBundle(String)} and {@link
017 * OsgiFramework#startBundles(java.util.List, boolean)} for a description of exception-safety issues to consider when
018 * installing bundles that use the {@link OsgiHeader#PREINSTALL_BUNDLE} manifest instruction.</p>
019 *
020 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
021 */
022public final class BundleInstallationException extends BundleException {
023
024    private final List<Bundle> installedBundles;
025
026    public BundleInstallationException(Collection<Bundle> installedBundles, Throwable cause) {
027        super(cause.getMessage(), cause);
028        this.installedBundles = ImmutableList.copyOf(installedBundles);
029    }
030
031    public List<Bundle> installedBundles() {
032        return installedBundles;
033    }
034}
