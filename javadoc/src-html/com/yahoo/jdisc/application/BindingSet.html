002package com.yahoo.jdisc.application;
003
004import com.google.common.collect.ImmutableList;
005
006import java.net.URI;
007import java.util.Collection;
008import java.util.Collections;
009import java.util.Comparator;
010import java.util.Iterator;
011import java.util.LinkedList;
012import java.util.List;
013import java.util.Map;
014
015/**
016 * <p>This is an immutable set of ordered bindings from {@link UriPattern}s to some target type T. To create an instance
017 * of this class, you must 1) create a {@link BindingRepository}, 2) configure it using the {@link
018 * BindingRepository#bind(String, Object)} method, and finally 3) call {@link BindingRepository#activate()}.</p>
019 *
020 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
021 */
022public class BindingSet<T> implements Iterable<Map.Entry<UriPattern, T>>  {
023
024    public static final String DEFAULT = "default";
025    private final Collection<Map.Entry<UriPattern, T>> bindings;
026
027    BindingSet(Collection<Map.Entry<UriPattern, T>> bindings) {
028        this.bindings = sort(bindings);
029    }
030
031    /**
032     * <p>Resolves the binding that best matches (see commentary on {@link BindingRepository#bind(String, Object)}) the
033     * given {@link URI}, and returns a {@link BindingMatch} object that describes the match and contains the
034     * matched target. If there is no binding that matches the given URI, this method returns null.</p>
035     *
036     * @param uri The URI to match against the bindings in this set.
037     * @return A {@link BindingMatch} object describing the match found, or null if not found.
038     */
039    public BindingMatch<T> match(URI uri) {
040        for (Map.Entry<UriPattern, T> entry : bindings) {
041            UriPattern.Match match = entry.getKey().match(uri);
042            if (match != null) {
043                return new BindingMatch<>(match, entry.getValue());
044            }
045        }
046        return null;
047    }
048
049    /**
050     * <p>Resolves the binding that best matches (see commentary on {@link BindingRepository#bind(String, Object)}) the
051     * given {@link URI}, and returns that target. If there is no binding that matches the given URI, this method
052     * returns null.</p>
053     *
054     * <p>Apart from a <em>null</em>-guard, this is equal to <code>return match(uri).target()</code>.</p>
055     *
056     * @param uri The URI to match against the bindings in this set.
057     * @return The best matched target, or null.
058     * @see #match(URI)
059     */
060    public T resolve(URI uri) {
061        BindingMatch<T> match = match(uri);
062        if (match == null) {
063            return null;
064        }
065        return match.target();
066    }
067
068    @Override
069    public Iterator<Map.Entry<UriPattern, T>> iterator() {
070        return bindings.iterator();
071    }
072
073    private static <T> Collection<Map.Entry<UriPattern, T>> sort(Collection<Map.Entry<UriPattern, T>> unsorted) {
074        List<Map.Entry<UriPattern, T>> ret = new LinkedList<>(unsorted);
075        Collections.sort(ret, new Comparator<Map.Entry<UriPattern, ?>>() {
076
077            @Override
078            public int compare(Map.Entry<UriPattern, ?> lhs, Map.Entry<UriPattern, ?> rhs) {
079                return lhs.getKey().compareTo(rhs.getKey());
080            }
081        });
082        return ImmutableList.copyOf(ret);
083    }
084}
