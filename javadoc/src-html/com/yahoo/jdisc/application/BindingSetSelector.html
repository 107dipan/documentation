002package com.yahoo.jdisc.application;
003
004import com.google.inject.ImplementedBy;
005import com.google.inject.Module;
006import com.yahoo.jdisc.Container;
007import com.yahoo.jdisc.core.DefaultBindingSelector;
008import com.yahoo.jdisc.service.CurrentContainer;
009import com.yahoo.jdisc.service.NoBindingSetSelectedException;
010
011import java.net.URI;
012
013/**
014 * This interface defines the component that is used by the {@link CurrentContainer} to assign a {@link BindingSet} to a
015 * newly created {@link Container} based on the given {@link URI}. The default implementation of this interface returns
016 * {@link BindingSet#DEFAULT} regardless of input. To specify your own selector you need to {@link
017 * GuiceRepository#install(Module) install} a Guice {@link Module} that provides a binding for this interface.
018 *
019 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
020 */
021@ImplementedBy(DefaultBindingSelector.class)
022public interface BindingSetSelector {
023
024    /**
025     * Returns the name of the {@link BindingSet} to assign to the {@link Container} for the given {@link URI}. If this
026     * method returns <em>null</em>, the corresponding call to {@link CurrentContainer#newReference(URI)} will throw a
027     * {@link NoBindingSetSelectedException}.
028     *
029     * @param uri The URI to select on.
030     * @return The name of selected BindingSet.
031     */
032    public String select(URI uri);
033}
