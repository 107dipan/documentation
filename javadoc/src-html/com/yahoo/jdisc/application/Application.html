002package com.yahoo.jdisc.application;
003
004import com.yahoo.jdisc.Container;
005import com.yahoo.jdisc.core.ApplicationLoader;
006import com.yahoo.jdisc.service.ClientProvider;
007import com.yahoo.jdisc.service.ServerProvider;
008
009/**
010 * <p>This interface defines the API of the singleton Application that runs in a jDISC instance. An Application instance
011 * will always have its {@link #destroy()} method called, regardless of whether {@link #start()} or {@link #stop()}
012 * threw any exceptions.</p>
013 *
014 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
015 */
016public interface Application {
017
018    /**
019     * <p>This method is called by the {@link ApplicationLoader} just after creating this Application instance. Use this
020     * method to start the Application's worker thread, and to activate a {@link Container}. If you attempt to call
021     * {@link ContainerActivator#activateContainer(ContainerBuilder)} before this method is invoked, that call will
022     * throw an {@link ApplicationNotReadyException}. If this method does not throw an exception, the {@link #stop()}
023     * method will be called at some time in the future.</p>
024     */
025    void start();
026
027    /**
028     * <p>This method is called by the {@link ApplicationLoader} after the corresponding signal has been issued by the
029     * controlling start script. Once this method returns, all calls to {@link
030     * ContainerActivator#activateContainer(ContainerBuilder)} will throw {@link ApplicationNotReadyException}s. Use
031     * this method to prepare for termination (see {@link #destroy()}).</p>
032     */
033    void stop();
034
035    /**
036     * <p>This method is called by the {@link ApplicationLoader} after first calling {@link #stop()}, and all previous
037     * {@link DeactivatedContainer}s have terminated. Use this method to shut down all Application components such as
038     * {@link ClientProvider}s and {@link ServerProvider}s.</p>
039     */
040    void destroy();
041
042}
