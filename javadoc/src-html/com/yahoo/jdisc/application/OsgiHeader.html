002package com.yahoo.jdisc.application;
003
004import org.osgi.framework.Bundle;
005
006import java.util.List;
007
008/**
009 * This interface acts as a namespace for the supported OSGi bundle headers.
010 *
011 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
012 */
013public abstract class OsgiHeader {
014
015    public static final String APPLICATION = "X-JDisc-Application";
016    public static final String PREINSTALL_BUNDLE = "X-JDisc-Preinstall-Bundle";
017    public static final String PRIVILEGED_ACTIVATOR = "X-JDisc-Privileged-Activator";
018
019    /**
020     * Returns true if the named header is present in the manifest of the given bundle.
021     *
022     * @param bundle     The bundle whose manifest to check.
023     * @param headerName The name of the header to check for.
024     * @return True if header is present.
025     */
026    public static boolean isSet(Bundle bundle, String headerName) {
027        return Boolean.valueOf(String.valueOf(bundle.getHeaders().get(headerName)));
028    }
029
030    /**
031     * This method reads the named header from the manifest of the given bundle, and parses it as a comma-separated list
032     * of values. If the header is not set, this method returns an empty list.
033     *
034     * @param bundle     The bundle whose manifest to parse the header from.
035     * @param headerName The name of the header to parse.
036     * @return A list of parsed header values, may be empty.
037     */
038    public static List<String> asList(Bundle bundle, String headerName) {
039        return ContainerBuilder.safeStringSplit(bundle.getHeaders().get(headerName), ",");
040    }
041}
