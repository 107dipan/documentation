002package com.yahoo.jdisc;
003
004import com.yahoo.jdisc.handler.RequestHandler;
005
006import java.util.concurrent.TimeUnit;
007
008/**
009 * <p>This interface provides a callback for when the {@link Request#setTimeout(long, TimeUnit)} is invoked. If no such
010 * handler is registered at the time where the target {@link RequestHandler} is called, the default timeout manager will
011 * be injected.</p>
012 *
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
014 */
015public interface TimeoutManager {
016
017    /**
018     * Schedule timeout management for a request.
019     * This is called by a request whenever {@link Request#setTimeout(long, TimeUnit)} is invoked;
020     * this may be called multiple times for the same {@link Request}.
021     *
022     * @param request the request whose timeout to schedule
023     */
024    public void scheduleTimeout(Request request);
025
026    /**
027     * Unschedule timeout management for a previously scheduled request.
028     * This is called whenever a request is cancelled, and the purpose is to free up
029     * resources taken by the implementation of this associated with the request.
030     * <p>
031     * This is only called once for a request, and only after at least one scheduleTimeout call.
032     * <p>
033     * The default implementation of this does nothing.
034     *
035     * @param request the previously scheduled timeout
036     */
037    default public void unscheduleTimeout(Request request) {
038    }
039
040}
