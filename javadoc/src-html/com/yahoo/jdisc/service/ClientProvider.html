002package com.yahoo.jdisc.service;
003
004import com.yahoo.jdisc.Container;
005import com.yahoo.jdisc.application.*;
006import com.yahoo.jdisc.handler.RequestHandler;
007
008/**
009 * <p>This interface defines a component that is capable of acting as a client to an external server. To activate a
010 * ClientProvider it must be {@link BindingRepository#bind(String, Object) bound} to a {@link UriPattern} within a
011 * {@link ContainerBuilder}, and that builder must be {@link ContainerActivator#activateContainer(ContainerBuilder)
012 * activated}.</p>
013 *
014 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
015 */
016public interface ClientProvider extends RequestHandler {
017
018    /**
019     * <p>This is a synchronous method to configure this ClientProvider. The {@link Container} does <em>not</em> call
020     * this method, instead it is a required step in the {@link Application} initialization code.</p>
021     */
022    void start();
023
024}
