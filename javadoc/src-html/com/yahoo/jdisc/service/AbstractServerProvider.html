002package com.yahoo.jdisc.service;
003
004import com.google.inject.Inject;
005import com.yahoo.jdisc.AbstractResource;
006import com.yahoo.jdisc.Request;
007
008import java.util.Objects;
009
010/**
011 * <p>This is a convenient parent class for {@link ServerProvider} with default implementations for all but the
012 * essential {@link #start()} and {@link #close()} methods. It requires that the {@link CurrentContainer} is injected in
013 * the constructor, since that interface is needed to dispatch {@link Request}s.</p>
014 *
015 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
016 */
017public abstract class AbstractServerProvider extends AbstractResource implements ServerProvider {
018
019    private final CurrentContainer container;
020
021    @Inject
022    protected AbstractServerProvider(CurrentContainer container) {
023        Objects.requireNonNull(container, "container");
024        this.container = container;
025    }
026
027    public final CurrentContainer container() {
028        return container;
029    }
030}
