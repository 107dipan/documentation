002package com.yahoo.jdisc.handler;
003
004import com.yahoo.jdisc.Request;
005import com.yahoo.jdisc.service.ClientProvider;
006
007import java.net.URI;
008
009/**
010 * <p>This exception is used to signal that a {@link Request} was rejected by the corresponding {@link ClientProvider}
011 * or {@link RequestHandler}. There is no automation in throwing an instance of this class, but all RequestHandlers are
012 * encouraged to use this where appropriate.</p>
013 *
014 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
015 */
016public final class RequestDeniedException extends RuntimeException {
017
018    private final Request request;
019
020    /**
021     * <p>Constructs a new instance of this class with a detail message that contains the {@link URI} of the {@link
022     * Request} that was denied.</p>
023     *
024     * @param request The Request that was denied.
025     */
026    public RequestDeniedException(Request request) {
027        super("Request with URI '" + request.getUri() + "' denied.");
028        this.request = request;
029    }
030
031    /**
032     * <p>Returns the {@link Request} that was denied.</p>
033     *
034     * @return The Request that was denied.
035     */
036    public Request request() {
037        return request;
038    }
039}
