002package com.yahoo.jdisc.handler;
003
004import com.yahoo.jdisc.Response;
005
006import java.util.concurrent.Callable;
007
008/**
009 * <p>This is a convenient subclass of {@link RequestDispatch} that implements the {@link Callable} interface. This
010 * should be used in place of {@link RequestDispatch} if you intend to schedule its execution. Because {@link #call()}
011 * does not return until a {@link Response} becomes available, you can use the <tt>Future</tt> return value of
012 * <tt>ExecutorService.submit(Callable)</tt> to wait for it.</p>
013 *
014 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
015 */
016public abstract class CallableRequestDispatch extends RequestDispatch implements Callable<Response> {
017
018    @Override
019    public final Response call() throws Exception {
020        return dispatch().get();
021    }
022}
