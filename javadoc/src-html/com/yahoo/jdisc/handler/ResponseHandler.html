002package com.yahoo.jdisc.handler;
003
004import com.yahoo.jdisc.Request;
005import com.yahoo.jdisc.Response;
006import com.yahoo.jdisc.service.ClientProvider;
007
008/**
009 * <p>This interface defines a component that is capable of acting as a handler for a {@link Response}. An
010 * implementation of this interface is required to be passed alongside every {@link Request} as part of the API (see
011 * {@link ClientProvider#handleRequest(Request, ResponseHandler)} and {@link RequestHandler#handleRequest(Request,
012 * ResponseHandler)}).</p>
013 *
014 * <p>The jDISC API has intentionally been designed as not to provide a implicit reference from Response to
015 * corresponding Request, but rather leave that to the implementation of context-aware ResponseHandlers. By creating
016 * light-weight ResponseHandlers on a per-Request basis, any necessary reference can be embedded within.</p>
017 *
018 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
019 */
020public interface ResponseHandler {
021
022    /**
023     * <p>This method will process the given {@link Response} and return a {@link ContentChannel} into which the caller
024     * can write the Response's content.</p>
025     *
026     * @param response The Response to handle.
027     * @return The ContentChannel to write the Response content to. Notice that the ContentChannel holds a Container
028     *         reference, so failure to close this will prevent the Container from ever shutting down.
029     */
030    ContentChannel handleResponse(Response response);
031
032}
