002package com.yahoo.jdisc.handler;
003
004import com.google.common.util.concurrent.AbstractFuture;
005
006/**
007 * <p>This class provides an implementation of {@link CompletionHandler} that allows you to wait for either {@link
008 * #completed()} or {@link #failed(Throwable)} to be called. If failed() was called, the corresponding Throwable will
009 * be rethrown when calling either of the get() methods. Unless an exception is thrown, the get() methods will always
010 * return Boolean.TRUE.</p>
011 *
012 * <p>Notice that calling {@link #cancel(boolean)} throws an UnsupportedOperationException.</p>
013 *
014 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
015 */
016public final class FutureCompletion extends AbstractFuture<Boolean> implements CompletionHandler {
017
018    @Override
019    public void completed() {
020        set(true);
021    }
022
023    @Override
024    public void failed(Throwable t) {
025        setException(t);
026    }
027
028    @Override
029    public final boolean cancel(boolean mayInterruptIfRunning) {
030        throw new UnsupportedOperationException();
031    }
032
033    @Override
034    public final boolean isCancelled() {
035        return false;
036    }
037}
