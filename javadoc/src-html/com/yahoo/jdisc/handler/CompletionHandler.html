002package com.yahoo.jdisc.handler;
003
004import com.yahoo.jdisc.Container;
005
006/**
007 * This interface defines a handler for consuming the result of an asynchronous I/O operation.
008 * <p>
009 * The asynchronous channels defined in this package allow a completion handler to be specified to consume the result of
010 * an asynchronous operation. The {@link #completed()} method is invoked when the I/O operation completes successfully.
011 * The {@link #failed(Throwable)} method is invoked if the I/O operations fails. The implementations of these methods
012 * should complete in a timely manner so as to avoid keeping the invoking thread from dispatching to other completion
013 * handlers.
014 * <p>
015 * Because a CompletionHandler might have a completely different lifespan than the originating ContentChannel objects,
016 * all instances of this interface are internally backed by a reference to the {@link Container} that was active when
017 * the initial Request was created. This ensures that the configured environment of the CompletionHandler is stable
018 * throughout its lifetime. This also means that the either {@link #completed()} or {@link #failed(Throwable)} MUST be
019 * called in order to release that reference. Failure to do so will prevent the Container from ever shutting down.
020 *
021 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
022 */
023public interface CompletionHandler {
024
025    /**
026     * Invoked when an operation has completed. Notice that you MUST call either this or {@link #failed(Throwable)} to
027     * release the internal {@link Container} reference. Failure to do so will prevent the Container from ever shutting
028     * down.
029     */
030    public void completed();
031
032    /**
033     * Invoked when an operation fails. Notice that you MUST call either this or {@link #completed()} to release the
034     * internal {@link Container} reference. Failure to do so will prevent the Container from ever shutting down.
035     *
036     * @param t The exception to indicate why the I/O operation failed.
037     */
038    public void failed(Throwable t);
039}
