002package com.yahoo.jdisc;
003
004import com.google.inject.ImplementedBy;
005import com.yahoo.jdisc.core.SystemTimer;
006
007/**
008 * <p>This class provides access to the current time in milliseconds, as viewed by the {@link Container}. Inject an
009 * instance of this class into any component that needs to access time, instead of using
010 * <code>System.currentTimeMillis()</code>.</p>
011 *
012 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
013 */
014@ImplementedBy(SystemTimer.class)
015public interface Timer {
016
017    /**
018     * <p>Returns the current time in milliseconds. Note that while the unit of time of the return value is a
019     * millisecond, the granularity of the value depends on the underlying operating system and may be larger. For
020     * example, many operating systems measure time in units of tens of milliseconds.</p>
021     *
022     * <p> See the description of the class <code>Date</code> for a discussion of slight discrepancies that may arise
023     * between "computer time" and coordinated universal time (UTC).</p>
024     *
025     * @return The difference, measured in milliseconds, between the current time and midnight, January 1, 1970 UTC.
026     * @see java.util.Date
027     */
028    public long currentTimeMillis();
029}
