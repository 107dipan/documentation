050    public DiscFilterRequest(ServletOrJdiscHttpRequest parent) {
051        this.parent = parent;
052
053        //save untreated headers from parent
054        untreatedHeaders = new HeaderFields();
055        parent.copyHeaders(untreatedHeaders);
056
057        untreatedParams = new HashMap<>(parent.parameters());
058
059        int port = parent.getUri().getPort();
060        if(port < 0) {
061            port  = 0;
062        }
063        localAddress = new InetSocketAddress(parent.getUri().getHost(), port);
064    }

208    public long getDateHeader(String name) {
209        String value = getHeader(name);
210        if (value == null)
211            return -1L;
212
213        Date date = null;
214        for (int i = 0; (date == null) && (i < formats.length); i++) {
215            try {
216                date = formats[i].parse(value);
217            } catch (ParseException e) {
218            }
219        }
220        if (date == null) {
221            return -1L;
222        }
223
224        return date.getTime();
225    }

344    public boolean isUserInRole(String role) {
345        if(overrideIsUserInRole) {
346            if(roles != null) {
347                for (String role1 : roles) {
348                    if (role1 != null && role1.trim().length() > 0) {
349                        String userRole = role1.trim();
350                        if (userRole.equals(role)) {
351                            return true;
352                        }
353                    }
354                }
355            }
356            return false;
357        }
358        else {
359            return false;
360        }
361    }

410    public void addCookie(JDiscCookieWrapper cookie) {
411        if(cookie != null) {
412            List<Cookie> cookies = new ArrayList<Cookie>();
413            //Get current set of cookies first
414            List<Cookie> c = getCookies();
415            if(c != null && !c.isEmpty()) {
416                cookies.addAll(c);
417            }
418            cookies.add(cookie.getCookie());
419            setCookies(cookies);
420        }
421    }

425    public JDiscCookieWrapper[] getWrappedCookies() {
426        List<Cookie> cookies = getCookies();
427        if(cookies == null)  {
428            return null;
429        }
430        List<JDiscCookieWrapper> cookieWrapper = new ArrayList<>(cookies.size());
431        for(Cookie cookie : cookies) {
432            cookieWrapper.add(JDiscCookieWrapper.wrap(cookie));
433        }
434
435        return cookieWrapper.toArray(new JDiscCookieWrapper[cookieWrapper.size()]);
436    }

438    private String setCharsetFromContentType(String contentType,String charset) {
439        String newContentType = "";
440        if (contentType == null)
441            return (null);
442        int start = contentType.indexOf("charset=");
443        if (start < 0) {
444            //No charset present:
445            newContentType = contentType + ";charset=" + charset;
446            return newContentType;
447        }
448        String encoding = contentType.substring(start + 8);
449        int end = encoding.indexOf(';');
450        if (end >= 0) {
451            newContentType = contentType.substring(0,start);
452            newContentType = newContentType + "charset=" + charset;
453            newContentType = newContentType + encoding.substring(end,encoding.length());
454        }
455        else {
456            newContentType = contentType.substring(0,start);
457            newContentType = newContentType + "charset=" + charset;
458        }
459
460        return (newContentType.trim());
461
462    }
463
464    private String getCharsetFromContentType(String contentType) {
465
466        if (contentType == null)
467            return (null);
468        int start = contentType.indexOf("charset=");
469        if (start < 0)
470            return (null);
471        String encoding = contentType.substring(start + 8);
472        int end = encoding.indexOf(';');
473        if (end >= 0)
474            encoding = encoding.substring(0, end);
475        encoding = encoding.trim();
476        if ((encoding.length() > 2) && (encoding.startsWith("\""))
477                && (encoding.endsWith("\"")))
478            encoding = encoding.substring(1, encoding.length() - 1);
479        return (encoding.trim());
480
481    }

483    public static boolean isMultipart(DiscFilterRequest request) {
484        if (request == null) {
485            return false;
486        }
487
488        String contentType = request.getContentType();
489
490        if (contentType == null) {
491            return false;
492        }
493
494        String[] parts = Pattern.compile(";").split(contentType);
495        if (parts.length == 0) {
496            return false;
497        }
498
499        for (String part : parts) {
500            if ("multipart/form-data".equals(part)) {
501                return true;
502            }
503        }
504
505        return false;
506    }

516    /**
517     * The set of SimpleDateFormat formats to use in getDateHeader().
518     *
519     * Notice that because SimpleDateFormat is not thread-safe, we can't declare
520     * formats[] as a static variable.
521     */
522    protected static final class ThreadLocalSimpleDateFormat extends
523            ThreadLocal<SimpleDateFormat> {
524        private final String format;
525        private final Locale locale;
526
527        public ThreadLocalSimpleDateFormat(String format, Locale locale) {
528            super();
529            this.format = format;
530            this.locale = locale;
531        }
532
533        // @see java.lang.ThreadLocal#initialValue()
534        @Override
535        protected SimpleDateFormat initialValue() {
536            return new SimpleDateFormat(format, locale);
537        }
538
539        public Date parse(String value) throws ParseException {
540            return get().parse(value);
541        }
