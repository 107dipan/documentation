002package com.yahoo.jdisc.http.filter;
003
004import java.util.concurrent.TimeUnit;
005
006import com.yahoo.jdisc.http.Cookie;
007
008/**
009 * Wrapper of Cookie.
010 *
011 * @author tejalk
012 *
013 */
014public class JDiscCookieWrapper {
015
016    private Cookie cookie;
017
018    protected JDiscCookieWrapper(Cookie cookie) {
019        this.cookie = cookie;
020    }
021
022    public static JDiscCookieWrapper wrap(Cookie cookie) {
023        return new JDiscCookieWrapper(cookie);
024    }
025
026    public String getComment() {
027        return cookie.getComment();
028    }
029
030    public String getDomain() {
031        return cookie.getDomain();
032    }
033
034    public int getMaxAge() {
035        return cookie.getMaxAge(TimeUnit.SECONDS);
036    }
037
038    public String getName() {
039        return cookie.getName();
040    }
041
042    public String getPath() {
043        return cookie.getPath();
044    }
045
046    public boolean getSecure() {
047        return cookie.isSecure();
048    }
049
050    public String getValue() {
051        return cookie.getValue();
052    }
053
054    public int getVersion() {
055        return cookie.getVersion();
056    }
057
058    public void setComment(String purpose) {
059        cookie.setComment(purpose);
060    }
061
062    public void setDomain(String pattern) {
063        cookie.setDomain(pattern);
064    }
065
066    public void setMaxAge(int expiry) {
067        cookie.setMaxAge(expiry, TimeUnit.SECONDS);
068    }
069
070    public void setPath(String uri) {
071        cookie.setPath(uri);
072    }
073
074    public void setSecure(boolean flag) {
075      cookie.setSecure(flag);
076    }
077
078    public void setValue(String newValue) {
079       cookie.setValue(newValue);
080    }
081
082    public void setVersion(int version) {
083       cookie.setVersion(version);
084    }
085
086    /**
087     * Return com.yahoo.jdisc.http.Cookie
088     *
089     * @return - cookie
090     */
091    public Cookie getCookie() {
092       return cookie;
093    }
094
095}
