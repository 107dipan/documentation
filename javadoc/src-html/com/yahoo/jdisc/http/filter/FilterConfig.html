002package com.yahoo.jdisc.http.filter;
003
004import java.util.Collection;
005
006/**
007 * Legacy filter config. Prefer to use a regular stringly typed config class for new filters.
008 *
009 * @author tejalk
010 */
011public interface FilterConfig {
012
013    /** Returns the filter-name of this filter */
014    String getFilterName();
015
016    /** Returns the filter-class of this filter */
017    String getFilterClass();
018
019    /**
020     * Returns a String containing the value of the
021     * named initialization parameter, or null if
022     * the parameter does not exist.
023     *
024     * @param name      a String specifying the name of the initialization parameter
025     * @return a String containing the value of the initialization parameter
026     */
027    String getInitParameter(String name);
028
029    /**
030     * Returns the boolean value of the init parameter. If not present returns default value
031     *
032     * @return boolean value of init parameter
033     */
034    boolean getBooleanInitParameter(String name, boolean defaultValue);
035
036    /**
037     * Returns the names of the filter's initialization parameters as an Collection of String objects,
038     * or an empty Collection if the filter has no initialization parameters.
039     */
040    Collection<String> getInitParameterNames();
041
042}
