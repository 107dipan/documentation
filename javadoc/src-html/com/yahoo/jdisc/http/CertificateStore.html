002package com.yahoo.jdisc.http;
003
004/**
005 * A store of certificates. An implementation can be plugged in to provide certificates to components who use it.
006 *
007 * @author bratseth
008 */
009public interface CertificateStore {
010
011    /** Returns a certificate for a given appid, using the default TTL and retry time */
012    default String getCertificate(String appid) { return getCertificate(appid, 0L, 0L); }
013
014    /** Returns a certificate for a given appid, using a TTL and default retry time */
015    default String getCertificate(String appid, long ttl) { return getCertificate(appid, ttl, 0L); }
016
017    /**
018     * Returns a certificate for a given appid, using a TTL and default retry time
019     *
020     * @param ttl certificate TTL in ms. Use the default TTL if set to 0
021     * @param retry if no certificate is found, allow access to cert DB again in
022     *              "retry" ms. Use the default retry time if set to 0.
023     */
024    String getCertificate(String appid, long ttl, long retry);
025
026}
