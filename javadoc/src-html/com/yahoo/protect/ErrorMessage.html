002package com.yahoo.protect;
003
004
005/**
006 * An error message with a code.
007 * This class should be treated as immutable.
008 *
009 * @author bratseth
010 */
011public class ErrorMessage {
012
013    /** An error code */
014    protected int code;
015
016    /** The short message of this error, always set */
017    protected String message;
018
019    /** The detailed instance message of this error, not always set */
020    protected String detailedMessage = null;
021
022    /** The cause of this error, or null if none is recorded */
023    protected Throwable cause = null;
024
025    /**
026     * Create an invalid instance for a subclass to initialize.
027     */
028    public ErrorMessage() {
029    }
030
031    public ErrorMessage(int code, String message) {
032        this.code = code;
033        this.message = message;
034    }
035
036    /**
037     * Create an application specific error message with an application
038     * specific code
039     */
040    public ErrorMessage(int code, String message, String detailedMessage) {
041        this(code, message);
042        this.detailedMessage = detailedMessage;
043    }
044
045    /** Create an application specific error message with an application specific code */
046    public ErrorMessage(int code, String message, String detailedMessage, Throwable cause) {
047        this(code, message, detailedMessage);
048        this.cause = cause;
049    }
050
051    public int getCode() {
052        return code;
053    }
054
055    public String getMessage() {
056        return message;
057    }
058
059    /** Returns the detailed message, or null if there is no detailed message */
060    public String getDetailedMessage() {
061        return detailedMessage;
062    }
063   /**
064     * Sets the cause of this. This should be set on errors which likely have their origin in plugin component code,
065     * not on others.
066     */
067    public void setCause(Throwable cause) { this.cause=cause; }
068
069    /** Returns the cause of this, or null if none is set */
070    public Throwable getCause() { return cause; }
071
072    public int hashCode() {
073        return code * 7 + message.hashCode() + (detailedMessage == null ? 0 : 17 * detailedMessage.hashCode());
074    }
075
076    /**
077     * Two error messages are equal if they have the same code and message.
078     * The cause is ignored in the comparison.
079     */
080    @Override
081    public boolean equals(Object o) {
082        if (!(o instanceof ErrorMessage)) return false;
083
084        ErrorMessage other = (ErrorMessage) o;
085
086        if (this.code != other.code) return false;
087
088        if (!this.message.equals(other.message)) return false;
089
090        if (this.detailedMessage==null) return other.detailedMessage==null;
091        if (other.detailedMessage==null) return false;
092
093        return this.detailedMessage.equals(other.detailedMessage);
094    }
095
096    @Override
097    public String toString() {
098        String details = "";
099
100        if (detailedMessage != null) {
101            details = detailedMessage;
102        }
103        if (cause !=null) {
104            if (details.length()>0)
105                details+=": ";
106            details+= com.yahoo.yolean.Exceptions.toMessageString(cause);
107        }
108        if (details.length()>0)
109            details=" (" + details + ")";
110
111        return "error : " + message + details;
112    }
113
114}
